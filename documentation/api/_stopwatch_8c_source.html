50void Stopwatch::reset()
51{
52 this->interrupts = 0;
53 this->milliSeconds = 0;
54 this->timerCounter = TimerManager::getTimerCounter();
55 this->timeProportion = TimerManager::getTargetTimePerInterruptInMS() / (float)this->timerCounter;
56 this->previousTimerCounter = this->timerCounter;
57}

68float Stopwatch::lap()
69{
70 extern uint8* const _hardwareRegisters;
71
72 TimerManager::disable();
73
74 uint16 currentTimerCounter = (_hardwareRegisters[__THR] << 8 ) | _hardwareRegisters[__TLR];
75
76 uint16 timerCounter = 0;
77
78 if(0 == this->interrupts)
79 {
80 if(currentTimerCounter > this->previousTimerCounter)
81 {
82 timerCounter = VUEngine::getGameFrameDuration() / this->timeProportion;
83 }
84 else
85 {
86 timerCounter = this->previousTimerCounter - currentTimerCounter;
87 }
88 }
89 else if(1 == this->interrupts)
90 {
91 timerCounter = this->previousTimerCounter + (this->timerCounter - currentTimerCounter);
92 }
93 else
94 {
95 timerCounter = this->previousTimerCounter + (this->timerCounter - currentTimerCounter);
96 timerCounter += (this->interrupts - 1) * this->timerCounter;
97 }
98
99 float elapsedTime = timerCounter * this->timeProportion;
100
101 this->interrupts = 0;
102
103 this->previousTimerCounter = currentTimerCounter;
104
105 this->milliSeconds += elapsedTime;
106
107 TimerManager::enable();
108
109 return elapsedTime;
110}
