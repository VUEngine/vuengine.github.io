203 fixed_t cameraNearPlane; // logical distance between the eyes and the screen

205 fixed_t horizontalViewPointCenter; // horizontal View point center
206 fixed_t verticalViewPointCenter; // vertical View point center

216 fixed_ext_t scalingMultiplier; // scaling multiplier
