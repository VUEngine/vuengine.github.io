45void Asterisk::render(Vector3D relativePosition)
46{
47 NM_ASSERT(NULL != this->transformation, "Asterisk::render: NULL transformation");
48
49 relativePosition = Vector3D::rotate(relativePosition, _previousCameraInvertedRotation);
50 this->position = PixelVector::projectVector3D(relativePosition, Optics::calculateParallax(relativePosition.z));
51 this->scaledLength = __METERS_TO_PIXELS(__FIXED_MULT(this->length, Vector3D::getScale(relativePosition.z, false)));
52}

56bool Asterisk::draw()
57{
58 NM_ASSERT(NULL != this->transformation, "Asterisk::draw: NULL transformation");
59
60 bool drawn = false;
61
62 if(this->bufferIndex)
63 {
64 drawn = DirectDraw::drawX(this->position, this->scaledLength, this->color, this->bufferIndex, this->interlaced);
65 }
66 else
67 {
68 drawn = DirectDraw::drawCross(this->position, this->scaledLength, this->color, this->bufferIndex, this->interlaced);
69 }
70
71 this->bufferIndex = !this->bufferIndex;
72
73 return drawn;
74}
