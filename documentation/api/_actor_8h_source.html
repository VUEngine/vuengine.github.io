315static inline bool
316Actor::isInsideFrustrum(Vector3D vector3D, RightBox rightBox)
317{
318 extern const CameraFrustum* _cameraFrustum;
319 vector3D = Vector3D::rotate(Vector3D::getRelativeToCamera(vector3D), *_cameraInvertedRotation);
320
321 if
322 (
323 vector3D.x + rightBox.x0 > __PIXELS_TO_METERS(_cameraFrustum->x1)
324 ||
325 vector3D.x + rightBox.x1 < __PIXELS_TO_METERS(_cameraFrustum->x0)
326 )
327 {
328 return false;
329 }
330
331 // Check y visibility
332 if
333 (
334 vector3D.y + rightBox.y0 > __PIXELS_TO_METERS(_cameraFrustum->y1)
335 ||
336 vector3D.y + rightBox.y1 < __PIXELS_TO_METERS(_cameraFrustum->y0)
337 )
338 {
339 return false;
340 }
341
342 // Check z visibility
343 if
344 (
345 vector3D.z + rightBox.z0 > __PIXELS_TO_METERS(_cameraFrustum->z1)
346 ||
347 vector3D.z + rightBox.z1 < __PIXELS_TO_METERS(_cameraFrustum->z0))
348 {
349 return false;
350 }
351
352 return true;
353}
