37static bool SoundManager::playSound
38(
39 const SoundSpec* soundSpec, const Vector3D* position, uint32 playbackType, EventListener soundReleaseListener, ListenerObject scope
40)
41{
42 SoundManager soundManager = SoundManager::getInstance();
43
44 if(soundManager->lock || NULL == soundSpec)
45 {
46 return false;
47 }
48
49 Sound sound = SoundManager::doGetSound(soundManager, soundSpec, soundReleaseListener, scope);
50
51 if(!isDeleted(sound))
52 {
53 Sound::autoReleaseOnFinish(sound, true);
54 Sound::play(sound, position, playbackType);
55
56 return true;
57 }
58
59 return false;
60}

78static Sound SoundManager::findSound(const SoundSpec* soundSpec, EventListener soundReleaseListener, ListenerObject scope)
79{
80 SoundManager soundManager = SoundManager::getInstance();
81
82 if(NULL == soundReleaseListener || NULL == scope)
83 {
84 return NULL;
85 }
86
87 for(VirtualNode node = soundManager->sounds->head; NULL != node; node = node->next)
88 {
89 Sound sound = Sound::safeCast(node->data);
90
91 if(!isDeleted(sound))
92 {
93 if(soundSpec == sound->soundSpec)
94 {
95 Sound::addEventListener(sound, scope, soundReleaseListener, kEventSoundReleased);
96 return sound;
97 }
98 }
99 }
100
101 return NULL;
102}

112static inline void SoundManager::updatePCM (Sound sound, uint32 elapsedMicroseconds, uint32 targetPCMUpdates)
113{
114 if(kSoundPlaying != sound->state)
115 {
116 return;
117 }
118
119 SoundTrack soundTrack = SoundTrack::safeCast(sound->soundTracks->head->data);
120
121 CACHE_ENABLE;
122
123 // Elapsed time during PCM playback is based on the cursor, track's ticks and target Hz
124 soundTrack->elapsedTicks += elapsedMicroseconds;
125
126 soundTrack->cursor = soundTrack->elapsedTicks / targetPCMUpdates;
127
128 VSUManager::applyPCMSampleToSoundSource(soundTrack->soundTrackSpec->SxLRV[soundTrack->cursor]);
129
130 CACHE_DISABLE;
131
132 if(soundTrack->cursor >= soundTrack->samples)
133 {
134 Sound::finishPlayback(sound);
135 }
136}

146secure void SoundManager::playSounds(uint32 elapsedMicroseconds)
147{
148#ifdef __RELEASE
149 // This is an aggressive optimization that bypasses the SoundTrack's methods altogether
150 // to keep the PCM playback viable on hardware
151 if(kPlaybackPCM == VSUManager::getMode() && NULL != this->sounds->head)
152 {
153 SoundManager::updatePCM(Sound::safeCast(this->sounds->head->data), elapsedMicroseconds, this->targetPCMUpdates);
154 }
155 else
156 {
157#endif
158 VSUManager::update(VSUManager::getInstance());
159
160 for(VirtualNode node = this->sounds->head; NULL != node; node = node->next)
161 {
162 Sound::update(Sound::safeCast(node->data), elapsedMicroseconds, this->targetPCMUpdates);
163 }
164#ifdef __RELEASE
165 }
166#endif
167}

171secure void SoundManager::reset()
172{
173 VSUManager::reset(VSUManager::getInstance());
174
175 for(VirtualNode node = this->sounds->head; NULL != node; node = node->next)
176 {
177 Sound sound = Sound::safeCast(node->data);
178
179 NM_ASSERT(!isDeleted(sound), "SoundManager::reset: deleted soundSpec wrapper");
180
181 delete sound;
182 }
183
184 VirtualList::clear(this->sounds);
185
186 SoundManager::setPCMTargetPlaybackRefreshRate(this, __DEFAULT_PCM_HZ);
187 SoundManager::stopAllSounds(this, false, NULL);
188 SoundManager::unlock(this);
189}

268void SoundManager::stopAllSounds(bool release, SoundSpec** excludedSounds)
269{
270 VirtualNode node = this->sounds->head;
271
272 for(; NULL != node; node = node->next)
273 {
274 Sound sound = Sound::safeCast(node->data);
275
276 if(!isDeleted(sound))
277 {
278 bool excludeSoundWrapper = false;
279
280 if(NULL != excludedSounds)
281 {
282 for(int16 i = 0; NULL != excludedSounds[i]; i++)
283 {
284 if(excludedSounds[i] == sound->soundSpec)
285 {
286 if(release)
287 {
288 Sound::removeAllEventListeners(sound);
289 }
290
291 excludeSoundWrapper = true;
292
293 break;
294 }
295 }
296 }
297
298 if(!excludeSoundWrapper)
299 {
300 if(release)
301 {
302 Sound::release(sound);
303 }
304 else
305 {
306 Sound::stop(sound);
307 }
308 }
309 }
310 }
311
312 if(release)
313 {
314 SoundManager::purgeReleasedSounds(this);
315 }
316
317 if(NULL == excludedSounds)
318 {
319 VSUManager::stopAllSounds(VSUManager::getInstance());
320 }
321}

392Sound SoundManager::doGetSound(const SoundSpec* soundSpec, EventListener soundReleaseListener, ListenerObject scope)
393{
394#ifdef __RELEASE
395 // This is an aggressive optimization that bypasses the SoundTrack's methods altogether
396 // to keep the PCM playback viable on hardware
397 if(kPlaybackPCM == VSUManager::getMode() && NULL != this->sounds->head)
398 {
399#ifndef __SHIPPING
400 Printing::setDebugMode();
401 Printing::clear();
402 Error::triggerException("SoundManager::doGetSound: a PCM sound is loaded, unload it first", NULL);
403#endif
404
405 return NULL;
406 }
407#endif
408
409 SoundManager::purgeReleasedSounds(this);
410
411 if(NULL == soundSpec)
412 {
413 return NULL;
414 }
415
416 Sound sound = new Sound(soundSpec, soundReleaseListener, scope);
417
418 VirtualList::pushBack(this->sounds, sound);
419
420 return sound;
421}

425void SoundManager::purgeReleasedSounds()
426{
427 for(VirtualNode node = this->sounds->head, nextNode = NULL; NULL != node; node = nextNode)
428 {
429 nextNode = node->next;
430
431 Sound sound = Sound::safeCast(node->data);
432
433 if(NULL == sound->soundSpec)
434 {
435 VirtualList::removeNode(this->sounds, node);
436
437 delete sound;
438 }
439 }
440}
