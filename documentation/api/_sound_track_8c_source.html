118bool SoundTrack::update
119(
120 uint32 elapsedMicroseconds, uint32 targetPCMUpdates, fix7_9_ext tickStep, fix7_9_ext targetTimerResolutionFactor,
121 fixed_t leftVolumeFactor, fixed_t rightVolumeFactor, int8 volumeReduction, uint8 volumenScalePower, uint16 frequencyDelta
122)
123{
124 if(this->finished)
125 {
126 return true;
127 }
128
129 if(kTrackPCM == this->soundTrackSpec->trackType)
130 {
131 this->finished = SoundTrack::updatePCM(this, elapsedMicroseconds, targetPCMUpdates, volumeReduction);
132 }
133 else /*if(kTrackNative == this->soundTrackSpec->trackType)*/
134 {
135 this->finished =
136 SoundTrack::updateNative
137 (
138 this, tickStep, targetTimerResolutionFactor, leftVolumeFactor, rightVolumeFactor, volumeReduction,
139 volumenScalePower, frequencyDelta
140 );
141 }
142
143 return this->finished;
144}

167uint32 SoundTrack::getTotalPlaybackMilliseconds(uint16 targetTimerResolutionUS)
168{
169 uint32 totalPlaybackMilliseconds = 0;
170
171 if(kTrackPCM == this->soundTrackSpec->trackType)
172 {
173 totalPlaybackMilliseconds = (this->samples * __MICROSECONDS_PER_MILLISECOND) / _pcmTargetPlaybackRefreshRate;
174 }
175 else if(kTrackNative == this->soundTrackSpec->trackType)
176 {
177 totalPlaybackMilliseconds = (uint32)((long)this->ticks * targetTimerResolutionUS / __MICROSECONDS_PER_MILLISECOND);
178 }
179
180 return totalPlaybackMilliseconds;
181}

191void SoundTrack::constructor(const SoundTrackSpec* soundTrackSpec)
192{
193 // Always explicitly call the base's constructor
194 Base::constructor();
195
196 this->soundTrackSpec = soundTrackSpec;
197 this->samples = this->soundTrackSpec->samples;
198
199 SoundTrack::reset(this);
200
201 if(0 == this->samples)
202 {

239
240void SoundTrack::computeLength()
241{
242 if(NULL == this->soundTrackSpec || NULL == this->soundTrackSpec->trackKeyframes || NULL == this->soundTrackSpec->SxLRV)
243 {
244 return;
245 }
246
247 this->ticks = 0;
248 this->samples = 0;
249
250 int32 keyframe = 0;
251
252 while(kSoundTrackEventEnd != this->soundTrackSpec->trackKeyframes[keyframe].events)
253 {
254 keyframe++;
255 this->samples++;
256 this->ticks += this->soundTrackSpec->trackKeyframes[keyframe].tick;
257 }
258
259 this->ticks += this->soundTrackSpec->trackKeyframes[keyframe].tick;
260}

264bool SoundTrack::updatePCM(uint32 elapsedMicroseconds, uint32 targetPCMUpdates, int8 volumeReduction)
265{
266 CACHE_ENABLE;
267
268 // Elapsed time during PCM playback is based on the cursor, track's ticks and target Hz
269 this->elapsedTicks += elapsedMicroseconds;
270
271 this->cursor = this->elapsedTicks / targetPCMUpdates;
272
273 VSUManager::applyPCMSampleToSoundSource(this->soundTrackSpec->SxLRV[this->cursor] - volumeReduction);
274
275 CACHE_DISABLE;
276
277 return this->cursor >= this->samples;
278}

281
282bool SoundTrack::updateNative
283(
284 fix7_9_ext tickStep, fix7_9_ext targetTimerResolutionFactor, fixed_t leftVolumeFactor, fixed_t rightVolumeFactor,
285 int8 volumeReduction, uint8 volumenScalePower, uint16 frequencyDelta
286)
287{
288 this->elapsedTicks += tickStep;
289
290 if(this->elapsedTicks < this->nextElapsedTicksTarget)
291 {
292 return false;
293 }
294
295 this->elapsedTicks -= this->nextElapsedTicksTarget;
296
297 SoundTrackKeyframe soundTrackKeyframe = this->soundTrackSpec->trackKeyframes[this->cursor];
298
299 this->nextElapsedTicksTarget = __I_TO_FIX7_9_EXT(soundTrackKeyframe.tick);
300
301 if(0 != (kSoundTrackEventSxINT & soundTrackKeyframe.events))
302 {
303 this->cursorSxINT++;
304 }
305
306 if(0 != (kSoundTrackEventSxLRV & soundTrackKeyframe.events))
307 {
308 this->cursorSxLRV++;
309 }
310
311 if(0 != (kSoundTrackEventSxFQ & soundTrackKeyframe.events))
312 {
313 this->cursorSxFQ++;
314 }
315
316 if(0 != (kSoundTrackEventSxEV0 & soundTrackKeyframe.events))
317 {
318 this->cursorSxEV0++;
319 }
320
321 if(0 != (kSoundTrackEventSxEV1 & soundTrackKeyframe.events))
322 {
323 this->cursorSxEV1++;
324 }
325
326 if(0 != (kSoundTrackEventSxRAM & soundTrackKeyframe.events))
327 {
328 this->cursorSxRAM++;
329 }
330
331 if(0 != (kSoundTrackEventSxSWP & soundTrackKeyframe.events))
332 {
333 this->cursorSxSWP++;
334 }
335
336 if(0 != (kSoundTrackEventSxMOD & soundTrackKeyframe.events))
337 {
338 this->cursorSxMOD++;
339 }
340
341 uint8 volume = this->soundTrackSpec->SxLRV[this->cursorSxLRV];
342
343 int16 leftVolume = volume >> 4;
344 int16 rightVolume = volume & 0xF;
345
346 if(0 != volume)
347 {
348 fixed_t volumeHelper = __I_TO_FIXED(volume);
349
350 if(0 <= leftVolumeFactor)
351 {
352 leftVolume = __FIXED_TO_I(__FIXED_MULT(volumeHelper, leftVolumeFactor));
353 }
354
355 if(0 <= rightVolumeFactor)
356 {
357 rightVolume = __FIXED_TO_I(__FIXED_MULT(volumeHelper, rightVolumeFactor));
358 }
359 }
360
361 leftVolume -= volumeReduction;
362 rightVolume -= volumeReduction;
363
364 if(leftVolume < 0)
365 {
366 leftVolume = 0;
367 }
368 else if(leftVolume > __MAXIMUM_VOLUME)
369 {
370 leftVolume = __MAXIMUM_VOLUME;
371 }
372
373 if(rightVolume < 0)
374 {
375 rightVolume = 0;
376 }
377 else if(rightVolume > __MAXIMUM_VOLUME)
378 {
379 rightVolume = __MAXIMUM_VOLUME;
380 }
381
382 leftVolume >>= volumenScalePower;
383 rightVolume >>= volumenScalePower;
384
385 uint16 note = this->soundTrackSpec->SxFQ[this->cursorSxFQ] + frequencyDelta;
386
387 VSUSoundSourceConfiguration vsuChannelConfiguration =
388 {
389 // Requester object
390 Object::safeCast(this),
391 // VSU sound source to configure
392 NULL,
393 // VSU waveform to use
394 NULL,
395 // Time when the configuration elapses
396 __FIX7_9_EXT_DIV(__I_TO_FIX7_9_EXT(soundTrackKeyframe.tick), targetTimerResolutionFactor),
397 // Sound source type
398 0 != (kSoundTrackEventSweepMod & soundTrackKeyframe.events) ?
399 kSoundSourceModulation:
400 0 != (kSoundTrackEventNoise & soundTrackKeyframe.events) ?
401 kSoundSourceNoise:
402 kSoundSourceNormal,
403 // SxINT values
404 this->soundTrackSpec->SxINT[this->cursorSxINT],
405 // SxLRV values
406 (leftVolume << 4) | rightVolume,
407 // SxFQL values
408 note & 0xFF,
409 // SxFQH values
410 note >> 8,
411 // SxEV0 values
412 this->soundTrackSpec->SxEV0[this->cursorSxEV0],
413 // SxEV1 values
414 this->soundTrackSpec->SxEV1[this->cursorSxEV1],
415 // SxRAM pointer
416 this->soundTrackSpec->SxRAM[this->cursorSxRAM],
417 // SxSWP values
418 this->soundTrackSpec->SxSWP[this->cursorSxSWP],
419 // SxMOD pointer
420 this->soundTrackSpec->SxMOD[this->cursorSxMOD],
421 // Skip if no sound source available?
422 this->soundTrackSpec->skippable
423 };
424
425 VSUManager::applySoundSourceConfiguration(&vsuChannelConfiguration);
426
427 return ++this->cursor >= this->samples;
428}
