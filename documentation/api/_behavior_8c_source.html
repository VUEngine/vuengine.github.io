22static Behavior Behavior::create(Entity owner, const BehaviorSpec* behaviorSpec)
23{
24 ASSERT(behaviorSpec, "Behavior::create: NULL behavior");
25 ASSERT(behaviorSpec->componentSpec.allocator, "Behavior::create: no behavior allocator");
26
27 if(NULL == behaviorSpec || !behaviorSpec->componentSpec.allocator)
28 {
29 return NULL;
30 }
31
32 return ((Behavior (*)(Entity, BehaviorSpec**)) behaviorSpec->componentSpec.allocator)(owner, (BehaviorSpec**)behaviorSpec);
33}
