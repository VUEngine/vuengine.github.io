60bool AnimationCoordinator::playAnimation(AnimationController animationController, const AnimationFunction** animationFunctions, const char* animationName)
61{
62 if(!isDeleted(this->animationControllers->head))
63 {
64 AnimationController firstAnimationController = AnimationController::safeCast(VirtualList::front(this->animationControllers));
65
66 if(animationController == firstAnimationController)
67 {
68 return true;
69 }
70
71 // Only if not playing already
72 if
73 (
74 !AnimationController::isPlaying(firstAnimationController)
75 ||
76 strncmp
77 (
78 animationName,
79 AnimationController::getPlayingAnimationFunction(firstAnimationController)->name, __MAX_ANIMATION_FUNCTION_NAME_LENGTH
80 )
81 )
82 {
83 // First animate the frame
84 AnimationController::play(firstAnimationController, animationFunctions, animationName, this->scope, NULL);
85 }
86
87 return false;
88 }
89
90 return true;
91}

95void AnimationCoordinator::addAnimationController(AnimationController animationController)
96{
97 ASSERT(animationController, "AnimationCoordinator::addAnimationController: null animationController");
98 ASSERT
99 (
100 !VirtualList::find(this->animationControllers, animationController),
101 "AnimationCoordinator::addAnimationController: animationController already registered"
102 );
103
104 if(!VirtualList::find(this->animationControllers, animationController))
105 {
106 VirtualList::pushBack(this->animationControllers, animationController);
107 }
108}

112void AnimationCoordinator::removeAnimationController(AnimationController animationController)
113{
114 ASSERT(this->animationControllers->head, "AnimationCoordinator::removeAnimationController: null this");
115
116 bool mustChangeLeader = animationController == AnimationController::safeCast(VirtualList::front(this->animationControllers));
117 VirtualList::removeData(this->animationControllers, animationController);
118
119 if(mustChangeLeader && !isDeleted(this->animationControllers->head))
120 {
121 AnimationController firstAnimationController = AnimationController::safeCast(VirtualList::front(this->animationControllers));
122
123 if(firstAnimationController)
124 {
125 if(AnimationController::isPlaying(animationController))
126 {
127 AnimationController::playAnimationFunction
128 (
129 firstAnimationController, AnimationController::getPlayingAnimationFunction(animationController), this->scope
130 );
131 int16 currentFrame = AnimationController::getActualFrame(animationController);
132 uint8 frameDuration = AnimationController::getFrameDuration(animationController);
133 AnimationController::setActualFrame(firstAnimationController, currentFrame);
134 AnimationController::setFrameDuration(firstAnimationController, frameDuration);
135 AnimationController::stop(animationController);
136 AnimationController::fireEvent(firstAnimationController, kEventAnimationStarted);
137 }
138 }
139 }
140}
