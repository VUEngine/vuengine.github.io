76static inline char* Utilities::itoa(uint32 number, uint32 base, int32 digits)
77{
78 int32 i = __ITOA_ARRAY_SIZE - 1;
79
80 for(; 0 < i;)
81 {
82 i--;
83 _itoaArray[i] = _itoaNumbers[number % base];
84 number /= base;
85
86 if(0 == number)
87 {
88 break;
89 }
90 }
91
92 while(__ITOA_ARRAY_SIZE - 1 - digits < i)
93 {
94 _itoaArray[--i] = '0';
95 }
96
97 _itoaArray[__ITOA_ARRAY_SIZE - 1] = '\0';
98
99 return _itoaArray + i;
100}

104static inline const char* Utilities::toUppercase(const char* string)
105{
106 int32 i = 0;
107 char* result = NULL;
108
109 while(string[i])
110 {
111 result[i] = (string[i] >= 'a' && string[i] <= 'z') ? string[i] - 32 : string[i];
112 i++;
113 }
114 result[i] = 0;
115
116 return result;
