48static void Error::triggerException(char* message __attribute__((unused)), char* detail __attribute__((unused)))
49{
50#ifndef __SHIPPING
51 static bool processingException = false;
52
53 if(processingException)
54 {
55 return;
56 }
57
58 processingException = true;
59
60 int32 lp = _vuengineLinkPointer;
61 int32 sp = _vuengineStackPointer;
62 int32 eipc = _vuengineEIPC;
63 int32 fepc = _vuengineFEPC;
64 int32 ecr = _vuengineECR;
65
66 int32 x = 0 <= __EXCEPTION_COLUMN && __EXCEPTION_COLUMN <= 24 ? __EXCEPTION_COLUMN : 0;
67 int32 y = 0 <= __EXCEPTION_LINE && __EXCEPTION_LINE <= 28 ? __EXCEPTION_LINE : 0;
68
69 // Disable vip interrupts
70 _vipRegisters[__INTENB]= 0;
71 _vipRegisters[__INTCLR] = _vipRegisters[__INTPND];
72
73 // Disable timer
74 _hardwareRegisters[__TCR] &= ~(__TIMER_ENB | __TIMER_INT);
75
76 // Turn on the display
77 _vipRegisters[__REST] = 0;
78 _vipRegisters[__DPCTRL] = _vipRegisters[__DPSTTS] | (__SYNCE | __RE | __DISP);
79 _vipRegisters[__FRMCYC] = 0;
80 _vipRegisters[__XPCTRL] = _vipRegisters[__XPSTTS] | __XPEN;
81
82 // Make sure the brightness is ok
83 _vipRegisters[__BRTA] = 32;
84 _vipRegisters[__BRTB] = 64;
85 _vipRegisters[__BRTC] = 32;
86
87 VIPManager::configureBackgroundColor(__COLOR_BLACK);
88
89 // Make sure there are fonts to show the exception
90 Printing::setDebugMode();
91
92 //print error message to screen
93 if(0 < y)
94 {
95 Printing::text(" ", x, y - 1, NULL);
96 }
97
98 Printing::text
99 (
100 "\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08 EXCEPTION "
101 "\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08\x08" , x, y++, NULL
102 );
103
104 Printing::text(" " , x, y++, NULL);
105 Printing::text(" Last process: ", x, y, NULL);
106 Printing::text(VUEngine::getProcessName(), x + 15, y++, NULL);
107 Printing::text(" LP: " , x, y, NULL);
108 Printing::hex(lp, x + 8, y, 8, NULL);
109 Printing::text(" SP: " , x, ++y, NULL);
110 Printing::hex(sp, x + 8, y, 8, NULL);
111
112 Printing::text(" EIPC: " , x, ++y, NULL);
113 Printing::hex(eipc, x + 8, y, 8, NULL);
114 Printing::text(" FEPC: " , x, ++y, NULL);
115 Printing::hex(fepc, x + 8, y, 8, NULL);
116 Printing::text(" ECR: " , x, ++y, NULL);
117 Printing::hex(ecr, x + 8, y, 8, NULL);
118
119 if(message)
120 {
121 Printing::text(" " , x, ++y + 1, NULL);
122 Printing::text(" Message: " , x, ++y, NULL);
123
124 int32 stringMaxLength = (__SCREEN_WIDTH_IN_CHARS) - 2;
125 int32 rowsAvailable = (__SCREEN_HEIGHT_IN_CHARS) - y;
126 int32 stringLength = strnlen(message, stringMaxLength * rowsAvailable) + 1;
127 int32 lines = stringLength / stringMaxLength + (stringLength % stringMaxLength ? 1 : 0);
128 int32 line = 0;
129
130 for(; line < lines; line++, message += stringMaxLength)
131 {
132 char messageLine[stringLength];
133 strncpy(messageLine, message, stringLength);
134
135 // TODO: fix me, termination character not working
136 messageLine[stringLength - 1] = (char)0;
137 Printing::text(" " , x, ++y, NULL);
138 Printing::text(messageLine, x + 1, y, NULL);
139 }
140
141 if(detail)
142 {
143 Printing::text(detail, x + 1, ++y, NULL);
144 }
145
146 if(y < (__SCREEN_HEIGHT_IN_CHARS) - 1)
147 {
148 Printing::text(" ", x, y + 3, NULL);
149 }
150 }
151
152#ifdef __SHOW_STACK_OVERFLOW_ALERT
153 HardwareManager::printStackStatus((__SCREEN_WIDTH_IN_CHARS) - 10, 0, true);
154#endif
155
156 // Prevent VIP's interrupts
157 HardwareManager::disableInterrupts();
158
159 // Error display message
160 WorldAttributes* worldPointer = &_worldAttributesBaseAddress[__EXCEPTIONS_WORLD];
161
162 worldPointer->mx = __PRINTING_BGMAP_X_OFFSET;
163 worldPointer->mp = __PRINTING_BGMAP_PARALLAX_OFFSET;
164 worldPointer->my = __PRINTING_BGMAP_Y_OFFSET;
165 worldPointer->gx = 0;
166 worldPointer->gp = 0;
167 worldPointer->gy = 0;
168 worldPointer->w = __SCREEN_WIDTH;
169 worldPointer->h = __SCREEN_HEIGHT;
170 worldPointer->head =
171 __WORLD_ON | __WORLD_BGMAP | __WORLD_OVR | BgmapTextureManager::getPrintingBgmapSegment(BgmapTextureManager::getInstance());
172
173 _worldAttributesBaseAddress[__EXCEPTIONS_WORLD - 1].head = __WORLD_END;
174
175 // Dimm game
176 _vipRegisters[__GPLT0] = 0xE4;
177 _vipRegisters[__GPLT1] = __DIMM_VALUE_2;
178 _vipRegisters[__GPLT2] = __DIMM_VALUE_1;
179 _vipRegisters[__GPLT3] = __DIMM_VALUE_1;
180 _vipRegisters[__JPLT0] = __DIMM_VALUE_1;
181 _vipRegisters[__JPLT1] = __DIMM_VALUE_1;
182 _vipRegisters[__JPLT2] = __DIMM_VALUE_1;
183 _vipRegisters[__JPLT3] = __DIMM_VALUE_1;
184
185 // Trap the game here
186 while(true);
187#endif
188}

192static void Error::zeroDivisionException()
193{
194#ifndef __SHIPPING
195 uint32 eipc = 0;
196 // Save EIPC
197 asm
198 (
199 "stsr eipc, r10 \n\t" \
200 "mov r10, %[eipc] \n\t"
201 : [eipc] "=r" (eipc)
202 : // No Input
203 : "r10" // regs used
204 );
205
206 _vuengineEIPC = eipc;
207
208 Error::triggerException("Zero division", NULL);
209#endif
210}
