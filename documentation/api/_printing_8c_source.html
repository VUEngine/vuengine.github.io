122{

125 if(!isDeleted(printing->printingSprites))

171static void Printing::loadFonts(FontSpec** fontSpecs)
172{
173 Printing printing = Printing::getInstance();
174
175 // Since fonts' charsets will be released, there is no reason to keep
176 // Anything in the printing area
177 Printing::clear(printing);
178
179 // Empty list of registered fonts
180 Printing::releaseFonts(printing);
181
182 // Prevent VIP's interrupt from calling render during printing process
183 HardwareManager::suspendInterrupts();
184
186 CharSetManager::writeCharSets(CharSetManager::getInstance());
187
188 // Iterate over all defined fonts and add to internal list
189 uint32 i = 0, j = 0;
190 for(i = 0; _fonts[i]; i++)
191 {
192 // Instance and initialize a new fontdata instance
193 FontData* fontData = new FontData;
194 fontData->fontSpec = _fonts[i];
195 fontData->charSet = NULL;
196
197 // Preload charset for font if in list of fonts to preload
198 if(fontSpecs)
199 {
200 // Find defined font in list of fonts to preload
201 for(j = 0; fontSpecs[j]; j++)
202 {
203 // Preload charset and save charset reference, if font was found
204 if(_fonts[i]->charSetSpec == fontSpecs[j]->charSetSpec)
205 {
206 fontData->charSet = CharSet::get(fontSpecs[j]->charSetSpec);
207
208 CharSet::addEventListener(fontData->charSet, ListenerObject::safeCast(printing), (EventListener)Printing::onFontCharChangedOffset, kEventCharSetChangedOffset);
209 }
210 }
211 }
212
213 // Add fontdata to internal list
214 VirtualList::pushBack(printing->fonts, fontData);
215 }
216
217 if(NULL == printing->activePrintingSprite)
218 {
219 Printing::addSprite(printing);
220 }
221
222 HardwareManager::resumeInterrupts();
223}

227static void Printing::releaseFonts()
228{
229 Printing printing = Printing::getInstance();
230
231 Printing::removeEventListeners(printing, NULL, kEventFontRewritten);
232
233 VirtualNode node = VirtualList::begin(printing->fonts);
234
235 for(; NULL != node; node = VirtualNode::getNext(node))
236 {
237 FontData* fontData = VirtualNode::getData(node);
238
239 if(!isDeleted(fontData))
240 {
241 if(!isDeleted(fontData->charSet))
242 {
243 CharSet::removeEventListener
244 (
245 fontData->charSet, ListenerObject::safeCast(printing), (EventListener)Printing::onFontCharChangedOffset,
246 kEventCharSetChangedOffset
247 );
248
249 while(!CharSet::release(fontData->charSet));
250 }
251
252 delete fontData;
253 }
254 }
255
256 VirtualList::clear(printing->fonts);
257
258 printing->lastUsedFont = NULL;
259 printing->lastUsedFontData = NULL;
260}

313static void Printing::float(float value, uint8 x, uint8 y, int32 precision, const char* font)
314{
315 if(1 > precision)
316 {
317 precision = 1;
318 }
319 else if(10 < precision)
320 {
321 precision = 10;
322 }
323
324 int32 decMultiplier = 1;
325
326 int32 decimals = 0;
327
328 for(; decimals < precision; decimals++)
329 {
330 decMultiplier *= 10;
331 }
332
333 // Round last precision digit
334 value += (0.5f / (decMultiplier));
335
336 char string[48] = "\0";
337
338 int32 i = 0;
339
340 // Handle negatives
341 if(0 > value)
342 {
343 string[i] = '-';
344 i++;
345
346 value *= -1;
347 }
348
349 // Get integral part
350 int32 floorValue = ((int32)(value * 10)) / 10;
351 char* integer = Utilities::itoa(floorValue, 10, Math::getDigitsCount(floorValue));
352
353 // Save it right away
354 for(int32 j = 0; integer[j];)
355 {
356 string[i++] = integer[j++];
357 }
358
359 // Start decimal part
360 string[i++] = '.';
361
362 // Get decimal part
363 float decimalValue = value - floorValue;
364
365 // Promote to integral all the decimals up to precision
366 decimalValue *= decMultiplier;
367
368 int32 zeros = 0;
369 int32 flooredDecimalValue = (int32)Math::floor(decimalValue);
370
371 while(10 <= decMultiplier)
372 {
373 decMultiplier /= 10;
374
375 if(0 != (flooredDecimalValue / decMultiplier))
376 {
377 break;
378 }
379
380 string[i++] = '0';
381 zeros++;
382 }
383
384 if(decimals <= precision && zeros < precision)
385 {
386 long roundedDecimalValue = (int32)(decimalValue * 10) / 10;
387
388 if(0 == roundedDecimalValue)
389 {
390 string[i] = 0;
391 }
392 else
393 {
394 int32 totalDecimalDigits = Math::getDigitsCount(roundedDecimalValue);
395
396 char* decimalString = Utilities::itoa((int32)(decimalValue * 10) / 10, 10, totalDecimalDigits);
397
398 int32 j = 0;
399
400 for(; j < totalDecimalDigits; j++)
401 {
402 string[i + j] = decimalString[j];
403 }
404
405 string[i + j] = 0;
406 }
407 }
408 else
409 {
410 string[i] = 0;
411 }
412
413 Printing::text(string, x, y, font);
414}

733
734static FontData* Printing::getFontByName(const char* font)
735{
736 Printing printing = Printing::getInstance();
737
738 FontData* result = NULL;
739
740 if(printing->mode == __PRINTING_MODE_DEBUG)
741 {
742 result = (FontData*)&VUENGINE_DEBUG_FONT_DATA;
743 }
744 else if(NULL != printing->fonts)
745 {
746 // Set first defined font as default
747 result = VirtualList::front(printing->fonts);
748
749 if(NULL != result)
750 {
751 if(NULL != font)
752 {
753 // Iterate over registered fonts to find spec of font to use
754 VirtualNode node = VirtualList::begin(printing->fonts);
755
756 for(; NULL != node; node = VirtualNode::getNext(node))
757 {
758 FontData* fontData = VirtualNode::getData(node);
759
760 if(!strcmp(fontData->fontSpec->name, font))
761 {
762 result = fontData;
763 break;
764 }
765 }
766 }
767
768 // If font's charset has not been preloaded, load it now
769 if(NULL == result->charSet)
770 {
771 result->charSet = CharSet::get(result->fontSpec->charSetSpec);
772
773 CharSet::addEventListener
774 (
775 result->charSet, ListenerObject::safeCast(printing), (EventListener)Printing::onFontCharChangedOffset,
776 kEventCharSetChangedOffset
777 );
778 }
779 }
780 }
781
782 return result;
783}

786
787static FontSize Printing::getTextSize(const char* string, const char* font)
788{
789 FontSize fontSize = {0, 0};
790 uint16 i = 0, currentLineLength = 0;
791
792 FontData* fontData = Printing::getFontByName(font);
793
794 if(NULL == fontData)
795 {
796 // Just to make sure that no client code does a 0 division with these results
797 fontSize = (FontSize){8, 8};
798 return fontSize;
799 }
800
801 fontSize.y = fontData->fontSpec->fontSize.y;
802
803 while('\0' != string[i])
804 {
805 switch(string[i])
806 {
807 // Line feed
808 case 13:
809
810 break;
811
812 // Tab
813 case 9:
814
815 currentLineLength += (currentLineLength / __TAB_SIZE + 1) * __TAB_SIZE * fontData->fontSpec->fontSize.x;
816 break;
817
818 // Carriage return
819 case 10:
820
821 fontSize.y += fontData->fontSpec->fontSize.y;
822 currentLineLength = 0;
823 break;
824
825 default:
826
827 currentLineLength += fontData->fontSpec->fontSize.x;
828 if(currentLineLength >= 64)
829 {
830 fontSize.y += fontData->fontSpec->fontSize.y;
831 currentLineLength = 0;
832 }
833
834 break;
835 }
836
837 if(currentLineLength > fontSize.x)
838 {
839 fontSize.x = currentLineLength;
840 }
841
842 i++;
843 }
844
845 return fontSize;
846}

880
881static void Printing::out(uint8 x, uint8 y, const char* string, const char* font)
882{
883 Printing printing = Printing::getInstance();
884
885#ifdef __DEFAULT_FONT
886 if(NULL == font)
887 {
888 font = __DEFAULT_FONT;
889 }
890#endif
891
892#ifdef __FORCE_FONT
893 font = __FORCE_FONT;
894#endif
895
896 if(-1 == printing->printingBgmapSegment)
897 {
898 printing->printingBgmapSegment = BgmapTextureManager::getPrintingBgmapSegment(BgmapTextureManager::getInstance());
899
900 if(-1 == printing->printingBgmapSegment)
901 {
902 return;
903 }
904 }
905
906 uint32 i = 0, position = 0, startColumn = x, temp = 0;
907 uint32 charOffset = 0, charOffsetX = 0, charOffsetY = 0;
908 FontData* fontData = printing->lastUsedFontData;
909
910 if(NULL == fontData)
911 {
912 fontData = Printing::getFontByName(font);
913 printing->lastUsedFontData = fontData;
914 printing->lastUsedFont = font;
915 }
916 else if(printing->lastUsedFont != font && strcmp(fontData->fontSpec->name, font))
917 {
918 fontData = Printing::getFontByName(font);
919 printing->lastUsedFontData = fontData;
920 printing->lastUsedFont = font;
921 }
922 else
923 {
924 printing->lastUsedFontData = fontData;
925 printing->lastUsedFont = font;
926 }
927
928 if(NULL == fontData || (__PRINTING_MODE_DEBUG != printing->mode && isDeleted(fontData->charSet)))
929 {
930 return;
931 }
932
933 uint16* const bgmapSpaceBaseAddress = (uint16*)__BGMAP_SPACE_BASE_ADDRESS;
934 uint32 offset = __PRINTING_MODE_DEBUG == printing->mode ? VUENGINE_DEBUG_FONT_CHARSET_OFFSET : CharSet::getOffset(fontData->charSet);
935
936 // Print text
937 while('\0' != string[i] && x < (__SCREEN_WIDTH_IN_CHARS))
938 {
939 // Do not allow printing outside of the visible area, since that would corrupt the param table
940 if(y > 27/* || y < 0*/)
941 {
942 break;
943 }
944
945 position = (y << 6) + x;
946
947 switch(string[i])
948 {
949 // Line feed
950 case 13:
951
952 break;
953
954 // Tab
955 case 9:
956
957 if(kPrintingOrientationHorizontal == printing->orientation)
958 {
959 x = (x / __TAB_SIZE + 1) * __TAB_SIZE * fontData->fontSpec->fontSize.x;
960 }
961 else
962 {
963 y = (y / __TAB_SIZE + 1) * __TAB_SIZE * fontData->fontSpec->fontSize.y;
964 }
965 break;
966
967 // Carriage return
968 case 10:
969
970 temp = fontData->fontSpec->fontSize.y;
971 y = (printing->direction == kPrintingDirectionLTR)
972 ? y + temp
973 : y - temp;
974 x = startColumn;
975 break;
976
977 default:
978 {
979 for(charOffsetX = 0; charOffsetX < fontData->fontSpec->fontSize.x; charOffsetX++)
980 {
981 for(charOffsetY = 0; charOffsetY < fontData->fontSpec->fontSize.y; charOffsetY++)
982 {
983 charOffset =
984 charOffsetX +
985 (charOffsetY * fontData->fontSpec->charactersPerLineInCharset * fontData->fontSpec->fontSize.x);
986
987 bgmapSpaceBaseAddress[(0x1000 * (printing->printingBgmapSegment + 1) -
988 __PRINTABLE_BGMAP_AREA) + position + charOffsetX + (charOffsetY << 6)] =
989 (
990 // Offset of charset in char memory
991 offset +
992 ((uint8)(string[i] - fontData->fontSpec->offset) * fontData->fontSpec->fontSize.x) +
993
994 (((uint8)(string[i] - fontData->fontSpec->offset)
995 / fontData->fontSpec->charactersPerLineInCharset
996 * fontData->fontSpec->charactersPerLineInCharset * fontData->fontSpec->fontSize.x)
997 * (fontData->fontSpec->fontSize.y - 1)) +
998
999 // Respective char of character
1000 charOffset
1001 )
1002 | (printing->palette << 14);
1003 }
1004 }
1005 }
1006
1007 if(kPrintingOrientationHorizontal == printing->orientation)
1008 {
1009 temp = fontData->fontSpec->fontSize.x;
1010 x = (printing->direction == kPrintingDirectionLTR)
1011 ? x + temp
1012 : x - temp;
1013 }
1014 else
1015 {
1016 temp = fontData->fontSpec->fontSize.y;
1017 y = (printing->direction == kPrintingDirectionLTR)
1018 ? y + temp
1019 : y - temp;
1020 }
1021
1022 if(x >= 48/* || x < 0*/)
1023 {
1024 // Wrap around when outside of the visible area
1025 temp = fontData->fontSpec->fontSize.y;
1026 y = (printing->direction == kPrintingDirectionLTR)
1027 ? y + temp
1028 : y - temp;
1029 x = startColumn;
1030 }
1031
1032 break;
1033 }
1034 i++;
1035 }
1036}
