39static void TimerManager::interruptHandler()
40{
41 TimerManager timerManager = TimerManager::getInstance();
42
43 //disable
44#ifndef __ENABLE_PROFILER
45 TimerManager::disable(timerManager);
46 TimerManager::clearStat(timerManager);
47#else
48 TimerManager::disableInterrupt(timerManager);
49
50 Profiler::lap(kProfilerLapTypeStartInterrupt, NULL);
51#endif
52
53 timerManager->interruptsPerSecond++;
54
55 timerManager->interruptsPerGameFrame++;
56
57 timerManager->elapsedMicroseconds += timerManager->elapsedMicrosecondsPerInterrupt;
58
59 if(timerManager->elapsedMicroseconds > __MICROSECONDS_PER_MILLISECOND)
60 {
61 uint32 elapsedMilliseconds = timerManager->elapsedMicroseconds / __MICROSECONDS_PER_MILLISECOND;
62
63 timerManager->elapsedMicroseconds = timerManager->elapsedMicroseconds % __MICROSECONDS_PER_MILLISECOND;
64
65 timerManager->elapsedMilliseconds += elapsedMilliseconds;
66
67 timerManager->totalElapsedMilliseconds += elapsedMilliseconds;
68 }
69
70 // Update sounds
71 SoundManager::playSounds(SoundManager::getInstance(), timerManager->elapsedMicrosecondsPerInterrupt);
72
73 // Update Stopwatchs: no use is being done of them so this is commented out for now since it affects PCM playback
74 //StopwatchManager::update();
75
76// enable
77#ifndef __ENABLE_PROFILER
78 TimerManager::enable();
79#else
80 TimerManager::enableInterrupt(timerManager);
81 Profiler::lap(kProfilerLapTypeTimerInterruptProcess, PROCESS_NAME_SOUND_PLAY);
82#endif
83}

135static void TimerManager::resetTimerCounter()
136{
137 TimerManager timerManager = TimerManager::getInstance();
138
139 uint16 timerCounter = TimerManager::computeTimerCounter(timerManager);
140
141 switch(timerManager->resolution)
142 {
143 case __TIMER_20US:
144
145 break;
146
147 case __TIMER_100US:
148
149 // Compensate for the difference in speed between 20US and 100US timer resolution
150 timerCounter += __TIMER_COUNTER_DELTA;
151 break;
152 }
153
154 _hardwareRegisters[__TLR] = (timerCounter & 0xFF);
155 _hardwareRegisters[__THR] = (timerCounter >> 8);
156}

160static void TimerManager::setResolution(uint16 resolution)
161{
162 TimerManager timerManager = TimerManager::getInstance();
163
164 switch(resolution)
165 {
166 case __TIMER_20US:
167
168 timerManager->resolution = resolution;
169 break;
170
171 case __TIMER_100US:
172
173 timerManager->resolution = resolution;
174 break;
175
176 default:
177
178 NM_ASSERT(false, "TimerManager::setResolution: wrong timer resolution");
179
180 timerManager->resolution = __TIMER_20US;
181 break;
182 }
183
184 uint32 targetTimePerInterrupt = timerManager->targetTimePerInterrupt;
185
186 switch(timerManager->targetTimePerInterrupttUnits)
187 {
188 case kUS:
189 {
190 uint32 residue = targetTimePerInterrupt % TimerManager::getResolutionInUS(timerManager);
191
192 if(targetTimePerInterrupt > residue)
193 {
194 targetTimePerInterrupt -= residue;
195 }
196 }
197 break;
198
199 case kMS:
200
201 break;
202
203 default:
204
205 ASSERT(false, "TimerManager::setResolution: wrong timer resolution scale");
206 break;
207 }
208
209 TimerManager::setTargetTimePerInterrupt(targetTimePerInterrupt);
210}

251static void TimerManager::setTargetTimePerInterrupt(uint16 targetTimePerInterrupt)
252{
253 TimerManager timerManager = TimerManager::getInstance();
254
255 int16 minimumTimePerInterrupt = 0;
256 int16 maximumTimePerInterrupt = 1000;
257
258 switch(timerManager->targetTimePerInterrupttUnits)
259 {
260 case kUS:
261
262 minimumTimePerInterrupt =
263 TimerManager::getResolutionInUS(timerManager) + TimerManager::getResolutionInUS(timerManager) * __TIMER_COUNTER_DELTA;
264 maximumTimePerInterrupt = __MAXIMUM_TIME_PER_INTERRUPT_US;
265 break;
266
267 case kMS:
268
269 minimumTimePerInterrupt = __MINIMUM_TIME_PER_INTERRUPT_MS;
270 maximumTimePerInterrupt = __MAXIMUM_TIME_PER_INTERRUPT_MS;
271 break;
272
273 default:
274
275 ASSERT(false, "TimerManager::setTargetTimePerInterrupt: wrong resolution scale");
276 break;
277 }
278
279 if((int16)targetTimePerInterrupt < minimumTimePerInterrupt)
280 {
281 targetTimePerInterrupt = minimumTimePerInterrupt;
282 }
283 else if((int16)targetTimePerInterrupt > maximumTimePerInterrupt)
284 {
285 targetTimePerInterrupt = maximumTimePerInterrupt;
286 }
287
288 timerManager->targetTimePerInterrupt = targetTimePerInterrupt;
289 timerManager->elapsedMicrosecondsPerInterrupt = TimerManager::getTargetTimePerInterruptInUS(timerManager);
290}

444static void TimerManager::wait(uint32 milliseconds)
445{
446 TimerManager timerManager = TimerManager::getInstance();
447
448 // Declare as volatile to prevent the compiler to optimize currentMilliseconds away
449 // Making the last assignment invalid
450 volatile uint32 currentMilliseconds = timerManager->totalElapsedMilliseconds;
451 uint32 waitStartTime = timerManager->totalElapsedMilliseconds;
452 volatile uint32 *totalElapsedMilliseconds = (uint32*)&timerManager->totalElapsedMilliseconds;
453
454 while ((*totalElapsedMilliseconds - waitStartTime) < milliseconds)
455 {
456 HardwareManager::halt();
457 }
458
459 timerManager->elapsedMilliseconds = currentMilliseconds;
460}

464static void TimerManager::repeatMethodCall(uint32 callTimes, uint32 duration, ListenerObject object, void (*method)(ListenerObject, uint32))
465{
466 TimerManager timerManager = TimerManager::getInstance();
467
468 if(!isDeleted(object) && method)
469 {
470 // Declare as volatile to prevent the compiler to optimize currentMilliseconds away
471 // Making the last assignment invalid
472 volatile uint32 currentMilliseconds = timerManager->elapsedMilliseconds;
473
474 uint32 i = 0;
475
476 for(; i < callTimes; i++)
477 {
478 TimerManager::wait(duration / callTimes);
479
480 if(isDeleted(object))
481 {
482 return;
483 }
484
485 method(object, i);
486 }
487
488 timerManager->elapsedMilliseconds = currentMilliseconds;
489 }
490}

494static void TimerManager::print(int32 x, int32 y)
495{
496 TimerManager timerManager = TimerManager::getInstance();
497
498 Printing::text("TIMER CONFIG", x, y++, NULL);
499 y++;
500
501 switch(timerManager->resolution)
502 {
503 case __TIMER_20US:
504
505 Printing::text("Resolution 20 US ", x, y++, NULL);
506 break;
507
508 case __TIMER_100US:
509
510 Printing::text("Resolution 100 US ", x, y++, NULL);
511 break;
512
513 default:
514
515 Printing::text("Resolution ? ", x, y++, NULL);
516 break;
517 }
518
519 switch(timerManager->targetTimePerInterrupttUnits)
520 {
521 case kUS:
522
523 Printing::text("US/interrupt ", x, y, NULL);
524 break;
525
526 case kMS:
527
528 Printing::text("MS/interrupt ", x, y, NULL);
529 break;
530
531 default:
532
533 Printing::text(" ?/interrupt ", x, y, NULL);
534 break;
535 }
536
537 Printing::int32(timerManager->targetTimePerInterrupt, x + 14, y++, NULL);
538
539 Printing::text("Timer counter ", x, y, NULL);
540 Printing::int32(TimerManager::getTimerCounter(timerManager), x + 14, y++, NULL);
541}

570secure void TimerManager::reset()
571{
572 this->tcrValue = 0;
573 this->elapsedMilliseconds = 0;
574 this->elapsedMicroseconds = 0;
575 this->resolution = __TIMER_100US;
576 this->targetTimePerInterrupt = 1;
577 this->targetTimePerInterrupttUnits = kMS;
578 this->interruptsPerGameFrame = 0;
579 this->elapsedMicrosecondsPerInterrupt = TimerManager::getTargetTimePerInterruptInUS(this);
580}

584secure void TimerManager::frameStarted(uint32 elapsedMicroseconds)
585{
586 this->elapsedMilliseconds = 0;
587 this->elapsedMicroseconds = 0;
588
589 if(0 >= this->interruptsPerGameFrame)
590 {
591 this->elapsedMicrosecondsPerInterrupt = TimerManager::getTargetTimePerInterruptInUS(this);
592 }
593 else
594 {
595 this->elapsedMicrosecondsPerInterrupt = elapsedMicroseconds / this->interruptsPerGameFrame;
596 }
597
598 this->interruptsPerGameFrame = 0;
599}

645
