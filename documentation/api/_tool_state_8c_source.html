46void ToolState::enter(void* owner __attribute__ ((unused)))
47{
48 Base::enter(this, owner);
49 GameState::pauseClocks(GameState::safeCast(VUEngine::getPreviousState()));
50 GameState::startClocks(this);
51
52 this->stream = false;
53 this->transform = false;
54 this->updatePhysics = false;
55 this->processCollisions = false;
56
57 if(!isDeleted(this->tool))
58 {
59 Tool::setStage(this->tool, GameState::getStage(GameState::safeCast(VUEngine::getPreviousState())));
60 Tool::show(this->tool);
61 }
62}
