44void Line::render(Vector3D relativePosition)
45{
46 Vector3D a = Vector3D::sum(relativePosition, ((LineSpec*)this->componentSpec)->a);
47 a = Vector3D::rotate(a, _previousCameraInvertedRotation);
48 this->a = PixelVector::projectVector3D(a, Optics::calculateParallax(a.z));
49
50 Vector3D b = Vector3D::sum(relativePosition, ((LineSpec*)this->componentSpec)->b);
51 b = Vector3D::rotate(b, _previousCameraInvertedRotation);
52 this->b = PixelVector::projectVector3D(b, Optics::calculateParallax(b.z));
53}
