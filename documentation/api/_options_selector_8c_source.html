90void OptionsSelector::constructor(uint16 cols, uint16 rows, char* font, char* leftMark, char* rightMark)
91{
92 // Always explicitly call the base's constructor
93 Base::constructor();
94
95 this->pages = NULL;
96 this->currentPage = NULL;
97 this->currentOption = NULL;
98 this->currentPageIndex = 0;
99 this->currentOptionIndex = 0;
100 this->x = 0;
101 this->y = 0;
102 this->optionsLength = 0;
103 this->alignment = kOptionsAlignLeft;
104 this->spacing = 0;
105 this->cols = ((0 < cols) && (cols <= __OPTIONS_SELECT_MAX_COLS)) ? cols : 1;
106 this->rows = ((0 < rows) && (rows <= __OPTIONS_SELECT_MAX_ROWS)) ? rows : __OPTIONS_SELECT_MAX_ROWS;
107 this->totalOptions = 0;
108 this->leftMark = NULL == leftMark ? __CHAR_SELECTOR_LEFT : leftMark;
109 this->rightMark = rightMark;
110 this->font = font;
111 this->columnWidth = (__SCREEN_WIDTH_IN_CHARS) / this->cols;
112}

151void OptionsSelector::setOptions(VirtualList options)
152{
153 ASSERT(options, "OptionsSelector::setOptions: null options");
154
155 OptionsSelector::flushPages(this);
156
157 this->pages = new VirtualList();
158
159 this->totalOptions = VirtualList::getCount(options);
160
161 int32 optionsPerPage = this->cols * this->rows;
162 int32 numberOfPages = (int32)(this->totalOptions / optionsPerPage);
163 numberOfPages += (0 < (this->totalOptions % optionsPerPage)) ? 1 : 0;
164
165 VirtualNode node = options->head;
166
167 ASSERT(VirtualList::getCount(options), "OptionsSelector::setOptions: empty options");
168
169 if(0 < VirtualList::getCount(options))
170 {
171 int32 page = 0;
172
173 for(; page < numberOfPages && node; page++)
174 {
175 VirtualList pageOptions = new VirtualList();
176
177 int32 counter = 0;
178 for(; node && counter < optionsPerPage; counter++, node = node->next)
179 {
180 VirtualList::pushBack(pageOptions, node->data);
181 }
182
183 VirtualList::pushBack(this->pages, pageOptions);
184 }
185
186 this->currentPage = this->pages->head;
187 ASSERT(VirtualList::getCount(this->pages), "OptionsSelector::setOptions: empty pages");
188
189 this->currentOption = this->currentPage ? (VirtualList::safeCast(VirtualNode::getData(this->currentPage)))->head : NULL;
190 }
191
192 this->currentPageIndex = 0;
193 this->currentOptionIndex = 0;
194}

198bool OptionsSelector::setSelectedOption(int32 optionIndex)
199{
200 bool changed = false;
201
202 // Check if desired option index is within bounds
203 if(optionIndex >= 0 && optionIndex <= this->totalOptions)
204 {
205 if(optionIndex < this->currentOptionIndex)
206 {
207 // If desired option index is smaller than the current one, select previous until desired option is set
208 while(this->currentOptionIndex != optionIndex)
209 {
210 OptionsSelector::selectNext(this);
211 changed = true;
212 }
213 }
214 else if(optionIndex > this->currentOptionIndex)
215 {
216 // If desired option index is larger than the current one, select next until desired option is set
217 while(this->currentOptionIndex != optionIndex)
218 {
219 OptionsSelector::selectPrevious(this);
220 changed = true;
221 }
222 }
223 }
224
225 return changed;
226}

230void OptionsSelector::selectNext()
231{
232 if(this->currentOption)
233 {
234 // Remove previous selection mark
235 OptionsSelector::printSelectorMark(this, " ", -this->optionsLength);
236 OptionsSelector::printSelectorMark(this, " ", this->optionsLength);
237
238 // Get next option
239 this->currentOption = this->currentOption->next;
240 this->currentOptionIndex++;
241
242 // If there's no next option on the current page
243 if(!this->currentOption)
244 {
245 // If there's more than 1 page, go to next page
246 if(VirtualList::getCount(this->pages) > 1)
247 {
248 // Select next page
249 this->currentPage = this->currentPage->next;
250 this->currentPageIndex++;
251
252 // If next page does not exist
253 if(!this->currentPage)
254 {
255 this->currentPage = this->pages->head;
256 this->currentPageIndex = 0;
257 this->currentOptionIndex = 0;
258 }
259
260 // Get new option
261 this->currentOption = (VirtualList::safeCast(VirtualNode::getData(this->currentPage)))->head;
262 ASSERT(this->currentOption, "selectNext: null current option");
263
264 // Render new page
265 OptionsSelector::print(this, this->x, this->y, this->alignment, this->spacing - 1);
266 }
267 else
268 {
269 // Wrap around and select first option
270 this->currentOption = (VirtualList::safeCast(VirtualNode::getData(this->currentPage)))->head;
271 this->currentOptionIndex = 0;
272 }
273 }
274
275 // Print new selection mark
276 OptionsSelector::printSelectorMark(this, this->leftMark, -this->optionsLength);
277 OptionsSelector::printSelectorMark(this, this->rightMark, this->optionsLength);
278 }
279}

283void OptionsSelector::selectPrevious()
284{
285 if(this->currentOption)
286 {
287 // Remove previous selection mark
288 OptionsSelector::printSelectorMark(this, " ", -this->optionsLength);
289 OptionsSelector::printSelectorMark(this, " ", this->optionsLength);
290
291 // Get previous option
292 this->currentOption = VirtualNode::getPrevious(this->currentOption);
293 this->currentOptionIndex--;
294
295 // If there's no previous option on the current page
296 if(!this->currentOption)
297 {
298 // If there's more than 1 page
299 if(VirtualList::getCount(this->pages) > 1)
300 {
301 // Select previous page
302 this->currentPage = VirtualNode::getPrevious(this->currentPage);
303 this->currentPageIndex--;
304
305 // If previous page does not exist, go to last page
306 if(!this->currentPage)
307 {
308 this->currentPage = this->pages->tail;
309 this->currentPageIndex = VirtualList::getCount(this->pages) - 1;
310 this->currentOptionIndex = this->totalOptions - 1;
311 }
312
313 // Get new option
314 this->currentOption = (VirtualList::safeCast(VirtualNode::getData(this->currentPage)))->tail;
315 ASSERT(this->currentOption, "selectPrevious: current option data");
316
317 // Render new page
318 OptionsSelector::print(this, this->x, this->y, this->alignment, this->spacing - 1);
319 }
320 else
321 {
322 // Wrap around and select last option
323 this->currentOption = (VirtualList::safeCast(VirtualNode::getData(this->currentPage)))->tail;
324 this->currentOptionIndex = this->totalOptions - 1;
325 }
326 }
327
328 // Print new selection mark
329 OptionsSelector::printSelectorMark(this, this->leftMark, -this->optionsLength);
330 OptionsSelector::printSelectorMark(this, this->rightMark, this->optionsLength);
331 }

350void OptionsSelector::print(uint8 x, uint8 y, uint32 alignment, uint8 spacing)
351{
352 spacing++;
353
354 if(this->currentPage && 0 < VirtualList::getCount(VirtualList::safeCast(VirtualNode::getData(this->currentPage))))
355 {
356 FontData* fontData = Printing::getFontByName(this->font);
357
358 this->x = (x < (__SCREEN_WIDTH_IN_CHARS)) ? x : 0;
359 this->y = (y < (__SCREEN_HEIGHT_IN_CHARS)) ? y : 0;
360 this->alignment = alignment;
361 this->spacing = spacing;
362
363 ASSERT(this->currentPage, "print: currentPage");
364 VirtualNode node = (VirtualList::safeCast(VirtualNode::getData(this->currentPage)))->head;
365
366 int8 jStart = 0;
367
368 switch(alignment)
369 {
370 case kOptionsAlignCenter:
371
372 jStart -= this->columnWidth / 2 - 1;
373 break;
374
375 case kOptionsAlignRight:
376
377 jStart -= this->columnWidth;
378 break;
379 }
380
381 for(int32 i = 0; i < (this->rows * fontData->fontSpec->fontSize.y) && y + i < __SCREEN_HEIGHT_IN_CHARS; i++)
382 {
383 for(int32 j = 0; (this->columnWidth * this->cols) > j && x + j < __SCREEN_WIDTH_IN_CHARS; j++)
384 {
385 Printing::text(" ", x + j + jStart, y + i * spacing, this->font);
386 }
387 }
388
389 for(int32 counter = 0; NULL != node; node = node->next, counter++)
390 {
391 ASSERT(node, "print: push null node");
392 ASSERT(node->data, "print: push null node data");
393
394 int8 optionsLength = 0;
395 int8 optionsLengthDivisor = 1;
396 Option* option = VirtualNode::getData(node);
397
398 switch(alignment)
399 {
400 case kOptionsAlignCenter:
401
402 optionsLengthDivisor = 2;
403 break;
404 }
405
406 if(NULL == option->value)
407 {
408 Printing::int32(counter, x + fontData->fontSpec->fontSize.x, y, this->font);
409 }
410 else
411 {
412 switch(option->type)
413 {
414 case kString:
415
416 optionsLength = alignment == kOptionsAlignLeft ? 0 : strnlen((char*)option->value, this->columnWidth);
417 Printing::text
418 (
419 (char*)option->value, x + fontData->fontSpec->fontSize.x - optionsLength / optionsLengthDivisor,
420 y, this->font
421 );
422 break;
423
424 case kFloat:
425
426 optionsLength = alignment == kOptionsAlignLeft ? 0 : Math::getDigitsCount(*((int32*)option->value)) - 3;
427 Printing::float
428 (
429 *((float*)option->value), x + fontData->fontSpec->fontSize.x - optionsLength / optionsLengthDivisor,
430 y, 2, this->font
431 );
432 break;
433
434 case kInt:
435
436 optionsLength = alignment == kOptionsAlignLeft ? 0 : Math::getDigitsCount(*((int32*)option->value));
437 Printing::int32
438 (
439 *((int32*)option->value), x + fontData->fontSpec->fontSize.x - optionsLength / optionsLengthDivisor,
440 y, this->font
441 );
442 break;
443
444 case kShortInt:
445
446 optionsLength = alignment == kOptionsAlignLeft ? 0 : Math::getDigitsCount(*((int32*)option->value));
447 Printing::int32
448 (
449 *((int16*)option->value), x + fontData->fontSpec->fontSize.x - optionsLength / optionsLengthDivisor,
450 y, this->font
451 );
452 break;
453
454 case kChar:
455
456 optionsLength = alignment == kOptionsAlignLeft ? 0 : Math::getDigitsCount(*((int32*)option->value));
457 Printing::int32
458 (
459 *((int8*)option->value), x + fontData->fontSpec->fontSize.x - optionsLength / optionsLengthDivisor,
460 y, this->font
461 );
462 break;
463 }
464 }
465
466 if(__ABS(optionsLength) > __ABS(this->optionsLength))
467 {
468 this->optionsLength = optionsLength;
469 }
470
471 y += fontData->fontSpec->fontSize.y * spacing;
472
473 if((y >= (this->rows * this->spacing * fontData->fontSpec->fontSize.y + this->y)) || (y >= (__SCREEN_HEIGHT_IN_CHARS)))
474 {
475 y = this->y;
476 x += this->columnWidth;
477 }
478 }
479
480 OptionsSelector::printSelectorMark(this, this->leftMark, -this->optionsLength);
481 OptionsSelector::printSelectorMark(this, this->rightMark, this->optionsLength);
482 }
483}

492
493void OptionsSelector::flushPages()
494{
495 if(this->pages)
496 {
497 VirtualNode node = this->pages->head;
498
499 for(; NULL != node; node = node->next)
500 {
501 ASSERT(node->data, "flushPages: null node data");
502
503 VirtualNode optionsNode = (VirtualList::safeCast(node->data))->head;
504
505 for(; optionsNode; optionsNode = optionsNode->next)
506 {
507 delete optionsNode->data;
508 }
509
510 delete node->data;
511 }
512
513 delete this->pages;
514 }
515
516 this->pages = NULL;
517}

520
521void OptionsSelector::printSelectorMark(char* mark, int8 optionsLength)
522{
523 if(this->currentPage && NULL != mark)
524 {
525 FontData* fontData = Printing::getFontByName(this->font);
526
527 ASSERT(this->currentPage, "printSelectorMark: current page");
528 ASSERT(VirtualNode::getData(this->currentPage), "printSelectorMark: null current data");
529
530 int32 indexOption =
531 this->currentOptionIndex -
532 this->currentPageIndex * VirtualList::getCount(VirtualList::safeCast(VirtualList::front(this->pages)));
533 int32 optionColumn = (int32)(indexOption / this->rows);
534 int32 optionRow = indexOption - optionColumn * this->rows;
535 optionColumn = this->columnWidth * optionColumn;
536
537 switch(this->alignment)
538 {
539 case kOptionsAlignCenter:
540
541 if(0 < optionsLength)
542 {
543 int8 optionsLengthHelper = optionsLength;
544 optionsLength = optionsLength / 2 + (0 == __MODULO(__ABS(optionsLengthHelper), 2) ? 1 : 2);
545 }
546 else
547 {
548 optionsLength /= 2;
549 }
550
551 break;
552 }
553
554 Printing::text
555 (
556 mark,
557 this->x + optionColumn + optionsLength,
558 this->y + (optionRow * this->spacing * fontData->fontSpec->fontSize.y),
559 this->font
560 );
561 }
562}
