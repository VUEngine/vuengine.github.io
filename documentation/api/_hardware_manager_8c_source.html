92static void HardwareManager::reset()
93{
94 // Set ROM waiting to 1 cycle
95 _hardwareRegisters[__WCR] |= 0x0001;
96
97 // Check memory map before anything else
98 HardwareManager::checkMemoryMap();
99
100 //setup timer interrupts
101 HardwareManager::setInterruptVectors();
102 HardwareManager::setInterruptLevel(0);
103 HardwareManager::setExceptionVectors();
104}

190static void HardwareManager::printStackStatus(int32 x, int32 y, bool resumed)
191{
192 Printing::setDebugMode();
193 Printing::clear();
194
195 int32 sp;
196 asm
197 (
198 "mov sp, %0"
199 : "=r" (sp)
200 );
201
202 int32 room = sp - (int32)&_bssEnd;
203
204 if(resumed)
205 {
206 if((__SCREEN_WIDTH_IN_CHARS) < x + Math::getDigitsCount(room) + 13)
207 {
208 x = (__SCREEN_WIDTH_IN_CHARS) - Math::getDigitsCount(room) - 13;
209 }
210
211 Printing::text(" STACK'S ROOM " , x - 3, y, NULL);
212 Printing::int32(room, x + 13, y, NULL);
213 }
214 else
215 {
216 if((__SCREEN_WIDTH_IN_CHARS) - 1 < Math::getDigitsCount(room) + 15)
217 {
218 x = (__SCREEN_WIDTH_IN_CHARS) - 1 - Math::getDigitsCount(room) - 11;
219 }
220
221 Printing::text(" STACK'S STATUS" , x - 3, y, NULL);
222 Printing::text("Bss' end:" , x, ++y, NULL);
223 Printing::hex((int32)&_bssEnd, x + 15, y, 4, NULL);
224 Printing::text("Stack Pointer:" , x, ++y, NULL);
225 Printing::hex(sp, x + 15, y, 4, NULL);
226 Printing::text("Minimum Room: " , x, ++y, NULL);
227 Printing::int32(__STACK_HEADROOM, x + 15, y, NULL);
228 Printing::text("Actual Room: " , x, ++y, NULL);
229 Printing::int32(room, x + 15, y, NULL);
230 Printing::text("Overflow: " , x, ++y, NULL);
231 Printing::int32(__STACK_HEADROOM - room, x + 15, y, NULL);
232 }
233}

244static void HardwareManager::checkMemoryMap()
245{
246 if((uint32)&_dramDataStart < __WORLD_SPACE_BASE_ADDRESS && (uint32)&_dramBssEnd >= __WORLD_SPACE_BASE_ADDRESS)
247 {
248 Printing::setDebugMode();
249 int32 y = 15;
250 uint32 missingSpace = (uint32)&_dramBssEnd - __WORLD_SPACE_BASE_ADDRESS;
251 uint32 recommendedDramStart = (uint32)&_dramDataStart - missingSpace;
252 uint32 recommendedDramSize = (__WORLD_SPACE_BASE_ADDRESS - recommendedDramStart);
253 uint32 recommendedBgmapSegments = (recommendedDramStart - __BGMAP_SPACE_BASE_ADDRESS) / 8192;
254
255 Printing::text("Increase the dram section in the vb.ld file", 1, y++, NULL);
256 Printing::text("Missing space: ", 1, ++y, NULL);
257 Printing::int32(missingSpace, 17, y, NULL);
258 Printing::text("Bytes ", 17 + Math::getDigitsCount(missingSpace) + 1, y++, NULL);
259 Printing::text("WORLD space: ", 1, ++y, NULL);
260 Printing::hex((uint32)__WORLD_SPACE_BASE_ADDRESS, 17, y, 4, NULL);
261 Printing::text("DRAM start: ", 1, ++y, NULL);
262 Printing::hex((uint32)&_dramDataStart, 17, y, 4, NULL);
263 Printing::text("DRAM end: ", 1, ++y, NULL);
264 Printing::hex((uint32)&_dramBssEnd, 17, y++, 4, NULL);
265 Printing::text("Suggested DRAM start: ", 1, ++y, NULL);
266 Printing::hex(recommendedDramStart, 25, y, 4, NULL);
267 Printing::text("Suggested DRAM size: ", 1, ++y, NULL);
268 Printing::int32(recommendedDramSize, 25, y, NULL);
269 Printing::text("Bytes ", 25 + Math::getDigitsCount(recommendedDramSize) + 1, y++, NULL);
270 Printing::text("Maximum BGMAP segments: ", 1, ++y, NULL);
271 Printing::int32(recommendedBgmapSegments, 25, y, NULL);
272
273 NM_ASSERT(false, "HardwareManager::checkMemoryMap: DRAM section overflow");
274 }
275}

307static int32 HardwareManager::getInterruptLevel()
308{
309 int32 level;
310
311 asm
312 (
313 "stsr sr5, r6 \n\t" \
314 "shr 0x10, r6 \n\t" \
315 "andi 0x000F, r6, r6 \n\t" \
316 "mov r6, %0 \n\t"
317 : "=r" (level) // Output
318 : // Input
319 : "r6" // Clobber
320 );
321
322 return level;
323}
