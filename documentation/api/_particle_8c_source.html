84void Particle::setup
85(
86 const ComponentSpec* visualComponentSpec, const ComponentSpec* physicsComponentSpec, const ComponentSpec* colliderComponentSpec,
87 int16 lifeSpan, const Vector3D* position, const Vector3D* force, uint32 movementType, const AnimationFunction** animationFunctions,
88 const char* animationName
89)
90{
91 this->expired = false;
92 this->lifeSpan = lifeSpan;
93
94 // Is this needed? It can be quite heavy.
95 // Particle::resetComponents(this);
96
97 if(NULL != visualComponentSpec)
98 {
99 Particle::removeComponents(this, kSpriteComponent);
100 Particle::removeComponents(this, kWireframeComponent);
101 Particle::addComponent(this, visualComponentSpec);
102 }
103
104 if(NULL != physicsComponentSpec && NULL == this->body)
105 {
106 Particle::removeComponents(this, kPhysicsComponent);
107 this->body = Body::safeCast(Particle::addComponent(this, physicsComponentSpec));
108 }
109
110 if(NULL != colliderComponentSpec)
111 {
112 Particle::removeComponents(this, kColliderComponent);
113 Particle::addComponent(this, colliderComponentSpec);
114 Particle::registerCollisions(this, false);
115 }
116
117 Particle::playAnimation(this, animationFunctions, animationName);
118
119 if(!isDeleted(this->body))
120 {
121 Body::setPosition(this->body, position, Entity::safeCast(this));
122 }
123
124 this->transformation.position = *position;
125
126 if(NULL != force)
127 {
128 if(0 != force->x || 0 != force->y || 0 != force->z)
129 {
130 Particle::applyForce(this, force, movementType);
131 }
132 }
133
134 Particle::show(this);
135}

164bool Particle::isVisible()
165{
166 PixelVector pixelVector;
167
168 int16 halfWidth = __PARTICLE_VISIBILITY_PADDING;
169 int16 halfHeight = __PARTICLE_VISIBILITY_PADDING;
170
171 Vector3D relativeGlobalPosition =
172 Vector3D::rotate(Vector3D::getRelativeToCamera(this->transformation.position), *_cameraInvertedRotation);
173
174 pixelVector = PixelVector::projectVector3D(relativeGlobalPosition, Optics::calculateParallax(relativeGlobalPosition.z));
175
176 // Check x visibility
177 if(pixelVector.x + halfWidth < _cameraFrustum->x0 || pixelVector.x - halfWidth > _cameraFrustum->x1)
178 {
179 return false;
180 }
181
182 // Check y visibility
183 if(pixelVector.y + halfHeight < _cameraFrustum->y0 || pixelVector.y - halfHeight > _cameraFrustum->y1)
184 {
185 return false;
186 }
187
188 // Check z visibility
189 if(pixelVector.z > __SCREEN_DEPTH || pixelVector.z < -(__SCREEN_DEPTH >> 1))
190 {
191 return false;
192 }
193
194 return true;
195}

215bool Particle::update(uint32 elapsedTime, void (* behavior)(Particle particle))
216{
217 if(0 <= this->lifeSpan)
218 {
219 this->lifeSpan -= elapsedTime;
220
221 if(0 > this->lifeSpan)
222 {
223 Particle::expire(this);
224
225 if(!isDeleted(this->body))
226 {
227 Body::stopMovement(this->body, __ALL_AXIS);
228 }
229
230 return true;
231 }
232
233 if(NULL != behavior)
234 {
235 behavior(this);
236 }
237 }
238
239 return false;
240}

250void Particle::applyForce(const Vector3D* force __attribute__ ((unused)), uint32 movementType __attribute__ ((unused)))
251{
252 if(isDeleted(this->body))
253 {
254 return;
255 }
256
257 if(__UNIFORM_MOVEMENT == movementType)
258 {
259 fixed_t mass = Body::getMass(this->body);
260
261 Vector3D acceleration =
262 {
263 force->x,
264 force->y,
265 force->z
266 };
267
268 if(mass && __1I_FIXED != mass)
269 {
270 acceleration.x = __FIXED_DIV(acceleration.x, mass);
271 acceleration.y = __FIXED_DIV(acceleration.y, mass);
272 acceleration.z = __FIXED_DIV(acceleration.z, mass);
273 }
274
275 Vector3D velocity =
276 {
277 acceleration.x,
278 acceleration.y,
279 acceleration.z
280 };
281
282 Body::setVelocity(this->body, &velocity);
283 }
284 else
285 {
286 Body::applyForce(this->body, force);
287 }
288}
