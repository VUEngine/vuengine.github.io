160void CharSet::addChar(uint32 charToAddTo, const uint32* newChar)
161{
162 if(NULL != newChar && charToAddTo < this->charSetSpec->numberOfChars)
163 {
164 Mem::combineWORDs
165 (
166 (uint32*)(__CHAR_SPACE_BASE_ADDRESS + (((uint32)this->offset + charToAddTo) << 4)),
167 (uint32*)&this->charSetSpec->tiles[__UINT32S_PER_CHARS(charToAddTo) + 1] + this->tilesDisplacement,
168 (uint32*)newChar,
169 __UINT32S_PER_CHARS(1)
170 );
171 }
172}

191void CharSet::putPixel(const uint32 charToReplace, const Pixel* charSetPixel, BYTE newPixelColor)
192{
193 if(charSetPixel && charToReplace < this->charSetSpec->numberOfChars && charSetPixel->x < 8 && charSetPixel->y < 8)
194 {
195 static BYTE auxChar[] =
196 {
197 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00,
198 };
199
200 Mem::copyBYTE(auxChar, (uint8*)__CHAR_SPACE_BASE_ADDRESS + (((uint32)this->offset) << 4) + (charToReplace << 4), (int32)(1 << 4));
201
202 uint16 displacement = (charSetPixel->y << 1) + (charSetPixel->x >> 2);
203 uint16 pixelToReplaceDisplacement = (charSetPixel->x % 4) << 1;
204
205 // TODO: review this, only works with non transparency pixels
206 auxChar[displacement] &= (~(0x03 << pixelToReplaceDisplacement) | ((uint16)newPixelColor << pixelToReplaceDisplacement));
207// auxChar[displacement] |= (uint16)newPixelColor << pixelToReplaceDisplacement;
208
209 Mem::copyBYTE
210 (
211 (uint8*)__CHAR_SPACE_BASE_ADDRESS + (((uint32)this->offset) << 4) + (charToReplace << 4), auxChar, (int32)(sizeof(BYTE) << 4)
212 );
213 }
214}

218void CharSet::setFrame(uint16 frame)
219{
220 uint32 tilesDisplacement = 0;
221
222 if(NULL != this->charSetSpec->frameOffsets)
223 {
224 tilesDisplacement = this->charSetSpec->frameOffsets[frame] - 1;
225 }
226 else
227 {
228 tilesDisplacement = __UINT32S_PER_CHARS(this->charSetSpec->numberOfChars * frame);
229 }
230
231 if(!this->written || this->tilesDisplacement != tilesDisplacement)
232 {
233 this->tilesDisplacement = tilesDisplacement;
234
235 CharSet::write(this);
236 }
237}

240
241void CharSet::write()
242{
243 NM_ASSERT(0 < this->charSetSpec->numberOfChars, "CharSet::write: 0 chars");
244
245 uint16 tilesToWrite = this->charSetSpec->numberOfChars;
246
247#ifdef __SHOW_SPRITES_PROFILING
248 extern int32 _writtenTiles;
249 _writtenTiles += tilesToWrite;
250#endif
251
252 switch(this->charSetSpec->tiles[0])
253 {
254 case __CHAR_SET_COMPRESSION_RLE:
255
256 CharSet::writeRLE(this);
257 break;
258
259 default:
260
261 if(100 < tilesToWrite)
262 {
263 CACHE_RESET;
264 }
265
266#ifndef __RELEASE
267 Mem::copyWORD
268 (
269 (uint32*)(__CHAR_SPACE_BASE_ADDRESS + (((uint32)this->offset) << 4)),
270 &this->charSetSpec->tiles[1] + this->tilesDisplacement,
271 __UINT32S_PER_CHARS(tilesToWrite)
272 );
273#else
274 // Based on dasi's memcpy
275 asm
276 (
277 "loop%=: \n\t" \
278 "ld.w 0[%1], r11 \n\t" \
279 "ld.w 4[%1], r12 \n\t" \
280 "ld.w 8[%1], r13 \n\t" \
281 "st.w r11, 0[%0] \n\t" \
282 "addi 16, %0, %0 \n\t" \
283 "ld.w 12[%1], r11 \n\t" \
284 "st.w r12, -12[%0] \n\t" \
285 "addi 16, %1, %1 \n\t" \
286 "st.w r13, -8[%0] \n\t" \
287 "add -1, %2 \n\t" \
288 "st.w r11, -4[%0] \n\t" \
289 "bne loop%= \n\t"
290 : // No Output
291 : "r" ((uint32*)(__CHAR_SPACE_BASE_ADDRESS + (((uint32)this->offset) << 4))),
292 "r" (&this->charSetSpec->tiles[1] + this->tilesDisplacement), "r" (__UINT32S_PER_CHARS(tilesToWrite) >> 2)
293 : "r11", "r12", "r13" // regs used
294 );
295#endif
296 break;
297 }
298
299 this->written = true;
300}

309
310void CharSet::writeRLE()
311{
312 // 1 poxel = 2 pixels = 4 bits = 1 hex digit
313 // So, each char has 32 poxels
314 uint32 totalPoxels = this->charSetSpec->numberOfChars << 5;
315
316 uint32* destination = (uint32*)(__CHAR_SPACE_BASE_ADDRESS + (((uint32)this->offset) << 4));
317 uint32* limit = destination + __UINT32S_PER_CHARS(this->charSetSpec->numberOfChars);
318 uint32* source = &this->charSetSpec->tiles[1] + this->tilesDisplacement;
319
320 uint32 uncompressedData = 0;
321 uint32 uncompressedDataSize = 0;
322
323 CACHE_RESET;
324
325 for(uint32 poxel = 0; poxel < totalPoxels; poxel++)
326 {
327 uint32 compressedData = source[poxel];
328
329 uint32 cycles = 4;
330
331 while(cycles--)
332 {
333 uint8 pack = ((BYTE*)&compressedData)[3];
334
335 uint8 counter = (pack >> 4) + 1;
336 uint8 data = 0x0F & (pack);
337
338 while(0 < counter--)
339 {
340 uncompressedData = (uncompressedData << 4) | data;
341 uncompressedDataSize++;
342
343 if(8 <= uncompressedDataSize)
344 {
345 if(limit <= destination)
346 {
347 return;
348 }
349
350 *destination = uncompressedData;
351 destination++;
352
353 uncompressedData = 0;
354 uncompressedDataSize = 0;
355 }
356 }
357
358 compressedData <<= 8;
359 }
360 }
361}
