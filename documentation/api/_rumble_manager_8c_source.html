29static void RumbleManager::startEffect(const RumbleEffectSpec* rumbleEffect)
30{
31 RumbleManager rumbleManager = RumbleManager::getInstance();
32
33 rumbleManager->rumbleCommandIndex = 0;
34
35 if(!rumbleManager->overridePreviousEffect && 0 != rumbleManager->rumbleCommandIndex)
36 {
37 return;
38 }
39
40 if(NULL == rumbleEffect)
41 {
42 return;
43 }
44
45 if(rumbleManager->rumbleEffectSpec == rumbleEffect)
46 {
47 if(rumbleEffect->stop)
48 {
49 RumbleManager::stop(rumbleManager);
50 }
51
52 RumbleManager::play(rumbleManager);
53 RumbleManager::execute(rumbleManager);
54 return;
55 }
56
57 rumbleManager->rumbleEffectSpec = rumbleEffect;
58
59 if(rumbleEffect->stop)
60 {
61 RumbleManager::stop(rumbleManager);
62 }
63
64 RumbleManager::setFrequency(rumbleManager, rumbleEffect->frequency);
65 RumbleManager::setSustainPositive(rumbleManager, rumbleEffect->sustainPositive);
66 RumbleManager::setSustainNegative(rumbleManager, rumbleEffect->sustainNegative);
67 RumbleManager::setOverdrive(rumbleManager, rumbleEffect->overdrive);
68 RumbleManager::setBreak(rumbleManager, rumbleEffect->breaking);
69 RumbleManager::setEffect(rumbleManager, rumbleEffect->effect);
70 RumbleManager::execute(rumbleManager);
71}

113secure void RumbleManager::reset()
114{
115 this->async = true;
116 this->overridePreviousEffect = true;
117 this->rumbleEffectSpec = NULL;
118 this->rumbleCommandIndex = 0;
119 this->cachedRumbleEffect.frequency = 0;
120 this->cachedRumbleEffect.sustainPositive = 0;
121 this->cachedRumbleEffect.sustainNegative = 0;
122 this->cachedRumbleEffect.overdrive = 0;
123 this->cachedRumbleEffect.breaking = 0;
124
125 for(int32 i = 0; i < __RUMBLE_TOTAL_COMMANDS; i++)
126 {
127 this->rumbleCommands[i] = 0;
128 }
129}

146void RumbleManager::execute()
147{
148// Rumble only is called in release mode since emulators that don't implement communications,
149// lock when trying to broadcast message throught the EXT port
150#ifdef __EMU_ONLY
151#if __EMU_ONLY
152 return;
153#endif
154#endif
155
156#ifdef __RELEASE
157 if(this->async)
158 {
159 if(this->overridePreviousEffect)
160 {
161 CommunicationManager::broadcastDataAsync
162 (
163 CommunicationManager::getInstance(), (BYTE*)this->rumbleCommands, this->rumbleCommandIndex, NULL, NULL
164 );
165 }
166 else
167 {
168 CommunicationManager::broadcastDataAsync
169 (
170 CommunicationManager::getInstance(), (BYTE*)this->rumbleCommands, this->rumbleCommandIndex,
171 (EventListener)RumbleManager::onBroadcastDataDone, ListenerObject::safeCast(this)
172 );
173 }
174 }
175 else
176 {
177 CommunicationManager::broadcastData(
178 CommunicationManager::getInstance(), (BYTE*)this->rumbleCommands, this->rumbleCommandIndex);
179 this->rumbleCommandIndex = 0;
180 }
181#endif
182}

212void RumbleManager::storeEffectChain(uint8 chainNumber, uint8* effectChain)
213{
214 uint8 i = 0;
215
216 RumbleManager::sendCode(this, __RUMBLE_CMD_WRITE_EFFECT_CHAIN);
217
218 RumbleManager::sendCode(this, chainNumber);
219
220 for(i=0; effectChain[i] != __RUMBLE_EFFECT_CHAIN_END && i < __RUMBLE_MAX_EFFECTS_IN_CHAIN; i++)
221 {
222 RumbleManager::setEffect(this, effectChain[i]);
223 }
224
225 RumbleManager::sendCode(this, __RUMBLE_EFFECT_CHAIN_END);
226}

229
230void RumbleManager::setEffectChain(uint8 effectChain)
231{
232 uint8 command = effectChain;
233
234 if(command <= __RUMBLE_CHAIN_EFFECT_4)
235 {
236 command += __RUMBLE_CMD_CHAIN_EFFECT_0;
237 }
238
239 if(command >= __RUMBLE_CMD_CHAIN_EFFECT_0 && command <= __RUMBLE_CMD_CHAIN_EFFECT_4)
240 {
241 RumbleManager::sendCode(this, command);
242 }
243}

247void RumbleManager::setFrequency(uint8 value)
248{
249 if(this->cachedRumbleEffect.frequency == value)
250 {
251 return;
252 }
253
254 this->cachedRumbleEffect.frequency = value;
255
256 if(value <= __RUMBLE_FREQ_400HZ)
257 {
258 value += __RUMBLE_CMD_FREQ_50HZ;
259 }
260
261 if(value >= __RUMBLE_CMD_FREQ_50HZ && value <= __RUMBLE_CMD_FREQ_400HZ)
262 {
263 RumbleManager::sendCode(this, value);
264 }
265}

269void RumbleManager::setOverdrive(uint8 value)
270{
271 if(this->cachedRumbleEffect.overdrive == value)
272 {
273 return;
274 }
275
276 if(__RUMBLE_MAX_OVERDRIVE < value)
277 {
278 value = __RUMBLE_MAX_OVERDRIVE;
279 }
280
281 this->cachedRumbleEffect.overdrive = value;
282
283 RumbleManager::sendCommandWithValue(this, __RUMBLE_CMD_OVERDRIVE, value);
284}
