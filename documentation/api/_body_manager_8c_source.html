85void BodyManager::update()
86{
87 if(__TARGET_FPS < ++this->cycle)
88 {
89 this->cycle = 1;
90 }
91
92 if(__1I_FIXED > this->timeScale)
93 {
94 if(__F_TO_FIXED(0.5f) < this->timeScale)
95 {
96 if(++this->remainingSkipCycles > this->skipCycles)
97 {
98 this->remainingSkipCycles = 0;
99 return;
100 }
101 }
102 else
103 {
104 if(++this->remainingSkipCycles <= this->skipCycles)
105 {
106 return;
107 }
108
109 this->remainingSkipCycles = 0;
110 }
111 }
112
113 this->dirty = false;
114
115 for(VirtualNode node = this->components->head, nextNode = NULL; NULL != node; node = nextNode)
116 {
117 nextNode = node->next;
118
119 Body body = Body::safeCast(node->data);
120
121 NM_ASSERT(!isDeleted(body), "BodyManager::update: deleted body");
122
123 if(body->deleteMe)
124 {
125 // Place in the removed bodies list
126 VirtualList::removeNode(this->components, node);
127
128 delete body;
129 continue;
130 }
131
132 if(!body->awake)
133 {
134 continue;
135 }
136
137 if(__NO_AXIS != body->axisSubjectToGravity && Entity::isSubjectToGravity(body->owner, this->gravity))
138 {
139 // Check if necessary to apply gravity
140 uint16 movingState = Body::getMovementOnAllAxis(body);
141
142 uint16 gravitySensibleAxis =
143 body->axisSubjectToGravity
144 &
145 (
146 (__X_AXIS & ~(__X_AXIS & movingState) ) | (__Y_AXIS & ~(__Y_AXIS & movingState)) |
147 (__Z_AXIS & ~(__Z_AXIS & movingState))
148 );
149
150 if(__NO_AXIS != gravitySensibleAxis)
151 {
152 fixed_t mass = Body::getMass(body);
153
154 Vector3D force =
155 {
156 __X_AXIS & gravitySensibleAxis ? __FIXED_MULT(this->gravity.x, mass) : 0,
157 __Y_AXIS & gravitySensibleAxis ? __FIXED_MULT(this->gravity.y, mass) : 0,
158 __Z_AXIS & gravitySensibleAxis ? __FIXED_MULT(this->gravity.z, mass) : 0,
159 };
160
161 Body::applyForce(body, &force);
162 }
163 }
164
165 Body::update(body, this->cycle, __PHYSICS_TIME_ELAPSED_STEP);
166 }
167
168#ifdef __SHOW_PHYSICS_PROFILING
169 BodyManager::print(this, 1, 1);
170#endif
171}

175Body BodyManager::createBody(Entity owner, const BodySpec* bodySpec)
176{
177 if(this->dirty)
178 {
179 for(VirtualNode node = this->components->head, nextNode = NULL; NULL != node; node = nextNode)
180 {
181 nextNode = node->next;
182
183 Body body = Body::safeCast(node->data);
184
185 if(body->deleteMe)
186 {
187 // Place in the removed bodies list
188 VirtualList::removeNode(this->components, node);
189
190 delete body;
191 }
192 }
193 }
194
195 // If the actor is already registered
196 Body body = new Body(owner, bodySpec);
197 VirtualList::pushFront(this->components, body);
198 ASSERT(Body::safeCast(VirtualList::front(this->components)), "BodyManager::createBody: bad class body");
199
200 // Return created body
201 return body;
202}

206void BodyManager::destroyBody(Body body)
207{
208 ASSERT(!isDeleted(body), "BodyManager::destroyBody: dead body");
209 ASSERT(VirtualList::find(this->components, body), "BodyManager::destroyBody: body not registered");
210
211 if(!isDeleted(body))
212 {
213 body->deleteMe = true;
214 this->dirty = true;
215 }

220void BodyManager::setTimeScale(fixed_t timeScale)
221{
222 this->timeScale = timeScale;
223
224 if(this->timeScale > __1I_FIXED)
225 {
226 this->timeScale = __1I_FIXED;
227 }
228 else if(0 >= timeScale)
229 {
230 this->timeScale = __F_TO_FIXED(0.1f);
231 }
232
233 this->remainingSkipCycles = 0;
234 this->skipCycles = 0;
235
236 if(__F_TO_FIXED(0.5f) < this->timeScale)
237 {
238 uint32 gameFramesPerSecond = __TARGET_FPS / __PHYSICS_TIME_ELAPSED_DIVISOR;
239 fixed_t targetUpdatesPerSecond = __FIXED_MULT(__I_TO_FIXED(gameFramesPerSecond), this->timeScale);
240 fixed_t targetSkipsPerSecond = __I_TO_FIXED(gameFramesPerSecond) - targetUpdatesPerSecond;
241
242 if(targetSkipsPerSecond)
243 {
244 this->skipCycles = __FIXED_TO_I(__FIXED_DIV(targetUpdatesPerSecond, targetSkipsPerSecond) + __05F_FIXED);
245 }
246 }
247 else
248 {
249 this->skipCycles = __FIXED_TO_I(__FIXED_DIV(__1I_FIXED, this->timeScale) - __1I_FIXED + __05F_FIXED);
250 }
251}

359
360Body BodyManager::getBody(Entity owner)
361{
362 ASSERT(this->components, "BodyManager::getBody: null bodies");
363
364 VirtualNode node = this->components->head;
365
366 for(; NULL != node; node = node->next)
367 {
368 // Current body
369 Body body = Body::safeCast(node->data);
370 ASSERT(body, "BodyManager::getBody: null body");
371
372 // Check if current body's owner is the same as the actor calling this method
373 if(owner == body->owner)
374 {
375 return body;
376 }
377 }
378
379 return NULL;
380}

387

389

391 {
392 // Current body

394
395 // Check if current body's owner is the same as the actor calling this method
396 if(Entity::safeCast(owner) == body->owner)
397 {
398 return true;
399 }
400 }
401
402 return false;
