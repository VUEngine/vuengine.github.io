52static inline Optical Optical::getFromPixelOptical(PixelOptical pixelOptical, CameraFrustum cameraFrustum)
53{
54 int16 maximumXViewDistancePower = -__PIXELS_PER_METER_2_POWER;
55 int16 maximumYViewDistancePower = -__PIXELS_PER_METER_2_POWER;
56
57 while(pixelOptical.maximumXViewDistance)
58 {
59 pixelOptical.maximumXViewDistance >>= 1;
60 maximumXViewDistancePower++;
61 }
62
63 while(pixelOptical.maximumYViewDistance)
64 {
65 pixelOptical.maximumYViewDistance >>= 1;
66 maximumYViewDistancePower++;
67 }
68
69 Optical optical;
70
71 optical.maximumXViewDistancePower =
72 __MINIMUM_X_VIEW_DISTANCE_POWER > maximumXViewDistancePower ? __MINIMUM_X_VIEW_DISTANCE_POWER : maximumXViewDistancePower;
73 optical.maximumYViewDistancePower =
74 __MINIMUM_Y_VIEW_DISTANCE_POWER > maximumYViewDistancePower ? __MINIMUM_Y_VIEW_DISTANCE_POWER : maximumYViewDistancePower;
75 optical.cameraNearPlane = __PIXELS_TO_METERS(pixelOptical.cameraNearPlane);
76 optical.baseDistance = __PIXELS_TO_METERS(pixelOptical.baseDistance);
77 optical.horizontalViewPointCenter = __PIXELS_TO_METERS(pixelOptical.horizontalViewPointCenter);
78 optical.verticalViewPointCenter = __PIXELS_TO_METERS(pixelOptical.verticalViewPointCenter);
79 optical.scalingFactor = __F_TO_FIXED(pixelOptical.scalingFactor);
80
81 return Optical::applyCameraFrustum(optical, cameraFrustum);
82}

86static inline Optical Optical::applyCameraFrustum(Optical optical, CameraFrustum cameraFrustum)
87{
88 Optical result = optical;
89
90 result.halfWidth = __PIXELS_TO_METERS((cameraFrustum.x1 - cameraFrustum.x0) >> 1);
91 result.halfHeight = __PIXELS_TO_METERS((cameraFrustum.y1 - cameraFrustum.y0) >> 1);
92 result.aspectRatio = __FIXED_EXT_DIV(result.halfWidth, result.halfHeight);
93 // this assumes a fov of 90 degrees (__CAMERA_FOV_DEGREES fix7_9) to speed up computations
94 // fov = 1 / tan(angle / 2)
95 result.fov =
96 __FIXED_EXT_DIV
97 (
98 __I_TO_FIXED_EXT(1),
99 __FIX7_9_TO_FIXED(__FIX7_9_DIV(__SIN(__CAMERA_FOV_DEGREES >> 1), __COS(__CAMERA_FOV_DEGREES >> 1)))
100 );
101 result.aspectRatioXfov = __FIXED_MULT(result.aspectRatio, result.fov);
102 // farRatio1Near = // (far + near) / (far - near)
103 result.farRatio1Near = __FIXED_EXT_DIV(cameraFrustum.z1 + cameraFrustum.z0, cameraFrustum.z1 - cameraFrustum.z0);
104 // farRatio2Near = // (2 * far * near) / (near - far)
105 result.farRatio2Near = __FIXED_EXT_DIV(__FIXED_EXT_MULT(cameraFrustum.z1, cameraFrustum.z0) << 1, cameraFrustum.z0 - cameraFrustum.z1);
106 result.projectionMultiplierHelper = __FIXED_EXT_MULT(result.halfWidth, result.aspectRatioXfov);
107 result.scalingMultiplier = __FIXED_EXT_MULT(result.halfWidth, result.scalingFactor);
108
109 if(0 == result.cameraNearPlane)
110 {
111 result.cameraNearPlane = result.projectionMultiplierHelper;
112 }
113
114 return result;
115}
