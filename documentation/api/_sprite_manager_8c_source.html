110void SpriteManager::reset()
111{
112 HardwareManager::suspendInterrupts();
113
114 Texture::reset();
115 Printing::reset();
116 CharSetManager::reset(CharSetManager::getInstance());
117 BgmapTextureManager::reset(BgmapTextureManager::getInstance());
118 ParamTableManager::reset(ParamTableManager::getInstance());
119
120 SpriteManager::cleanUp(this);
121 ObjectSpriteContainer::reset();
122
123 int32 i = 0;
124 // Clean OBJ memory
125 for(; i < __TOTAL_OBJECTS; i++)
126 {
127 _vipRegisters[__SPT3 - i] = 0;
128 _objectAttributesCache[i].jx = 0;
129 _objectAttributesCache[i].head = 0;
130 _objectAttributesCache[i].jy = 0;
131 _objectAttributesCache[i].tile = 0;
132 }
133
134 this->animationsClock = NULL;
135 this->bgmapSprites = new VirtualList();
136 this->objectSpriteContainers = new VirtualList();
137 this->specialSprites = new VirtualList();
138
139 this->freeLayer = __TOTAL_LAYERS - 1;
140 this->deferTextureUpdating = false;
141 this->texturesMaximumRowsToWrite = -1;
142 this->sortingSpriteNode = NULL;
143 this->completeSort = true;
144
145 SpriteManager::stopRendering(this);
146
147 this->evenFrame = __TRANSPARENCY_EVEN;
148
149 VIPManager::addEventListener
150 (
151 VIPManager::getInstance(), ListenerObject::safeCast(this), (EventListener)SpriteManager::onVIPGAMESTART,
152 kEventVIPManagerGAMESTART
153 );
154
155 VIPManager::addEventListener
156 (
157 VIPManager::getInstance(), ListenerObject::safeCast(this), (EventListener)SpriteManager::onVIPXPEND,
158 kEventVIPManagerXPEND

187Sprite SpriteManager::createSprite(Entity owner, const SpriteSpec* spriteSpec)
188{
189 NM_ASSERT(NULL != spriteSpec, "SpriteManager::createSprite: null spriteSpec");
190
191 if(NULL == spriteSpec)
192 {
193 return NULL;
194 }
195
196 Sprite sprite = ((Sprite (*)(Entity, const SpriteSpec*)) ((ComponentSpec*)spriteSpec)->allocator)(owner, spriteSpec);
197 ASSERT(!isDeleted(sprite), "SpriteManager::createSprite: failed creating sprite");
198
199 VirtualList::pushBack(this->components, sprite);
200
201 Sprite::transform(sprite);
202
203 ClassPointer managerClassPointer = Sprite::getManagerClass(sprite);
204
205 if(typeofclass(SpriteManager) == managerClassPointer)
206 {
207 SpriteManager::registerSprite(this, sprite);
208 }
209 else if(typeofclass(ObjectSpriteContainer) == managerClassPointer)
210 {
211 int16 z = 0;
212
213 if(NULL != sprite->transformation)
214 {
215 z = __METERS_TO_PIXELS(sprite->transformation->position.z);
216 }
217
218 ObjectSpriteContainer objectSpriteContainer = SpriteManager::getObjectSpriteContainer(this, z + sprite->displacement.z);
219
220 NM_ASSERT(!isDeleted(objectSpriteContainer), "SpriteManager::createSprite: couldn't get a manager");
221 ObjectSpriteContainer::registerSprite(objectSpriteContainer, ObjectSprite::safeCast(sprite));
222 }
223
224 return sprite;
225}

227//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
228
229void SpriteManager::destroySprite(Sprite sprite)
230{
231 NM_ASSERT(!isDeleted(sprite), "SpriteManager::destroySprite: trying to dispose dead sprite");
232 NM_ASSERT(__GET_CAST(Sprite, sprite), "SpriteManager::destroySprite: trying to dispose a non sprite");
233
234 if(isDeleted(sprite))
235 {
236 return;
237 }
238
239 VirtualList::removeData(this->components, sprite);
240
241 Sprite::hide(sprite);
242
243 ClassPointer managerClassPointer = Sprite::getManagerClass(sprite);
244
245 if(typeofclass(SpriteManager) == managerClassPointer)
246 {
247 SpriteManager::unregisterSprite(this, sprite);
248 }
249 else if(typeofclass(ObjectSpriteContainer) == managerClassPointer)
250 {
251 ObjectSpriteContainer objectSpriteContainer = ObjectSpriteContainer::safeCast(Sprite::getManager(sprite));
252
253 NM_ASSERT(!isDeleted(objectSpriteContainer), "SpriteManager::destroySprite: couldn't get a manager");
254 ObjectSpriteContainer::unregisterSprite(objectSpriteContainer, ObjectSprite::safeCast(sprite));
255 }
256
257 delete sprite;
258}

260//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
261
262bool SpriteManager::registerSprite(Sprite sprite)
263{
264#ifndef __RELEASE
265 bool registeringSprite = false;
266
267 NM_ASSERT(!registeringSprite, "SpriteManager::registerSprite: already registering a sprite!");
268
269 if(registeringSprite)
270 {
271 return false;
272 }
273
274 registeringSprite = true;
275
276 ASSERT(Sprite::safeCast(sprite), "SpriteManager::registerSprite: adding no sprite");
277
278 ASSERT(!__GET_CAST(ObjectSprite, sprite), "SpriteManager::registerSprite: trying to register an object sprite");
279
280 if(VirtualList::find(this->bgmapSprites, sprite))
281 {
282 Printing::setDebugMode();
283 Printing::clear();
284 Printing::text(__GET_CLASS_NAME(sprite), 1, 20, NULL);
285 NM_ASSERT(false, "SpriteManager::registerSprite: sprite already registered");
286 }
287#endif
288
289 if(!isDeleted(sprite))
290 {
291 if(SpriteManager::doRegisterSprite(this, sprite) && Sprite::hasSpecialEffects(sprite))
292 {
293 VirtualList::pushBack(this->specialSprites, sprite);
294 }
295
296#ifndef __RELEASE
297 registeringSprite = false;
298#endif
299 return true;
300 }
301
302#ifndef __RELEASE
303 registeringSprite = false;
304#endif
305
306 NM_ASSERT(sprite, "SpriteManager::registerSprite: null sprite");
307 return false;
308}

334void SpriteManager::configureObjectSpriteContainers(int16 size[__TOTAL_OBJECT_SEGMENTS], int16 z[__TOTAL_OBJECT_SEGMENTS])
335{
336#ifndef __RELEASE
337 int16 previousZ = z[__TOTAL_OBJECT_SEGMENTS - 1];
338#endif
339
340 if(isDeleted(this->objectSpriteContainers) || 0 < VirtualList::getCount(this->objectSpriteContainers))
341 {
342 return;
343 }
344
345 for(int32 i = __TOTAL_OBJECT_SEGMENTS; i--; )
346 {
347 NM_ASSERT(z[i] <= previousZ, "SpriteManager::configureObjectSpriteContainers: wrong z");
348
349 if(0 < size[i])
350 {
351 ObjectSpriteContainer objectSpriteContainer = new ObjectSpriteContainer();
352
353 SpriteManager::registerSprite(this, Sprite::safeCast(objectSpriteContainer));
354 VirtualList::pushBack(this->objectSpriteContainers, objectSpriteContainer);
355
356 PixelVector position =
357 {
358 0, 0, z[i], 0
359 };
360
361 if(size[i])
362 {
363 ObjectSpriteContainer::setPosition(objectSpriteContainer, &position);
364 }
365
366#ifndef __RELEASE
367 previousZ = z[i];
368#endif
369 }
370 }
371}

424//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
425
426void SpriteManager::prepareAll()
427{
428 bool deferTextureUpdating = this->deferTextureUpdating;
429
430 // Prevent VIP's interrupt from calling render during spriteManager process
431 VIPManager::suspendDrawing(VIPManager::getInstance());
432
433 // Must make sure that all textures are completely written
434 SpriteManager::deferParamTableEffects(this, false);
435
436 // Make sure all textures are written right now
437 SpriteManager::writeTextures(this);
438
439 // Sort all sprites' layers
440 SpriteManager::sortSprites(this);
441
442 // Render and draw sprites as soon as possible
443 SpriteManager::renderAndDraw(this);
444
445 // Sort all sprites' layers again
446 // Don't remove me, some custom sprites depend on others
447 // To have been setup up before
448 SpriteManager::sortSprites(this);
449
450 // Render and draw sprites as soon as possible again
451 SpriteManager::renderAndDraw(this);
452
453 // Defer rendering again
454 SpriteManager::deferParamTableEffects(this, true);
455
456 // Restore drawing
457 VIPManager::resumeDrawing(VIPManager::getInstance());
458
459 this->deferTextureUpdating = deferTextureUpdating;
460}

462//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
463
464void SpriteManager::render()
465{
466#ifdef __SHOW_SPRITES_PROFILING
467 _renderedSprites = 0;
468#endif
469
470 this->completeSort = SpriteManager::sortProgressively(this, this->completeSort);
471
472 ParamTableManager::defragment(ParamTableManager::getInstance(), true);
473
474 // Switch between even and odd frame
475 this->evenFrame = __TRANSPARENCY_EVEN == this->evenFrame ? __TRANSPARENCY_ODD : __TRANSPARENCY_EVEN;
476
477 this->freeLayer = __TOTAL_LAYERS - 1;
478
479 bool updateAnimations = true;
480
481 if(!isDeleted(this->animationsClock))
482 {
483 updateAnimations = !Clock::isPaused(this->animationsClock);
484 }
485
486 for(VirtualNode node = this->bgmapSprites->tail; NULL != node && 0 < this->freeLayer; node = node->previous)
487 {
488 NM_ASSERT(!isDeleted(node->data), "SpriteManager::render: NULL node's data");
489
490 Sprite sprite = Sprite::safeCast(node->data);
491
492 // Saves on method calls quite a bit when there are lots of
493 // Sprites. Don't remove.
494 if(__HIDE == sprite->show || (sprite->transparency & this->evenFrame))
495 {
496 sprite->index = __NO_RENDER_INDEX;
497 continue;
498 }
499
500 if(Sprite::render(sprite, this->freeLayer, updateAnimations) == this->freeLayer)
501 {
502 this->freeLayer--;
503 }
504 }
505
506 NM_ASSERT(0 <= this->freeLayer, "SpriteManager::render: more sprites than WORLDs");
507
508 ObjectSpriteContainer::prepareForRendering();
509
510 for(VirtualNode node = this->objectSpriteContainers->head; NULL != node; node = node->next)
511 {
512 ObjectSpriteContainer objectSpriteContainer = ObjectSpriteContainer::safeCast(node->data);
513
514 ObjectSpriteContainer::renderSprites(objectSpriteContainer, this->evenFrame, updateAnimations);
515 }
516
517 ObjectSpriteContainer::finishRendering();
518
519 SpriteManager::stopRendering(this);
520
521#ifdef __SHOW_SPRITES_PROFILING
522 SpriteManager::computeTotalPixelsDrawn(this);
523#endif
524}

546//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
547
548void SpriteManager::writeDRAM()
549{
550#ifdef __SHOW_SPRITES_PROFILING
551 _writtenTiles = 0;
552 _writtenTextureTiles = 0;
553 _writtenObjectTiles = 0;
554#endif
555
556 // Update all graphical data
557
558 // Update CHAR memory
559 CharSetManager::defragment(CharSetManager::getInstance(), true);
560
561 // Update DRAM memory
562 Texture::updateTextures(this->texturesMaximumRowsToWrite, this->deferTextureUpdating);
563
564 // Update param tables
565 SpriteManager::applySpecialEffects(this);
566
567 // Finally, write OBJ and WORLD attributes to DRAM
568 ObjectSpriteContainer::writeDRAM();
569
570 // Finally, write OBJ and WORLD attributes to DRAM
571 SpriteManager::writeWORLDAttributesToDRAM(this);
572
573#ifdef __SHOW_SPRITES_PROFILING
574 int32 counter = __TARGET_FPS / 5;
575
576 if(0 >= --counter)
577 {
578 counter = __TARGET_FPS / 10;
579 SpriteManager::print(this, 1, 15, true);
580 }
581#endif
582}

597void SpriteManager::showAllSprites(Sprite spareSprite, bool showPrinting)
598{
599 for(VirtualNode node = this->components->tail; NULL != node; node = node->previous)
600 {
601 NM_ASSERT(!isDeleted(node->data), "SpriteManager::showSprites: NULL node's data");
602
603 Sprite sprite = Sprite::safeCast(node->data);
604
605 if(sprite == spareSprite)
606 {
607 Sprite::forceHide(spareSprite);
608 continue;
609 }
610
611 Sprite::forceShow(sprite);
612
613 Sprite::setPosition(sprite, &sprite->position);
614
615 _worldAttributesBaseAddress[sprite->index].head &= ~__WORLD_END;
616 }
617
618 if(showPrinting)
619 {
620 Printing::show();
621 }
622 else
623 {
624 Printing::hide();
625 }
626
627 SpriteManager::stopRendering(this);
628}

702//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
703
704ObjectSpriteContainer SpriteManager::getObjectSpriteContainer(fixed_t z)
705{
706 ObjectSpriteContainer suitableObjectSpriteContainer = NULL;
707
708 NM_ASSERT
709 (
710 !isDeleted(this->objectSpriteContainers),
711 "SpriteManager::getObjectSpriteContainer: no ObjectSpriteContainers created"
712 );
713
714 NM_ASSERT
715 (
716 0 < VirtualList::getCount(this->objectSpriteContainers),
717 "SpriteManager::getObjectSpriteContainer: no ObjectSpriteContainers available"
718 );
719
720 if(isDeleted(this->objectSpriteContainers))
721 {
722 return NULL;
723 }
724
725 for(VirtualNode node = this->objectSpriteContainers->head; NULL != node; node = node->next)
726 {
727 ObjectSpriteContainer objectSpriteContainer = ObjectSpriteContainer::safeCast(node->data);
728
729 if(NULL == suitableObjectSpriteContainer)
730 {
731 suitableObjectSpriteContainer = objectSpriteContainer;
732 }
733 else
734 {
735 if(__ABS(Sprite::getPosition(objectSpriteContainer)->z - z) < __ABS(Sprite::getPosition(suitableObjectSpriteContainer)->z - z))
736 {
737 suitableObjectSpriteContainer = objectSpriteContainer;
738 }
739 }
740 }
741
742 NM_ASSERT(suitableObjectSpriteContainer, "SpriteManager::getObjectSpriteContainer: no suitable ObjectSpriteContainers found");
743
744 return suitableObjectSpriteContainer;
745}

774//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
775
776void SpriteManager::print(int32 x, int32 y, bool resumed)
777{
778 Printing::setWorldCoordinates(0, 0, Printing::getSpriteIndex().z, 0);
779#ifndef __SHOW_SPRITES_PROFILING
780 SpriteManager::computeTotalPixelsDrawn(this);
781#endif
782
783 Printing::text("SPRITES USAGE", x, y++, NULL);
784 Printing::text("Total pixels: ", x, ++y, NULL);
785 Printing::int32(this->totalPixelsDrawn, x + 22, y, NULL);
786 Printing::text("Used layers: ", x, ++y, NULL);

904bool SpriteManager::doRegisterSprite(Sprite sprite)
905{
906 for(VirtualNode node = this->bgmapSprites->head; NULL != node; node = node->next)
907 {
908 NM_ASSERT(!isDeleted(node->data), "SpriteManager::doRegisterSprite: NULL node's data");
909
910 Sprite otherSprite = Sprite::safeCast(node->data);
911
912 NM_ASSERT(otherSprite != sprite, "SpriteManager::doRegisterSprite: sprite already registered");
913
914 if(otherSprite == sprite)
915 {
916 return false;
917 }
918
919 if(sprite->position.z + sprite->displacement.z <= otherSprite->position.z + otherSprite->displacement.z)
920 {
921 this->sortingSpriteNode = VirtualList::insertBefore(this->bgmapSprites, node, sprite);
922 return true;
923 }
924 }
925
926 this->sortingSpriteNode = VirtualList::pushBack(this->bgmapSprites, sprite);
927
928 return true;

931//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
932
933bool SpriteManager::sortProgressively(bool complete)
934{
935 if(NULL == this->sortingSpriteNode)
936 {
937 this->sortingSpriteNode = this->bgmapSprites->head;
938
939 if(NULL == this->sortingSpriteNode)
940 {
941 return false;
942 }
943 }
944
945 bool swapped = false;
946
947 do
948 {
949 swapped = false;
950
951 for
952 (
953 VirtualNode node = complete ? this->bgmapSprites->head : this->sortingSpriteNode; NULL != node && NULL != node->next;
954 node = node->next
955 )
956 {
957 VirtualNode nextNode = node->next;
958
959 NM_ASSERT(!isDeleted(node->data), "SpriteManager::sortProgressively: NULL node's data");
960 ASSERT(__GET_CAST(Sprite, nextNode->data), "SpriteManager::sortProgressively: node's data isn't a sprite");
961
962 Sprite sprite = Sprite::safeCast(node->data);
963
964 NM_ASSERT(!isDeleted(nextNode->data), "SpriteManager::sortProgressively: NULL nextNode's data");
965 ASSERT(__GET_CAST(Sprite, nextNode->data), "SpriteManager::sortProgressively: NULL nextNode's data cast");
966
967 Sprite nextSprite = Sprite::safeCast(nextNode->data);
968
969 // Check if z positions are swapped
970 if(nextSprite->position.z + nextSprite->displacement.z < sprite->position.z + sprite->displacement.z)
971 {
972 // Swap nodes' data
973 node->data = nextSprite;
974 nextNode->data = sprite;
975
976 node = nextNode;
977
978 swapped = true;
979 }
980
981 if(!complete)
982 {
983 break;
984 }
985 }
986 }
987 while(complete && swapped);
988
989 if(!complete)
990 {
991 this->sortingSpriteNode = this->sortingSpriteNode->next;
992 }
993
994 if(!isDeleted(this->objectSpriteContainers))
995 {
996 for(VirtualNode node = this->objectSpriteContainers->head; NULL != node; node = node->next)
997 {
998 ObjectSpriteContainer objectSpriteContainer = ObjectSpriteContainer::safeCast(node->data);
999
1000 swapped = swapped || ObjectSpriteContainer::sortProgressively(objectSpriteContainer, complete);
1001 }
1002 }
1003
1004 return swapped;
1005}
