141static inline Rotation Rotation::clamp(fixed_ext_t x, fixed_ext_t y, fixed_ext_t z)
142{
143 if(0 > x)
144 {
145 x += __FULL_ROTATION_DEGREES;
146 }
147 else if(__FULL_ROTATION_DEGREES <= x)
148 {
149 x -= __FULL_ROTATION_DEGREES;
150 }
151
152 if(0 > y)
153 {
154 y += __FULL_ROTATION_DEGREES;
155 }
156 else if(__FULL_ROTATION_DEGREES <= y)
157 {
158 y -= __FULL_ROTATION_DEGREES;
159 }
160
161 if(0 > z)
162 {
163 z += __FULL_ROTATION_DEGREES;
164 }
165 else if(__FULL_ROTATION_DEGREES <= z)
166 {
167 z -= __FULL_ROTATION_DEGREES;
168 }
169
170 return (Rotation){__FIXED_EXT_TO_FIXED(x), __FIXED_EXT_TO_FIXED(y), __FIXED_EXT_TO_FIXED(z)};
171}

175static inline fixed_t Rotation::getShortestDifferce(fixed_t angleFrom, fixed_t angleTo)
176{
177 int32 rotationDifference = (__FIXED_TO_I(angleTo) - __FIXED_TO_I(angleFrom) + 256 ) % 512 - 256;
178 return __I_TO_FIXED(-256 > rotationDifference ? rotationDifference + 512 : rotationDifference);
179}

218{
219 if(0 != scalar)
220 {
221 return
222 Rotation::clamp(__FIXED_EXT_DIV(rotation.x, scalar), __FIXED_EXT_DIV(rotation.y, scalar), __FIXED_EXT_DIV(rotation.z, scalar));
223 }
