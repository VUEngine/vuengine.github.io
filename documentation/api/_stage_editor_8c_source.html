116void StageEditor::processUserInput(uint16 pressedKey)
117{
118 if(isDeleted(this->stage))
119 {
120 return;
121 }
122
123 if(pressedKey & K_SEL)
124 {
125 this->state++;
126
127 if(kLastState <= this->state)
128 {
129 this->state = kFirstState + 1;
130 }
131
132 StageEditor::configureState(this);
133 return;
134 }
135
136 switch(this->state)
137 {
138 case kMoveCamera:
139
140 StageEditor::moveCamera(this, pressedKey);
141 break;
142
143 case kChangeProjection:
144
145 StageEditor::changeProjection(this, pressedKey);
146 break;
147
148 case kTranslateActors:
149
150 StageEditor::translateActor(this, pressedKey);
151 break;
152
153 case kAddObjects:
154
155 StageEditor::selectUserObject(this, pressedKey);
156 break;
157 }
158}

168void StageEditor::constructor()
169{
170 // Always explicitly call the base's constructor
171 Base::constructor();
172
173 this->actorNode = NULL;
174 this->userActorSprite = NULL;
175 this->state = kFirstState + 1;
176 this->wireframe = NULL;
177 this->userActorSelector = new OptionsSelector(2, 12, NULL, NULL, NULL);
178
179 VirtualList userObjects = new VirtualList();
180
181 int32 i = 0;
182 for(; _userObjects[i].actorSpec; i++)
183 {
184 Option* option = new Option;
185 option->value = _userObjects[i].name;
186 option->type = kString;
187 VirtualList::pushBack(userObjects, option);
188 }
189
190 if(VirtualList::getCount(userObjects))
191 {
192 OptionsSelector::setOptions(this->userActorSelector, userObjects);
193 }
194
195 delete userObjects;
196
197 this->translationStepSize = __DEFAULT_TRANSLATION_STEP;
198}

230//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
231
232void StageEditor::configureState()
233{
234 Printing::clear();
235 StageEditor::printHeader(this);
236 StageEditor::removePreviousSprite(this);
237
238 switch(this->state)
239 {
240 case kAddObjects:
241
242 if(OptionsSelector::getNumberOfOptions(this->userActorSelector))
243 {
244 StageEditor::releaseWireframe(this);
245 StageEditor::printUserObjects(this);
246 StageEditor::showSelectedUserObject(this);
247 break;
248 }
249
250 this->state = kMoveCamera;
251
252 case kMoveCamera:
253
254 StageEditor::releaseWireframe(this);
255 StageEditor::printCameraPosition(this);
256 StageEditor::printTranslationStepSize(this, 38, 7);
257 break;
258
259 case kChangeProjection:
260
261 StageEditor::releaseWireframe(this);
262 StageEditor::printProjectionValues(this);
263 StageEditor::printTranslationStepSize(this, 38, 10);
264 break;
265
266 case kTranslateActors:
267
268 if(!this->actorNode)
269 {
270 StageEditor::selectNextActor(this);
271 }
272 else
273 {
274 StageEditor::highLightActor(this);
275 }
276
277 StageEditor::printActorPosition(this);
278 StageEditor::printTranslationStepSize(this, 38, 8);
279 break;
280 }
281}

297void StageEditor::highLightActor()
298{
299 if(NULL != this->actorNode)
300 {
301 StageEditor::printActorPosition(this);
302
303 Actor actor = Actor::safeCast(VirtualNode::getData(this->actorNode));
304
305 if(!isDeleted(actor))
306 {
307 int16 width = __METERS_TO_PIXELS(Actor::getWidth(actor)) << 2;
308 int16 height = __METERS_TO_PIXELS(Actor::getHeight(actor)) << 2;
309 fixed_t parallax = Optics::calculateParallax(Actor::getPosition(actor)->z);
310
311 const PixelVector MeshesSegments[][2]=
312 {
313 {
314 PixelVector::getFromVector3D((Vector3D){-width / 2, -height / 2, 0}, parallax),
315 PixelVector::getFromVector3D((Vector3D){width / 2, -height / 2, 0}, parallax),
316 },
317 {
318 PixelVector::getFromVector3D((Vector3D){-width / 2, height / 2, 0}, parallax),
319 PixelVector::getFromVector3D((Vector3D){width / 2, height / 2, 0}, parallax),
320 },
321 {
322 PixelVector::getFromVector3D((Vector3D){-width / 2, -height / 2, 0}, parallax),
323 PixelVector::getFromVector3D((Vector3D){-width / 2, height / 2, 0}, parallax),
324 },
325 {
326 PixelVector::getFromVector3D((Vector3D){width / 2, -height / 2, 0}, parallax),
327 PixelVector::getFromVector3D((Vector3D){width / 2, height / 2, 0}, parallax),
328 },
329
330 // Limiter
331 {
332 {0, 0, 0, 0},
333 {0, 0, 0, 0}
334 },
335 };
336
337 MeshSpec meshSpec =
338 {
339 {
340 {
341 // Allocator
342 __TYPE(Mesh),
343
344 // Component type
345 kWireframeComponent
346 },
347
349 {0, 0, 0},
350
352 __COLOR_BRIGHT_RED,
353
355 __TRANSPARENCY_NONE,
356
358 false
359 },
360
361 // Segments
362 (PixelVector(*)[2])MeshesSegments
363 };
364
365 this->wireframe =
366 WireframeManager::createWireframe
367 (
368 Entity::safeCast(actor), (WireframeSpec*)&meshSpec
369 );
370 }
371 }
372 else
373 {
374 Printing::text("No actors in stage", 1, 4, NULL);
375 }
376}

380void StageEditor::selectPreviousActor()
381{
382 if(isDeleted(this->stage))
383 {
384 return;
385 }
386
387 StageEditor::releaseWireframe(this);
388
389 VirtualList stageActors = (Container::safeCast(this->stage))->children;
390
391 if(!this->actorNode)
392 {
393 this->actorNode = stageActors ? stageActors->tail : NULL;
394 }
395 else
396 {
397 this->actorNode = VirtualNode::getPrevious(this->actorNode);
398
399 if(!this->actorNode)
400 {
401 this->actorNode = stageActors ? stageActors->tail : NULL;
402 }
403 }
404
405 if(this->actorNode)
406 {
407 StageEditor::highLightActor(this);
408 }

472 Vector3D translation =
473 {
474 0,
475 __PIXELS_TO_METERS(-this->translationStepSize),
476 0
477 };
478
479 StageEditor::applyTranslationToCamera(this, translation);
480 }
481 else if(pressedKey & K_LD)
482 {
483 Vector3D translation =
484 {
485 0,
486 __PIXELS_TO_METERS(this->translationStepSize),
487 0
488 };
489
490 StageEditor::applyTranslationToCamera(this, translation);
491 }
492 else if(pressedKey & K_RU)
493 {
494 Vector3D translation =
495 {
496 0,
497 0,
498 __PIXELS_TO_METERS(this->translationStepSize),
499 };
500
501 StageEditor::applyTranslationToCamera(this, translation);
502 }
503 else if(pressedKey & K_RD)

505 Vector3D translation =
506 {
507 0,

535
536void StageEditor::changeProjection(uint32 pressedKey)
537{
538 if(isDeleted(this->stage))
539 {
540 return;
541 }
542
543 Optical optical = *_optical;
544
545 if(pressedKey & K_LL)
546 {
547 optical.horizontalViewPointCenter -= __PIXELS_TO_METERS(this->translationStepSize);
548 }
549 else if(pressedKey & K_LR)
550 {
551 optical.horizontalViewPointCenter += __PIXELS_TO_METERS(this->translationStepSize);
552 }
553 else if(pressedKey & K_LU)
554 {
555 optical.verticalViewPointCenter -= __PIXELS_TO_METERS(this->translationStepSize);
556 }
557 else if(pressedKey & K_LD)
558 {
559 optical.verticalViewPointCenter += __PIXELS_TO_METERS(this->translationStepSize);
560 }
561 else if(pressedKey & K_A)
562 {
563 optical.maximumXViewDistancePower += __MAXIMUM_VIEW_DISTANCE_STEP;
564 optical.maximumYViewDistancePower += __MAXIMUM_VIEW_DISTANCE_STEP;
565 }
566 else if(pressedKey & K_B)
567 {
568 optical.maximumXViewDistancePower -= __MAXIMUM_VIEW_DISTANCE_STEP;
569 optical.maximumYViewDistancePower -= __MAXIMUM_VIEW_DISTANCE_STEP;
570
571 if(0 >= optical.maximumXViewDistancePower)
572 {
573 optical.maximumXViewDistancePower = 1;
574 }
575
576 if(0 >= optical.maximumYViewDistancePower)
577 {
578 optical.maximumYViewDistancePower = 1;
579 }
580 }
581 else if(pressedKey & K_LT)
582 {
583 optical.baseDistance -= __PIXELS_TO_METERS(this->translationStepSize);
584 }
585 else if(pressedKey & K_RT)
586 {
587 optical.baseDistance += __PIXELS_TO_METERS(this->translationStepSize);
588 }
589 else if(pressedKey & K_RR)
590 {
591 if(__MAX_TRANSLATION_STEP < ++this->translationStepSize)
592 {
593 this->translationStepSize = __MAX_TRANSLATION_STEP;
594 }
595
596 StageEditor::printTranslationStepSize(this, 38, 10);
597 }
598 else if(pressedKey & K_RL)
599 {
600 if(1 > --this->translationStepSize)
601 {
602 this->translationStepSize = 1;
603 }
604
605 StageEditor::printTranslationStepSize(this, 38, 10);
606 }
607
608 Camera::setOptical(optical);
609
610 StageEditor::printProjectionValues(this);
611
612 Stage::transform(this->stage, &_neutralEnvironmentTransformation, Camera::getTransformationFlags());
613}

715void StageEditor::applyTranslationToActor(Vector3D translation)
716{
717 if(isDeleted(this->stage))

758void StageEditor::showSelectedUserObject()
759{
760 StageEditor::removePreviousSprite(this);
761
762 SpriteSpec* spriteSpec =
763 (SpriteSpec*)_userObjects[OptionsSelector::getSelectedOption(this->userActorSelector)].actorSpec->componentSpecs[0];
764
765 if(spriteSpec)
766 {
767 this->userActorSprite = ComponentManager::createComponent(NULL, (ComponentSpec*)spriteSpec);
768 ASSERT(this->userActorSprite, "AnimationInspector::createSprite: null animatedSprite");
769 ASSERT(Sprite::getTexture(this->userActorSprite), "AnimationInspector::createSprite: null texture");
770
771 PixelVector spritePosition = Sprite::getDisplacedPosition(this->userActorSprite);
772 spritePosition.x = __I_TO_FIXED((__HALF_SCREEN_WIDTH) - (Texture::getCols(Sprite::getTexture(this->userActorSprite)) << 2));
773 spritePosition.y = __I_TO_FIXED((__HALF_SCREEN_HEIGHT) - (Texture::getRows(Sprite::getTexture(this->userActorSprite)) << 2));
774 spritePosition.parallax = Optics::calculateParallax(spritePosition.z);
775
776 Rotation spriteRotation = {0, 0, 0};
777 PixelScale spriteScale = {1, 1};
778 Sprite::setPosition(this->userActorSprite, &spritePosition);
779 Sprite::setRotation(this->userActorSprite, &spriteRotation);
780 Sprite::setScale(this->userActorSprite, &spriteScale);
781
782 this->userActorSprite->updateAnimationFrame = true;
783 SpriteManager::writeTextures();
784 SpriteManager::sortSprites();
785 SpriteManager::deferParamTableEffects(false);
786 SpriteManager::render();
787 SpriteManager::deferParamTableEffects(true);
788 }

793void StageEditor::selectUserObject(uint32 pressedKey)
794{
795 if(isDeleted(this->stage))
796 {
797 return;
798 }
799
800 if(pressedKey & K_LU)
801 {
802 OptionsSelector::selectPrevious(this->userActorSelector);
803 StageEditor::showSelectedUserObject(this);
804 }
805 else if(pressedKey & K_LD)
806 {
807 OptionsSelector::selectNext(this->userActorSelector);
808 StageEditor::showSelectedUserObject(this);
809 }
810 else if(pressedKey & K_A)
811 {
812 if(1 >= SpriteManager::getFreeLayer())
813 {
814 Printing::text("No more WORLDs", 48 - 15, 5, NULL);
815 Printing::text("available ", 48 - 15, 6, NULL);
816 return;
817 }
818
819 Vector3D cameraPosition = Camera::getPosition();
820
821 PositionedActor DUMMY_ENTITY =
822 {
823 (ActorSpec*)_userObjects[OptionsSelector::getSelectedOption(this->userActorSelector)].actorSpec,
824 {
825 __METERS_TO_PIXELS(cameraPosition.x) + __HALF_SCREEN_WIDTH,
826 __METERS_TO_PIXELS(cameraPosition.y) + __HALF_SCREEN_HEIGHT,
827 __METERS_TO_PIXELS(cameraPosition.z)
828 },
829 {0, 0, 0},
830 {1, 1, 1},
831 0,
832 NULL,
833 NULL,
834 NULL,
835 false
836 };
837
838 Stage::spawnChildActor(this->stage, &DUMMY_ENTITY, false);
839 SpriteManager::sortSprites();
840
841 VirtualList stageActors = (Container::safeCast(this->stage))->children;
842 this->actorNode = stageActors ? stageActors->tail : NULL;
843
844 // Select the added actor
845 this->state = kTranslateActors;
846 StageEditor::configureState(this);
847
848 StageEditor::removePreviousSprite(this);
849 SpriteManager::sortSprites();
850 SpriteManager::writeTextures();
851 SpriteManager::deferParamTableEffects(false);
852 SpriteManager::render();
853 SpriteManager::deferParamTableEffects(true);
854 }
855}

859void StageEditor::printActorPosition()
860{
861 int32 x = 1;
862 int32 y = 2;
863 uint8 controlsXPos = 38;
864 uint8 controlsYPos = 2;
865
