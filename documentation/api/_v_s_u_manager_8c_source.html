96static void VSUManager::applySoundSourceConfiguration(const VSUSoundSourceConfiguration* vsuSoundSourceConfiguration)
97{
98 VSUManager vsuManager = VSUManager::getInstance();
99
100 int16 vsuSoundSourceIndex =
101 VSUManager::findAvailableSoundSource
102 (
103 vsuSoundSourceConfiguration->requester, vsuSoundSourceConfiguration->type, !vsuSoundSourceConfiguration->skippable
104 );
105
106 if(0 > vsuSoundSourceIndex)
107 {
108 if(vsuManager->allowQueueingSoundRequests && !vsuSoundSourceConfiguration->skippable)
109 {
110 VSUManager::registerQueuedSoundSourceConfiguration(vsuSoundSourceConfiguration);
111 }
112 }
113 else
114 {
115 Waveform* waveform = VSUManager::findWaveform(vsuSoundSourceConfiguration->SxRAM);
116
117 VSUManager::configureSoundSource(vsuSoundSourceIndex, vsuSoundSourceConfiguration, waveform);
118 }
119}

123static void VSUManager::applyPCMSampleToSoundSource(int8 sample)
124{
125 int16 vsuSoundSourceIndex = 0;
126
127 while(true)
128 {
129 if(__MAXIMUM_VOLUME <= sample)
130 {
131 _vsuSoundSources[vsuSoundSourceIndex].SxLRV = 0xFF;
132 sample -= __MAXIMUM_VOLUME;
133 }
134 else
135 {
136 _vsuSoundSources[vsuSoundSourceIndex].SxLRV = ((sample << 4) | sample);
137 break;
138 }
139
140 vsuSoundSourceIndex++;
141 }
142}

146static void VSUManager::reset()
147{
148 VSUManager vsuManager = VSUManager::getInstance();
149
150 VSUManager::stopAllSounds();
151 VSUManager::enableQueue();
152
153 vsuManager->ticks = 0;
154 vsuManager->haveUsedSoundSources = false;
155 vsuManager->haveQueuedRequests = false;
156
157 VirtualList::deleteData(vsuManager->queuedVSUSoundSourceConfigurations);
158
159 for(int16 i = 0; i < __TOTAL_WAVEFORMS; i++)
160 {
161 vsuManager->waveforms[i].index = i;
162 vsuManager->waveforms[i].usageCount = 0;
163 vsuManager->waveforms[i].wave = __WAVE_ADDRESS(i);
164 vsuManager->waveforms[i].data = NULL;
165 vsuManager->waveforms[i].overwrite = true;
166
167 for(uint32 j = 0; j < 128; j++)
168 {
169 vsuManager->waveforms[i].wave[j] = 0;
170 }
171 }
172
173 uint8* modulationData = __MODULATION_DATA;
174
175 for(int16 i = 0; i <= 32 * 4; i++)
176 {
177 modulationData[i] = 0;
178 }
179
180 for(int16 i = 0; i < __TOTAL_NORMAL_CHANNELS; i++)
181 {
182 vsuManager->vsuSoundSourceConfigurations[i].type = kSoundSourceNormal;
183 }
184
185 for(int16 i = __TOTAL_NORMAL_CHANNELS; i < __TOTAL_NORMAL_CHANNELS + __TOTAL_MODULATION_CHANNELS; i++)
186 {
187 vsuManager->vsuSoundSourceConfigurations[i].type = kSoundSourceModulation | kSoundSourceNormal;
188 }
189
190 for
191 (
192 int16 i = __TOTAL_NORMAL_CHANNELS + __TOTAL_MODULATION_CHANNELS;
193 i < __TOTAL_NORMAL_CHANNELS + __TOTAL_MODULATION_CHANNELS + __TOTAL_NOISE_CHANNELS;
194 i++
195 )
196 {
197 vsuManager->vsuSoundSourceConfigurations[i].type = kSoundSourceNoise;
198 }
199
200 for(int16 i = 0; i < __TOTAL_SOUND_SOURCES; i++)
201 {
202 vsuManager->vsuSoundSourceConfigurations[i].requester = NULL;
203 vsuManager->vsuSoundSourceConfigurations[i].vsuSoundSource = &_vsuSoundSources[i];
204 vsuManager->vsuSoundSourceConfigurations[i].waveform = NULL;
205 vsuManager->vsuSoundSourceConfigurations[i].timeout = -1;
206 vsuManager->vsuSoundSourceConfigurations[i].SxLRV = 0;
207 vsuManager->vsuSoundSourceConfigurations[i].SxFQL = 0;
208 vsuManager->vsuSoundSourceConfigurations[i].SxFQH = 0;
209 vsuManager->vsuSoundSourceConfigurations[i].SxEV0 = kPlaybackPCM == vsuManager->playbackMode ? 0xFF : 0x00;
210 vsuManager->vsuSoundSourceConfigurations[i].SxEV1 = 0;
211 vsuManager->vsuSoundSourceConfigurations[i].SxRAM = kPlaybackPCM == vsuManager->playbackMode ? PCMWaveForm : NULL;
212 vsuManager->vsuSoundSourceConfigurations[i].SxSWP = 0;
213 vsuManager->vsuSoundSourceConfigurations[i].SxINT = kPlaybackPCM == vsuManager->playbackMode ? 0x9F : 0;
214
215 Waveform* waveform = VSUManager::findWaveform(vsuManager->vsuSoundSourceConfigurations[i].SxRAM);
216
217 vsuManager->vsuSoundSourceConfigurations[i].waveform = waveform;
218 vsuManager->vsuSoundSourceConfigurations[i].vsuSoundSource->SxLRV = vsuManager->vsuSoundSourceConfigurations[i].SxLRV;
219 vsuManager->vsuSoundSourceConfigurations[i].vsuSoundSource->SxFQL = vsuManager->vsuSoundSourceConfigurations[i].SxFQL;
220 vsuManager->vsuSoundSourceConfigurations[i].vsuSoundSource->SxFQH = vsuManager->vsuSoundSourceConfigurations[i].SxFQH;
221 vsuManager->vsuSoundSourceConfigurations[i].vsuSoundSource->SxEV0 = vsuManager->vsuSoundSourceConfigurations[i].SxEV0;
222 vsuManager->vsuSoundSourceConfigurations[i].vsuSoundSource->SxEV1 = vsuManager->vsuSoundSourceConfigurations[i].SxEV1;
223 vsuManager->vsuSoundSourceConfigurations[i].vsuSoundSource->SxRAM = NULL == waveform ? 0 : waveform->index;
224 vsuManager->vsuSoundSourceConfigurations[i].vsuSoundSource->SxSWP = vsuManager->vsuSoundSourceConfigurations[i].SxSWP;
225 vsuManager->vsuSoundSourceConfigurations[i].vsuSoundSource->SxINT = vsuManager->vsuSoundSourceConfigurations[i].SxINT;
226 }
227}

347static void VSUManager::configureSoundSource
348(
349 int16 vsuSoundSourceIndex, const VSUSoundSourceConfiguration* vsuSoundSourceConfiguration, Waveform* waveform
350)
351{
352 VSUManager vsuManager = VSUManager::getInstance();
353
354 int16 i = vsuSoundSourceIndex;
355 VSUSoundSource* vsuSoundSource = vsuManager->vsuSoundSourceConfigurations[i].vsuSoundSource;
356
357 vsuManager->haveUsedSoundSources = true;
358
359 bool setSxINT = vsuManager->vsuSoundSourceConfigurations[i].SxINT != vsuSoundSourceConfiguration->SxINT;
360
361 vsuManager->vsuSoundSourceConfigurations[i].requester = vsuSoundSourceConfiguration->requester;
362 vsuManager->vsuSoundSourceConfigurations[i].waveform = waveform;
363 vsuManager->vsuSoundSourceConfigurations[i].timeout = vsuManager->ticks + vsuSoundSourceConfiguration->timeout;
364 vsuManager->vsuSoundSourceConfigurations[i].SxINT = vsuSoundSourceConfiguration->SxINT;
365 vsuManager->vsuSoundSourceConfigurations[i].SxLRV = vsuSoundSourceConfiguration->SxLRV;
366 vsuManager->vsuSoundSourceConfigurations[i].SxFQL = vsuSoundSourceConfiguration->SxFQL;
367 vsuManager->vsuSoundSourceConfigurations[i].SxFQH = vsuSoundSourceConfiguration->SxFQH;
368 vsuManager->vsuSoundSourceConfigurations[i].SxEV0 = vsuSoundSourceConfiguration->SxEV0;
369 vsuManager->vsuSoundSourceConfigurations[i].SxEV1 = vsuSoundSourceConfiguration->SxEV1;
370 vsuManager->vsuSoundSourceConfigurations[i].SxRAM = vsuSoundSourceConfiguration->SxRAM;
371 vsuManager->vsuSoundSourceConfigurations[i].SxSWP = vsuSoundSourceConfiguration->SxSWP;
372 vsuManager->vsuSoundSourceConfigurations[i].skippable = vsuSoundSourceConfiguration->skippable;
373
374 vsuSoundSource->SxLRV = vsuSoundSourceConfiguration->SxLRV;
375 vsuSoundSource->SxFQL = vsuSoundSourceConfiguration->SxFQL;
376 vsuSoundSource->SxFQH = vsuSoundSourceConfiguration->SxFQH;
377 vsuSoundSource->SxEV0 = vsuSoundSourceConfiguration->SxEV0;
378 vsuSoundSource->SxEV1 = vsuSoundSourceConfiguration->SxEV1;
379 vsuSoundSource->SxRAM = waveform->index;
380 vsuSoundSource->SxSWP = vsuSoundSourceConfiguration->SxSWP;
381
384 if(setSxINT)
385 {
386 vsuSoundSource->SxINT = vsuSoundSourceConfiguration->SxINT;
387 }
388}

392static int16 VSUManager::findAvailableSoundSource(Object requester, uint32 soundSourceType, bool force)
393{
394 VSUManager vsuManager = VSUManager::getInstance();
395
396 // First try to find a sound source that has previously assigned to the same requester
397 for(int16 i = 0; i < __TOTAL_SOUND_SOURCES; i++)
398 {
399 if(0 == (soundSourceType & vsuManager->vsuSoundSourceConfigurations[i].type))
400 {
401 continue;
402 }
403
404 if(requester == vsuManager->vsuSoundSourceConfigurations[i].requester)
405 {
406 return i;
407 }
408 }
409
410 // Now try to find a sound source whose timeout has just expired
411 for(int16 i = 0; i < __TOTAL_SOUND_SOURCES; i++)
412 {
413 if(0 == (soundSourceType & vsuManager->vsuSoundSourceConfigurations[i].type))
414 {
415 continue;
416 }
417
418 if(vsuManager->ticks >= vsuManager->vsuSoundSourceConfigurations[i].timeout)
419 {
420 return i;
421 }
422 }
423
424 if(force)
425 {
426 int16 soonestFreeSoundSource = -1;
427
428 // Now try to find a sound source whose timeout has just expired
429 for(int16 i = 0; i < __TOTAL_SOUND_SOURCES; i++)
430 {
431 if(0 == (soundSourceType & vsuManager->vsuSoundSourceConfigurations[i].type))
432 {
433 continue;
434 }
435
436 if(!vsuManager->vsuSoundSourceConfigurations[i].skippable)
437 {
438 continue;
439 }
440
441 if
442 (
443 0 > soonestFreeSoundSource
444 ||
445 vsuManager->vsuSoundSourceConfigurations[i].timeout < vsuManager->vsuSoundSourceConfigurations[soonestFreeSoundSource].timeout
446 )
447 {
448 soonestFreeSoundSource = i;
449 }
450 }
451
452 return soonestFreeSoundSource;
453 }
454
455 return -1;
456}

460static void VSUManager::releaseSoundSources()
461{
462 VSUManager vsuManager = VSUManager::getInstance();
463
464 vsuManager->haveUsedSoundSources = false;
465
466 for(int16 i = 0; i < __TOTAL_WAVEFORMS; i++)
467 {
468 vsuManager->waveforms[i].usageCount = 0;
469 }
470
471 for(int16 i = 0; i < __TOTAL_SOUND_SOURCES; i++)
472 {
473 if(0 > vsuManager->vsuSoundSourceConfigurations[i].timeout)
474 {
475 continue;
476 }
477
480 if(vsuManager->ticks >= vsuManager->vsuSoundSourceConfigurations[i].timeout)
481 {
482 vsuManager->vsuSoundSourceConfigurations[i].timeout = -1;
483 vsuManager->vsuSoundSourceConfigurations[i].SxINT |= __SOUND_WRAPPER_STOP_SOUND;
484 vsuManager->vsuSoundSourceConfigurations[i].waveform = NULL;
485 vsuManager->vsuSoundSourceConfigurations[i].vsuSoundSource->SxEV1 |= 0x01;
486 }
487 else if(NULL != vsuManager->vsuSoundSourceConfigurations[i].waveform)
488 {
489 vsuManager->waveforms[vsuManager->vsuSoundSourceConfigurations[i].waveform->index].usageCount++;
490
491 vsuManager->haveUsedSoundSources = true;
492 }
493 }
494}

498static void VSUManager::dispatchQueuedSoundSourceConfigurations()
499{
500 VSUManager vsuManager = VSUManager::getInstance();
501
502 if(isDeleted(vsuManager->queuedVSUSoundSourceConfigurations) || NULL == vsuManager->queuedVSUSoundSourceConfigurations->head)
503 {
504 return;
505 }
506
507 for(VirtualNode node = vsuManager->queuedVSUSoundSourceConfigurations->head, nextNode = NULL; NULL != node; node = nextNode)
508 {
509 nextNode = node->next;
510
511 VSUSoundSourceConfiguration* queuedVSUSoundSourceConfiguration = (VSUSoundSourceConfiguration*)node->data;
512
513 int16 vsuSoundSourceIndex =
514 VSUManager::findAvailableSoundSource
515 (
516 queuedVSUSoundSourceConfiguration->requester, queuedVSUSoundSourceConfiguration->type,
517 !queuedVSUSoundSourceConfiguration->skippable
518 );
519
520 if(0 <= vsuSoundSourceIndex)
521 {
522 Waveform* waveform = VSUManager::findWaveform(queuedVSUSoundSourceConfiguration->SxRAM);
523
524 VSUManager::configureSoundSource(vsuSoundSourceIndex, queuedVSUSoundSourceConfiguration, waveform);
525
526 VirtualList::removeNode(vsuManager->queuedVSUSoundSourceConfigurations, node);
527
528 delete queuedVSUSoundSourceConfiguration;
529 }
530 }
531
532 vsuManager->haveQueuedRequests = NULL == vsuManager->queuedVSUSoundSourceConfigurations->head;
533}

557static Waveform* VSUManager::findWaveform(const int8* waveFormData)
558{
559 VSUManager vsuManager = VSUManager::getInstance();
560
561 if(NULL == waveFormData)
562 {
563 return NULL;
564 }
565
566 for(int16 i = 0; i < __TOTAL_WAVEFORMS; i++)
567 {
568 if(waveFormData == vsuManager->waveforms[i].data)
569 {
570 return &vsuManager->waveforms[i];
571 }
572 }
573
574 for(int16 i = 0; i < __TOTAL_WAVEFORMS; i++)
575 {
576 if(NULL == vsuManager->waveforms[i].data || 0 == vsuManager->waveforms[i].usageCount)
577 {
578 VSUManager::setWaveform(&vsuManager->waveforms[i], waveFormData);
579
580 return &vsuManager->waveforms[i];
581 }
582 }
583
585 return &vsuManager->waveforms[0];
586}
