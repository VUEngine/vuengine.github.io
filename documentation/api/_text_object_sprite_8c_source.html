62int16 TextObjectSprite::doRender(int16 index)
63{
64 NM_ASSERT(!isDeleted(this->texture), "TextObjectSprite::doRender: null texture");
65
66 if(!this->printed)
67 {
68 TextObjectSprite::out(this, index);
69 }
70
71 int32 cols = strlen(this->text);
72
73 // TODO: add support for multiline fonts
74 int32 rows = 1;
75
76 int32 xDirection = this->head & 0x2000 ? -1 : 1;
77 int32 yDirection = this->head & 0x1000 ? -1 : 1;
78
79 int32 x = this->position.x - this->halfWidth * xDirection + this->displacement.x - (__LEFT == xDirection ? __OBJECT_SPRITE_FLIP_X_DISPLACEMENT : 0);
80
81 // TODO: the halfHeight should be calculted based on the font's height
82 int32 y = this->position.y - this->halfHeight * yDirection + this->displacement.y - (__UP == yDirection ? __OBJECT_SPRITE_FLIP_Y_DISPLACEMENT : 0);
83
84 //FontData* fontData = Printing::getFontByName(this->font);
85
86 int32 i = 0;
87 uint16 secondWordValue = (this->head & __OBJECT_SPRITE_CHAR_SHOW_MASK) | ((this->position.parallax + this->displacement.parallax) & ~__OBJECT_SPRITE_CHAR_SHOW_MASK);
88 uint16 fourthWordValue = (this->head & 0x3000);
89 ObjectAttributes* objectPointer = NULL;
90
91 for(; i < rows; i++)
92 {
93 // TODO: Account for multiline characters
94 //int32 outputY = y + (i << 3) * yDirection;
95 int32 outputY = y;
96
97 if((unsigned)(outputY - _cameraFrustum->y0 + 4) > (unsigned)(_cameraFrustum->y1 - _cameraFrustum->y0))
98 {
99 int32 j = 0;
100 for(; j < cols; j++)
101 {
102 int32 objectIndex = index + j;
103
104 objectPointer = &_objectAttributesCache[objectIndex];
105 objectPointer->head = __OBJECT_SPRITE_CHAR_HIDE_MASK;
106 }
107
108 continue;
109 }
110
111 int32 j = 0;
112
113 for(; j < cols; j++)
114 {
115 int32 objectIndex = index + j;
116
117 // TODO: Account for character's size
118// int32 outputX = x + (j * fontData->fontSpec->fontSize.x) * xDirection;
119 int32 outputX = x + (j * 8) * xDirection;
120
121 // Add 8 to the calculation to avoid char's cut off when scrolling hide the object if outside
122 // Screen's bounds
123 if((unsigned)(outputX - _cameraFrustum->x0 + 4) > (unsigned)(_cameraFrustum->x1 - _cameraFrustum->x0))
124 {
125 objectPointer->head = __OBJECT_SPRITE_CHAR_HIDE_MASK;
126 continue;
127 }
128
129 objectPointer = &_objectAttributesCache[objectIndex];
130
131 objectPointer->jx = outputX;
132 objectPointer->head = secondWordValue;
133 objectPointer->jy = outputY;
134 objectPointer->tile |= fourthWordValue;
135 }
136 }
137
138 return index;
139}

149void TextObjectSprite::out(uint16 index)
150{
151 uint32 i = 0;
152 uint32 charOffset = 0, charOffsetX = 0, charOffsetY = 0;
153
154 FontData* fontData = Printing::getFontByName(this->font);
155
156 if(!fontData)
157 {
158 return;
159 }
160
161 uint32 offset = CharSet::getOffset(fontData->charSet);
162
163 // Print text
164 while(this->text[i])
165 {
166 switch(this->text[i])
167 {
168 default:
169 {
170 for(charOffsetX = 0; charOffsetX < fontData->fontSpec->fontSize.x; charOffsetX++)
171 {
172 for(charOffsetY = 0; charOffsetY < fontData->fontSpec->fontSize.y; charOffsetY++)
173 {
174 charOffset = charOffsetX + (charOffsetY * fontData->fontSpec->charactersPerLineInCharset * fontData->fontSpec->fontSize.x);
175
176 int32 objectIndex = index + i;
177
178 uint16 charNumber =
179 // Offset of charset in char memory
180 offset +
181
182 // Offset of character in charset
183 ((uint8)(this->text[i] - fontData->fontSpec->offset) * fontData->fontSpec->fontSize.x) +
184
185 // Additional y offset in charset
186 (((uint8)(this->text[i] - fontData->fontSpec->offset)
187 / fontData->fontSpec->charactersPerLineInCharset
188 * fontData->fontSpec->charactersPerLineInCharset * fontData->fontSpec->fontSize.x)
189 * (fontData->fontSpec->fontSize.y - 1)) +
190
191 // Respective char of character
192 charOffset;
193
194 ObjectAttributes* objectPointer = &_objectAttributesCache[objectIndex];
195
196 objectPointer->tile = this->palette | (charNumber & 0x7FF);
197 }
198 }
199 }
200
201 break;
202 }
203
204 i++;
205 }
206
207 this->printed = true;
208}
