82bool StateMachine::transitionTo(State state, int16 transition)
83{
84 if(kStateMachineIdle != this->transition || NULL == state || kStateMachineIdle == transition)
85 {
86 return false;
87 }
88
89 this->nextState = state;
90 this->transition = transition;
91
92 if(NULL == this->currentState && NULL != this->nextState)
93 {
94 StateMachine::applyTransition(this);
95 }
96
97 return true;
98}

102void StateMachine::swapState(State newState)
103{
104 if(isDeleted(newState))
105 {
106 return;
107 }
108
109 NM_ASSERT(!isDeleted(newState), "StateMachine::swapState: null newState");
110
111 // Update the stack
112 // Remove current state
113 VirtualList::popFront(this->stateStack);
114
115 // Finalize current state
116 if(NULL != this->currentState)
117 {
118 this->previousState = this->currentState;
119
120 // Call the exit method from current state
121 State::exit(this->currentState, this->owner);
122 }
123
124 NM_ASSERT(!isDeleted(newState), "StateMachine::swapState: newState deleted during exit of previous state");
125
126 this->currentState = newState;
127 this->previousState = NULL;
128
129 // Push new state in the top of the stack
130 VirtualList::pushFront(this->stateStack, (BYTE*)this->currentState);
131
132 // Call enter method from new state
133 State::enter(this->currentState, this->owner);
134}

138void StateMachine::pushState(State newState)
139{
140 if(isDeleted(newState))
141 {
142 return;
143 }
144
145 // Finalize current state
146 if(NULL != this->currentState)
147 {
148 // Call the pause method from current state
149 State::suspend(this->currentState, this->owner);
150 }
151
152 // Set new state
153 this->currentState = newState;
154
155 NM_ASSERT(!isDeleted(this->currentState), "StateMachine::pushState: null currentState");
156
157 // Push new state in the top of the stack
158 VirtualList::pushFront(this->stateStack, (BYTE*)this->currentState);
159
160 // Call enter method from new state
161 State::enter(this->currentState, this->owner);
162}

166void StateMachine::popState()
167{
168 // Return in case the stack is empty
169 if(StateMachine::getStackSize(this) == 0)
170 {
171 return;
172 }
173
174 // Finalize current state
175 if(NULL != this->currentState)
176 {
177 // Call the exit method from current state
178 State::exit(this->currentState, this->owner);
179 }
180
181 // Update the stack
182 // Remove the state in the top of the stack
183 VirtualList::popFront(this->stateStack);
184
185 // Update current state
186 this->currentState = VirtualList::front(this->stateStack);
187
188 // Call resume method from new state
189 if(NULL != this->currentState)
190 {
191 State::resume(this->currentState, this->owner);
192 }
193}

204State StateMachine::update()
205{
206 if(kStateMachineIdle != this->transition)
207 {
208 StateMachine::applyTransition(this);
209 }
210 else if(!isDeleted(this->currentState))
211 {
212 State::execute(this->currentState, this->owner);
213 }
214
215 return this->currentState;
216}

284uint32 StateMachine::popStateWithoutResume()
285{
286 // Return in case the stack is empty
287 if(StateMachine::getStackSize(this) == 0)
288 {
289 return 0;
290 }
291
292 // Finalize current state
293 if(NULL != this->currentState)
294 {
295 // Call the exit method from current state
296 State::exit(this->currentState, this->owner);
297 }
298
299 // Remove the state in the top of the stack
300 VirtualList::popFront(this->stateStack);
301
302 // Update current state
303 this->currentState = VirtualList::front(this->stateStack);
304
305 // Return the resulting stack size
306 return StateMachine::getStackSize(this);
307}

310
311void StateMachine::applyTransition()
312{
313 if(kStateMachineIdle != this->transition)
314 {
315 switch(this->transition)
316 {
317 case kStateMachineCleanStack:
318
319 if(NULL != this->events)
320 {
321 StateMachine::fireEvent(this, kEventStateMachineWillCleanStack);
322 }
323
324 StateMachine::popAllStates(this);
325
326 if(NULL != this->nextState)
327 {
328 StateMachine::pushState(this, this->nextState);
329 }
330
331 if(NULL != this->events)
332 {
333 StateMachine::fireEvent(this, kEventStateMachineCleanedStack);
334 }
335 break;
336
337 case kStateMachineSwapState:
338
339 if(NULL != this->events)
340 {
341 StateMachine::fireEvent(this, kEventStateMachineWillSwapState);
342 }
343
344 StateMachine::swapState(this, this->nextState);
345
346 if(NULL != this->events)
347 {
348 StateMachine::fireEvent(this, kEventStateMachineSwapedState);
349 }
350
351 break;
352
353 case kStateMachinePushState:
354
355 if(NULL != this->events)
356 {
357 StateMachine::fireEvent(this, kEventStateMachineWillPushState);
358 }
359
360 StateMachine::pushState(this, this->nextState);
361
362 if(NULL != this->events)
363 {
364 StateMachine::fireEvent(this, kEventStateMachinePushedState);
365 }
366
367 break;
368
369 case kStateMachinePopState:
370
371 if(NULL != this->events)
372 {
373 StateMachine::fireEvent(this, kEventStateMachineWillPopState);
374 }
375
376 StateMachine::popState(this);
377
378 if(NULL != this->events)
379 {
380 StateMachine::fireEvent(this, kEventStateMachinePoppedState);
381 }
382 break;
383 }
384
385 this->nextState = NULL;
386 this->transition = kStateMachineIdle;
387 }
388}
