90VirtualNode VirtualCircularList::find(const void* const data)
91{
92 if(this->head)
93 {
94 VirtualNode node = this->head;
95
96 do
97 {
98 if(node->data == (void*)data)
99 {
100 return node;
101 }
102
103 node = node->next;
104 }
105 while(node != this->head);
106 }
107
108 return NULL;
109}

113VirtualNode VirtualCircularList::getNode(int32 item)
114{
115 int32 counter = 0;
116
117 int32 listSize = VirtualCircularList::getCount(this);
118
119 VirtualNode node = this->head;
120
121 // If not null head
122 if(node)
123 {
124 // If item hasn't reached list's size
125 if(item < listSize)
126 {
127 // Increase counter while node hasn't reached list's end
128 // And counter hasn't reached the item requested
129 while(node && counter < item)
130 {
131 // Increase counter
132 counter++;
133
134 // Load next node
135 node = node->next;
136
137 // If item reached
138 if(counter == item)
139 {
140 // Return node's data
141 return node;
142 }
143 }
144
145 // If item reached
146 if(counter == item)
147 {
148 // Return node's data
149 return node;
150 }
151
152 return NULL;
153 }
154 }
155
156 return NULL;
157}

212int32 VirtualCircularList::getCount()
213{
214 int32 counter = 0;
215
216 if(this->head)
217 {
218 // Point to the head
219 VirtualNode node = this->head;
220
221 // While node doesn't reach the head again
222 do
223 {
224 counter++;
225
226 // Increment counter
227 ASSERT(counter < LIST_MAX_SIZE, "VirtualCircularList::getCount: endless list getting size");
228
229 // Move the node to the head
230 node = node->next;
231 }
232 while(node != this->head);
233 }
234
235 return counter;
236}

240VirtualNode VirtualCircularList::pushFront(const void* const data)
241{
242 VirtualNode newNode = new VirtualNode(data);
243
244 // Set previous if list isn't empty
245 if(this->head)
246 {
247 this->head->previous = newNode;
248 }
249
250 // Assign the node to the head of the list
251 newNode->next = this->head;
252 newNode->previous = this->tail;
253
254 // Move the head
255 this->head = newNode;
256
257 // Set the tail
258 if(this->tail)
259 {
260 this->tail->next = this->head;
261 }
262 else
263 {
264 this->tail = this->head;
265 newNode->next = newNode->previous = newNode;
266 }
267
268 return newNode;
269}

306VirtualNode VirtualCircularList::insertAfter(VirtualNode node, const void* const data)
307{
308 if(!VirtualCircularList::checkThatNodeIsPresent(this, node))
309 {
310 return NULL;
311 }
312
313 VirtualNode newNode = NULL;
314
315 if(!node || node == this->tail)
316 {
317 VirtualCircularList::pushBack(this, data);
318
319 newNode = this->tail;
320 }
321 else
322 {
323 newNode = new VirtualNode(data);
324
325 if(!newNode)
326 {
327 return false;
328 }
329
330 // Set previous if list isn't empty
331 newNode->next = node->next;
332
333 if(node->next)
334 {
335 node->next->previous = newNode;
336 }
337
338 node->next = newNode;
339
340 newNode->previous = node;
341 }
342
343 return newNode;
344}

348void* VirtualCircularList::popFront()
349{
350 if(this->head)
351 {
352 VirtualNode node = this->head;
353 void* data = node->data;
354
355 if(this->head != this->tail)
356 {
357 this->head = this->head->next;
358
359 // Move head's previous pointer
360 this->head->previous = this->tail;
361 this->tail->next = this->head;
362 }
363 else
364 {
365 // Set head
366 this->head = NULL;
367 this->tail = NULL;
368 }
369
370 // Free dynamic memory
371 delete node;
372
373 return data;
374 }
375
376 return NULL;
377}

414bool VirtualCircularList::removeNode(VirtualNode node)
415{
416 // If node isn't null
417 if(VirtualCircularList::checkThatNodeIsPresent(this, node))
418 {
419 // If the node is the head of the list
420 if(this->head == this->tail)
421 {
422 if(node == this->head)
423 {
424 this->head = this->tail = NULL;
425 }
426 }
427 else
428 {
429 if(node == this->head)
430 {
431 this->head = this->head->next;
432 this->head->previous = this->tail;
433 this->tail->next = this->head;
434 }
435 else if(node == this->tail)
436 {
437 this->tail = this->tail->previous;
438 this->tail->next = this->head;
439 this->head->previous = this->tail;
440 }
441 else
442 {
443 // Make sure that node belongs to this list
444 // Join the previous and next nodes
445 node->previous->next = node->next;
446 node->next->previous = node->previous;
447 }
448 }
449
450 delete node;
451 return true;
452 }
453
454 NM_ASSERT(false, "VirtualCircularList::removeNode: removing invalid node");
455
456 return false;
457}

468void VirtualCircularList::reverse()
469{
470 if(NULL != this->head && this->tail != this->head)
471 {
472 VirtualNode node = this->head;
473 VirtualNode nextNode = node->next;
474
475 do
476 {
477 VirtualNode helper = node->next;
478 node->next = node->previous;
479 node->previous = helper;
480
481 node = nextNode;
482 nextNode = nextNode->next;
483 }
484 while(nextNode != this->head);
485
486 VirtualNode helper = node->next;
487 node->next = node->previous;
488 node->previous = helper;
489
490 helper = this->head;
491 this->head = this->tail;
492 this->tail = helper;
493 }
494}

498void VirtualCircularList::copy(VirtualCircularList sourceList)
499{
500#ifdef __DEBUG
501 int32 counter = 0;
502#endif
503
504 if(sourceList->head)
505 {
506 VirtualNode node = sourceList->head;
507
508 VirtualCircularList::clear(this);
509
510 do
511 {
512 VirtualCircularList::pushBack(this, node->data);
513
514 node = node->next;
515
516 ASSERT(++counter < LIST_MAX_SIZE, "VirtualCircularList::copy: endless list copying");
517 }
518 while(node != sourceList->head);

524void VirtualCircularList::clear()
525{
526 if(this->head)
527 {
528 // Point to the head
529 VirtualNode node = this->head;
530
531 // While node doesn't reach the head again
532 do
533 {
534 // Call destructor
535 delete node;
536
537 // Move the node to the head
538 node = node->next;
539 }
540 while(node != this->head);
541
542 this->head = NULL;
543 this->tail = NULL;

549void VirtualCircularList::deleteData()
550{
551 if(!isDeleted(this->head))
552 {
553 HardwareManager::suspendInterrupts();
554
555 // Point to the head
556 VirtualNode node = this->head;
557
558 // While node doesn't reach the head again
559 do
560 {
561 delete node->data;
562
563 VirtualNode aux = node;
564
565 node = node->next;
566
567#ifndef __SHPPING
568 if(isDeleted(aux))
569 {
570 continue;
571 }
572#endif
573
574 delete aux;
575 }
576 while(node != this->head);
577
578 this->head = NULL;
579 this->tail = NULL;
580
581 HardwareManager::resumeInterrupts();
582 }
583}

593bool VirtualCircularList::checkThatNodeIsPresent(VirtualNode node)
594{
595 if(!node)
596 {
597 return false;
598 }
599
600 if(this->head)
601 {
602 // Point to the head
603 VirtualNode auxNode = this->head;
604
605 // While node doesn't reach the head again
606 do
607 {
608 if(auxNode == node)
609 {
610 return true;
611 }
612
613 // Move the node to the head
614 auxNode = auxNode->next;
615 }
616 while(auxNode != this->head);
617 }
618
619 return false;
620}
