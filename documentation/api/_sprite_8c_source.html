49void Sprite::constructor(Entity owner, const SpriteSpec* spriteSpec)
50{
51 // Always explicitly call the base's constructor
52 Base::constructor(owner, (const VisualComponentSpec*)&spriteSpec->visualComponentSpec);
53
54 // Clear values
55 this->index = __NO_RENDER_INDEX;
56 this->head = 0;
57 this->texture = NULL;
58 this->halfWidth = 0;
59 this->halfHeight = 0;
60 this->transparency = __TRANSPARENCY_NONE;
61 this->checkIfWithinScreenSpace = true;
62 this->position = (PixelVector){0, 0, 0, 0};
63 this->rotation = (Rotation){0, 0, 0};
64 this->scale = (PixelScale){__1I_FIX7_9, __1I_FIX7_9};
65 this->displacement = PixelVector::zero();
66 this->hasTextures = true;
67
68 if(NULL != spriteSpec)
69 {
70 this->transparency = spriteSpec->transparency;
71
72 if(NULL != spriteSpec->textureSpec)
73 {
74 this->halfWidth = spriteSpec->textureSpec->cols << 2;
75 this->halfHeight = spriteSpec->textureSpec->rows << 2;
76 }
77 }
78}

90RightBox Sprite::getRightBox()
91{
92#ifndef __RELEASE
93 if(0 == this->halfWidth || 0 == this->halfHeight)
94 {
95 Printing::setDebugMode();
96 Printing::clear();
97 PRINT_HEX((uint32)this->componentSpec, 1, 27);
98 Error::triggerException("Sprite::getRightBox: 0 size sprite", NULL);
99 }
100#endif
101
102 return (RightBox)
103 {
104 __PIXELS_TO_METERS(-this->halfWidth + this->displacement.x),
105 __PIXELS_TO_METERS(-this->halfHeight + this->displacement.y),
106 __PIXELS_TO_METERS(-SPRITE_HALF_DEPTH),
107 __PIXELS_TO_METERS(this->halfWidth + this->displacement.x),
108 __PIXELS_TO_METERS(this->halfHeight + this->displacement.y),
109 __PIXELS_TO_METERS(SPRITE_HALF_DEPTH),
110 };
111}

165int16 Sprite::render(int16 index, bool updateAnimation)
166{
167 // If the client code makes these checks before calling this method,
168 // It saves on method calls quite a bit when there are lots of
169 // Sprites. Don't uncomment.
170/*
171 if(__HIDE == this->show)
172 {
173 return __NO_RENDER_INDEX;
174 }
175*/
176
177 if(!this->hasTextures)
178 {
179 this->index = Sprite::doRender(this, index);
180 return this->index;
181 }
182
183 if(kTextureInvalid == this->texture->status || isDeleted(this->texture->charSet))
184 {
185 this->index = __NO_RENDER_INDEX;
186 return this->index;
187 }
188
189 if(kTexturePendingWriting == this->texture->status)
190 {
191 this->index = __NO_RENDER_INDEX;
192 return this->index;
193 }
194
195 if(NULL != this->owner)
196 {
197 if(NULL == this->transformation)
198 {
199 return __NO_RENDER_INDEX;
200 }
201 else if(__NON_TRANSFORMED == this->transformation->invalid)
202 {
203 return __NO_RENDER_INDEX;
204 }
205
206 Sprite::position(this);
207
208 if
209 (
210 this->rotation.x != this->transformation->rotation.x
211 ||
212 this->rotation.y != this->transformation->rotation.y
213 ||
214 this->rotation.z != this->transformation->rotation.z
215 )
216 {
217 Sprite::rotate(this);
218 }
219
220 if
221 (
222 this->scale.x != this->transformation->scale.x
223 ||
224 this->scale.y != this->transformation->scale.y
225 )
226 {
227 Sprite::scale(this);
228 }
229 }
230
231 // If the client code makes these checks before calling this method,
232 // It saves on method calls quite a bit when there are lots of
233 // Sprites. Don't uncomment.
234/*
235 if
236 (
237 !(((this->transparency == __TRANSPARENCY_NONE) || (0x01 & (this->transparency ^ evenFrame)))
238 &&
239 Sprite::isWithinScreenSpace(this))
240 )
241 {
242 return this->index;
243 }
244*/
245
246 // Do not remove this check, it prevents sprites from looping
247 if(this->checkIfWithinScreenSpace && !Sprite::isWithinScreenSpace(this))
248 {
249 this->index = __NO_RENDER_INDEX;
250 return this->index;
251 }
252
253 if(updateAnimation)
254 {
255 Sprite::update(this);
256 }
257
258 if(!this->rendered || this->index != index)
259 {
260 this->rendered = true;
261
262 this->index = Sprite::doRender(this, index);
263
264#ifdef __SHOW_SPRITES_PROFILING
265 extern int32 _renderedSprites;
266 _renderedSprites++;
267#endif
268 }
269
270 return this->index;
271}

636void Sprite::print(int32 x, int32 y)
637{
638 // Allow normal rendering once for WORLD values to populate properly
639 uint8 transparency = this->transparency;
640 this->transparency = __TRANSPARENCY_NONE;
641
642 Printing::text("SPRITE ", x, y++, NULL);
643 Printing::text("Class: ", x, ++y, NULL);
644 Printing::text(__GET_CLASS_NAME(this), x + 18, y, NULL);
645 Printing::text("Mode:", x, ++y, NULL);
646
647 if(Sprite::isObject(this))
648 {
649 Printing::text("OBJECT ", x + 18, y, NULL);
650 }
651 else if(Sprite::isAffine(this))
652 {
653 Printing::text("AFFINE ", x + 18, y, NULL);
654 }
655 else if(Sprite::isHBias(this))
656 {
657 Printing::text("H-BIAS ", x + 18, y, NULL);
658 }
659 else if(Sprite::isBgmap(this))
660 {
661 Printing::text("BGMAP ", x + 18, y, NULL);
662 }
663
664 Printing::text("Index: ", x, ++y, NULL);
665 Printing::int32(this->index, x + 18, y, NULL);
666 Printing::text("Head: ", x, ++y, NULL);
667 Printing::hex(Sprite::getEffectiveHead(this), x + 18, y, 8, NULL);
668 Printing::text("Transparent: ", x, ++y, NULL);
669 Printing::text(transparency > 0 ? __CHAR_CHECKBOX_CHECKED : __CHAR_CHECKBOX_UNCHECKED, x + 18, y, NULL);
670 Printing::text(transparency == 1 ? "(Even)" : (transparency == 2) ? "(Odd)" : "", x + 20, y, NULL);
671 Printing::text("Shown: ", x, ++y, NULL);
672 Printing::text(__HIDE != this->show ? __CHAR_CHECKBOX_CHECKED : __CHAR_CHECKBOX_UNCHECKED, x + 18, y, NULL);
673
674 Printing::text("Pos. (x,y,z,p): ", x, ++y, NULL);
675 Printing::int32(this->position.x, x + 18, y, NULL);
676 Printing::int32(this->position.y, x + 24, y, NULL);
677 Printing::int32(this->position.z, x + 30, y, NULL);
678 Printing::int32(this->position.parallax, x + 36, y, NULL);
679 Printing::text("Displ. (x,y,z,p): ", x, ++y, NULL);
680 Printing::int32(this->displacement.x, x + 18, y, NULL);
681 Printing::int32(this->displacement.y, x + 24, y, NULL);
682 Printing::int32(this->displacement.z, x + 30, y, NULL);
683 Printing::int32(this->displacement.parallax, x + 36, y, NULL);
684 Printing::text("FPos. (x,y,z,p): ", x, ++y, NULL);
685 Printing::int32(this->position.x + this->displacement.x, x + 18, y, NULL);
686 Printing::int32(this->position.y + this->displacement.y, x + 24, y, NULL);
687 Printing::int32(this->position.z + this->displacement.z, x + 30, y, NULL);
688 Printing::int32(this->position.parallax + this->displacement.parallax, x + 36, y, NULL);
689 Printing::text("G (x,y,p): ", x, ++y, NULL);
690 Printing::int32(Sprite::getEffectiveX(this), x + 18, y, NULL);
691 Printing::int32(Sprite::getEffectiveY(this), x + 24, y, NULL);
692 Printing::int32(Sprite::getEffectiveP(this), x + 30, y, NULL);
693 Printing::text("M (x,y,p): ", x, ++y, NULL);
694 Printing::int32(Sprite::getEffectiveMX(this), x + 18, y, NULL);
695 Printing::int32(Sprite::getEffectiveMY(this), x + 24, y, NULL);
696 Printing::int32(Sprite::getEffectiveMP(this), x + 30, y, NULL);
697 Printing::text("Size (w,h): ", x, ++y, NULL);
698 Printing::int32(Sprite::getEffectiveWidth(this), x + 18, y, NULL);
699 Printing::int32(Sprite::getEffectiveHeight(this), x + 24, y++, NULL);
700 Printing::text("Pixels: ", x, y, NULL);
701 Printing::int32(Sprite::getTotalPixels(this), x + 18, y++, NULL);
702
703 if(NULL != Sprite::getTexture(this))
704 {
705 y++;
706 Printing::text("TEXTURE ", x, ++y, NULL);
707 y++;
708 Printing::text("Spec: ", x, ++y, NULL);
709 Printing::hex((int32)Texture::getSpec(Sprite::getTexture(this)), x + 18, y, 8, NULL);
710 Printing::text("Size (w,h): ", x, ++y, NULL);
711 Printing::int32(this->halfWidth * 2, x + 18, y, NULL);
712 Printing::int32(this->halfHeight * 2, x + 24, y, NULL);
713
714 if(Sprite::getTexture(this) && __GET_CAST(BgmapTexture, Sprite::getTexture(this)))
715 {
716 BgmapTexture bgmapTexture = __GET_CAST(BgmapTexture, Sprite::getTexture(this));
717
718 Printing::text("Segment: ", x, ++y, NULL);
719 Printing::int32(BgmapTexture::getSegment(bgmapTexture), x + 18, y, NULL);
720 Printing::text("Written: ", x, ++y, NULL);
721 Printing::text
722 (
723
724 Texture::isWritten(bgmapTexture) ? __CHAR_CHECKBOX_CHECKED : __CHAR_CHECKBOX_UNCHECKED, x + 18, y, NULL
725 );
726
727 Printing::text("Rows remaining: ", x, ++y, NULL);
728 Printing::int32(BgmapTexture::getRemainingRowsToBeWritten(bgmapTexture), x + 18, y, NULL);
729 }
730 }
731
732 this->transparency = transparency;
733}

741//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
742
743void Sprite::position()
744{
745#ifdef __SPRITE_ROTATE_IN_3D
746 PixelVector position = PixelVector::transformVector3D(this->transformation->position);
747
748 if(position.z != this->position.z)
749 {
750 this->scale.x = this->scale.y = 0;
751 }
752
753#else
754 PixelVector position =
755 PixelVector::projectVector3D(Vector3D::sub(this->transformation->position, *_cameraPosition), this->position.parallax);
756
757 if(position.z != this->position.z)
758 {
759 position.parallax = Optics::calculateParallax(this->transformation->position.z - _cameraPosition->z);
760
761 this->scale.x = this->scale.y = 0;
762 }
763#endif
764
765 Sprite::setPosition(this, &position);
766}
