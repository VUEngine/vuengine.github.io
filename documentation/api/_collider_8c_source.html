46void Collider::constructor(Entity owner, const ColliderSpec* colliderSpec)
47{
48 // Always explicitly call the base's constructor
49 Base::constructor(owner, (const ComponentSpec*)&colliderSpec->componentSpec);
50
51 // Not setup yet
52 this->deleteMe = false;
53 this->enabled = true;
54
55 this->wireframe = NULL;
56
57 // Set flag
58 this->checkForCollisions = false;
59 this->layers = colliderSpec->layers;
60 this->layersToIgnore = colliderSpec->layersToIgnore;
61 this->otherColliders = NULL;
62 this->registerCollisions = colliderSpec->checkForCollisions;
63
64 this->position = Vector3D::sum(this->transformation->position, Vector3D::getFromPixelVector(colliderSpec->displacement));
65 this->invalidPosition = true;
66}

70void Collider::destructor()
71{
72 Collider::hide(this);
73
74 if(NULL != this->events)
75 {
76 Collider::fireEvent(this, kEventColliderDeleted);
77 NM_ASSERT(!isDeleted(this), "Collider::destructor: deleted this during kEventColliderDeleted");
78 }
79
80 if(NULL != this->otherColliders)
81 {
82 VirtualNode node = this->otherColliders->head;
83
84 for(; NULL != node; node = node->next)
85 {
86 OtherColliderRegistry* otherColliderRegistry = (OtherColliderRegistry*)node->data;
87
88 ASSERT(!isDeleted(otherColliderRegistry), "Collider::destructor: dead otherColliderRegistry");
89
90 if(!isDeleted(otherColliderRegistry->collider))
91 {
92 Collider::removeEventListener
93 (
94 otherColliderRegistry->collider, ListenerObject::safeCast(this), (EventListener)Collider::onOtherColliderChanged,
95 kEventColliderDeleted
96 );
97
98 Collider::removeEventListener
99 (
100 otherColliderRegistry->collider, ListenerObject::safeCast(this), (EventListener)Collider::onOtherColliderChanged,
101 kEventColliderChanged
102 );
103 }
104
105 delete otherColliderRegistry;
106 }
107
108 delete this->otherColliders;
109 this->otherColliders = NULL;
110 }
111
112 // Always explicitly call the base's destructor
113 Base::destructor();
114}

257CollisionResult Collider::collides(Collider collider)
258{
259 if(isDeleted(this->owner))
260 {
261 return kNoCollision;
262 }
263
264 Collision collision;
265 collision.result = kNoCollision;
266 collision.collisionInformation.collider = NULL;
267 collision.collisionInformation.otherCollider = NULL;
268
269 OtherColliderRegistry* otherColliderRegistry = Collider::findOtherColliderRegistry(this, collider);
270
271 // Test if new collision
272 if(NULL == otherColliderRegistry)
273 {
274 // Check for new overlap
275 CollisionTester::testOverlaping(this, collider, &collision.collisionInformation, 0);
276
277 if(NULL != collision.collisionInformation.collider && 0 != collision.collisionInformation.solutionVector.magnitude)
278 {
279 // New collision
280 collision.result = kCollisionStarts;
281
282 if(this->registerCollisions)
283 {
284 otherColliderRegistry =
285 Collider::registerOtherCollider(this, collider, collision.collisionInformation.solutionVector, false);
286 }
287 }
288 }
289 // Impenetrable registered colliding colliders require another test
290 // To determine if I'm not colliding against them anymore
291 else if(otherColliderRegistry->isImpenetrable && otherColliderRegistry->solutionVector.magnitude)
292 {
293 CollisionTester::testOverlaping(this, collider, &collision.collisionInformation, __STILL_COLLIDING_CHECK_SIZE_INCREMENT);
294
295 if
296 (
297 collision.collisionInformation.collider == this
298 &&
299 collision.collisionInformation.solutionVector.magnitude >= __STILL_COLLIDING_CHECK_SIZE_INCREMENT
300 )
301 {
302 collision.result = kCollisionPersists;
303 }
304 else
305 {
306 collision.result = kCollisionEnds;
307 collision.collisionInformation.collider = this;
308 collision.collisionInformation.otherCollider = collider;
309 }
310 }
311 else
312 {
313 // Otherwise make a normal collision test
314 CollisionTester::testOverlaping(this, collider, &collision.collisionInformation, 0);
315
316 if(collision.collisionInformation.collider == this && 0 != collision.collisionInformation.solutionVector.magnitude)
317 {
318 collision.result = kCollisionPersists;
319 }
320 else
321 {
322 collision.result = kCollisionEnds;
323 collision.collisionInformation.collider = this;
324 collision.collisionInformation.otherCollider = collider;
325 }
326 }
327
328 switch(collision.result)
329 {
330 case kCollisionStarts:
331
332 Collider::collisionStarts(this, &collision);
333 break;
334
335 case kCollisionPersists:
336
337 Collider::collisionPersists(this, &collision);
338 break;
339
340 case kCollisionEnds:
341
342 Collider::collisionEnds(this, &collision);
343 break;
344
345 default:
346 break;
347 }
348
349 return collision.result;
350}

354void Collider::resolveCollision(const CollisionInformation* collisionInformation)
355{
356 ASSERT(collisionInformation->collider, "Collider::resolveCollision: null collider");
357 ASSERT(collisionInformation->otherCollider, "Collider::resolveCollision: null collidingActors");
358
359 if(isDeleted(this->owner))
360 {
361 return;
362 }
363
364 SolutionVector solutionVector = collisionInformation->solutionVector;
365
366 if(collisionInformation->collider == this && solutionVector.magnitude)
367 {
368 Collider::displaceOwner(this, Vector3D::scalarProduct(solutionVector.direction, solutionVector.magnitude));
369
370 // Need to invalidate solution vectors for other colliding colliders
371 //Collider::checkPreviousCollisions(this, collisionInformation->otherCollider);
372
373 if(this->registerCollisions)
374 {
375 OtherColliderRegistry* otherColliderRegistry =
376 Collider::registerOtherCollider
377 (
378 this, collisionInformation->otherCollider, collisionInformation->solutionVector, true
379 );
380
381 ASSERT(!isDeleted(otherColliderRegistry), "Collider::resolveCollision: dead otherColliderRegistry");
382 otherColliderRegistry->frictionCoefficient = Entity::getFrictionCoefficient(collisionInformation->otherCollider->owner);
383 }
384 }
385}

388
389bool Collider::canMoveTowards(Vector3D displacement)
390{
391 if(!this->otherColliders || NULL == this->otherColliders->head)
392 {
393 return true;
394 }
395
396 bool canMove = true;
397
398 Vector3D normalizedDisplacement = Vector3D::normalize(displacement);
399
400 for(VirtualNode node = this->otherColliders->head; canMove && node; node = node->next)
401 {
402 OtherColliderRegistry* otherColliderRegistry = (OtherColliderRegistry*)node->data;
403
404 ASSERT(!isDeleted(otherColliderRegistry), "Collider::canMoveTowards: dead otherColliderRegistry");
405
406 if(otherColliderRegistry->isImpenetrable)
407 {
408 // Check if solution is valid
409 if(otherColliderRegistry->solutionVector.magnitude)
410 {
411 fixed_t cosAngle = Vector3D::dotProduct(otherColliderRegistry->solutionVector.direction, normalizedDisplacement);
412 canMove &= -(__1I_FIXED - __COLLIDER_ANGLE_TO_PREVENT_DISPLACEMENT) < cosAngle;
413 }
414 }
415 }
416
417 // Not colliding anymore
418 return canMove;
419}

423void Collider::discardCollisions()
424{
425 if(NULL != this->otherColliders)
426 {
427 VirtualNode node = this->otherColliders->head;
428
429 for(; NULL != node; node = node->next)
430 {
431 OtherColliderRegistry* otherColliderRegistry = (OtherColliderRegistry*)node->data;
432
433 ASSERT(!isDeleted(otherColliderRegistry), "Collider::reset: dead otherColliderRegistry");
434
435 if(!isDeleted(otherColliderRegistry->collider))
436 {
437 Collider::removeEventListener
438 (
439 otherColliderRegistry->collider, ListenerObject::safeCast(this), (EventListener)Collider::onOtherColliderChanged,
440 kEventColliderDeleted
441 );
442
443 Collider::removeEventListener
444 (
445 otherColliderRegistry->collider, ListenerObject::safeCast(this), (EventListener)Collider::onOtherColliderChanged,
446 kEventColliderChanged
447 );
448 }
449
450 delete otherColliderRegistry;
451 }
452
453 delete this->otherColliders;
454 this->otherColliders = NULL;
455 }
456}

460fixed_t Collider::getCollidingFrictionCoefficient()
461{
462 if(!this->otherColliders)
463 {
464 return 0;
465 }
466
467 fixed_t totalFrictionCoefficient = 0;
468
469 VirtualNode node = this->otherColliders->head;
470
471 for(; NULL != node; node = node->next)
472 {
473 OtherColliderRegistry* otherColliderRegistry = (OtherColliderRegistry*)node->data;
474 ASSERT(!isDeleted(otherColliderRegistry), "Collider::getCollidingFriction: dead otherColliderRegistry");
475
476 ASSERT(otherColliderRegistry->collider, "Collider::getCollidingFriction: null otherCollider");
477
478 if(!isDeleted(otherColliderRegistry->collider->owner))
479 {
480 totalFrictionCoefficient += otherColliderRegistry->frictionCoefficient;
481 }
482 }
483
484 return totalFrictionCoefficient;
485}

586bool Collider::onOtherColliderChanged(ListenerObject eventFirer)
587{
588 if(isDeleted(this->owner) || isDeleted(eventFirer))
589 {
590 return false;
591 }
592
593 Collider otherCollider = Collider::safeCast(eventFirer);
594
595 OtherColliderRegistry* otherColliderRegistry = Collider::findOtherColliderRegistry(this, otherCollider);
596 NM_ASSERT(!isDeleted(otherColliderRegistry), "Collider::onOtherColliderChanged: dead otherColliderRegistry");
597
598 if(NULL == otherColliderRegistry)
599 {
600 return false;
601 }
602
603 if(Collider::unregisterOtherCollider(this, otherCollider))
604 {
605 CollisionInformation collisionInformation;
606 collisionInformation.collider = this;
607 collisionInformation.otherCollider = otherCollider;
608
609 Entity::collisionEnds(this->owner, &collisionInformation);
610 }
611
612 return true;
613}

617OtherColliderRegistry* Collider::registerOtherCollider(Collider otherCollider, SolutionVector solutionVector, bool isImpenetrable)
618{
619 if(!this->otherColliders)
620 {
621 this->otherColliders = new VirtualList();
622 }
623
624 bool newEntry = false;
625 OtherColliderRegistry* otherColliderRegistry = Collider::findOtherColliderRegistry(this, Collider::safeCast(otherCollider));
626
627 if(NULL == otherColliderRegistry)
628 {
629 newEntry = true;
630 otherColliderRegistry = new OtherColliderRegistry;
631 }
632
633 otherColliderRegistry->collider = otherCollider;
634 otherColliderRegistry->solutionVector = solutionVector;
635 otherColliderRegistry->isImpenetrable = isImpenetrable;
636 otherColliderRegistry->frictionCoefficient = 0;
637
638 if(newEntry)
639 {
640 VirtualList::pushBack(this->otherColliders, otherColliderRegistry);
641
642 Collider::addEventListener
643 (
644 otherCollider, ListenerObject::safeCast(this), (EventListener)Collider::onOtherColliderChanged, kEventColliderDeleted
645 );
646
647 Collider::addEventListener
648 (
649 otherCollider, ListenerObject::safeCast(this), (EventListener)Collider::onOtherColliderChanged, kEventColliderChanged
650 );
651 }
652
653 return otherColliderRegistry;
654}

658bool Collider::unregisterOtherCollider(Collider otherCollider)
659{
660 ASSERT(!isDeleted(otherCollider), "Collider::removeOtherCollider: dead otherCollider");
661
662 OtherColliderRegistry* otherColliderRegistry = Collider::findOtherColliderRegistry(this, Collider::safeCast(otherCollider));
663 NM_ASSERT(!isDeleted(otherColliderRegistry), "Collider::removeOtherCollider: dead otherColliderRegistry");
664
665 if(isDeleted(otherColliderRegistry))
666 {
667 return false;
668 }
669
670 VirtualList::removeData(this->otherColliders, otherColliderRegistry);
671 delete otherColliderRegistry;
672
673 if(!isDeleted(otherCollider))
674 {
675 Collider::removeEventListener
676 (
677 otherCollider, ListenerObject::safeCast(this), (EventListener)Collider::onOtherColliderChanged, kEventColliderDeleted
678 );
679
680 Collider::removeEventListener
681 (
682 otherCollider, ListenerObject::safeCast(this), (EventListener)Collider::onOtherColliderChanged, kEventColliderChanged
683 );
684 }
685
686 return true;
687}

690
691OtherColliderRegistry* Collider::findOtherColliderRegistry(Collider collider)
692{
693 ASSERT(collider, "Collider::findOtherColliderRegistry: null collider");
694
695 if(NULL == this->otherColliders || isDeleted(collider))
696 {
697 return NULL;
698 }
699
700 for(VirtualNode node = this->otherColliders->head; NULL != node; node = node->next)
701 {
702 ASSERT(!isDeleted(node->data), "Collider::findOtherColliderRegistry: deleted registry");
703
704 if(collider == ((OtherColliderRegistry*)node->data)->collider)
705 {
706 return (OtherColliderRegistry*)node->data;
707 }
708 }
709
710 return NULL;
711}
