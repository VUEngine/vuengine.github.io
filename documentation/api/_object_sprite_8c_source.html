32void ObjectSprite::constructor(Entity owner, const ObjectSpriteSpec* objectSpriteSpec)
33{
34 NM_ASSERT(NULL != objectSpriteSpec, "ObjectSprite::constructor: NULL objectSpriteSpec");
35
36 // Always explicitly call the base's constructor
37 Base::constructor(owner, (SpriteSpec*)objectSpriteSpec);
38
39 this->head = objectSpriteSpec->display & __OBJECT_SPRITE_CHAR_SHOW_MASK;
40 this->objectSpriteContainer = NULL;
41 this->totalObjects = 0;
42
43 this->displacement = objectSpriteSpec->spriteSpec.displacement;
44 this->xDisplacementIncrement = 8;
45 this->yDisplacementIncrement = 8;
46 this->xDisplacementDelta = 0;
47 this->yDisplacementDelta = 0;
48
49 this->objectTextureSource.displacement = 0;
50
51 ASSERT(objectSpriteSpec->spriteSpec.textureSpec, "ObjectSprite::constructor: null textureSpec");
52
53 if(NULL != objectSpriteSpec->spriteSpec.textureSpec)
54 {
55 this->texture = Texture::get(typeofclass(ObjectTexture), (ObjectTextureSpec*)objectSpriteSpec->spriteSpec.textureSpec, 0, 0, 0);
56
57 NM_ASSERT(this->texture, "ObjectSprite::constructor: null texture");
58
59 Texture::addEventListener
60 (
61 this->texture, ListenerObject::safeCast(this), (EventListener)ObjectSprite::onTextureRewritten, kEventTextureRewritten);
62
63 this->totalObjects = objectSpriteSpec->spriteSpec.textureSpec->cols * objectSpriteSpec->spriteSpec.textureSpec->rows;
64
65 NM_ASSERT(this->texture, "ObjectSprite::constructor: null texture");
66 }
67 else
68 {
69 NM_ASSERT(this->texture, "ObjectSprite::constructor: null texture spec");
70 }
71
72 this->cols = this->halfWidth >> 2;
73 this->rows = this->halfHeight >> 2;
74
75 this->fourthWordValue = (this->head & 0x3000) | (this->texture->palette << 14);
76}

91 );
92

94 }
95
96 this->texture = NULL;
97
98 // Always explicitly call the base's destructor
99 Base::destructor();
100}

118int16 ObjectSprite::doRender(int16 index)
119{
120 NM_ASSERT(!isDeleted(this->texture), "ObjectSprite::doRender: null texture");
121 NM_ASSERT(!isDeleted(this->texture->charSet), "ObjectSprite::doRender: null char set");
122
123 int16 x = this->position.x - this->halfWidth + this->displacement.x - this->xDisplacementDelta;
124 int16 y = this->position.y - this->halfHeight + this->displacement.y - this->yDisplacementDelta;
125
126 uint16 secondWordValue = this->head | (this->position.parallax + this->displacement.parallax);
127 uint16 fourthWordValue =
128 this->fourthWordValue | (CharSet::getOffset(this->texture->charSet) + this->objectTextureSource.displacement);
129
130 int16 yDisplacement = 0;
131 int16 jDisplacement = 0;
132
133 uint16* framePointer =
134 (uint16*)(this->texture->textureSpec->map + this->texture->mapDisplacement + this->objectTextureSource.displacement);
135 uint16 result = index;
136
137 ObjectAttributes* objectPointer = NULL;
138
139 for(int16 i = 0; i < this->rows; i++, jDisplacement += this->cols, yDisplacement += this->yDisplacementIncrement)
140 {
141 int16 outputY = y + yDisplacement;
142
143 int16 objectIndexStart = index + jDisplacement;
144
145 if((unsigned)(outputY - _cameraFrustum->y0 + 8) > (unsigned)(_cameraFrustum->y1 - _cameraFrustum->y0 + 8))
146 {
147 for(int16 j = 0; j < this->cols; j++)
148 {
149 int16 objectIndex = objectIndexStart + j;
150
151 objectPointer = &_objectAttributesCache[objectIndex];
152 objectPointer->head = __OBJECT_SPRITE_CHAR_HIDE_MASK;
153 }
154
155 continue;
156 }
157
158 for(int16 j = 0, xDisplacement = 0; j < this->cols; j++, xDisplacement += this->xDisplacementIncrement)
159 {
160 int16 objectIndex = objectIndexStart + j;
161 objectPointer = &_objectAttributesCache[objectIndex];
162
163 int16 outputX = x + xDisplacement;
164
165 // Add 8 to the calculation to avoid char's cut off when scrolling hide the object if outside
166 // Screen's bounds
167 if((unsigned)(outputX - _cameraFrustum->x0 + 4) > (unsigned)(_cameraFrustum->x1 - _cameraFrustum->x0 + 4))
168 {
169 objectPointer->head = __OBJECT_SPRITE_CHAR_HIDE_MASK;
170 continue;
171 }
172
173 objectPointer->jx = outputX;
174 objectPointer->head = secondWordValue;
175 objectPointer->jy = outputY;
176
177 objectPointer->tile = fourthWordValue + framePointer[jDisplacement + j];
178
179 result = index;
180 }
181 }
182
183 return result;
184}

188void ObjectSprite::setRotation(const Rotation* rotation)
189{
190 this->rotation = *rotation;
191
192 NormalizedDirection normalizedDirection =
193 {
194 __QUARTER_ROTATION_DEGREES < __ABS(rotation->y) || __QUARTER_ROTATION_DEGREES < __ABS(rotation->z) ? __LEFT : __RIGHT,
195 __QUARTER_ROTATION_DEGREES < __ABS(rotation->x) || __QUARTER_ROTATION_DEGREES < __ABS(rotation->z) ? __UP : __DOWN,
196 __FAR,
197 };
198
199 if(__LEFT == normalizedDirection.x)
200 {
201 this->head |= 0x2000;
202 }
203 else if(__RIGHT == normalizedDirection.x)
204 {
205 this->head &= 0xDFFF;
206 }
207
208 if(__UP == normalizedDirection.y)
209 {
210 this->head |= 0x1000;
211 }
212 else if(__DOWN == normalizedDirection.y)
213 {
214 this->head &= 0xEFFF;
215 }
216
217 this->xDisplacementIncrement = 8;
218 this->yDisplacementIncrement = 8;
219 this->xDisplacementDelta = 0;
220 this->yDisplacementDelta = 0;
221
222 this->halfWidth = this->texture->textureSpec->cols << 2;
223 this->halfHeight = this->texture->textureSpec->rows << 2;
224
225 this->cols = this->halfWidth >> 2;
226 this->rows = this->halfHeight >> 2;
227
228 if(this->head & 0x2000)
229 {
230 this->xDisplacementIncrement = -8;
231 this->halfWidth = -this->halfWidth;
232 this->xDisplacementDelta = __OBJECT_SPRITE_FLIP_X_DISPLACEMENT;
233 }
234
235 if(this->head & 0x1000)
236 {
237 this->yDisplacementIncrement = -8;
238 this->halfHeight = -this->halfHeight;
239 this->yDisplacementDelta = __OBJECT_SPRITE_FLIP_Y_DISPLACEMENT;
240 }
241
242 this->fourthWordValue = (this->head & 0x3000) | (this->texture->palette << 14);
243}

283void ObjectSprite::removeFromCache()
284{
285 if(__NO_RENDER_INDEX != this->index)
286 {
287 int16 jDisplacement = 0;
288 ObjectAttributes* objectPointer = NULL;
289
290 for(int16 i = 0; i < this->rows; i++, jDisplacement += this->cols)
291 {
292 int16 objectIndexStart = this->index + jDisplacement;
293
294 for(int16 j = 0; j < this->cols; j++)
295 {
296 int16 objectIndex = objectIndexStart + j;
297 objectPointer = &_objectAttributesCache[objectIndex];
298 objectPointer->head = __OBJECT_SPRITE_CHAR_HIDE_MASK;
299 }
300 }
301 }
302}

306void ObjectSprite::rewrite()
307{
308 if(__HIDE == this->show)
309 {
310 return;
311 }
312
313 if(__NO_RENDER_INDEX == this->index)
314 {
315 return;
316 }
317
318 NM_ASSERT(!isDeleted(this->texture), "ObjectSprite::rewrite: null texture");
319 NM_ASSERT(!isDeleted(this->texture->charSet), "ObjectSprite::rewrite: null char set");
320
321 uint16 fourthWordValue =
322 (this->head & 0x3000) | (this->texture->palette << 14) |
323 (CharSet::getOffset(this->texture->charSet) + this->objectTextureSource.displacement);
324
325 int16 jDisplacement = 0;
326
327 uint16* framePointer =
328 (uint16*)(this->texture->textureSpec->map + this->texture->mapDisplacement + this->objectTextureSource.displacement);
329
330 ObjectAttributes* objectPointer = NULL;
331
332 for(int16 i = 0; i < this->rows; i++, jDisplacement += this->cols)
333 {
334 int16 objectIndexStart = this->index + jDisplacement;
335
336 for(int16 j = 0; j < this->cols; j++)
337 {
338 int16 objectIndex = objectIndexStart + j;
339 objectPointer = &_objectAttributesCache[objectIndex];
340
341 objectPointer->tile = fourthWordValue + framePointer[jDisplacement + j];
342 }
343 }
344}
