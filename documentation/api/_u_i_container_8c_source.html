65Actor UIContainer::spawnChildActor(const PositionedActor* const positionedActor)
66{
67 if(NULL != positionedActor)
68 {
69 Actor actor = Actor::createActor(positionedActor, !isDeleted(this->children) ? VirtualList::getCount(this->children) : 0);
70 ASSERT(actor, "UIContainer::doAddChildActor: actor not loaded");
71
72 if(!isDeleted(actor))
73 {
74 // Create the actor and add it to the world
75 UIContainer::addChild(this, Container::safeCast(actor));
76 }
77
78 return actor;
79 }
80
81 return NULL;
82}
