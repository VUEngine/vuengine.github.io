27static void LineField::project(Vector3D center, fixed_t radius, Vector3D vector, fixed_t* min, fixed_t* max)
28{
29 // Project this onto the current normal
30 fixed_t dotProduct = Vector3D::dotProduct(vector, center);
31
32 *min = dotProduct - radius;
33 *max = dotProduct + radius;
34
35 if(*min > *max)
36 {
37 fixed_t aux = *min;
38 *min = *max;
39 *max = aux;
40 }
41}

91void LineField::configureWireframe()
92{
93 if(!isDeleted(this->wireframe))
94 {
95 return;
96 }
97
98 this->meshSpec = new MeshSpec;
99
100 const PixelVector MeshesSegments[][2]=
101 {
102 // Line
103 {
104 PixelVector::getFromVector3D(this->a, 0),
105 PixelVector::getFromVector3D(this->b, 0),
106 },
107
108 // Normal
109 {
110 PixelVector::getFromVector3D(Vector3D::intermediate(this->a, this->b), 0),
111 PixelVector::getFromVector3D
112 (
113 Vector3D::sum(Vector3D::intermediate(this->a, this->b), Vector3D::scalarProduct(this->normal, this->normalLength)), 0
114 ),
115 },
116
117 // Limiter
118 {
119 {0, 0, 0, 0},
120 {0, 0, 0, 0}
121 },
122 };
123
124 *this->meshSpec = (MeshSpec)
125 {
126 {
127 // Component
128 {
129 // Allocator
130 __TYPE(Mesh),
131
132 // Component type
133 kWireframeComponent
134 },
135
136 // Displacement
137 {0, 0, 0},
138
140 __COLOR_BRIGHT_RED,
141
143 __TRANSPARENCY_NONE,
144
146 true
147 },
148
150 (PixelVector(*)[2])MeshesSegments
151 };
152
153 // Create a wireframe
154 this->wireframe = Wireframe::safeCast(new Mesh(this->owner, this->meshSpec));
155
156 if(!isDeleted(this->wireframe))
157 {
158 Line::setDisplacement(this->wireframe, Vector3D::getFromPixelVector(((ColliderSpec*)this->componentSpec)->displacement));
159 }
160}

195void LineField::displace(fixed_t displacement)
196{
197 this->a = Vector3D::sum(this->a, Vector3D::scalarProduct(this->normal, displacement));
198 this->b = Vector3D::sum(this->b, Vector3D::scalarProduct(this->normal, displacement));
199}

250void LineField::computeSize()
251{
252 if(NULL == this->transformation)
253 {
254 return;
255 }
256
257 fix7_9 normalScale = __I_TO_FIX7_9(1);
258
259 if(this->transformation->scale.x > normalScale)
260 {
261 normalScale = this->transformation->scale.x;
262 }
263
264 if(this->transformation->scale.y > normalScale)
265 {
266 normalScale = this->transformation->scale.y;
267 }
268
269 if(this->transformation->scale.z > normalScale)
270 {
271 normalScale = this->transformation->scale.z;
272 }
273
274 this->normalLength = __FIXED_MULT(this->normalLength, __FIX7_9_TO_FIXED(normalScale));
275
276 Rotation rotation =
277 Rotation::sum
278 (
279 Rotation::getFromPixelRotation(((ColliderSpec*)this->componentSpec)->pixelRotation), this->transformation->rotation
280 );
281
282 Size size = Size::getFromPixelSize(((ColliderSpec*)this->componentSpec)->pixelSize);
283
284 if(0 != size.x)
285 {
286 this->a.x = size.x >> 1;
287 this->a.y = 0;
288 this->a.z = 0;
289
290 if(0 != rotation.y)
291 {
292 this->a.x = __FIXED_MULT((size.x >> 1), __FIX7_9_TO_FIXED(__COS(__FIXED_TO_I(rotation.y))));
293 this->a.z = __FIXED_MULT((size.x >> 1), __FIX7_9_TO_FIXED(__SIN(__FIXED_TO_I(rotation.y))));
294 this->a.y = 0;
295 }
296 else if(0 != rotation.z)
297 {
298 this->a.x = __FIXED_MULT((size.x >> 1), __FIX7_9_TO_FIXED(__COS(__FIXED_TO_I(rotation.z))));
299 this->a.y = __FIXED_MULT((size.x >> 1), __FIX7_9_TO_FIXED(__SIN(__FIXED_TO_I(rotation.z))));
300 this->a.z = 0;
301 }
302 }
303 else if(0 != size.y)
304 {
305 this->a.x = 0;
306 this->a.y = size.y >> 1;
307 this->a.z = 0;
308
309 if(0 != rotation.x)
310 {
311 this->a.x = __FIXED_MULT((size.y >> 1), __FIX7_9_TO_FIXED(__COS(__FIXED_TO_I(rotation.x))));
312 this->a.z = __FIXED_MULT((size.y >> 1), __FIX7_9_TO_FIXED(__SIN(__FIXED_TO_I(rotation.x))));
313 this->a.y = 0;
314 }
315 else if(0 != rotation.z)
316 {
317 this->a.x = __FIXED_MULT((size.y >> 1), __FIX7_9_TO_FIXED(__COS(__FIXED_TO_I(rotation.z))));
318 this->a.y = __FIXED_MULT((size.y >> 1), __FIX7_9_TO_FIXED(__SIN(__FIXED_TO_I(rotation.z))));
319 this->a.z = 0;
320 }
321 }
322 else if(0 != size.z)
323 {
324 this->a.x = 0;
325 this->a.y = 0;
326 this->a.z = size.z >> 1;
327
328 if(0 != rotation.x)
329 {
330 this->a.x = __FIXED_MULT((size.z >> 1), __FIX7_9_TO_FIXED(__COS(__FIXED_TO_I(rotation.y))));
331 this->a.y = __FIXED_MULT((size.z >> 1), __FIX7_9_TO_FIXED(__SIN(__FIXED_TO_I(rotation.y))));
332 this->a.z = 0;
333 }
334 else if(0 != rotation.y)
335 {
336 this->a.x = __FIXED_MULT((size.z >> 1), __FIX7_9_TO_FIXED(__COS(__FIXED_TO_I(rotation.x))));
337 this->a.y = 0;
338 this->a.z = __FIXED_MULT((size.z >> 1), __FIX7_9_TO_FIXED(__SIN(__FIXED_TO_I(rotation.x))));
339 }
340 }
341
342 this->b = Vector3D::scalarProduct(this->a, __I_TO_FIXED(-1));
343
344 fixed_t dx = this->b.x - this->a.x;
345 fixed_t dy = this->b.y - this->a.y;
346 fixed_t dz = this->b.z - this->a.z;
347
348 this->normal = Vector3D::normalize((Vector3D){dy, -dx, dz});
349}
