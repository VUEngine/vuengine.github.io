31static int16 Affine::transform
32(
33 int32 maximumParamTableRowsToComputePerCall, uint32 param, int16 paramTableRow, fixed_t targetHalfWidth,
34 fixed_t targetHalfHeight, fix13_3 mx, fix13_3 my, fixed_t halfWidth, fixed_t halfHeight, const Rotation* rotation
35)
36{
37 fixed_t finalScaleX = __FIXED_MULT(__FIX7_9_TO_FIXED(__COS(__FIXED_TO_I(rotation->y))), __FIXED_DIV(targetHalfWidth, halfWidth));
38 fixed_t finalScaleY = __FIXED_MULT(__FIX7_9_TO_FIXED(__COS(__FIXED_TO_I(rotation->x))), __FIXED_DIV(targetHalfHeight, halfHeight));
39
40 if(0 == finalScaleX)
41 {
42 finalScaleX = __F_TO_FIX7_9(0.01f);
43 }
44
45 if(0 == finalScaleY)
46 {
47 finalScaleY = __F_TO_FIX7_9(0.01f);
48 }
49
50 fixed_t highPrecisionPa = __FIXED_DIV(__FIX7_9_TO_FIXED(__COS(-__FIXED_TO_I(rotation->z))), finalScaleX);
51 fixed_t highPrecisionPb = -__FIXED_DIV(__FIX7_9_TO_FIXED(__SIN(-__FIXED_TO_I(rotation->z))), finalScaleX);
52 fixed_t highPrecisionPc = __FIXED_DIV(__FIX7_9_TO_FIXED(__SIN(-__FIXED_TO_I(rotation->z))), finalScaleY);
53 fixed_t highPrecisionPd = __FIXED_DIV(__FIX7_9_TO_FIXED(__COS(-__FIXED_TO_I(rotation->z))), finalScaleY);
54
55 FixedAffineMatrix fixedAffineMatrix;
56 fixedAffineMatrix.pa = __FIXED_TO_FIX7_9(highPrecisionPa);
57 fixedAffineMatrix.pc = __FIXED_TO_FIX7_9(highPrecisionPc);
58
59 // BgX + bgWidth - pa * dispX - pb * dispY
60 fixedAffineMatrix.dx =
61 mx
62 +
63 __FIXED_TO_FIX13_3
64 (
65 halfWidth
66 -
67 (
68 __FIXED_MULT(highPrecisionPa, targetHalfWidth)
69 +
70 __FIXED_MULT(highPrecisionPb, targetHalfHeight)
71 )
72 );
73
74 // BgY + bgHeight - pc * dispX - pd * dispY
75 fixedAffineMatrix.dy =
76 my
77 +
78 __FIXED_TO_FIX13_3
79 (
80 halfHeight
81 -
82 (
83 __FIXED_MULT(highPrecisionPc, targetHalfWidth)
84 +
85 __FIXED_MULT(highPrecisionPd, targetHalfHeight)
86 )
87 );
88
89 fixedAffineMatrix.parallax = 0;
90
91 AffineEntry* affine = (AffineEntry*)param;
92
93 int16 i = 0 <= paramTableRow ? paramTableRow : 0;
94 int32 lastRow = __FIXED_TO_I(__FIXED_MULT((halfHeight << 1), finalScaleY)) + 1;
95 int32 counter = maximumParamTableRowsToComputePerCall;
96
97 if(rotation->x)
98 {
99 fix19_13 topEdgeSizeZ = __FIX19_13_MULT(__FIXED_TO_FIX19_13(halfHeight), __FIX7_9_TO_FIX19_13(__SIN(__FIXED_TO_I(rotation->x))));
100 fix19_13 bottomEdgeSizeZ = -topEdgeSizeZ;
101
102 fix19_13 scaleXDifference =
103 __FIX19_13_MULT
104 (
105 __I_TO_FIX19_13(1),
106 __FIXED_TO_FIX19_13(__FIXED_DIV(__PIXELS_TO_METERS(__FIX19_13_TO_I(topEdgeSizeZ)), _optical->scalingFactor))
107 );
108
109 fix19_13 scaleYDifference = __FIX7_9_TO_FIX19_13(__SIN(__FIXED_TO_I(rotation->x)));
110
111 fix19_13 scaleXFactor = __FIXED_TO_FIX19_13(finalScaleX);
112 fix19_13 scaleYFactor = __FIXED_TO_FIX19_13(finalScaleX);
113
114 fix19_13 topEdgeScaleX = __FIX19_13_MULT(__I_TO_FIX19_13(1) - scaleXDifference, (scaleXFactor));
115 fix19_13 bottomEdgeScaleX = __FIX19_13_MULT(__I_TO_FIX19_13(1) + scaleXDifference, (scaleXFactor));
116
117 fix19_13 topEdgeScaleY = __FIX19_13_MULT(__I_TO_FIX19_13(1) - scaleYDifference, (scaleYFactor));
118 fix19_13 bottomEdgeScaleY = __FIX19_13_MULT(__I_TO_FIX19_13(1) + scaleYDifference, (scaleYFactor));
119
120 fix19_13 scaleXIncrement = __FIX19_13_DIV(bottomEdgeScaleX - topEdgeScaleX, __I_TO_FIX19_13(lastRow));
121 fix19_13 scaleYIncrement = __FIX19_13_DIV(bottomEdgeScaleY - topEdgeScaleY, __I_TO_FIX19_13(lastRow));
122
123 fix19_13 scaleX = topEdgeScaleX;
124 fix19_13 scaleY = topEdgeScaleY;
125
126 fixed_t parallaxIncrement = -__FIX19_13_TO_FIXED(__FIX19_13_DIV((bottomEdgeSizeZ - topEdgeSizeZ), __I_TO_FIX19_13(lastRow)));
127 fixed_t parallax = -__FIXED_MULT(__I_TO_FIXED(lastRow / 2), parallaxIncrement) + __FIXED_MULT(parallaxIncrement, __I_TO_FIXED(i));
128
129/*
130PRINT_TEXT(" ", 1, 5);
131PRINT_INT(rotation->x, 1, 5);
132PRINT_TEXT(" ", 1, 7);
133PRINT_INT(__FIX19_13_TO_F(topEdgeScaleX)*1000, 1, 7);
134PRINT_TEXT(" ", 10, 7);
135PRINT_INT(__FIX19_13_TO_F(topEdgeScaleY)*1000, 10, 7);
136PRINT_TEXT(" ", 1, 8);
137PRINT_INT(__FIX19_13_TO_F(bottomEdgeScaleX)*1000, 1, 8);
138PRINT_TEXT(" ", 10, 8);
139PRINT_INT(__FIX19_13_TO_F(bottomEdgeScaleY)*1000, 10, 8);
140
141PRINT_TEXT(" ", 1, 9);
142PRINT_INT(__FIX19_13_TO_F(topEdgeSizeZ)*1000, 1, 9);
143PRINT_TEXT(" ", 10, 9);
144PRINT_INT(__FIX19_13_TO_F(bottomEdgeSizeZ)*1000, 10, 9);
145
146PRINT_TEXT(" ", 1, 10);
147PRINT_INT(__FIX19_13_TO_F(scaleX)*1000, 1, 10);
148PRINT_TEXT(" ", 10, 10);
149PRINT_INT(__FIX19_13_TO_F(scaleY)*1000, 10, 10);
150
151PRINT_TEXT(" ", 1, 16);
152PRINT_INT(lastRow, 1, 16);
153//PRINT_INT((_optical->maximumXViewDistancePower), 1, 14);
154//PRINT_INT(__METERS_TO_PIXELS(_optical->scalingFactor), 1, 15);
155*/
156
157 for(; 0!= counter && i <= lastRow; i++, counter--)
158 {
159 if(0 == scaleX)
160 {
161 if(!scaleXIncrement)
162 {
163 affine[i].pb_y = 0;
164 affine[i].parallax = 0;
165 affine[i].pd_y = 0;
166 affine[i].pa = 0;
167 affine[i].pc = 0;
168
169 continue;
170 }
171
172 scaleX += scaleXIncrement;
173 }
174
175 if(0 == scaleY)
176 {
177 if(!scaleYIncrement)
178 {
179 affine[i].pb_y = 0;
180 affine[i].parallax = 0;
181 affine[i].pd_y = 0;
182 affine[i].pa = 0;
183 affine[i].pc = 0;
184
185 continue;
186 }
187
188 scaleY += scaleYIncrement;
189 }
190
191 highPrecisionPa = (__FIX19_13_DIV(__FIX7_9_TO_FIX19_13(__COS(-__FIXED_TO_I(rotation->z))), scaleX));
192 highPrecisionPb = -(__FIX19_13_DIV(__FIX7_9_TO_FIX19_13(__SIN(-__FIXED_TO_I(rotation->z))), scaleX));
193 highPrecisionPc = (__FIX19_13_DIV(__FIX7_9_TO_FIX19_13(__SIN(-__FIXED_TO_I(rotation->z))), scaleY));
194 highPrecisionPd = (__FIX19_13_DIV(__FIX7_9_TO_FIX19_13(__COS(-__FIXED_TO_I(rotation->z))), scaleY));
195
196 fixedAffineMatrix.pa = __FIX19_13_TO_FIX7_9(highPrecisionPa);
197 fixedAffineMatrix.pc = __FIX19_13_TO_FIX7_9(highPrecisionPc);
198
199 // BgX + bgWidth - pa * dispX - pb * dispY
200 fixedAffineMatrix.dx =
201 mx
202 +
203 __FIX19_13_TO_FIX13_3
204 (
205 __FIXED_TO_FIX19_13(halfWidth)
206 -
207 (
208 __FIX19_13_MULT(highPrecisionPa, __FIXED_TO_FIX19_13(targetHalfWidth))
209 +
210 __FIX19_13_MULT(highPrecisionPb, __FIXED_TO_FIX19_13(targetHalfHeight))
211 )
212 );
213
214 // BgY + bgHeight - pc * dispX - pd * dispY
215 fixedAffineMatrix.dy =
216 my
217 +
218 __FIX19_13_TO_FIX13_3
219 (
220 __FIXED_TO_FIX19_13(halfHeight)
221 -
222 (
223 __FIX19_13_MULT(highPrecisionPc, __FIXED_TO_FIX19_13(targetHalfWidth))
224 +
225 __FIX19_13_MULT(highPrecisionPd, __FIXED_TO_FIX19_13(targetHalfHeight))
226 )
227 );
228
229 affine[i].pb_y = __FIX19_13_TO_FIX13_3(__FIXED_MULT(__I_TO_FIXED(i), highPrecisionPb)) + fixedAffineMatrix.dx;
230 affine[i].parallax = __FIXED_TO_I(parallax);
231 affine[i].pd_y = __FIX19_13_TO_FIX13_3(__FIX19_13_MULT(__I_TO_FIX19_13(i), highPrecisionPd)) + fixedAffineMatrix.dy;
232 affine[i].pa = fixedAffineMatrix.pa;
233 affine[i].pc = fixedAffineMatrix.pc;
234
235 parallax += parallaxIncrement;
236 scaleX += scaleXIncrement;
237 scaleY += scaleYIncrement;
238 }
239 }
240 else
241 {
242 for(; 0!= counter && i <= lastRow; i++, counter--)
243 {
244 affine[i].pb_y = __FIXED_TO_FIX13_3(__FIXED_MULT(__I_TO_FIXED(i), highPrecisionPb)) + fixedAffineMatrix.dx;
245 affine[i].parallax = fixedAffineMatrix.parallax;
246 affine[i].pd_y = __FIXED_TO_FIX13_3(__FIXED_MULT(__I_TO_FIXED(i), highPrecisionPd)) + fixedAffineMatrix.dy;
247 affine[i].pa = fixedAffineMatrix.pa;
248 affine[i].pc = fixedAffineMatrix.pc;
249 }
250 }
251
252 if(i <= lastRow)
253 {
254 return i;
255
256 }
257
258 affine[i].pb_y = 0;
259 affine[i].parallax = 0;
260 affine[i].pd_y = 0;
261 affine[i].pa = 0;
262 affine[i].pc = 0;
263
264 return -1;
265}
