37static void BgmapTextureManager::reset()
38{
39 BgmapTextureManager bgmapTextureManager = BgmapTextureManager::getInstance();
40
41 NM_ASSERT(__BGMAP_SPACE_BASE_ADDRESS < __PARAM_TABLE_END, "BgmapTextureManager::reset: bgmap address space is negative");
42
43 VirtualList::deleteData(bgmapTextureManager->bgmapTextures);
44
45 bgmapTextureManager->availableBgmapSegmentsForTextures = (uint32)((__PARAM_TABLE_END - __BGMAP_SPACE_BASE_ADDRESS) / __BGMAP_SEGMENT_SIZE);
46
47 if(bgmapTextureManager->availableBgmapSegmentsForTextures > __MAX_NUMBER_OF_BGMAPS_SEGMENTS)
48 {
49 bgmapTextureManager->availableBgmapSegmentsForTextures = __MAX_NUMBER_OF_BGMAPS_SEGMENTS;
50 }
51
52 bgmapTextureManager->printingBgmapSegment = bgmapTextureManager->availableBgmapSegmentsForTextures - 1;
53
54 // Clear each bgmap segment usage
55 for(int32 i = 0; i < __MAX_NUMBER_OF_BGMAPS_SEGMENTS; i++)
56 {
57 bgmapTextureManager->usedTiles[i] = 0;
58
59 // Clear the offsets
60 for(int32 j = 0; j <__NUM_BGMAPS_PER_SEGMENT; j++)
61 {
62 bgmapTextureManager->xOffset[i][j] = 0;
63 bgmapTextureManager->yOffset[i][j] = 0;
64 }
65 }
66
67 for(int32 i = 0; i < __MAX_NUMBER_OF_BGMAPS_SEGMENTS * __NUM_BGMAPS_PER_SEGMENT; i++)
68 {
69 bgmapTextureManager->offset[i][kXOffset] = -1;
70 bgmapTextureManager->offset[i][kYOffset] = -1;
71 bgmapTextureManager->offset[i][kCols] = 0;
72 bgmapTextureManager->offset[i][kRows] = 0;
73 }
74}

121static void BgmapTextureManager::loadTextures(const TextureSpec** textureSpecs)
122{
123 // Textures
124 if(NULL != textureSpecs)
125 {
126 VirtualList sortedTextureSpecs = new VirtualList();
127 VirtualList preloadedTextures = new VirtualList();
128
129 for(int16 i = 0; NULL != textureSpecs[i]; i++)
130 {
131 if(NULL == textureSpecs[i]->charSetSpec)
132 {
133 continue;
134 }
135
136 if
137 (
138 textureSpecs[i]->recyclable
139 ||
140 (
141 textureSpecs[i]->charSetSpec->shared
142 )
143 )
144 {
145 VirtualNode node = NULL;
146
147 for(node = sortedTextureSpecs->head; NULL != node; node = node->next)
148 {
149 TextureSpec* sortedTextureSpec = (TextureSpec*)node->data;
150
151 if(textureSpecs[i] == sortedTextureSpec)
152 {
153 break;
154 }
155
156 if(textureSpecs[i]->rows < sortedTextureSpec->rows)
157 {
158 continue;
159 }
160
161 VirtualList::insertBefore(sortedTextureSpecs, node, textureSpecs[i]);
162 break;
163 }
164
165 if(NULL == node)
166 {
167 VirtualList::pushBack(sortedTextureSpecs, textureSpecs[i]);
168 }
169 }
170 }
171
172 // Must not release the texture just after loading them so
173 // The recyclable ones are not recycled immediately
174 for(VirtualNode node = sortedTextureSpecs->head; NULL != node; node = node->next)
175 {
176 TextureSpec* textureSpec = (TextureSpec*)node->data;
177
178 BgmapTexture bgmapTexture = BgmapTextureManager::getTexture(textureSpec, 0, false, __WORLD_1x1);
179
180 NM_ASSERT(!isDeleted(bgmapTexture), "BgmapTextureManager::loadTextures: failed to load bgmapTexture");
181 VirtualList::pushBack(preloadedTextures, bgmapTexture);
182 }
183
184 delete sortedTextureSpecs;
185
186 for(VirtualNode node = VirtualList::begin(preloadedTextures); NULL != node; node = node->next)
187 {
188 BgmapTextureManager::releaseTexture(BgmapTexture::safeCast(node->data));
189 }
190
191 delete preloadedTextures;
192 }
193}

197static BgmapTexture BgmapTextureManager::getTexture
198(
199 BgmapTextureSpec* bgmapTextureSpec, int16 minimumSegment, bool mustLiveAtEvenSegment, uint32 scValue
200)
201{
202 if(NULL == bgmapTextureSpec)
203 {
204 NM_ASSERT(false, "BgmapTextureManager::getTexture: NULL spec provided");
205 return NULL;
206 }
207
208 BgmapTexture bgmapTexture = NULL;
209
210 if(NULL == bgmapTextureSpec->charSetSpec)
211 {
212 bgmapTexture = BgmapTextureManager::allocateTexture(bgmapTextureSpec, minimumSegment, mustLiveAtEvenSegment, scValue);
213
214 ASSERT(!isDeleted(bgmapTexture), "BgmapTextureManager::getTexture: (animated) texture no allocated");
215 }
216 else
217 {
218 if(!bgmapTextureSpec->charSetSpec->shared)
219 {
220 if(bgmapTextureSpec->recyclable)
221 {
222 bgmapTexture = BgmapTextureManager::findTexture(bgmapTextureSpec, true);
223 }
224
225 if(NULL != bgmapTexture)
226 {
227 BgmapTexture::increaseUsageCount(bgmapTexture);
228 }
229 else
230 {
231 // Load a new texture
232 bgmapTexture =
233 BgmapTextureManager::allocateTexture(bgmapTextureSpec, minimumSegment, mustLiveAtEvenSegment, scValue);
234 }
235
236 ASSERT(bgmapTexture, "BgmapTextureManager::getTexture: (animated) texture no allocated");
237 }
238 else
239 {
240 // First try to find an already created texture
241 bgmapTexture = BgmapTextureManager::findTexture(bgmapTextureSpec, false);
242
243 // If couldn't find the texture
244 if(NULL != bgmapTexture)
245 {
246 BgmapTexture::increaseUsageCount(bgmapTexture);
247 }
248 else
249 {
250 // Load it
251 bgmapTexture =
252 BgmapTextureManager::allocateTexture(bgmapTextureSpec, minimumSegment, mustLiveAtEvenSegment, scValue);
253 }
254
255 ASSERT(!isDeleted(bgmapTexture), "BgmapTextureManager::getTexture: (shared) texture no allocated");
256 }
257 }
258
259 if(!isDeleted(bgmapTexture) && kTextureWritten != bgmapTexture->status)
260 {
261 BgmapTexture::prepare(bgmapTexture);
262 }
263
264 return bgmapTexture;
265}

371static BgmapTexture BgmapTextureManager::findTexture(const BgmapTextureSpec* bgmapTextureSpec, bool recyclableOnly)
372{
373 BgmapTextureManager bgmapTextureManager = BgmapTextureManager::getInstance();
374
375 TextureSpec* textureSpec = (TextureSpec*)bgmapTextureSpec;
376 BgmapTexture selectedBgmapTexture = NULL;
377
378 CACHE_RESET;
379
380 // Try to find a texture with the same bgmap spec
381 for(VirtualNode node = bgmapTextureManager->bgmapTextures->head; NULL != node; node = node->next)
382 {
383 BgmapTexture allocatedBgmapTexture = BgmapTexture::safeCast(node->data);
384 const TextureSpec* allocatedTextureSpec = allocatedBgmapTexture->textureSpec;
385
386 if(!recyclableOnly && allocatedTextureSpec == textureSpec)
387 {
388 if
389 (
390 (
391 NULL == allocatedBgmapTexture->charSet
392 ||
393 allocatedTextureSpec->charSetSpec->shared == bgmapTextureSpec->charSetSpec->shared
394 )
395 &&
396 (
397 allocatedTextureSpec->padding.cols == bgmapTextureSpec->padding.cols
398 &&
399 allocatedTextureSpec->padding.rows == bgmapTextureSpec->padding.rows
400 )
401 )
402 {
403 // Return if found
404 return allocatedBgmapTexture;
405 }
406 }
407
408 if(!textureSpec->recyclable)
409 {
410 continue;
411 }
412
413 if(allocatedTextureSpec->recyclable && 0 == allocatedBgmapTexture->usageCount)
414 {
415 uint16 id = allocatedBgmapTexture->id;
416 uint16 cols = bgmapTextureManager->offset[id][kCols];
417 uint16 rows = bgmapTextureManager->offset[id][kRows];
418
419 if
420 (
421 textureSpec->cols > (cols >> 2) && textureSpec->cols <= cols
422 &&
423 (textureSpec->rows > (rows >> 2)) && textureSpec->rows <= rows
424 )
425 {
426 if(textureSpec->cols == cols && textureSpec->rows == rows)
427 {
428 selectedBgmapTexture = allocatedBgmapTexture;
429 break;
430 }
431 else if(NULL == selectedBgmapTexture)
432 {
433 selectedBgmapTexture = allocatedBgmapTexture;
434 }
435 else
436 {
437 uint16 selectedBgmapTextureId = selectedBgmapTexture->id;
438 uint16 selectedBgmapTextureCols = bgmapTextureManager->offset[selectedBgmapTextureId][kCols];
439 uint16 selectedBgmapTextureRows = bgmapTextureManager->offset[selectedBgmapTextureId][kRows];
440
441 if(cols < selectedBgmapTextureCols || rows < selectedBgmapTextureRows)
442 {
443 selectedBgmapTexture = allocatedBgmapTexture;
444 }
445 }
446 }
447 }
448 }
449
450 if(!isDeleted(selectedBgmapTexture))
451 {
452 Texture::setSpec(selectedBgmapTexture, textureSpec);
453 }
454
455 return selectedBgmapTexture;
456}

460static BgmapTexture BgmapTextureManager::allocateTexture
461(
462 BgmapTextureSpec* bgmapTextureSpec, int16 minimumSegment, bool mustLiveAtEvenSegment, uint32 scValue
463)
464{
465 BgmapTextureManager bgmapTextureManager = BgmapTextureManager::getInstance();
466
467 uint16 id = VirtualList::getCount(bgmapTextureManager->bgmapTextures);
468
469 //if not, then allocate
470 int32 segment =
471 BgmapTextureManager::doAllocate(id, (TextureSpec*)bgmapTextureSpec, minimumSegment, mustLiveAtEvenSegment, scValue);
472
473 if(0 > segment)
474 {
475 return NULL;
476 }
477
478 // Create new texture and register it
479 BgmapTexture bgmapTexture = new BgmapTexture(bgmapTextureSpec, id);
480 BgmapTexture::setSegment(bgmapTexture, segment);
481 BgmapTexture::setOffsets(bgmapTexture, BgmapTextureManager::getXOffset(id), BgmapTextureManager::getYOffset(id));
482
483 VirtualList::pushBack(bgmapTextureManager->bgmapTextures, bgmapTexture);
484
485 return bgmapTexture;
486}

489
490static int32 BgmapTextureManager::doAllocate
491(
492 uint16 id, TextureSpec* textureSpec, int16 minimumSegment, bool mustLiveAtEvenSegment, uint32 scValue
493)
494{
495 BgmapTextureManager bgmapTextureManager = BgmapTextureManager::getInstance();
496
497 int32 i = 0;
498 int32 j = 0;
499 int32 aux = 0;
500
501 int32 cols = Texture::getTotalCols(textureSpec);
502 int32 rows = Texture::getTotalRows(textureSpec);
503
504 uint16 colsPad = (textureSpec->padding.cols << 1);// + (cols < 64 ? 1 : 0);
505 uint16 rowsPad = (textureSpec->padding.rows << 1);// + (rows < 64 ? 1 : 0);
506
507 int32 area = (cols + colsPad) * (rows + rowsPad);
508
509 int16 segmentStep = 1;
510
511 if(mustLiveAtEvenSegment)
512 {
513 switch(scValue)
514 {
515 case __WORLD_8x1:
516 case __WORLD_4x2:
517 case __WORLD_2x4:
518 case __WORLD_1x8:
519
520 segmentStep = 8;
521 break;
522
523 case __WORLD_4x1:
524 case __WORLD_2x2:
525 case __WORLD_1x4:
526
527 segmentStep = 4;
528 break;
529
530 case __WORLD_2x1:
531 case __WORLD_1x2:
532
533 segmentStep = 2;
534 break;
535 }
536
537 if(0 != minimumSegment % segmentStep)
538 {
539 minimumSegment += (segmentStep - minimumSegment % segmentStep);
540 }
541
542 NM_ASSERT(0 == minimumSegment % 2, "BgmapTextureManager::doAllocate: cannot honor request for even bgmap");
543 }
544
545 CACHE_RESET;
546
547 for(i = minimumSegment; i < __MAX_NUMBER_OF_BGMAPS_SEGMENTS && i < bgmapTextureManager->availableBgmapSegmentsForTextures; i += segmentStep)
548 {
549 int32 maximumRow = i == bgmapTextureManager->printingBgmapSegment ? 64 - __SCREEN_HEIGHT_IN_CHARS : 64;
550
551 // If there is space in the segment memory
552 // There are 4096 chars in each bgmap segment
553 if((int32)(4096 - bgmapTextureManager->usedTiles[i]) >= (int32)area )
554 {
555 // Check if there is space within the segment
556 // We check the next so don't go to the last element
557 for(j = 0; j < __NUM_BGMAPS_PER_SEGMENT - 1; j++)
558 {
559 // Determine the y offset inside the bgmap segment
560 if(!bgmapTextureManager->yOffset[i][j + 1])
561 {
562 aux = maximumRow;
563 }
564 else
565 {
566 aux = bgmapTextureManager->yOffset[i][j + 1];
567 }
568
569 // Determine if there is still mem space (columns) in the current y offset
570 if(rows + rowsPad <= aux - bgmapTextureManager->yOffset[i][j] || (!bgmapTextureManager->yOffset[i][j + 1]))
571 {
572 if(rows + rowsPad <= maximumRow - bgmapTextureManager->yOffset[i][j])
573 {
574 if(cols + colsPad <= 64 - bgmapTextureManager->xOffset[i][j])
575 {
576 // Register bgmap spec
577 bgmapTextureManager->offset[id][kXOffset] = bgmapTextureManager->xOffset[i][j] + (colsPad >> 1);
578 bgmapTextureManager->offset[id][kYOffset] = bgmapTextureManager->yOffset[i][j] + (rowsPad >> 1);
579 bgmapTextureManager->offset[id][kCols] = cols;
580 bgmapTextureManager->offset[id][kRows] = rows;
581
582 NM_ASSERT
583 (
584 !mustLiveAtEvenSegment || 0 == (i % 2),
585 "BgmapTextureManager::doAllocate: cannot honor request for even bgmap"
586 );
587
588 // Increment the x offset
589 bgmapTextureManager->xOffset[i][j] += cols + colsPad;
590
591 // If the number of rows of the bgmap spec is greater than the
592 // Next y offset defined, increase the next y offset
593 if(bgmapTextureManager->yOffset[i][j + 1] - bgmapTextureManager->yOffset[i][j] < rows + rowsPad)
594 {
595 bgmapTextureManager->yOffset[i][j + 1] = bgmapTextureManager->yOffset[i][j] + rows + rowsPad;
596 }
597
598 // Update the number of chars defined inside the bgmap segment
599 bgmapTextureManager->usedTiles[i] += area;
600
601 // If there is a free bgmap segment
602 return i;
603 }
604 }
605 else
606 {
607 break;
608 }
609 }
610 else
611 {
612 if(rows + rowsPad > 64 - bgmapTextureManager->yOffset[i][j])
613 {
614 break;
615 }
616 }
617 }
618 }
619 }
620
621#ifdef __DEBUG
622 Printing::setDebugMode();
623 Printing::clear();
624 BgmapTextureManager::print(1, 10);
625 Printing::text("Texture ", 1, 19, NULL);
626 Printing::text(" Address: ", 1, 21, NULL);
627 Printing::hex((WORD)textureSpec, 14, 21, 8, NULL);
628
629 // Throw an exception if there is no enough space to allocate the bgmap spec
630 Error::triggerException("BgmapTextureManager::doAllocate: bgmap segments depleted", NULL);
631#endif
632
633 return -1;
634}
