26static void Box::project(Vector3D vertexes[__BOX_VERTEXES], Vector3D vector, fixed_t* min, fixed_t* max)
27{
28 int32 vertexIndex = 0;
29
30 // Project this onto the current normal
31 fixed_t dotProduct = Vector3D::dotProduct(vector, vertexes[vertexIndex]);
32
33 fixed_t finalMin = dotProduct;
34 fixed_t finalMax = dotProduct;
35
36 // Project this onto the current normal
37 for(; vertexIndex < __BOX_VERTEXES; vertexIndex++)
38 {
39 dotProduct = Vector3D::dotProduct(vector, vertexes[vertexIndex]);
40
41 if(dotProduct < finalMin)
42 {
43 finalMin = dotProduct;
44 }
45 else if(dotProduct > finalMax)
46 {
47 finalMax = dotProduct;
48 }
49 }
50
51 *min = finalMin;
52 *max = finalMax;
53}

63void Box::constructor(Entity owner, const ColliderSpec* colliderSpec)
64{
65 // Always explicitly call the base's constructor
66 Base::constructor(owner, colliderSpec);
67
68 this->classIndex = kColliderBoxIndex;
69
70 this->rotationVertexDisplacement = Vector3D::zero();
71
72 this->normals = NULL;
73
74 int32 normalIndex = 0;
75 for(; normalIndex < __COLLIDER_NORMALS; normalIndex++)
76 {
77 this->vertexProjections[normalIndex].min = 0;
78 this->vertexProjections[normalIndex].max = 0;
79 }
80
81 Box::computeRightBox(this);
82 Box::projectOntoItself(this);
83}

214void Box::getVertexes(Vector3D vertexes[__BOX_VERTEXES])
215{
216 Vector3D leftTopNear = {this->rightBox.x0, this->rightBox.y0, this->rightBox.z0};
217 Vector3D rightTopNear = {this->rightBox.x1, this->rightBox.y0, this->rightBox.z0};
218 Vector3D leftBottomNear = {this->rightBox.x0, this->rightBox.y1, this->rightBox.z0};
219 Vector3D rightBottomNear = {this->rightBox.x1, this->rightBox.y1, this->rightBox.z0};
220 Vector3D leftTopFar = {this->rightBox.x0, this->rightBox.y0, this->rightBox.z1};
221 Vector3D rightTopFar = {this->rightBox.x1, this->rightBox.y0, this->rightBox.z1};
222 Vector3D leftBottomFar = {this->rightBox.x0, this->rightBox.y1, this->rightBox.z1};
223 Vector3D rightBottomFar = {this->rightBox.x1, this->rightBox.y1, this->rightBox.z1};
224
225 if(0 != this->rotationVertexDisplacement.x || 0 != this->rotationVertexDisplacement.y || 0 != this->rotationVertexDisplacement.z)
226 {
227 if(0 == this->rotationVertexDisplacement.z)
228 {
229 leftTopNear.y += this->rotationVertexDisplacement.y;
230 rightTopNear.x -= this->rotationVertexDisplacement.x;
231 leftBottomNear.x += this->rotationVertexDisplacement.x;
232 rightBottomNear.y -= this->rotationVertexDisplacement.y;
233
234 leftTopFar.y += this->rotationVertexDisplacement.y;
235 rightTopFar.x -= this->rotationVertexDisplacement.x;
236 leftBottomFar.x += this->rotationVertexDisplacement.x;
237 rightBottomFar.y -= this->rotationVertexDisplacement.y;
238 }
239 else if(0 == this->rotationVertexDisplacement.y)
240 {
241 leftTopNear.x += this->rotationVertexDisplacement.x;
242 rightTopNear.z += this->rotationVertexDisplacement.z;
243 leftBottomNear.x += this->rotationVertexDisplacement.x;
244 rightBottomNear.z += this->rotationVertexDisplacement.z;
245
246 leftTopFar.z -= this->rotationVertexDisplacement.z;
247 rightTopFar.x -= this->rotationVertexDisplacement.x;
248 leftBottomFar.z -= this->rotationVertexDisplacement.z;
249 rightBottomFar.x -= this->rotationVertexDisplacement.x;
250 }
251 else if(0 == this->rotationVertexDisplacement.x)
252 {
253 leftTopNear.z += this->rotationVertexDisplacement.z;
254 rightTopNear.z += this->rotationVertexDisplacement.z;
255 leftBottomNear.y -= this->rotationVertexDisplacement.y;
256 rightBottomNear.y -= this->rotationVertexDisplacement.y;
257
258 leftTopFar.y += this->rotationVertexDisplacement.y;
259 rightTopFar.y += this->rotationVertexDisplacement.y;
260 leftBottomFar.z -= this->rotationVertexDisplacement.z;
261 rightBottomFar.z -= this->rotationVertexDisplacement.z;
262 }
263 }
264
265 vertexes[0] = Vector3D::sum(leftTopNear, this->transformation->position);
266 vertexes[1] = Vector3D::sum(rightTopNear, this->transformation->position);
267 vertexes[2] = Vector3D::sum(leftBottomNear, this->transformation->position);
268 vertexes[3] = Vector3D::sum(rightBottomNear, this->transformation->position);
269
270 vertexes[4] = Vector3D::sum(leftTopFar, this->transformation->position);
271 vertexes[5] = Vector3D::sum(rightTopFar, this->transformation->position);
272 vertexes[6] = Vector3D::sum(leftBottomFar, this->transformation->position);
273 vertexes[7] = Vector3D::sum(rightBottomFar, this->transformation->position);
274}

278void Box::projectOntoItself()
279{
280 Vector3D vertexes[__BOX_VERTEXES];
281 Box::getVertexes(this, vertexes);
282
283 // Compute normals
284 Box::computeNormals(this, vertexes);
285
286 // Has to project all points on all the normals of the tilted box
287 int32 normalIndex = 0;
288
289 // Initialize vertex projections
290 for(; normalIndex < __COLLIDER_NORMALS; normalIndex++)
291 {
292 Box::project
293 (
294 vertexes, this->normals->vectors[normalIndex], &this->vertexProjections[normalIndex].min,
295 &this->vertexProjections[normalIndex].max
296 );
297 }
298}

308void Box::computeRightBox()
309{
310 this->rotationVertexDisplacement.x = 0;
311 this->rotationVertexDisplacement.y = 0;
312 this->rotationVertexDisplacement.z = 0;
313
314 Size surroundingBoxSize = Size::getFromPixelSize(((ColliderSpec*)this->componentSpec)->pixelSize);
315
316 // Angle | theta | psi
317 if(0 != this->transformation->rotation.z || 0 != this->transformation->rotation.y || 0 != this->transformation->rotation.x)
318 {
319 fixed_t width = surroundingBoxSize.x >> 1;
320 fixed_t height = surroundingBoxSize.y >> 1;
321 fixed_t depth = surroundingBoxSize.z >> 1;
322
323 // Allow only one this->transformation->rotation
324 if(0 != this->transformation->rotation.z && 256 != this->transformation->rotation.z)
325 {
326 // Clamp value around 256 degrees (180) to avoid conditionals later when calculating rotationVertexDisplacement
327 int16 angle = this->transformation->rotation.z - ((this->transformation->rotation.z / 256) << 8);
328 angle = angle < 0 ? 256 + angle : angle;
329
330 // Calculate position of box's right-bottom corner
331 fixed_t sinAngle = __FIX7_9_TO_FIXED(__SIN(angle));
332 fixed_t cosAngle = __FIX7_9_TO_FIXED(__COS(angle));
333
334 // Use vectors (x1, y0, z1) and (x1, y1, z1)
335 Vector3D topRight =
336 {
337 __FIXED_MULT(width, cosAngle) - __FIXED_MULT(-height, sinAngle),
338 __FIXED_MULT(width, sinAngle) + __FIXED_MULT(-height, cosAngle),
339 depth
340 };
341
342 Vector3D bottomRight =
343 {
344 __FIXED_MULT(width, cosAngle) - __FIXED_MULT(height, sinAngle),
345 __FIXED_MULT(width, sinAngle) + __FIXED_MULT(height, cosAngle),
346 depth
347 };
348
349 Vector3D topRightHelper =
350 {
351 __ABS(topRight.x),
352 __ABS(topRight.y),
353 __ABS(topRight.z),
354 };
355
356 Vector3D bottomRightHelper =
357 {
358 __ABS(bottomRight.x),
359 __ABS(bottomRight.y),
360 __ABS(bottomRight.z),
361 };
362
363 surroundingBoxSize.x = (bottomRightHelper.x > topRightHelper.x ? bottomRightHelper.x : topRightHelper.x) << 1;
364 surroundingBoxSize.y = (bottomRightHelper.y > topRightHelper.y ? bottomRightHelper.y : topRightHelper.y) << 1;
365
366 // Find the displacement over each axis for the rotated box
367 this->rotationVertexDisplacement.x = bottomRightHelper.x < topRightHelper.x ? bottomRight.x : topRight.x;
368 this->rotationVertexDisplacement.y = bottomRightHelper.y < topRightHelper.y ? bottomRight.y : topRight.y;
369 this->rotationVertexDisplacement.y = angle >= 128 ? -this->rotationVertexDisplacement.y : this->rotationVertexDisplacement.y;
370
371 this->rotationVertexDisplacement.x = (surroundingBoxSize.x >> 1) + this->rotationVertexDisplacement.x;
372 this->rotationVertexDisplacement.y = (surroundingBoxSize.y >> 1) - this->rotationVertexDisplacement.y;
373
374 if(!(__MODULO(angle, 128)))
375 {
376 this->rotationVertexDisplacement.x = 0;
377 this->rotationVertexDisplacement.y = 0;
378 }
379 }
380 else if(0 != this->transformation->rotation.y && 256 != this->transformation->rotation.y)
381 {
382 // Clamp value around 256 degrees (180) to avoid conditionals later when calculating rotationVertexDisplacement
383 int16 angle = this->transformation->rotation.y - ((this->transformation->rotation.y / 256) << 8);
384 angle = angle < 0 ? 256 + angle : angle;
385
386 // Calculate position of box's right-bottom corner
387 fixed_t sinAngle = __FIX7_9_TO_FIXED(__SIN(angle));
388 fixed_t cosAngle = __FIX7_9_TO_FIXED(__COS(0));
389
390 // Use vectors (x0, y1, z0) and (x1, y1, z0)
391 Vector3D bottomLeft =
392 {
393 __FIXED_MULT(-width, cosAngle) + __FIXED_MULT(-depth, sinAngle),
394 height,
395 -__FIXED_MULT(-width, sinAngle) + __FIXED_MULT(-depth, cosAngle),
396 };
397
398 Vector3D bottomRight =
399 {
400 __FIXED_MULT(width, cosAngle) + __FIXED_MULT(-depth, sinAngle),
401 height,
402 -__FIXED_MULT(width, sinAngle) + __FIXED_MULT(-depth, cosAngle),
403 };
404
405 Vector3D bottomLeftHelper =
406 {
407 __ABS(bottomLeft.x),
408 __ABS(bottomLeft.y),
409 __ABS(bottomLeft.z),
410 };
411
412 Vector3D bottomRightHelper =
413 {
414 __ABS(bottomRight.x),
415 __ABS(bottomRight.y),
416 __ABS(bottomRight.z),
417 };
418
419 surroundingBoxSize.x = (bottomLeftHelper.x > bottomRightHelper.x ? bottomLeftHelper.x : bottomRightHelper.x) << 1;
420 surroundingBoxSize.z = (bottomLeftHelper.z > bottomRightHelper.z ? bottomLeftHelper.z : bottomRightHelper.z) << 1;
421
422 // Find the displacement over each axis for the rotated box
423 this->rotationVertexDisplacement.x = bottomLeftHelper.x < bottomRightHelper.x ? bottomLeft.x : bottomRight.x;
424 this->rotationVertexDisplacement.x = angle >= 128 ? -this->rotationVertexDisplacement.x : this->rotationVertexDisplacement.x;
425 this->rotationVertexDisplacement.z = bottomLeftHelper.z < bottomRightHelper.z ? bottomLeft.z : bottomRight.z;
426
427 this->rotationVertexDisplacement.x = (surroundingBoxSize.x >> 1) - this->rotationVertexDisplacement.x;
428 this->rotationVertexDisplacement.z = (surroundingBoxSize.z >> 1) + this->rotationVertexDisplacement.z;
429
430 if(!(__MODULO(angle, 128)))
431 {
432 this->rotationVertexDisplacement.x = 0;
433 this->rotationVertexDisplacement.z = 0;
434 }
435 }
436 else if(0 != this->transformation->rotation.x && 256 != this->transformation->rotation.x)
437 {
438 // Clamp value around 256 degrees (180) to avoid conditionals later when calculating rotationVertexDisplacement
439 int16 angle = this->transformation->rotation.x - ((this->transformation->rotation.x / 256) << 8);
440 angle = angle < 0 ? 256 + angle : angle;
441
442 // Calculate position of box's right-bottom corner
443 fixed_t sinAngle = __FIX7_9_TO_FIXED(__SIN(angle));
444 fixed_t cosAngle = __FIX7_9_TO_FIXED(__COS(angle));
445
446 // Use vectors (x1, y1, z0) and (x1, y1, z1)
447 Vector3D bottomNear =
448 {
449 width,
450 __FIXED_MULT(height, cosAngle) - __FIXED_MULT(-depth, sinAngle),
451 __FIXED_MULT(height, sinAngle) + __FIXED_MULT(-depth, cosAngle),
452 };
453
454 Vector3D bottomFar =
455 {
456 width,
457 __FIXED_MULT(height, cosAngle) - __FIXED_MULT(depth, sinAngle),
458 __FIXED_MULT(height, sinAngle) + __FIXED_MULT(depth, cosAngle),
459 };
460
461 Vector3D bottomNearHelper =
462 {
463 __ABS(bottomNear.x),
464 __ABS(bottomNear.y),
465 __ABS(bottomNear.z),
466 };
467
468 Vector3D bottomFarHelper =
469 {
470 __ABS(bottomFar.x),
471 __ABS(bottomFar.y),
472 __ABS(bottomFar.z),
473 };
474
475 surroundingBoxSize.y = (bottomFarHelper.y > bottomNearHelper.y ? bottomFarHelper.y : bottomNearHelper.y) << 1;
476 surroundingBoxSize.z = (bottomFarHelper.z > bottomNearHelper.z ? bottomFarHelper.z : bottomNearHelper.z) << 1;
477
478 // Find the displacement over each axis for the rotated box
479 this->rotationVertexDisplacement.y = bottomFarHelper.y < bottomNearHelper.y ? bottomFar.y : bottomNear.y;
480 this->rotationVertexDisplacement.z = bottomFarHelper.z < bottomNearHelper.z ? bottomFar.z : bottomNear.z;
481 this->rotationVertexDisplacement.z = angle >= 128 ? -this->rotationVertexDisplacement.z : this->rotationVertexDisplacement.z;
482
483 this->rotationVertexDisplacement.y = (surroundingBoxSize.y >> 1) - this->rotationVertexDisplacement.y;
484 this->rotationVertexDisplacement.z = (surroundingBoxSize.z >> 1) + this->rotationVertexDisplacement.z;
485
486 if(!(__MODULO(angle, 128)))
487 {
488 this->rotationVertexDisplacement.y = 0;
489 this->rotationVertexDisplacement.z = 0;
490 }
491 }
492 }
493
494 Vector3D displacement = Vector3D::getFromPixelVector(((ColliderSpec*)this->componentSpec)->displacement);
495
496 // Box's center if placed on P(0, 0, 0)
497 this->rightBox.x1 = surroundingBoxSize.x >> 1;
498 this->rightBox.y1 = surroundingBoxSize.y >> 1;
499 this->rightBox.z1 = surroundingBoxSize.z >> 1;
500
501 this->rightBox.x0 = -this->rightBox.x1;
502 this->rightBox.y0 = -this->rightBox.y1;
503 this->rightBox.z0 = -this->rightBox.z1;
504
505 this->rightBox.x0 += displacement.x;
506 this->rightBox.y0 += displacement.y;
507 this->rightBox.z0 += displacement.z;
508
509 this->rightBox.x1 += displacement.x;
510 this->rightBox.y1 += displacement.y;
511 this->rightBox.z1 += displacement.z;
512
513 if(NULL != this->normals)
514 {
515 Box::projectOntoItself(this);
516 }
517}

521void Box::computeNormals(Vector3D vertexes[__BOX_VERTEXES])
522{
523/*
524 // Generic way
525 normals[0] = Vector3D::getPlaneNormal(vertexes[6], vertexes[4], vertexes[0]);
526 normals[1] = Vector3D::getPlaneNormal(vertexes[0], vertexes[4], vertexes[5]);
527 normals[2] = Vector3D::getPlaneNormal(vertexes[0], vertexes[1], vertexes[3]);
528*/
529
530 if(NULL == this->normals)
531 {
532 this->normals = new Normals;
533 }
534
535 // Fast way given that the cubes are regular
536 this->normals->vectors[0] = (Vector3D)
537 {
538 vertexes[1].x - vertexes[0].x,
539 vertexes[1].y - vertexes[0].y,
540 vertexes[1].z - vertexes[0].z,
541 };
542
543 this->normals->vectors[1] = (Vector3D)
544 {
545 vertexes[2].x - vertexes[0].x,
546 vertexes[2].y - vertexes[0].y,
547 vertexes[2].z - vertexes[0].z,
548 };
549
550 this->normals->vectors[2] = (Vector3D)
551 {
552 vertexes[4].x - vertexes[0].x,
553 vertexes[4].y - vertexes[0].y,
554 vertexes[4].z - vertexes[0].z,
555 };
556
557 this->normals->vectors[0] = Vector3D::normalize(this->normals->vectors[0]);
558 this->normals->vectors[1] = Vector3D::normalize(this->normals->vectors[1]);
559 this->normals->vectors[2] = Vector3D::normalize(this->normals->vectors[2]);
560}
