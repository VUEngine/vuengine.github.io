67void VisualComponent::handleCommand(int32 command, va_list args)
68{
69 switch(command)
70 {
71 case cVisualComponentCommandShow:
72
73 VisualComponent::show(this);
74 break;
75
76 case cVisualComponentCommandHide:
77
78 VisualComponent::hide(this);
79 break;
80
81 case cVisualComponentCommandSetTransparency:
82
83 VisualComponent::setTransparency(this, (uint8)va_arg(args, uint32));
84 break;
85
86 case cVisualComponentCommandPlay:
87
88 VisualComponent::play
89 (
90 this, (const AnimationFunction**)va_arg(args, AnimationFunction**), (const char*)va_arg(args, char*),
91 va_arg(args, ListenerObject), (EventListener)va_arg(args, EventListener)
92 );
93 break;
94
95 case cVisualComponentCommandPause:
96
97 VisualComponent::pause(this, (bool)va_arg(args, uint32));
98 break;
99
100 case cVisualComponentCommandStop:
101
102 VisualComponent::stop(this);
103 break;
104
105 case cVisualComponentCommandSetFrame:
106
107 VisualComponent::setActualFrame(this, (int16)va_arg(args, uint32));
108 break;
109
110 case cVisualComponentCommandNextFrame:
111
112 VisualComponent::nextFrame(this);
113 break;
114
115 case cVisualComponentCommandPreviousFrame:
116
117 VisualComponent::previousFrame(this);
118 break;
119
120 default:
121
122 Base::handleCommand(this, command, args);
123 break;
124 }
125}

146bool VisualComponent::play
147(
148 const AnimationFunction* animationFunctions[], const char* animationName, ListenerObject scope, EventListener callback
149)
150{
151 if(NULL == animationFunctions || NULL == animationName)
152 {
153 return false;
154 }
155
156 ASSERT(NULL != animationFunctions, "VisualComponent::play: null animationFunctions");
157 ASSERT(NULL != animationName, "VisualComponent::play: null animationName");
158
159 bool playBackStarted = false;
160
161 if(!isDeleted(this->animationController))
162 {
163 playBackStarted = AnimationController::play(this->animationController, animationFunctions, animationName, scope, callback);
164 this->rendered = this->rendered && !this->updateAnimationFrame;
165 }
166
167 return playBackStarted;
168}

180 }
181
182 return false;
183}
