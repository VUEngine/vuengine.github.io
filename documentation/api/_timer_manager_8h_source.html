31// Use with 20us timer (range = 0 to 1300)
32#define __TIME_US(n) (((n) / TimerManager::getResolutionInUS()) \
33 - __TIMER_COUNTER_DELTA)
34#define __TIME_INVERSE_US(n) ((n + __TIMER_COUNTER_DELTA) * \
35 TimerManager::getResolutionInUS())
36
37// Use with 100us timer (range = 0 to 6500, and 0 to 6.5)
38#define __TIME_MS(n) ((((n) * __MICROSECONDS_PER_MILLISECOND) / \
39 TimerManager::getResolutionInUS()) - __TIMER_COUNTER_DELTA)
40#define __TIME_INVERSE_MS(n) ((n + __TIMER_COUNTER_DELTA) * \
41 TimerManager::getResolutionInUS() / 1000)
