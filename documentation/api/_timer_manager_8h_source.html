31// Use with 20us timer (range = 0 to 1300)
32#define __TIME_US(n) (((n) / TimerManager::getResolutionInUS()) \
33 - __TIMER_COUNTER_DELTA)
34#define __TIME_INVERSE_US(n) ((n + __TIMER_COUNTER_DELTA) * \
35 TimerManager::getResolutionInUS())
