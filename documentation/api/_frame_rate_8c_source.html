89static void FrameRate::gameFrameStarted(bool gameCycleEnded, bool printFPS)
90{
91 FrameRate frameRate = FrameRate::getInstance();
92
93 if(!gameCycleEnded)
94 {
95 frameRate->unevenFPS++;
96 }
97
98 frameRate->gameFrameStarts++;
99
100 if(frameRate->targetFPS <= frameRate->gameFrameStarts)
101 {
102 frameRate->seconds++;
103 frameRate->totalFPS += frameRate->FPS;
104 frameRate->totalUnevenFPS += frameRate->unevenFPS;
105
106 if(!isDeleted(frameRate->events))
107 {
108 FrameRate::fireEvent(frameRate, kEventFramerateReady);
109 }
110
111 if(frameRate->targetFPS > frameRate->FPS)
112 {
113 if(!isDeleted(frameRate->events))
114 {
115 FrameRate::fireEvent(frameRate, kEventFramerateDipped);
116 }
117 }
118
119 if(printFPS)
120 {
121#ifdef __TOOLS
122 if(!VUEngine::isInToolState())
123#endif
124 {
125 FrameRate::print(__PRINT_FRAMERATE_AT_X, __PRINT_FRAMERATE_AT_Y);
126 }
127 }
128
129 frameRate->FPS = 0;
130 frameRate->unevenFPS = 0;
131 frameRate->gameFrameStarts = 0;
132 }
133}

137static void FrameRate::print(int32 x, int32 y)
138{
139 FrameRate frameRate = FrameRate::getInstance();
140
141 Printing::text("FPS |TORN |AVR ", x, y, NULL);
142 Printing::int32(frameRate->FPS, x + 4, y, NULL);
143 Printing::int32(frameRate->unevenFPS, x + 14, y, NULL);
144 Printing::int32(((float)frameRate->totalFPS / frameRate->seconds) + 0.5f, x + 20, y, NULL);
145}
