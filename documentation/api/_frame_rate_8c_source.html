64void FrameRate::gameFrameStarted(bool gameCycleEnded, bool printFPS)
65{
66 if(!gameCycleEnded)
67 {
68 this->unevenFPS++;
69 }
70
71 this->gameFrameStarts++;
72
73 if(this->targetFPS <= this->gameFrameStarts)
74 {
75 this->seconds++;
76 this->totalFPS += this->FPS;
77 this->totalUnevenFPS += this->unevenFPS;
78
79 if(!isDeleted(this->events))
80 {
81 FrameRate::fireEvent(this, kEventFramerateReady);
82 }
83
84 if(this->targetFPS > this->FPS)
85 {
86 if(!isDeleted(this->events))
87 {
88 FrameRate::fireEvent(this, kEventFramerateDipped);
89 }
90 }
91
92 if(printFPS)
93 {
94 FrameRate::print(this, __PRINT_FRAMERATE_AT_X, __PRINT_FRAMERATE_AT_Y);
95 }
96
97 this->FPS = 0;
98 this->unevenFPS = 0;
99 this->gameFrameStarts = 0;
100 }
101}
