140static int32 Math::aSin(fix7_9 sin)
141{
142 int32 angle = 0;
143 int32 entry = 0;
144 int32 lastEntry = __ENTRIES_PER_QUADRANT;
145 int32 difference = 1024;
146
147 for(; entry < lastEntry; entry++)
148 {
149 if(__ABS(sin - _sinLut[entry]) <= difference)
150 {
151 difference = __ABS(sin - _sinLut[entry]);
152 angle = entry;
153 }
154 else if(__ABS(sin - _sinLut[entry]) > difference)
155 {
156 break;
157 }
158 }
159
160 return angle;
161}

165static int32 Math::getAngle(fix7_9 cos, fix7_9 sin)
166{
167 int16 firstEntry = 0;
168 int32 lastEntry = 0;
169 bool ascendent = false;
170
171 // Determine the quadrant
172 if(0 == cos)
173 {
174 // First quadrant
175 if(0 < sin)
176 {
177 return __ENTRIES_PER_QUADRANT * 1;
178 }
179 // Fourth quadrant
180 else if(0 > sin)
181 {
182 return __ENTRIES_PER_QUADRANT * 3;
183 }
184
185 return 0;
186 }
187 else if(0 == sin)
188 {
189 // First quadrant
190 if(0 < cos)
191 {
192 return __ENTRIES_PER_QUADRANT * 0;
193 }
194 // Fourth quadrant
195 else if(0 > cos)
196 {
197 return __ENTRIES_PER_QUADRANT * 2;
198 }
199
200 return 0;
201 }
202 else if(0 < cos)
203 {
204 // First quadrant
205 if(0 < sin)
206 {
207 firstEntry = 0;
208 lastEntry = __ENTRIES_PER_QUADRANT;
209 }
210 // Fourth quadrant
211 else
212 {
213 firstEntry = __ENTRIES_PER_QUADRANT * 3;
214 lastEntry = __TOTAL_ENTRIES;
215 }
216
217 ascendent = true;
218 }
219 // Second quadrant
220 else if(0 < sin)
221 {
222 firstEntry = __ENTRIES_PER_QUADRANT;
223 lastEntry = firstEntry + __ENTRIES_PER_QUADRANT;
224 }
225 // Third quadrant
226 else
227 {
228 firstEntry = __ENTRIES_PER_QUADRANT * 2;
229 lastEntry = __TOTAL_ENTRIES - __ENTRIES_PER_QUADRANT;
230 }
231
232 int16 pivotEntry = firstEntry + (__ENTRIES_PER_QUADRANT >> 1);
233 int16 angle = pivotEntry;
234 int16 distance = 1024;
235
236 while(true)
237 {
238 int16 difference = ascendent ? sin - _sinLut[pivotEntry] : _sinLut[pivotEntry] - sin;
239
240 if(distance > __ABS(difference))
241 {
242 distance = __ABS(difference);
243 }
244 else
245 {
246 break;
247 }
248
249 angle = pivotEntry;
250
251 if(0 < difference)
252 {
253 firstEntry = pivotEntry;
254 pivotEntry = lastEntry - ((lastEntry - pivotEntry) >> 1);
255 }
256 else if(0 > difference)
257 {
258 lastEntry = pivotEntry;
259 pivotEntry = firstEntry + ((pivotEntry - firstEntry) >> 1);
260 }
261 }
262
263 return angle;
264}

275static int32 Math::powerFast(int32 base, int32 power)
276{
277 int32 i=0;
278 int32 j=0;
279 int32 sum = base;
280 int32 result = 0;
281 int32 limit = base;
282
283 power = 0 > power ? -power : power;
284
285 for(i = 1; i < power; i++)
286 {
287 for(j = 0; j < limit; j++)
288 {
289 result += sum;
290 }
291 limit = base -1;
292 sum = result;
293 }
294
295 return 0 == power ? 1 : 1 == power ? base : result;
296}
