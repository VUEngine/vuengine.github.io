36static void SRAMManager::save(const BYTE* const source, int32 memberOffset, int32 dataSize)
37{
38 SRAMManager sramManager = SRAMManager::getInstance();
39
40 int32 i = 0;
41
42 uint16* destination = sramManager->spaceAddress + memberOffset;
43 ASSERT(0 == ((int32)destination % 2), "SRAMManager::save: odd destination");
44
45 for(; i < dataSize; i++)
46 {
47 destination[i] = source[i];
48 }
49}
