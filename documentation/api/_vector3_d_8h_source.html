478static inline Vector3D Vector3D::projectOnto(Vector3D p, Vector3D a, Vector3D b)
479{
480 Vector3D ap = Vector3D::get(a, p);
481 Vector3D ab = Vector3D::get(a, b);
482 fixed_ext_t dotApAb = Vector3D::dotProduct(ap, ab);
483 fixed_ext_t dotAbAb = Vector3D::dotProduct(ab, ab);
484
485 if(!dotAbAb)
486 {
487 if(a.x == b.x)
488 {
489 p.x = a.x;
490 }
491
492 if(a.y == b.y)
493 {
494 p.y = a.y;
495 }
496
497 if(a.z == b.z)
498 {
499 p.z = a.z;
500 }
501
502 return p;
503 }
504
505 Vector3D projection =
506 {
507 a.x + __FIXED_EXT_MULT(__FIXED_TO_FIXED_EXT(ab.x), __FIXED_EXT_DIV(dotApAb, dotAbAb)),
508 a.y + __FIXED_EXT_MULT(__FIXED_TO_FIXED_EXT(ab.y), __FIXED_EXT_DIV(dotApAb, dotAbAb)),
509 a.z + __FIXED_EXT_MULT(__FIXED_TO_FIXED_EXT(ab.z), __FIXED_EXT_DIV(dotApAb, dotAbAb)),
510 };
511
512 return projection;
513}

517static inline Vector3D Vector3D::projectOntoHighPrecision(Vector3D p, Vector3D a, Vector3D b)
518{
519 Vector3D ap = Vector3D::get(a, p);
520 Vector3D ab = Vector3D::get(a, b);
521 fix19_13 dotApAb = Vector3D::dotProductHighPrecision(ap, ab);
522 fix19_13 dotAbAb = Vector3D::dotProductHighPrecision(ab, ab);
523
524 if(!dotAbAb)
525 {
526 if(a.x == b.x)
527 {
528 p.x = a.x;
529 }
530
531 if(a.y == b.y)
532 {
533 p.y = a.y;
534 }
535
536 if(a.z == b.z)
537 {
538 p.z = a.z;
539 }
540
541 return p;
542 }
543
544 Vector3D projection =
545 {
546 a.x + __FIX19_13_TO_FIXED
547 (
548 __FIX19_13_MULT(__FIXED_TO_FIX19_13(ab.x), __FIX19_13_DIV(dotApAb, dotAbAb))
549 ),
550 a.y + __FIX19_13_TO_FIXED
551 (
552 __FIX19_13_MULT(__FIXED_TO_FIX19_13(ab.y), __FIX19_13_DIV(dotApAb, dotAbAb))
553 ),
554 a.z + __FIX19_13_TO_FIXED
555 (
556 __FIX19_13_MULT(__FIXED_TO_FIX19_13(ab.z), __FIX19_13_DIV(dotApAb, dotAbAb))
557 ),
558 };
559
560 return projection;
561}

566{
567 Vector3D result = vector;
568

570 {
571 result = Vector3D::rotateXAxis(result, __FIXED_TO_I(rotation.x + __05F_FIXED));
572 }
573
574 if(0 != rotation.y)
575 {
576 result = Vector3D::rotateYAxis(result, __FIXED_TO_I(rotation.y + __05F_FIXED));
577 }
578
579 if(0 != rotation.z)
580 {
581 result = Vector3D::rotateZAxis(result, __FIXED_TO_I(rotation.z + __05F_FIXED));
582 }
583
584 return result;

589static inline Vector3D Vector3D::rotateXAxis(Vector3D vector, int16 degrees)
590{
591 fix7_9_ext cos = __FIX7_9_TO_FIX7_9_EXT(__COS(degrees));
592 fix7_9_ext sin = __FIX7_9_TO_FIX7_9_EXT(__SIN(degrees));
593
594 fix7_9_ext y = __FIXED_TO_FIX7_9_EXT(vector.y);
595 fix7_9_ext z = __FIXED_TO_FIX7_9_EXT(vector.z);
596
597 return (Vector3D)
598 {
599 vector.x,
600 __FIX7_9_EXT_TO_FIXED(__FIX7_9_EXT_MULT(y , cos) - __FIX7_9_EXT_MULT(z , sin)),
601 __FIX7_9_EXT_TO_FIXED(__FIX7_9_EXT_MULT(y , sin) + __FIX7_9_EXT_MULT(z , cos))
602 };
603}

734
735static inline fixed_t Vector3D::getScale(fixed_t z, bool applyScalingMultiplier)
736{
737 if(0 == _optical->halfWidth)
738 {
739 return __1I_FIXED;
740 }
741
742 fixed_ext_t projectedWidth = 0;
743
744 if(applyScalingMultiplier)
745 {
746 if(0 == z + _optical->halfWidth)
747 {
748 return __1I_FIXED;
749 }
750
751 projectedWidth =
752 __FIXED_EXT_DIV
753 (
754 __FIXED_EXT_MULT(_optical->halfWidth, _optical->scalingMultiplier),
755 z + _optical->halfWidth
756 );
757 }
758 else
759 {
760 if(0 == z + _optical->cameraNearPlane)
761 {
762 return __1I_FIXED;
763 }
764
765 projectedWidth =
766 __FIXED_EXT_DIV
767 (
768 __FIXED_EXT_MULT(_optical->halfWidth, _optical->projectionMultiplierHelper),
769 z + _optical->cameraNearPlane
770 );
771 }
772
773 return __FIXED_EXT_DIV(projectedWidth, _optical->halfWidth);
774}
