36static RightBox Mesh::getRightBoxFromSpec(MeshSpec* meshSpec)
37{
38 RightBox rightBox = {0, 0, 0, 0, 0, 0};
39
40 bool isEndSegment = false;
41 uint16 i = 0;
42
43 do
44 {
45 Vector3D startVector = Vector3D::getFromPixelVector(meshSpec->segments[i][0]);
46 Vector3D endVector = Vector3D::getFromPixelVector(meshSpec->segments[i][1]);
47
48 isEndSegment = Vector3D::areEqual(Vector3D::zero(), startVector) && Vector3D::areEqual(Vector3D::zero(), endVector);
49
50 if(!isEndSegment)
51 {
52 if(startVector.x < endVector.x)
53 {
54 if(startVector.x < rightBox.x0)
55 {
56 rightBox.x0 = startVector.x;
57 }
58 }
59 else
60 {
61 if(endVector.x < rightBox.x0)
62 {
63 rightBox.x0 = endVector.x;
64 }
65 }
66
67 if(startVector.x > endVector.x)
68 {
69 if(startVector.x > rightBox.x1)
70 {
71 rightBox.x1 = startVector.x;
72 }
73 }
74 else
75 {
76 if(endVector.x > rightBox.x1)
77 {
78 rightBox.x1 = endVector.x;
79 }
80 }
81
82 if(startVector.y < endVector.y)
83 {
84 if(startVector.y < rightBox.y0)
85 {
86 rightBox.y0 = startVector.y;
87 }
88 }
89 else
90 {
91 if(endVector.y < rightBox.y0)
92 {
93 rightBox.y0 = endVector.y;
94 }
95 }
96
97 if(startVector.y > endVector.y)
98 {
99 if(startVector.y > rightBox.y1)
100 {
101 rightBox.y1 = startVector.y;
102 }
103 }
104 else
105 {
106 if(endVector.y > rightBox.y1)
107 {
108 rightBox.y1 = endVector.y;
109 }
110 }
111
112 if(startVector.z < endVector.z)
113 {
114 if(startVector.z < rightBox.z0)
115 {
116 rightBox.z0 = startVector.z;
117 }
118 }
119 else
120 {
121 if(endVector.z < rightBox.z0)
122 {
123 rightBox.z0 = endVector.z;
124 }
125 }
126
127 if(startVector.z > endVector.z)
128 {
129 if(startVector.z > rightBox.z1)
130 {
131 rightBox.z1 = startVector.z;
132 }
133 }
134 else
135 {
136 if(endVector.z > rightBox.z1)
137 {
138 rightBox.z1 = endVector.z;
139 }
140 }
141 }
142
143 i++;
144 }
145 while(!isEndSegment);
146
147 return rightBox;
148}

186RightBox Mesh::getRightBox()
187{
188 RightBox rightBox = {-1, -1, -1, 1, 1, 1};
189
190 for(VirtualNode node = this->vertices->head; NULL != node; node = node->next)
191 {
192 Vertex* vertex = (Vertex*)node->data;
193
194 Vector3D vector3D = Vector3D::sum(vertex->vector, this->displacement);
195
196 if(vector3D.x < rightBox.x0)
197 {
198 rightBox.x0 = vector3D.x;
199 }
200
201 if(vector3D.x > rightBox.x1)
202 {
203 rightBox.x1 = vector3D.x;
204 }
205
206 if(vector3D.y < rightBox.y0)
207 {
208 rightBox.y0 = vector3D.y;
209 }
210
211 if(vector3D.y > rightBox.y1)
212 {
213 rightBox.y1 = vector3D.y;
214 }
215
216 if(vector3D.z < rightBox.z0)
217 {
218 rightBox.z0 = vector3D.z;
219 }
220
221 if(vector3D.z > rightBox.z1)
222 {
223 rightBox.z1 = vector3D.z;
224 }
225 }
226
227 return rightBox;
228}

239void Mesh::render(Vector3D relativePosition)
240{
241 NM_ASSERT(NULL != this->transformation, "Mesh::render: NULL transformation");
242
243 bool scale =
244 (__1I_FIX7_9 != this->transformation->scale.x) + (__1I_FIX7_9 != this->transformation->scale.y) +
245 (__1I_FIX7_9 != this->transformation->scale.z);
246
247 bool rotate =
248 (0 != this->transformation->rotation.x) + (0 != this->transformation->rotation.y) + (0 != this->transformation->rotation.z);
249
250 if(!scale && !rotate)
251 {
252 CACHE_RESET;
253
254 for(VirtualNode node = this->vertices->head; NULL != node; node = node->next)
255 {
256 Vertex* vertex = (Vertex*)node->data;
257
258 Vector3D vector = Vector3D::rotate(Vector3D::sum(relativePosition, vertex->vector), _previousCameraInvertedRotation);
259
260 vertex->pixelVector = PixelVector::projectVector3D(vector, Optics::calculateParallax(vector.z));
261 }
262 }
263 else if(scale && rotate)
264 {
265 CACHE_RESET;
266
267 Rotation rotation = this->transformation->rotation;
268 Scale scale = this->transformation->scale;
269
270 for(VirtualNode node = this->vertices->head; NULL != node; node = node->next)
271 {
272 Vertex* vertex = (Vertex*)node->data;
273
274 Vector3D vector =
275 Vector3D::rotate
276 (
277 Vector3D::sum(relativePosition, Vector3D::rotate(Vector3D::scale(vertex->vector, scale), rotation)),
278 _previousCameraInvertedRotation
279 );
280
281 vertex->pixelVector = PixelVector::projectVector3D(vector, Optics::calculateParallax(vector.z));
282 }
283 }
284 else if(rotate)
285 {
286 CACHE_RESET;
287
288 Rotation rotation = this->transformation->rotation;
289
290 for(VirtualNode node = this->vertices->head; NULL != node; node = node->next)
291 {
292 Vertex* vertex = (Vertex*)node->data;
293
294 Vector3D vector =
295 Vector3D::rotate
296 (
297 Vector3D::sum(relativePosition, Vector3D::rotate(vertex->vector, rotation)), _previousCameraInvertedRotation
298 );
299
300 vertex->pixelVector = PixelVector::projectVector3D(vector, Optics::calculateParallax(vector.z));
301 }
302 }
303 else if(scale)
304 {
305 CACHE_RESET;
306
307 Scale scale = this->transformation->scale;
308
309 for(VirtualNode node = this->vertices->head; NULL != node; node = node->next)
310 {
311 Vertex* vertex = (Vertex*)node->data;
312
313 Vector3D vector =
314 Vector3D::rotate
315 (
316 Vector3D::sum(relativePosition, Vector3D::scale(vertex->vector, scale)), _previousCameraInvertedRotation
317 );
318
319 vertex->pixelVector = PixelVector::projectVector3D(vector, Optics::calculateParallax(vector.z));
320 }
321 }
322}

326bool Mesh::draw()
327{
328 NM_ASSERT(NULL != this->transformation, "Mesh::draw: NULL transformation");
329
330 bool drawn = false;
331
332 for(VirtualNode node = this->segments->head; NULL != node; node = node->next)
333 {
334 MeshSegment* meshSegment = (MeshSegment*)node->data;
335
336 // Draw the line in both buffers
337 drawn |=
338 DirectDraw::drawLine
339 (
340 meshSegment->fromVertex->pixelVector, meshSegment->toVertex->pixelVector, this->color,
341 this->bufferIndex, this->interlaced
342 );
343 }
344
345 this->bufferIndex = !this->bufferIndex;
346
347 return drawn;
348}

352void Mesh::addSegments(PixelVector (*segments)[2], Vector3D displacement)
353{
354 if(NULL == segments)
355 {
356 return;
357 }
358
359 bool isEndSegment = false;
360 uint16 i = 0;
361
362 // Prevent rendering when modifying the mesh because if segments are added after the initial render
363 // There can be graphical glitches if XPEND kicks in in the mist of adding new vertexes
364 this->rendered = false;
365
366 do
367 {
368 Vector3D startVector = Vector3D::getFromPixelVector(segments[i][0]);
369 Vector3D endVector = Vector3D::getFromPixelVector(segments[i][1]);
370
371 isEndSegment = Vector3D::areEqual(Vector3D::zero(), startVector) && Vector3D::areEqual(Vector3D::zero(), endVector);
372
373 if(!isEndSegment)
374 {
375 Mesh::addSegment(this, Vector3D::sum(startVector, displacement), Vector3D::sum(endVector, displacement));
376 }
377
378 i++;
379 }
380 while(!isEndSegment);
381}

385void Mesh::addSegment(Vector3D startVector, Vector3D endVector)
386{
387 MeshSegment* newMeshSegment = new MeshSegment;
388 newMeshSegment->fromVertex = NULL;
389 newMeshSegment->toVertex = NULL;
390
391 for(VirtualNode node = this->vertices->head; NULL != node; node = node->next)
392 {
393 Vertex* vertex = (Vertex*)node->data;
394
395 if(NULL == newMeshSegment->fromVertex && Vector3D::areEqual(vertex->vector, startVector))
396 {
397 newMeshSegment->fromVertex = vertex;
398 }
399 else if(NULL == newMeshSegment->toVertex && Vector3D::areEqual(vertex->vector, endVector))
400 {
401 newMeshSegment->toVertex = vertex;
402 }
403
404 if(NULL != newMeshSegment->fromVertex && NULL != newMeshSegment->toVertex)
405 {
406 break;
407 }
408 }
409
410 if(NULL == newMeshSegment->fromVertex)
411 {
412 newMeshSegment->fromVertex = new Vertex;
413 newMeshSegment->fromVertex->vector = startVector;
414 newMeshSegment->fromVertex->pixelVector = (PixelVector){0, 0, 0, 0};
415
416 VirtualList::pushBack(this->vertices, newMeshSegment->fromVertex);
417 }
418
419 if(NULL == newMeshSegment->toVertex)
420 {
421 newMeshSegment->toVertex = new Vertex;
422 newMeshSegment->toVertex->vector = endVector;
423 newMeshSegment->toVertex->pixelVector = (PixelVector){0, 0, 0, 0};
424
425 VirtualList::pushBack(this->vertices, newMeshSegment->toVertex);
426 }
427
428 VirtualList::pushBack(this->segments, newMeshSegment);
429}

433bool Mesh::drawInterlaced()
434{
435 bool drawn = false;
436
437 for(VirtualNode node = this->segments->head; NULL != node; node = node->next)
438 {
439 MeshSegment* meshSegment = (MeshSegment*)node->data;
440
441 // Draw the line in both buffers
442 drawn |=
443 DirectDraw::drawLine
444 (
445 meshSegment->fromVertex->pixelVector, meshSegment->toVertex->pixelVector, this->color, this->bufferIndex, true
446 )
447 ||
448 this->drawn;
449 }
450
451 this->bufferIndex = !this->bufferIndex;
452
453 return drawn;
454}
