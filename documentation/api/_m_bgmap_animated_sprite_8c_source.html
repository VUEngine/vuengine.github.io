54void MBgmapAnimatedSprite::updateAnimation()
55{
56 CharSet charSet = Texture::getCharSet(this->texture, true);
57
58 NM_ASSERT(!isDeleted(charSet), "MBgmapAnimatedSprite::updateAnimation: deleted charset");
59 NM_ASSERT(!isDeleted(this->animationController), "MBgmapAnimatedSprite::updateAnimation: null animation controller");
60
61 if(isDeleted(charSet) || isDeleted(this->animationController))
62 {
63 return;
64 }
65
66 if(Texture::isMultiframe(this->texture))
67 {
68 MBgmapAnimatedSprite::setMultiframe(this, AnimationController::getActualFrameIndex(this->animationController));
69 MBgmapAnimatedSprite::invalidateParamTable(this);
70 }
71 else
72 {
73 Texture::setFrame(this->texture, AnimationController::getActualFrameIndex(this->animationController));
74 }
75}

79void MBgmapAnimatedSprite::setMultiframe(uint16 frame)
80{
81 int16 mx = BgmapTexture::getXOffset(this->texture);
82 int16 my = BgmapTexture::getYOffset(this->texture);
83 int32 totalColumns = 64 - mx;
84 int32 frameColumn = Texture::getCols(this->texture) * frame;
85 this->bgmapTextureSource.mx = (mx + (frameColumn % totalColumns)) << 3;
86 this->bgmapTextureSource.my = (my + (frameColumn % totalColumns)) << 3;
87}
