106void ColliderManager::purgeDestroyedColliders()
107{
108 for(VirtualNode auxNode = this->components->head, auxNextNode = NULL; auxNode; auxNode = auxNextNode)
109 {
110 auxNextNode = auxNode->next;
111
112 Collider collider = Collider::safeCast(auxNode->data);
113
114 if(collider->deleteMe)
115 {
116 VirtualList::removeNode(this->components, auxNode);
117
118 delete collider;
119 }
120 }
121}

125uint32 ColliderManager::update()
126{
127 uint32 returnValue = false;
128
129#ifdef __SHOW_PHYSICS_PROFILING
130 _lastCycleCheckProducts = 0;
131 _lastCycleCollisionChecks = 0;
132 _lastCycleCollisions = 0;
133 _checkCycles++;
134#endif
135
136 this->dirty = false;
137
138 for(VirtualNode auxNode = this->components->head, auxNextNode = NULL; NULL != auxNode; auxNode = auxNextNode)
139 {
140 auxNextNode = auxNode->next;
141
142 Collider collider = Collider::safeCast(auxNode->data);
143
144#ifndef __RELEASE
145 if(isDeleted(collider) || collider->deleteMe)
146#else
147 if(collider->deleteMe)
148#endif
149 {
150 VirtualList::removeNode(this->components, auxNode);
151
152 delete collider;
153 continue;
154 }
155
156 collider->invalidPosition = true;
157 }
158
159 for(VirtualNode auxNode = this->components->head; NULL != auxNode; auxNode = auxNode->next)
160 {
161 Collider collider = Collider::safeCast(auxNode->data);
162
163 #ifdef __DRAW_SHAPES
164 if(collider->enabled)
165 {
166 Collider::show(collider);
167 }
168 else
169 {
170 Collider::hide(collider);
171 }
172 #endif
173
174 if(!(collider->enabled && collider->checkForCollisions) || __NON_TRANSFORMED == collider->transformation->invalid)
175 {
176 continue;
177 }
178
179 if(collider->invalidPosition)
180 {
181 Vector3D displacement =
182 Vector3D::rotate
183 (
184 Vector3D::getFromPixelVector(((ColliderSpec*)collider->componentSpec)->displacement), collider->transformation->rotation
185 );
186
187 collider->position = Vector3D::sum(collider->transformation->position, displacement);
188 collider->invalidPosition = false;
189 }
190
191 Vector3D colliderPosition = collider->transformation->position;
192
193 for(VirtualNode node = this->components->head; NULL != node; node = node->next)
194 {
195 Collider colliderToCheck = Collider::safeCast(node->data);
196
197 if(!colliderToCheck->enabled || __NON_TRANSFORMED == colliderToCheck->transformation->invalid)
198 {
199 continue;
200 }
201
202#ifdef __SHOW_PHYSICS_PROFILING
203 _lastCycleCheckProducts++;
204#endif
205
206 if(0 != (collider->layersToIgnore & colliderToCheck->layers))
207 {
208 continue;
209 }
210
211 if(collider->owner == colliderToCheck->owner)
212 {
213 continue;
214 }
215
216 fixed_ext_t distanceVectorSquareLength =
217 Vector3D::squareLength(Vector3D::get(colliderToCheck->transformation->position, colliderPosition));
218
219 if(__FIXED_SQUARE(__COLLIDER_MAXIMUM_SIZE) < distanceVectorSquareLength)
220 {
221 continue;
222 }
223
224#ifdef __SHOW_PHYSICS_PROFILING
225 _lastCycleCollisionChecks++;
226#endif
227
228 if(colliderToCheck->invalidPosition)
229 {
230 Vector3D displacement =
231 Vector3D::rotate
232 (
233 Vector3D::getFromPixelVector(((ColliderSpec*)colliderToCheck->componentSpec)->displacement),
234 colliderToCheck->transformation->rotation
235 );
236
237 colliderToCheck->position = Vector3D::sum(colliderToCheck->transformation->position, displacement);
238 colliderToCheck->invalidPosition = false;
239 }
240
241#ifdef __SHOW_PHYSICS_PROFILING
242 if(kNoCollision != Collider::collides(collider, colliderToCheck))
243 {
244 _lastCycleCollisions++;
245 }
246#else
247 Collider::collides(collider, colliderToCheck);
248#endif
249 if(this->dirty)
250 {
251 NM_ASSERT(false, "ColliderManager::update: added a collider as a response to a collision");
252 node = this->components->head;
253 }
254 }
255 }
256
257#ifdef __SHOW_PHYSICS_PROFILING
258 _collisionChecks += _lastCycleCollisionChecks;
259 _collisions += _lastCycleCollisions;
260
261 ColliderManager::print(this, 25, 1);
262#endif
263
264 return returnValue;
265}

269Collider ColliderManager::createCollider(Entity owner, const ColliderSpec* colliderSpec)
270{
271 if(NULL == colliderSpec)
272 {
273 return NULL;
274 }
275
276 ColliderManager::purgeDestroyedColliders(this);
277
278 Collider collider = ((Collider (*)(Entity, const ColliderSpec*)) ((ComponentSpec*)colliderSpec)->allocator)(owner, colliderSpec);
279
280 this->dirty = true;
281
282 VirtualList::pushFront(this->components, collider);
283
284 return collider;
285}

289void ColliderManager::destroyCollider(Collider collider)
290{
291 if(!isDeleted(collider))
292 {
293#ifndef __ENABLE_PROFILER
294 NM_ASSERT(NULL != __GET_CAST(Collider, collider), "ColliderManager::destroyCollider: invalide collider");
295 NM_ASSERT(NULL != VirtualList::find(this->components, collider), "ColliderManager::destroyCollider: non registerd collider");
296#endif
297 collider->deleteMe = true;
298 }
299}

407int32 ColliderManager::getNumberOfEnabledColliders()
408{
409 int32 count = 0;
410
411 for(VirtualNode node = this->components->head; NULL != node; node = node->next)
412 {
413 Collider collider = Collider::safeCast(node->data);
414
415 if(collider->enabled)
416 {
417 count++;
418 }
419 }
420
421 return count;
422}
