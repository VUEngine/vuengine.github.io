27static void Clock::printTime(uint32 milliseconds, int32 x, int32 y, const char* font, uint32 precision)
28{
29 char output[] = "00:00";
30 uint32 minutes = (uint32)(milliseconds / (__MILLISECONDS_PER_SECOND * 60));
31 uint32 seconds = (uint32)(milliseconds / __MILLISECONDS_PER_SECOND);
32
33 char* minutesString = Utilities::itoa(minutes, 10, 2);
34
35 output[0] = minutesString[0];
36 output[1] = minutesString[1];
37
38 char* secondsString = Utilities::itoa((seconds - minutes * 60) % 60, 10, 2);
39
40 output[3] = secondsString[0];
41 output[4] = secondsString[1];
42
43 Printing::text(output, x, y, font);
44
45 switch(precision)
46 {
47 case kTimePrecision1:
48
49 Clock::printDeciseconds(milliseconds, x + 6, y, font);
50 break;
51
52 case kTimePrecision2:
53
54 Clock::printCentiseconds(milliseconds, x + 6, y, font);
55 break;
56
57 case kTimePrecision3:
58
59 Clock::printMilliseconds(milliseconds, x + 6, y, font);
60 break;
61 }
62}

66static void Clock::printDeciseconds(uint32 milliseconds, int32 col, int32 row, const char* font)
67{

76static void Clock::printCentiseconds(uint32 milliseconds, int32 col, int32 row, const char* font)
77{
78 uint32 centiSeconds = ((milliseconds + 5) / 10);
79 centiSeconds -= ((centiSeconds / 100) * 100);
80
81 if(centiSeconds >= 10)
82 {
83 Printing::int32(centiSeconds, col, row, font);
84 }
85 else
86 {
87 Printing::int32(0, col, row, font);
88 Printing::int32(centiSeconds, col + 1, row, font);
89 }
90}

94static void Clock::printMilliseconds(uint32 milliseconds, int32 col, int32 row, const char* font)
95{
96 milliseconds -= ((milliseconds / 1000) * 1000);
97
98 if(milliseconds >= 100)
99 {
100 Printing::int32(milliseconds, col, row, font);
101 }
102 else if(milliseconds >= 10)
103 {
104 Printing::int32(0, col, row, font);
105 Printing::int32(milliseconds, col + 1, row, font);
106 }
107 else
108 {
109 Printing::int32(0, col, row, font);
110 Printing::int32(0, col + 1, row, font);
111 Printing::int32(milliseconds, col + 2, row, font);
112 }
113}

186void Clock::update(uint32 elapsedMilliseconds)
187{
188 // Increase count
189 if(this->paused)
190 {
191 return;
192 }
193
194 this->milliseconds += elapsedMilliseconds;
195
196 if(NULL != this->events)
197 {
198 uint32 currentSecond = Clock::getSeconds(this);
199
200 if(currentSecond != this->previousSecond)
201 {
202 this->previousSecond = currentSecond;
203
204 Clock::fireEvent(this, kEventSecondChanged);
205 NM_ASSERT(!isDeleted(this), "Clock::update: deleted this during kEventSecondChanged");
206
207 uint32 currentMinute = Clock::getMinutes(this);
208
209 if(currentMinute != this->previousMinute)
210 {
211 this->previousMinute = currentMinute;
212
213 Clock::fireEvent(this, kEventMinuteChanged);
214 NM_ASSERT(!isDeleted(this), "Clock::update: deleted this during kEventMinuteChanged");
215 }
216 }
217 }
218}
