131static inline void HardwareManager::setInterruptLevel(uint8 level)
132{
133 asm
134 (
135 "stsr sr5, r6 \n\t" \
136 "movhi 0xFFF1, r0, r7 \n\t" \
137 "movea 0xFFFF, r7, r7 \n\t" \
138 "and r6, r7 \n\t" \
139 "mov %0,r6 \n\t" \
140 "andi 0x000F, r6, r6 \n\t" \
141 "shl 0x10, r6 \n\t" \
142 "or r7, r6 \n\t" \
143 "ldsr r6, sr5 \n\t"
144 : // Output
145 : "r" (level) // Input
146 : "r6", "r7" // Clobber
147 );
148}

152static inline void HardwareManager::enableInterrupts()
153{
154 if(!_enabledInterrupts || 0 != _suspendInterruptRequest)
155 {
156 _enabledInterrupts = true;
157 _suspendInterruptRequest = 0;
158
159 asm("cli");
160 HardwareManager::setInterruptLevel(0);
161 }
162}
163

166static inline void HardwareManager::disableInterrupts()
167{
168 if(_enabledInterrupts)
169 {
170 _enabledInterrupts = false;
171 _suspendInterruptRequest = 0;
172
173 asm("sei");
174 HardwareManager::setInterruptLevel(5);
175 }
176}

178//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
179
180static inline void HardwareManager::resumeInterrupts()
181{
182 if(_enabledInterrupts)
183 {
184 if(0 >= --_suspendInterruptRequest)
185 {
186 _suspendInterruptRequest = 0;
187 asm("cli");
188 HardwareManager::setInterruptLevel(0);
189 }
190 }
191}

193//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————

195static inline void HardwareManager::suspendInterrupts()
196{
197 if(_enabledInterrupts)
198 {
199 _suspendInterruptRequest++;
200 asm("sei");
201 HardwareManager::setInterruptLevel(5);
202 }
203}

207static inline void HardwareManager::enableMultiplexedInterrupts()
208{
209 uint32 psw;
210
211 asm
212 (
213 "stsr psw, %0"
214 : "=r" (psw) // Output
215 );
216
217 psw &= 0xFFF0BFFF;
218
219 asm
220 (
221 "ldsr %0, psw \n\t" \
222 "cli \n\t"
223 : // Output
224 : "r" (psw) // Input
225 : // Clobber
226 );
227}

239 );
