27void Sphere::constructor(Entity owner, const SphereSpec* sphereSpec)
28{
29 // Always explicitly call the base's constructor
30 Base::constructor(owner, &sphereSpec->wireframeSpec);
31
32 this->position = PixelVector::zero();
33
34 if(NULL == sphereSpec)

58void Sphere::render(Vector3D relativePosition)
59{
60 NM_ASSERT(NULL != this->transformation, "Sphere::render: NULL transformation");
61
62 relativePosition = Vector3D::rotate(relativePosition, _previousCameraInvertedRotation);
63
64 this->position = PixelVector::projectVector3D(relativePosition, Optics::calculateParallax(relativePosition.z));
65 this->scaledRadius = __METERS_TO_PIXELS(__FIXED_MULT(this->radius, Vector3D::getScale(relativePosition.z, false)));
66}

70bool Sphere::draw()
71{
72 NM_ASSERT(NULL != this->transformation, "Sphere::render: NULL transformation");
73
74 bool drawn = false;
75
76 drawn = DirectDraw::drawCircle(this->position, this->scaledRadius, this->color, this->bufferIndex, this->interlaced);
77
78 if(this->drawCenter)
79 {
80 DirectDraw::drawPoint(this->position, this->color, this->bufferIndex, this->interlaced);
81 }
82
83 this->bufferIndex = !this->bufferIndex;
84
85 return drawn;
86}
