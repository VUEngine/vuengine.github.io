27static void Ball::project(Vector3D center, fixed_t radius, Vector3D vector, fixed_t* min, fixed_t* max)
28{
29 // Project this onto the current normal
30 fixed_t dotProduct = Vector3D::dotProduct(vector, center);
31
32 *min = dotProduct - radius;
33 *max = dotProduct + radius;
34
35 if(*min > *max)
36 {
37 fixed_t aux = *min;
38 *min = *max;
39 *max = aux;
40 }
41}

79void Ball::configureWireframe()
80{
81 if(!isDeleted(this->wireframe))
82 {
83 return;
84 }
85
86 // Create a wireframe
87 this->wireframe = Wireframe::safeCast(new Sphere(this->owner, NULL));
88
89 if(!isDeleted(this->wireframe))
90 {
91 if(NULL != this->componentSpec)
92 {
93 Sphere::setDisplacement(this->wireframe, Vector3D::getFromPixelVector(((ColliderSpec*)this->componentSpec)->displacement));
94 }
95
96 Sphere::setRadius(this->wireframe, this->radius);
97 }
98}

139void Ball::computeRadius()
140{
141 NM_ASSERT(NULL != this->componentSpec, "Ball::computeRadius: NULL componentSpec");
142
143 if(NULL == this->componentSpec)
144 {
145 return;
146 }
147
148 Size size = Size::getFromPixelSize(((ColliderSpec*)this->componentSpec)->pixelSize);
149
150 this->radius = size.z >> 1;
151
152 if(size.x > size.y)
153 {
154 if(size.x > size.z)
155 {
156 this->radius = size.x >> 1;
157 }
158 }
159 else if(size.y > size.z)
160 {
161 this->radius = size.y >> 1;
162 }
163}
