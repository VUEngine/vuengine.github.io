246static inline void VSUManager::applyPCMSampleToSoundSource(int8 sample)
247{
248 extern VSUSoundSource* const _vsuSoundSources;
249
250 int16 vsuSoundSourceIndex = 0;
251
252 while(true)
253 {
254 if(__MAXIMUM_VOLUME <= sample)
255 {
256 _vsuSoundSources[vsuSoundSourceIndex].SxLRV = 0xFF;
257 sample -= __MAXIMUM_VOLUME;
258 }
259 else
260 {
261 _vsuSoundSources[vsuSoundSourceIndex].SxLRV = ((sample << 4) | sample);
262 break;
263 }
264
265 vsuSoundSourceIndex++;
266 }
267}
