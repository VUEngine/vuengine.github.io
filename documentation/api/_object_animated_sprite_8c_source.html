45void ObjectAnimatedSprite::updateAnimation()
46{
47 NM_ASSERT(!isDeleted(this->animationController), "ObjectAnimatedSprite::updateAnimation: null animation controller");
48
49 if(isDeleted(this->animationController))
50 {
51 return;
52 }
53
54 if(Texture::isMultiframe(this->texture))
55 {
56 this->objectTextureSource.displacement = AnimationController::getActualFrameIndex(this->animationController) * this->rows * this->cols;
57 }
58 else
59 {
60 Texture::setFrame(this->texture, AnimationController::getActualFrameIndex(this->animationController));
61 }
62}
