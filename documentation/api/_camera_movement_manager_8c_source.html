60void CameraMovementManager::setFocusActor(Actor focusActor)
61{
62 this->focusActor = focusActor;
63 this->focusActorPosition = NULL;
64 this->focusActorRotation = NULL;
65
66 if(!isDeleted(this->focusActor))
67 {
68 Actor::addEventListener
69 (
70 this->focusActor, ListenerObject::safeCast(this), (EventListener)CameraMovementManager::onFocusActorDeleted,
71 kEventContainerDeleted
72 );
73
74 this->focusActorPosition = Actor::getPosition(this->focusActor);
75 this->focusActorRotation = Actor::getRotation(this->focusActor);
76 }
77}

108Vector3D CameraMovementManager::focus(Camera camera)
109{
110 if(isDeleted(camera))
111 {
112 return Vector3D::zero();
113 }
114
115 if(isDeleted(this->focusActor))
116 {
117 return Camera::getPosition(camera);
118 }
119
120 NormalizedDirection normalizedDirection = Actor::getNormalizedDirection(this->focusActor);
121
122 // Calculate the target position
123 Vector3D cameraNewPosition =
124 {
125 this->focusActorPosition->x + normalizedDirection.x * this->focusActorPositionDisplacement.x - __HALF_SCREEN_WIDTH_METERS,
126 this->focusActorPosition->y + normalizedDirection.y * this->focusActorPositionDisplacement.y - __HALF_SCREEN_HEIGHT_METERS,
127 this->focusActorPosition->z + normalizedDirection.z * this->focusActorPositionDisplacement.z - __HALF_SCREEN_DEPTH_METERS,
128 };
129
130 return cameraNewPosition;
131}

141bool CameraMovementManager::onFocusActorDeleted(ListenerObject eventFirer)
142{
143 if(ListenerObject::safeCast(this->focusActor) == eventFirer)
144 {
145 CameraMovementManager::setFocusActor(this, NULL);
146 }
147
148 return false;
