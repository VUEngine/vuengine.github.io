226static inline PixelVector PixelVector::projectVector3D(Vector3D vector3D, int16 parallax)
227{
228 fixed_ext_t x = (fixed_ext_t)(vector3D.x);
229 fixed_ext_t y = (fixed_ext_t)(vector3D.y);
230 fixed_ext_t z = (fixed_ext_t)(vector3D.z);
231
232#ifdef __LEGACY_COORDINATE_PROJECTION
233 if(0 != z)
234 {
235 x -= (__FIXED_EXT_MULT(x - _optical->horizontalViewPointCenter, z) >> _optical->maximumXViewDistancePower);
236 y -= (__FIXED_EXT_MULT(y - _optical->verticalViewPointCenter, z) >> _optical->maximumYViewDistancePower);
237 }
238#else
239 if(0 == z + _optical->cameraNearPlane)
240 {
241 x = 0 > x ? -(1 << 15) : (1 << 15);
242 y = 0 > y ? -(1 << 15) : (1 << 15);
243 }
244 else
245 {
246 /*
247 // Mathematically correct version
248 // but produces distorted results on the y axis
249 // x = x * aspect ratio * fov
250 x = __FIXED_EXT_MULT(x, _optical->aspectRatioXfov);
251 // y = y * fov
252 // since fov = 1 because it is assumed and angle of 90, there is no
253 // need to make the computation
254 y = __FIXED_EXT_MULT(y, _optical->fov);
255 // z = z * (far + near) / (far - near) + (2 * far * near) / (near - far)
256 // since the near plane will always be 0, there is no need to perform
257 // this product
258 z = __FIXED_EXT_MULT(z, _optical->farRatio1Near) + _optical->farRatio2Near;
259
260 x = __FIXED_EXT_DIV(__FIXED_EXT_MULT(x, _optical->halfHeight), z) + _optical->horizontalViewPointCenter;
261 y = __FIXED_EXT_DIV(__FIXED_EXT_MULT(y, _optical->halfWidth), z) + _optical->verticalViewPointCenter;
262 */
263
264 // Fast and produces the expected result
265 // x = x * aspect ratio * fov
266
267 // To reduce from 4 products and 2 divisions to 3 products, 1 division and 3 bit shifts
268 fixed_ext_t factor = __FIXED_EXT_DIV(_optical->projectionMultiplierHelper, z + _optical->cameraNearPlane);
269
270 x = (__FIXED_EXT_MULT(x, factor)) + _optical->horizontalViewPointCenter;
271 y = (__FIXED_EXT_MULT(y, factor)) + _optical->verticalViewPointCenter;
272 }
273#endif
274
275 PixelVector projection =
276 {
277 __METERS_TO_PIXELS(x),
278 __METERS_TO_PIXELS(y),
279 __METERS_TO_PIXELS(z),
280 parallax
281 };
282
283 return projection;
284}

288static inline PixelVector PixelVector::transformVector3D(Vector3D vector)
289{
290 vector = Vector3D::rotate(Vector3D::sub(vector, *_cameraPosition), *_cameraInvertedRotation);
291
292 return PixelVector::projectVector3D(vector, Optics::calculateParallax(vector.z));
293}

311static inline bool PixelVector::isVisible(PixelVector vector, PixelRightBox pixelRightBox, int16 padding)
312{
313 extern const CameraFrustum* _cameraFrustum;
314
315#ifndef __LEGACY_COORDINATE_PROJECTION
316 vector = PixelVector::sum(vector, (PixelVector){__HALF_SCREEN_WIDTH, __HALF_SCREEN_HEIGHT, 0, 0});
317#endif
318
319 if(vector.x + pixelRightBox.x0 > _cameraFrustum->x1 + padding || vector.x + pixelRightBox.x1 < _cameraFrustum->x0 - padding)
320 {
321 return false;
322 }
323
324 // Check y visibility
325 if(vector.y + pixelRightBox.y0 > _cameraFrustum->y1 + padding || vector.y + pixelRightBox.y1 < _cameraFrustum->y0 - padding)
326 {
327 return false;
328 }
329
330 // Check z visibility
331 if(vector.z + pixelRightBox.z0 > _cameraFrustum->z1 + padding || vector.z + pixelRightBox.z1 < _cameraFrustum->z0 - padding)
332 {
333 return false;
334 }
335
336 return true;
337}

341static inline bool PixelVector::isVector3DVisible(Vector3D vector, PixelRightBox pixelRightBox, int16 padding)
342{
343 extern const CameraFrustum* _cameraFrustum;
344 vector = Vector3D::rotate(Vector3D::getRelativeToCamera(vector), *_cameraInvertedRotation);
345 PixelVector pixelVector = PixelVector::projectVector3D(vector, 0);
346
347 if(pixelVector.x + pixelRightBox.x0 > _cameraFrustum->x1 + padding || pixelVector.x + pixelRightBox.x1 < _cameraFrustum->x0 - padding)
348 {
349 return false;
350 }
351
352 // Check y visibility
353 if(pixelVector.y + pixelRightBox.y0 > _cameraFrustum->y1 + padding || pixelVector.y + pixelRightBox.y1 < _cameraFrustum->y0 - padding)
354 {
355 return false;
356 }
357
358 // Check z visibility
359 if(pixelVector.z + pixelRightBox.z0 > _cameraFrustum->z1 + padding || pixelVector.z + pixelRightBox.z1 < _cameraFrustum->z0 - padding)
360 {
361 return false;
362 }
363
364 return true;
365}
