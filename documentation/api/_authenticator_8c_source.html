72void Authenticator::initialize(ClassPointer* requesterClasses)
73{
74 if(!isDeleted(this->authorizedRequesters))
75 {
76 return;
77 }
78
79 this->authorizedRequesters = new VirtualList();
80
81 for(int16 i = 0; NULL != requesterClasses[i]; i++)
82 {
83 AuthorizationTicket* authorizationTicket = new AuthorizationTicket;
84 authorizationTicket->classPointer = requesterClasses[i];
85
86 VirtualList::pushBack(this->authorizedRequesters, authorizationTicket);
87 }
88}

92bool Authenticator::authorize(ClassPointer requesterClass)
93{
94 for(VirtualNode node = this->authorizedRequesters->head; NULL != node; node = node->next)
95 {
96 AuthorizationTicket* authorizationTicket = (AuthorizationTicket*)node->data;
97
98 if(requesterClass == authorizationTicket->classPointer)
99 {
100 return true;
101 }
102 }
103
104 return false;
105}
