70void Sound::constructor(const SoundSpec* soundSpec, EventListener soundReleaseListener, ListenerObject scope)
71{
72 // Always explicitly call the base's constructor
73 Base::constructor();

75 this->state = kSoundOff;
76 this->soundSpec = soundSpec;
77 this->speed = __I_TO_FIX7_9_EXT(1);
78 this->previouslyElapsedTicks = 0;

130void Sound::play(const Vector3D* position, uint32 playbackType)
131{
132 if(NULL == this->soundSpec)
133 {
134 return;
135 }
136
137 if(isDeleted(this->soundTracks))
138 {
139 return;
140 }
141
142 switch(playbackType)
143 {
144 case kSoundPlaybackFadeIn:
145
146 if(kSoundPlaying != this->state)
147 {
148 Sound::setVolumeReduction(this, __MAXIMUM_VOLUME * this->volumeReductionMultiplier);
149 }
150 else
151 {
152 return;
153 }
154
155 break;
156
157 // Intentional fall through
158 case kSoundPlaybackNormal:
159
160 Sound::setVolumeReduction(this, 0);
161 break;
162 }
163
164 this->playbackType = playbackType;
165
166 switch(playbackType)
167 {
168 case kSoundPlaybackFadeIn:
169 case kSoundPlaybackNormal:
170 {
171 bool wasPaused = kSoundPaused == this->state;
172
173 this->state = kSoundPlaying;
174
175 this->position = position;
176
177 for(VirtualNode node = this->soundTracks->head; NULL != node; node = node->next)
178 {
179 SoundTrack soundTrack = SoundTrack::safeCast(node->data);
180
181 SoundTrack::start(soundTrack, wasPaused);
182 }
183
184 if(!wasPaused)
185 {
186 this->previouslyElapsedTicks = 0;
187 }
188 }
189
190 break;
191 }
192}

488
489void Sound::update(uint32 elapsedMicroseconds, uint32 targetPCMUpdates)
490{
491 if(kSoundPlaying != this->state)
492 {
493 return;
494 }
495
496 bool finished = true;
497
498 fixed_t leftVolumeFactor = -1;
499 fixed_t rightVolumeFactor = -1;
500
501 if(NULL != this->position)
502 {
503#ifndef __LEGACY_COORDINATE_PROJECTION
504 Vector3D relativePosition = Vector3D::rotate(Vector3D::getRelativeToCamera(*this->position), *_cameraInvertedRotation);
505#else
506 Vector3D relativePosition =
507 Vector3D::rotate
508 (
509 Vector3D::sub
510 (
511 Vector3D::getRelativeToCamera
512 (
513 *this->position), (Vector3D){__HALF_SCREEN_WIDTH_METERS, __HALF_SCREEN_HEIGHT_METERS, 0}
514 ),
515 *_cameraInvertedRotation
516 );
517#endif
518 if(_mirror.x)
519 {
520 relativePosition.x = -relativePosition.x;
521 }
522
523 if(_mirror.y)
524 {
525 relativePosition.y = -relativePosition.y;
526 }
527
528 if(_mirror.z)
529 {
530 relativePosition.z = -relativePosition.z;
531 }
532
533 Vector3D leftEar = (Vector3D){__PIXELS_TO_METERS(-__EAR_DISPLACEMENT), 0, 0};
534 Vector3D rightEar = (Vector3D){__PIXELS_TO_METERS(__EAR_DISPLACEMENT), 0, 0};
535
536 fixed_ext_t squaredDistanceToLeftEar = Vector3D::squareLength(Vector3D::get(leftEar, relativePosition));
537 fixed_ext_t squaredDistanceToRightEar = Vector3D::squareLength(Vector3D::get(rightEar, relativePosition));
538
539 leftVolumeFactor =
540 __1I_FIXED -
541 __FIXED_EXT_DIV(squaredDistanceToLeftEar, __FIXED_SQUARE(__PIXELS_TO_METERS(__SOUND_STEREO_ATTENUATION_DISTANCE)));
542
543 rightVolumeFactor =
544 __1I_FIXED -
545 __FIXED_EXT_DIV(squaredDistanceToRightEar, __FIXED_SQUARE(__PIXELS_TO_METERS(__SOUND_STEREO_ATTENUATION_DISTANCE)));
546
547 if(leftVolumeFactor > rightVolumeFactor)
548 {
549 rightVolumeFactor -= (leftVolumeFactor - rightVolumeFactor);
550 }
551 else
552 {
553 leftVolumeFactor -= (rightVolumeFactor - leftVolumeFactor);
554 }
555
556 leftVolumeFactor = 0 > leftVolumeFactor ? 0 : leftVolumeFactor;
557 rightVolumeFactor = 0 > rightVolumeFactor ? 0 : rightVolumeFactor;
558 }
559
560 for(VirtualNode node = this->soundTracks->head; NULL != node; node = node->next)
561 {
562 SoundTrack soundTrack = SoundTrack::safeCast(node->data);
563
564 finished =
565 SoundTrack::update
566 (
567 soundTrack,
568 elapsedMicroseconds,
569 targetPCMUpdates,
570 this->tickStep,
571 this->targetTimerResolutionFactor,
572 leftVolumeFactor,
573 rightVolumeFactor,
574 this->volumeReduction,
575 this->volumenScalePower,
576 this->frequencyDelta
577 ) && finished;
578 }
579
580 if(finished)
581 {
582 Sound::completedPlayback(this);
583 }
584
585 if(kSoundPlaybackNormal != this->playbackType)
586 {
587 Sound::updateVolumeReduction(this);
588 }
589}

593void Sound::print(int32 x, int32 y)
594{
595 PRINT_TEXT(" ", x, y);
596 PRINT_TEXT(this->soundSpec->name, x, y++);
597 y++;
598
599 Sound::printPlaybackProgress(this, x, y++);
600
601 uint8 trackInfoXOffset = x + 22;
602 uint8 trackInfoValuesXOffset = 9;
603 uint16 speed = __FIX7_9_EXT_TO_I(__FIX7_9_EXT_MULT(this->speed, __I_TO_FIX7_9_EXT(100)));
604
605 y++;
606
607 Sound::printTiming(this, 0, x + 23, y);
608 PRINT_TEXT("/", x + 27, y);
609 Sound::printTiming(this, this->totalPlaybackMilliseconds / __MILLISECONDS_PER_SECOND, x + 28, y);
610
611 PRINT_TEXT("Speed", x, y);
612 PRINT_TEXT(" ", x + 6, y);
613 PRINT_INT(speed, x + 6, y);
614 PRINT_TEXT("%", x + 6 + ((speed < 10) ? 1 : (speed < 100) ? 2 : 3), y);
615
616 PRINT_TEXT(kSoundPlaying == this->state ? "\x07\x07" : " \x0B ", x + 15, y++);
617
618 y+=2;
619
620 PRINT_TEXT("TRACK INFO", trackInfoXOffset, y++);
621
622 PRINT_TEXT("Tracks", trackInfoXOffset, ++y);
623 PRINT_INT(VirtualList::getCount(this->soundTracks), trackInfoXOffset + trackInfoValuesXOffset, y);
624
625 PRINT_TEXT("Loop", trackInfoXOffset, ++y);
626 PRINT_TEXT
627 (
628 this->soundSpec->loop ? __CHAR_CHECKBOX_CHECKED : __CHAR_CHECKBOX_UNCHECKED, trackInfoXOffset + trackInfoValuesXOffset, y++
629 );
630}

635{

647 if(previousSecond > currentSecond)

651
652 if(currentSecond > previousSecond)
653 {
654 previousSecond = currentSecond;
655
656 Sound::printTiming(this, currentSecond, x, y);

704//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
705
706fix7_9_ext Sound::computeTimerResolutionFactor()
707{
708 uint32 timerCounter = TimerManager::getTimerCounter() + __TIMER_COUNTER_DELTA;
709 uint32 timerUsPerInterrupt = timerCounter * __SOUND_TARGET_US_PER_TICK;
710 uint32 targetTimerResolutionUS = 0 != this->soundSpec->targetTimerResolutionUS ? this->soundSpec->targetTimerResolutionUS : 1000;
711 uint32 soundTargetUsPerInterrupt = (__TIME_US(targetTimerResolutionUS) + __TIMER_COUNTER_DELTA) * __SOUND_TARGET_US_PER_TICK;
712
713 NM_ASSERT(0 < soundTargetUsPerInterrupt, "Sound::computeTimerResolutionFactor: zero soundTargetUsPerInterrupt");
714
715 return __FIX7_9_EXT_DIV(__I_TO_FIX7_9_EXT(timerUsPerInterrupt), __I_TO_FIX7_9_EXT(soundTargetUsPerInterrupt));
716}

732void Sound::configureTracks()
733{
734 if(NULL == this->soundSpec || NULL == this->soundSpec->soundTrackSpecs)
735 {
736 return;
737 }
738
739 if(NULL == this->soundTracks)
740 {
741 this->soundTracks = new VirtualList();
742 }
743
744 SoundTrack longestSoundTrack = NULL;
745
746 for(int16 i = 0; NULL != this->soundSpec->soundTrackSpecs[i]; i++)
747 {
748 SoundTrack soundTrack = new SoundTrack(this->soundSpec->soundTrackSpecs[i]);
749
750 VirtualList::pushBack(this->soundTracks, soundTrack);
751
752 if(NULL == longestSoundTrack || SoundTrack::getTicks(longestSoundTrack) < SoundTrack::getTicks(soundTrack))
753 {
754 longestSoundTrack = soundTrack;
755 }
756
757 }
758
759 this->mainSoundTrack = longestSoundTrack;
760
761#ifdef __SOUND_TEST
762 this->totalPlaybackMilliseconds =
763 SoundTrack::getTotalPlaybackMilliseconds(this->mainSoundTrack, this->soundSpec->targetTimerResolutionUS);
764#endif
765
766}

796
797__attribute__((noinline))
798void Sound::updateVolumeReduction()
799{
800 uint32 elapsedMilliseconds =
801 this->soundSpec->targetTimerResolutionUS * (this->mainSoundTrack->elapsedTicks
802 - this->previouslyElapsedTicks) / __MICROSECONDS_PER_MILLISECOND;
803
804 if(VUEngine::getGameFrameDuration() <= elapsedMilliseconds)
805 {
806 switch(this->playbackType)
807 {
808 case kSoundPlaybackFadeIn:
809
810 this->volumeReduction -= (this->volumeReductionMultiplier >> 1) + 1;
811

845 }

848 }
