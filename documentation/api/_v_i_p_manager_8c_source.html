89static void VIPManager::interruptHandler()
90{
91 VIPManager vipManager = VIPManager::getInstance();
92
93 vipManager->currrentInterrupt = _vipRegisters[__INTPND];
94
95 // Clear interrupts
96 VIPManager::clearInterrupts();
97
98#ifndef __DEBUG
99 if(kVIPNoMultiplexedInterrupts != vipManager->enabledMultiplexedInterrupts)
100 {
101 if(kVIPOnlyVIPMultiplexedInterrupts == vipManager->enabledMultiplexedInterrupts)
102 {
103 HardwareManager::setInterruptLevel(vipManager->enabledMultiplexedInterrupts);
104 }
105
106 HardwareManager::enableMultiplexedInterrupts();
107 }
108#endif
109
110 // Handle the interrupt
111 VIPManager::processInterrupt(vipManager->currrentInterrupt);
112
113#ifndef __DEBUG
114 if(kVIPNoMultiplexedInterrupts != vipManager->enabledMultiplexedInterrupts)
115 {
116 HardwareManager::disableMultiplexedInterrupts();
117
118 if(kVIPOnlyVIPMultiplexedInterrupts == vipManager->enabledMultiplexedInterrupts)
119 {
120 HardwareManager::setInterruptLevel(0);
121 }
122 }
123#endif
124
125 vipManager->currrentInterrupt = 0;
126}

335static void VIPManager::setupColumnTable(ColumnTableSpec* columnTableSpec)
336{
337 int32 i, value;
338
339 // Use the default column table as fallback
340 if(columnTableSpec == NULL)
341 {
342 columnTableSpec = (ColumnTableSpec*)&DefaultColumnTableSpec;
343 }
344
345 // Write column table
346 for(i = 0; i < 256; i++)
347 {
348 value = (columnTableSpec->mirror && (i > (__COLUMN_TABLE_ENTRIES / 2 - 1)))
349 ? columnTableSpec->columnTable[(__COLUMN_TABLE_ENTRIES - 1) - i]
350 : columnTableSpec->columnTable[i];
351
352 _columnTableBaseAddressLeft[i] = value;
353 _columnTableBaseAddressRight[i] = value;
354 }
355}

359static void VIPManager::configureBrightness(Brightness* brightness)
360{
361 while(_vipRegisters[__XPSTTS] & __XPBSY);
362 _vipRegisters[__BRTA] = brightness->darkRed;
363 _vipRegisters[__BRTB] = brightness->mediumRed;
364 _vipRegisters[__BRTC] = brightness->brightRed - (brightness->mediumRed + brightness->darkRed);
365}

367//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
368
369static void VIPManager::configureBrightnessRepeat(BrightnessRepeatSpec* brightnessRepeatSpec)
370{
371 // Use the default repeat values as fallback
372 if(brightnessRepeatSpec == NULL)
373 {
374 brightnessRepeatSpec = (BrightnessRepeatSpec*)&DefaultBrightnessRepeatSpec;
375 }
376 // Column table offsets
377 int16 leftCta = _vipRegisters[__CTA] & 0xFF;
378 int16 rightCta = _vipRegisters[__CTA] >> 8;
379
380 CACHE_RESET;
381
382 // Write repeat values to column table
383 for(int16 i = 0; i < 96; i++)
384 {
385 int16 value = (brightnessRepeatSpec->mirror && (i > (__BRIGHTNESS_REPEAT_ENTRIES / 2 - 1)))
386 ? brightnessRepeatSpec->brightnessRepeat[__BRIGHTNESS_REPEAT_ENTRIES - 1 - i] << 8
387 : brightnessRepeatSpec->brightnessRepeat[i] << 8;
388
389 _columnTableBaseAddressLeft[leftCta - i] = (_columnTableBaseAddressLeft[leftCta - i] & 0xff) | value;
390 _columnTableBaseAddressRight[rightCta - i] = (_columnTableBaseAddressRight[rightCta - i] & 0xff) | value;
391 }
392}

542
543static void VIPManager::processInterrupt(uint16 interrupt)
544{
545 VIPManager vipManager = VIPManager::getInstance();
546
547 static uint16 interruptTable[] =
548 {
549 __FRAMESTART,
550 __GAMESTART,
551 __XPEND,
552#ifndef __SHIPPING
553 __TIMEERR,
554 __SCANERR
555#endif
556 };
557
558 for(uint32 i = 0; i < sizeof(interruptTable) / sizeof(uint16); i++)
559 {
560 switch(interrupt & interruptTable[i])
561 {
562 case __FRAMESTART:
563
564 vipManager->FRAMESTARTDuringXPEND = vipManager->processingXPEND;
565
566 VIPManager::fireEvent(vipManager, kEventVIPManagerFRAMESTART);
567 break;
568
569 case __GAMESTART:
570
571#ifdef __ENABLE_PROFILER
572 Profiler::lap(kProfilerLapTypeStartInterrupt, NULL);
573#endif
574
575 vipManager->processingGAMESTART = true;
576
577 VIPManager::registerCurrentDrawingFrameBufferSet();
578
579 VIPManager::fireEvent
580 (
581 vipManager,
582 vipManager->processingXPEND ?
583 kEventVIPManagerGAMESTARTDuringXPEND
584 :
585 kEventVIPManagerGAMESTART
586 );
587
588 vipManager->processingGAMESTART = false;
589
590#ifdef __ENABLE_PROFILER
591 Profiler::lap(kProfilerLapTypeVIPInterruptGAMESTARTProcess, PROCESS_NAME_RENDER);
592#endif
593 break;
594
595 case __XPEND:
596
597#ifdef __ENABLE_PROFILER
598 Profiler::lap(kProfilerLapTypeStartInterrupt, NULL);
599#endif
600 VIPManager::suspendDrawing();
601
602 vipManager->processingXPEND = true;
603
604 VIPManager::fireEvent
605 (
606 vipManager,
607 vipManager->processingGAMESTART ?
608 kEventVIPManagerXPENDDuringGAMESTART
609 :
610 kEventVIPManagerXPEND
611 );
612
613 VIPManager::applyPostProcessingEffects();
614
615 vipManager->processingXPEND = false;
616
617 VIPManager::resumeDrawing();
618
619#ifdef __ENABLE_PROFILER
620 Profiler::lap(kProfilerLapTypeVIPInterruptXPENDProcess, PROCESS_NAME_VRAM_WRITE);
621#endif
622 break;
623
624#ifndef __SHIPPING
625 case __TIMEERR:
626
627 VIPManager::fireEvent(vipManager, kEventVIPManagerTimeError);
628 break;
629
630 case __SCANERR:
631
632 VIPManager::fireEvent(vipManager, kEventVIPManagerScanError);
633
634 Error::triggerException("VIPManager::servo error", NULL);
635 break;
636#endif
637 }
638 }
639}

673
674static void VIPManager::applyPostProcessingEffects()
675{
676 VIPManager vipManager = VIPManager::getInstance();
677
678 for
679 (
680 VirtualNode node = vipManager->postProcessingEffects->tail, previousNode = NULL;
681 !vipManager->FRAMESTARTDuringXPEND && NULL != node;
682 node = previousNode
683 )
684 {
685 previousNode = node->previous;
686
687 PostProcessingEffectRegistry* postProcessingEffectRegistry = (PostProcessingEffectRegistry*)node->data;
688
689 if(isDeleted(postProcessingEffectRegistry) || postProcessingEffectRegistry->remove)
690 {
691 VirtualList::removeNode(vipManager->postProcessingEffects, node);
692
693 if(!isDeleted(postProcessingEffectRegistry))
694 {
695 delete postProcessingEffectRegistry;
696 }
697 }
698 else
699 {
700 postProcessingEffectRegistry->postProcessingEffect
701 (
702 vipManager->currentDrawingFrameBufferSet, postProcessingEffectRegistry->entity
703 );
704 }
705 }

707static void VIPManager::clearDRAM()
708{
709 uint8* bgmapStartAddress = (uint8*)__BGMAP_SPACE_BASE_ADDRESS;
710
711 // Clear every bgmap segment
712 for(bgmapStartAddress = 0; bgmapStartAddress < (uint8*)__PARAM_TABLE_END; bgmapStartAddress++)
713 {
714 *bgmapStartAddress = 0;
715 }
716
717 Mem::clear((BYTE*) __CHAR_SPACE_BASE_ADDRESS, 8192 * 4);
718
719 for(int32 i = 0; i < __TOTAL_LAYERS; i++)
720 {
721 _worldAttributesCache[i].head = 0;
722 _worldAttributesCache[i].gx = 0;
723 _worldAttributesCache[i].gp = 0;
724 _worldAttributesCache[i].gy = 0;
725 _worldAttributesCache[i].mx = 0;
726 _worldAttributesCache[i].mp = 0;
727 _worldAttributesCache[i].my = 0;
728 _worldAttributesCache[i].w = 0;
729 _worldAttributesCache[i].h = 0;
730 _worldAttributesCache[i].param = 0;
731 _worldAttributesCache[i].ovr = 0;
732
733 _worldAttributesBaseAddress[i].head = 0;
734 _worldAttributesBaseAddress[i].gx = 0;
735 _worldAttributesBaseAddress[i].gp = 0;
736 _worldAttributesBaseAddress[i].gy = 0;
737 _worldAttributesBaseAddress[i].mx = 0;
738 _worldAttributesBaseAddress[i].mp = 0;
739 _worldAttributesBaseAddress[i].my = 0;
740 _worldAttributesBaseAddress[i].w = 0;
741 _worldAttributesBaseAddress[i].h = 0;

743 _worldAttributesBaseAddress[i].ovr = 0;
744 }
745
746 for(int32 i = 0; i < __TOTAL_OBJECTS; i++)
747 {

749 _objectAttributesCache[i].head = 0;
