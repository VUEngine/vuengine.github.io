72static void VIPManager::interruptHandler()
73{
74 VIPManager vipManager = VIPManager::getInstance();
75
76 vipManager->currrentInterrupt = _vipRegisters[__INTPND];
77
78 // Clear interrupts
79 VIPManager::clearInterrupts(vipManager);
80
81#ifndef __DEBUG
82 if(kVIPNoMultiplexedInterrupts != vipManager->enabledMultiplexedInterrupts)
83 {
84 if(kVIPOnlyVIPMultiplexedInterrupts == vipManager->enabledMultiplexedInterrupts)
85 {
86 HardwareManager::setInterruptLevel(vipManager->enabledMultiplexedInterrupts);
87 }
88
89 HardwareManager::enableMultiplexedInterrupts();
90 }
91#endif
92
93 // Handle the interrupt
94 VIPManager::processInterrupt(vipManager, vipManager->currrentInterrupt);
95
96#ifndef __DEBUG
97 if(kVIPNoMultiplexedInterrupts != vipManager->enabledMultiplexedInterrupts)
98 {
99 HardwareManager::disableMultiplexedInterrupts();
100
101 if(kVIPOnlyVIPMultiplexedInterrupts == vipManager->enabledMultiplexedInterrupts)
102 {
103 HardwareManager::setInterruptLevel(0);
104 }
105 }
106#endif
107
108 vipManager->currrentInterrupt = 0;
109}

113static void VIPManager::pushFrontPostProcessingEffect(PostProcessingEffect postProcessingEffect, Entity entity)
114{
115 VIPManager vipManager = VIPManager::getInstance();
116
117 PostProcessingEffectRegistry* postProcessingEffectRegistry =
118 VIPManager::isPostProcessingEffectRegistered(vipManager, postProcessingEffect, entity);
119
120 if(!isDeleted(postProcessingEffectRegistry))
121 {
122 postProcessingEffectRegistry->remove = false;
123 return;
124 }
125
126 postProcessingEffectRegistry = new PostProcessingEffectRegistry;
127 postProcessingEffectRegistry->postProcessingEffect = postProcessingEffect;
128 postProcessingEffectRegistry->entity = entity;
129 postProcessingEffectRegistry->remove = false;
130
131 VirtualList::pushFront(vipManager->postProcessingEffects, postProcessingEffectRegistry);
132}

159static void VIPManager::removePostProcessingEffect(PostProcessingEffect postProcessingEffect, Entity entity)
160{
161 VIPManager vipManager = VIPManager::getInstance();
162
163 for(VirtualNode node = vipManager->postProcessingEffects->head; NULL != node; node = node->next)
164 {
165 PostProcessingEffectRegistry* postProcessingEffectRegistry = (PostProcessingEffectRegistry*)node->data;
166
167 if
168 (
169 postProcessingEffectRegistry->postProcessingEffect == postProcessingEffect
170 &&
171 postProcessingEffectRegistry->entity == entity
172 )
173 {
174 postProcessingEffectRegistry->remove = true;
175 return;
176 }
177 }

270static void VIPManager::setupColumnTable(ColumnTableSpec* columnTableSpec)
271{
272 int32 i, value;
273
274 // Use the default column table as fallback
275 if(columnTableSpec == NULL)
276 {
277 columnTableSpec = (ColumnTableSpec*)&DefaultColumnTableSpec;
278 }
279
280 // Write column table
281 for(i = 0; i < 256; i++)
282 {
283 value = (columnTableSpec->mirror && (i > (__COLUMN_TABLE_ENTRIES / 2 - 1)))
284 ? columnTableSpec->columnTable[(__COLUMN_TABLE_ENTRIES - 1) - i]
285 : columnTableSpec->columnTable[i];
286
287 _columnTableBaseAddressLeft[i] = value;
288 _columnTableBaseAddressRight[i] = value;
289 }
290}

294static void VIPManager::configureBrightness(Brightness* brightness)
295{
296 while(_vipRegisters[__XPSTTS] & __XPBSY);
297 _vipRegisters[__BRTA] = brightness->darkRed;
298 _vipRegisters[__BRTB] = brightness->mediumRed;
299 _vipRegisters[__BRTC] = brightness->brightRed - (brightness->mediumRed + brightness->darkRed);
300}

302//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
303
304static void VIPManager::configureBrightnessRepeat(BrightnessRepeatSpec* brightnessRepeatSpec)
305{
306 // Use the default repeat values as fallback
307 if(brightnessRepeatSpec == NULL)
308 {
309 brightnessRepeatSpec = (BrightnessRepeatSpec*)&DefaultBrightnessRepeatSpec;
310 }
311 // Column table offsets
312 int16 leftCta = _vipRegisters[__CTA] & 0xFF;
313 int16 rightCta = _vipRegisters[__CTA] >> 8;
314
315 CACHE_RESET;
316
317 // Write repeat values to column table
318 for(int16 i = 0; i < 96; i++)
319 {
320 int16 value = (brightnessRepeatSpec->mirror && (i > (__BRIGHTNESS_REPEAT_ENTRIES / 2 - 1)))
321 ? brightnessRepeatSpec->brightnessRepeat[__BRIGHTNESS_REPEAT_ENTRIES - 1 - i] << 8
322 : brightnessRepeatSpec->brightnessRepeat[i] << 8;
323
324 _columnTableBaseAddressLeft[leftCta - i] = (_columnTableBaseAddressLeft[leftCta - i] & 0xff) | value;
325 _columnTableBaseAddressRight[rightCta - i] = (_columnTableBaseAddressRight[rightCta - i] & 0xff) | value;
326 }
327}

548secure void VIPManager::processInterrupt(uint16 interrupt)
549{
550 static uint16 interruptTable[] =
551 {
552 __FRAMESTART,
553 __GAMESTART,
554 __XPEND,
555#ifndef __SHIPPING
556 __TIMEERR,
557 __SCANERR
558#endif
559 };
560
561 for(uint32 i = 0; i < sizeof(interruptTable) / sizeof(uint16); i++)
562 {
563 switch(interrupt & interruptTable[i])
564 {
565 case __FRAMESTART:
566
567 this->FRAMESTARTDuringXPEND = this->processingXPEND;
568
569 VIPManager::fireEvent(this, kEventVIPManagerFRAMESTART);
570 break;
571
572 case __GAMESTART:
573
574#ifdef __ENABLE_PROFILER
575 Profiler::lap(kProfilerLapTypeStartInterrupt, NULL);
576#endif
577
578 this->processingGAMESTART = true;
579
580 VIPManager::registerCurrentDrawingFrameBufferSet(this);
581
582 VIPManager::fireEvent
583 (
584 this,
585 this->processingXPEND ?
586 kEventVIPManagerGAMESTARTDuringXPEND
587 :
588 kEventVIPManagerGAMESTART
589 );
590
591 this->processingGAMESTART = false;
592
593#ifdef __ENABLE_PROFILER
594 Profiler::lap(kProfilerLapTypeVIPInterruptGAMESTARTProcess, PROCESS_NAME_RENDER);
595#endif
596 break;
597
598 case __XPEND:
599
600#ifdef __ENABLE_PROFILER
601 Profiler::lap(kProfilerLapTypeStartInterrupt, NULL);
602#endif
603 VIPManager::suspendDrawing(this);
604
605 this->processingXPEND = true;
606
607 VIPManager::fireEvent
608 (
609 this,
610 this->processingGAMESTART ?
611 kEventVIPManagerXPENDDuringGAMESTART
612 :
613 kEventVIPManagerXPEND
614 );
615
616 VIPManager::applyPostProcessingEffects(this);
617
618 this->processingXPEND = false;
619
620 VIPManager::resumeDrawing(this);
621
622#ifdef __ENABLE_PROFILER
623 Profiler::lap(kProfilerLapTypeVIPInterruptXPENDProcess, PROCESS_NAME_VRAM_WRITE);
624#endif
625 break;
626
627#ifndef __SHIPPING
628 case __TIMEERR:
629
630 VIPManager::fireEvent(this, kEventVIPManagerTimeError);
631 break;
632
633 case __SCANERR:
634
635 VIPManager::fireEvent(this, kEventVIPManagerScanError);
636
637 Error::triggerException("VIPManager::servo error", NULL);
638 break;
639#endif
640 }
641 }
642}

673//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
674
675secure void VIPManager::applyPostProcessingEffects()
676{
677 for
678 (
679 VirtualNode node = this->postProcessingEffects->tail, previousNode = NULL;
680 !this->FRAMESTARTDuringXPEND && NULL != node;
681 node = previousNode
682 )
683 {
684 previousNode = node->previous;
685
686 PostProcessingEffectRegistry* postProcessingEffectRegistry = (PostProcessingEffectRegistry*)node->data;
687
688 if(isDeleted(postProcessingEffectRegistry) || postProcessingEffectRegistry->remove)
689 {
690 VirtualList::removeNode(this->postProcessingEffects, node);
691
692 if(!isDeleted(postProcessingEffectRegistry))
693 {
694 delete postProcessingEffectRegistry;
695 }
696 }
697 else
698 {
699 postProcessingEffectRegistry->postProcessingEffect
700 (
701 this->currentDrawingFrameBufferSet, postProcessingEffectRegistry->entity
702 );
703 }
704 }
705}

709secure void VIPManager::clearDRAM()
710{
711 uint8* bgmapStartAddress = (uint8*)__BGMAP_SPACE_BASE_ADDRESS;
712
713 // Clear every bgmap segment
714 for(bgmapStartAddress = 0; bgmapStartAddress < (uint8*)__PARAM_TABLE_END; bgmapStartAddress++)
715 {
716 *bgmapStartAddress = 0;
717 }
718
719 Mem::clear((BYTE*) __CHAR_SPACE_BASE_ADDRESS, 8192 * 4);
720
721 for(int32 i = 0; i < __TOTAL_LAYERS; i++)
722 {
723 _worldAttributesCache[i].head = 0;
724 _worldAttributesCache[i].gx = 0;
725 _worldAttributesCache[i].gp = 0;
726 _worldAttributesCache[i].gy = 0;
727 _worldAttributesCache[i].mx = 0;
728 _worldAttributesCache[i].mp = 0;
729 _worldAttributesCache[i].my = 0;
730 _worldAttributesCache[i].w = 0;
731 _worldAttributesCache[i].h = 0;
732 _worldAttributesCache[i].param = 0;
733 _worldAttributesCache[i].ovr = 0;
734
735 _worldAttributesBaseAddress[i].head = 0;
736 _worldAttributesBaseAddress[i].gx = 0;
737 _worldAttributesBaseAddress[i].gp = 0;
738 _worldAttributesBaseAddress[i].gy = 0;
739 _worldAttributesBaseAddress[i].mx = 0;
740 _worldAttributesBaseAddress[i].mp = 0;
741 _worldAttributesBaseAddress[i].my = 0;
742 _worldAttributesBaseAddress[i].w = 0;
743 _worldAttributesBaseAddress[i].h = 0;

746 }
747

749 {

751 _objectAttributesCache[i].head = 0;

765{
