59bool BgmapTexture::write(int16 maximumTextureRowsToWrite)
60{
61 if(isDeleted(this->charSet))
62 {
63 // Make sure to force full writing if no char set
64 this->remainingRowsToBeWritten = this->textureSpec->rows;
65 }
66
67 uint8 status = this->status;
68
69 if(!Base::write(this, maximumTextureRowsToWrite))
70 {
71 return false;
72 }
73
74 if(0 >= this->remainingRowsToBeWritten)
75 {
76 this->remainingRowsToBeWritten = this->textureSpec->rows;
77 }
78
79 uint16 charSetOffset = (uint16)CharSet::getOffset(this->charSet);
80
81 if(BgmapTexture::isMultiframe(this))
82 {
83 BgmapTexture::writeAllFrames(this, maximumTextureRowsToWrite, this->xOffset, this->yOffset, charSetOffset);
84 }
85 else
86 {
87 BgmapTexture::writeFrame
88 (
89 this, maximumTextureRowsToWrite, kTexturePendingWriting < status && kTextureFrameChanged >= status,
90 this->xOffset, this->yOffset, charSetOffset, 0
91 );
92 }
93
94 if(kTexturePendingRewriting == status)
95 {
96 this->status = 0 >= this->remainingRowsToBeWritten ? kTextureWritten : kTexturePendingRewriting;
97 }
98 else
99 {
100 this->status = 0 >= this->remainingRowsToBeWritten ? kTextureWritten : kTexturePendingWriting;
101 }
102
103 return true;
104}

204void BgmapTexture::writeAllFrames(int16 maximumTextureRowsToWrite, int16 xOffset, int16 yOffset, uint16 charSetOffset)
205{
206 if((0 > xOffset) | (0 > yOffset))
207 {
208 return;
209 }
210
211 bool isCharSetOptimized = CharSet::isOptimized(this->charSet);
212
213 int16 currentXOffset = xOffset;
214 int16 currentYOffset = yOffset;
215 int16 charSetOffsetDelta = isCharSetOptimized ? 0 : this->textureSpec->cols * this->textureSpec->rows;
216
217 this->mapDisplacement = 0;
218
219 for(int16 frame = 0; frame < this->textureSpec->numberOfFrames; frame++)
220 {
221 this->remainingRowsToBeWritten = this->textureSpec->rows;
222
223 BgmapTexture::writeFrame
224 (
225 this, maximumTextureRowsToWrite, true, currentXOffset, currentYOffset, charSetOffset, isCharSetOptimized ? frame : 0
226 );
227
228 charSetOffset += charSetOffsetDelta;
229
230 currentXOffset += this->textureSpec->cols;
231
232 if(64 <= currentXOffset + this->textureSpec->cols)
233 {
234 currentXOffset = xOffset;
235 currentYOffset += this->textureSpec->rows;
236
237 if(64 <= currentYOffset)
238 {
239 NM_ASSERT(false, "BgmapTexture::writeAllFrames: no more space");
240 return;
241 }
242 }
243 }
244}

249static inline void BgmapTexture::addHWORD
250(
251 HWORD* destination, const HWORD* source, uint32 numberOfHWORDS, uint32 offset, uint16 flip, int16 increment
252)
253{
254#ifdef __SHOW_SPRITES_PROFILING
255 extern int32 _writtenTextureTiles;
256 _writtenTextureTiles += numberOfHWORDS;
257#endif
258
259 const HWORD* finalSource = source + numberOfHWORDS;
260
261 asm
262 (
263 "jr end%= \n\t" \
264 "loop%=: \n\t" \
265 "ld.h 0[%1], r10 \n\t" \
266 "xor %4, r10 \n\t" \
267 "add %3, r10 \n\t" \
268 "st.h r10, 0[%0] \n\t" \
269 "add %5, %0 \n\t" \
270 "add 2, %1 \n\t" \
271 "end%=: \n\t" \
272 "cmp %1, %2 \n\t" \
273 "bgt loop%= \n\t"
274 : /*"+r" (destination), "+r" (source)*/ // <- causes the graphics not to be writen
275 : "r" (destination), "r" (source), "r" (finalSource), "r" (offset), "r" (flip), "r" (increment)
276 : "r10"
277 );
278}

309void BgmapTexture::writeFrame
310(
311 int16 maximumTextureRowsToWrite, bool forceFullRewrite, int16 xOffset, int16 yOffset, uint16 charSetOffset, uint16 frame
312)
313{
314 if((0 > xOffset) || (0 > yOffset))
315 {
316 return;
317 }
318
319 int16 cols = this->textureSpec->cols;
320 int16 rows = this->textureSpec->rows;
321 HWORD* offsetDisplacement = (HWORD*)__BGMAP_SEGMENT(this->segment) + xOffset + (yOffset << 6);
322 const HWORD* mapDisplacement = (HWORD*)this->textureSpec->map + this->mapDisplacement + cols * rows * frame;
323 int32 counter = forceFullRewrite ? -1 : maximumTextureRowsToWrite;
324 uint16 flip = ((this->horizontalFlip << 1) | this->verticalFlip) << 12;
325 int8 remainingRowsToBeWritten = this->remainingRowsToBeWritten;
326 uint16 offset = charSetOffset | (this->palette << 14);
327
328 if(forceFullRewrite)
329 {
330 CACHE_RESET;
331 }
332
333 if(this->horizontalFlip)
334 {
335 if(this->verticalFlip)
336 {
337 //put the map into memory calculating the number of char for each reference
338 for(; 0 != counter && 0 != remainingRowsToBeWritten--; counter--)
339 {
340 BgmapTexture::addHWORD
341 (
342 (HWORD*)offsetDisplacement + ((remainingRowsToBeWritten) << 6) + cols - 1,
343 mapDisplacement + ((rows - remainingRowsToBeWritten - 1) * cols),
344 cols,
345 offset,
346 flip,
347 -2
348 );
349 }
350 }
351 else
352 {
353 //put the map into memory calculating the number of char for each reference
354 for(; 0 != counter && 0 != remainingRowsToBeWritten--; counter--)
355 {
356 BgmapTexture::addHWORD
357 (
358 (HWORD*)offsetDisplacement + (remainingRowsToBeWritten << 6) + cols - 1,
359 mapDisplacement + ((remainingRowsToBeWritten) * cols),
360 cols,
361 offset,
362 flip,
363 -2
364 );
365 }
366 }
367 }
368 else
369 {
370 if(this->verticalFlip)
371 {
372 //put the map into memory calculating the number of char for each reference
373 for(; 0 != counter && 0 != remainingRowsToBeWritten--; counter--)
374 {
375 BgmapTexture::addHWORD
376 (
377 (HWORD*)offsetDisplacement + (remainingRowsToBeWritten << 6),
378 mapDisplacement + ((rows - remainingRowsToBeWritten - 1) * cols),
379 cols,
380 offset,
381 flip,
382 2
383 );
384 }
385 }
386 else
387 {
388 //put the map into memory calculating the number of char for each reference
389 for(; 0 != counter && 0 != remainingRowsToBeWritten--; counter--)
390 {
391 BgmapTexture::addHWORD
392 (
393 (HWORD*)offsetDisplacement + (remainingRowsToBeWritten << 6),
394 mapDisplacement + ((remainingRowsToBeWritten) * cols),
395 cols,
396 offset,
397 flip,
398 2
399 );
400 }
401 }
402 }
403
404 this->remainingRowsToBeWritten = remainingRowsToBeWritten;
405
406 if(0 < this->textureSpec->padding.rows && -1 == this->remainingRowsToBeWritten)
407 {
408 BgmapTexture::addHWORD
409 (
410 (HWORD*)offsetDisplacement + (rows << 6),
411 (const HWORD*)_emptyTextureRow,
412 cols,
413 0,
414 false,
415 2
416 );
417 }
418}
