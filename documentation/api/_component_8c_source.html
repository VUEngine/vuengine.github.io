42void Component::constructor(Entity owner, const ComponentSpec* componentSpec)
43{
44 // Always explicitly call the base's constructor
45 Base::constructor();
46
47#ifndef __SHIPPING
48 extern uint32 _textStart __attribute__((unused));
49 extern uint32 _dataLma __attribute__((unused));
50
51 if(NULL != componentSpec && !(&_textStart < (uint32*)componentSpec && (uint32*)componentSpec < &_dataLma))
52 {
53 Printing::setDebugMode();
54 Printing::clear();
55 Printing::text(__GET_CLASS_NAME(this), 1, 25, NULL);
56 Printing::hex((WORD)componentSpec, 1, 26, 8, NULL);
57 NM_ASSERT(false, "Component::constructor: the provided spec lives in WRAM");
58 }
59#endif
60
61 this->componentSpec = componentSpec;
62 this->owner = owner;
63 this->deleteMe = false;
64
65 if(isDeleted(this->owner))
66 {
67 this->owner = NULL;
68 this->transformation = &_dummyTransformation;
69 }
70 else
71 {
72 this->transformation = Entity::getTransformation(this->owner);
73 }
74}
