44void MBgmapSprite::constructor(Entity owner, const MBgmapSpriteSpec* mBgmapSpriteSpec)
45{
46 // Always explicitly call the base's constructor
47 Base::constructor(owner, &mBgmapSpriteSpec->bgmapSpriteSpec);
48
49 this->checkIfWithinScreenSpace = false;
50
51 if(!isDeleted(this->texture))
52 {
53 Texture::release(this->texture);
54 this->texture = NULL;
55 }
56
57 this->textures = NULL;
58 MBgmapSprite::loadTextures(this);
59 MBgmapSprite::calculateSize(this);

61 if(0 == (__WORLD_BGMAP & this->head) && 0 == this->param)
62 {

79int16 MBgmapSprite::doRender(int16 index)
80{
81 NM_ASSERT(!isDeleted(this->texture), "MBgmapSprite::doRender: null texture");
82
83 BgmapTextureSource bgmapTextureSource = this->bgmapTextureSource;
84
85 PixelVector position = this->position;
86
87 if(((MBgmapSpriteSpec*)this->componentSpec)->xLoop)
88 {
89 bgmapTextureSource.mx = -this->position.x;
90 position.x = 0;
91 }
92 else
93 {
94 bgmapTextureSource.mx = this->textureXOffset;
95 }
96
97 if(((MBgmapSpriteSpec*)this->componentSpec)->yLoop)
98 {
99 bgmapTextureSource.my = -this->position.y;
100 position.y = 0;
101 }
102 else
103 {
104 bgmapTextureSource.my = this->textureYOffset;
105 }
106
107 WorldAttributes* worldPointer = &_worldAttributesCache[index];
108
109 // Get coordinates
110 int16 gx = position.x - this->halfWidth;
111 int16 gy = position.y - this->halfHeight;
112 int16 gp = position.parallax;
113
114 int32 mxDisplacement = 0;
115 if(_cameraFrustum->x0 > gx)
116 {
117 mxDisplacement = _cameraFrustum->x0 - gx;
118 gx = _cameraFrustum->x0;
119 }
120
121 int32 myDisplacement = 0;
122 if(_cameraFrustum->y0 > gy)
123 {
124 myDisplacement = _cameraFrustum->y0 - gy;
125 gy = _cameraFrustum->y0;
126 }
127
128 int16 mx = bgmapTextureSource.mx + mxDisplacement;
129 int16 my = bgmapTextureSource.my + myDisplacement;
130 int16 mp = bgmapTextureSource.mp;
131
132 int16 w = 0;
133 int16 h = 0;
134
135 // Set the world size
136 if(!((MBgmapSpriteSpec*)this->componentSpec)->xLoop)
137 {
138 w = (this->halfWidth << 1) - mxDisplacement;
139
140 if(w + gx >= _cameraFrustum->x1)
141 {
142 w = _cameraFrustum->x1 - gx;
143 }
144
145 if(0 >= w)
146 {
147 return __NO_RENDER_INDEX;
148 }
149 }
150 else
151 {
152 mp = - gp;
153 gp = 0;
154 gx = _cameraFrustum->x0;
155 w = _cameraFrustum->x1 - _cameraFrustum->x0 - __WORLD_SIZE_DISPLACEMENT;
156 }
157
158 if(!((MBgmapSpriteSpec*)this->componentSpec)->yLoop)
159 {
160 h = (this->halfHeight << 1) - myDisplacement;
161
162 if(h + gy >= _cameraFrustum->y1)
163 {
164 h = _cameraFrustum->y1 - gy;
165 }
166
167#ifdef __HACK_BGMAP_SPRITE_HEIGHT
168 if (__MINIMUM_BGMAP_SPRITE_HEIGHT >= h)
169 {
170 if (0 >= h)
171 {
172 return __NO_RENDER_INDEX;
173 }
174
175 my -= __MINIMUM_BGMAP_SPRITE_HEIGHT - h;
176 }
177#else
178 if (0 >= h)
179 {
180 return __NO_RENDER_INDEX;
181 }
182#endif
183 }
184 else
185 {
186 h = _cameraFrustum->y1 - myDisplacement - __WORLD_SIZE_DISPLACEMENT;
187
188 if(0 > h || h + gy >= _cameraFrustum->y1)
189 {
190 h = _cameraFrustum->y1 - gy - __WORLD_SIZE_DISPLACEMENT;
191 }
192 }
193
194 worldPointer->gx = gx;
195 worldPointer->gy = gy;
196 worldPointer->gp = gp;
197
198 worldPointer->mx = mx - this->displacement.x;
199 worldPointer->my = my - this->displacement.y;
200 worldPointer->mp = mp - this->displacement.parallax;
201
202 worldPointer->w = w - __WORLD_SIZE_DISPLACEMENT;
203 worldPointer->h = h - __WORLD_SIZE_DISPLACEMENT;
204
205 worldPointer->head = this->head | (BgmapTexture::safeCast(this->texture))->segment | ((MBgmapSpriteSpec*)this->componentSpec)->scValue;
206
207 if(0 < this->param)
208 {
209 worldPointer->param = (uint16)((((this->param)) - 0x20000) >> 1) & 0xFFF0;
210 }
211
212 return index;
213}

223void MBgmapSprite::loadTextures()
224{
225 if(NULL != ((MBgmapSpriteSpec*)this->componentSpec))
226 {
227 if(NULL == this->texture && NULL == this->textures)
228 {
229 this->textures = new VirtualList();
230
231 for(int32 i = 0; NULL != ((MBgmapSpriteSpec*)this->componentSpec)->textureSpecs[i]; i++)
232 {
233 MBgmapSprite::loadTexture(this, ((MBgmapSpriteSpec*)this->componentSpec)->textureSpecs[i], 0 == i && ((MBgmapSpriteSpec*)this->componentSpec)->textureSpecs[i + 1]);
234 }
235
236 this->textureXOffset = BgmapTexture::getXOffset(this->texture) << 3;
237 this->textureYOffset = BgmapTexture::getYOffset(this->texture) << 3;
238 }
239 else
240 {
241 NM_ASSERT(this->texture, "MBgmapSprite::loadTextures: textures already loaded");
242 }
243 }
244}

248void MBgmapSprite::loadTexture(TextureSpec* textureSpec, bool isFirstTextureAndHasMultipleTextures)
249{
250 ASSERT(textureSpec, "MBgmapSprite::loadTexture: null textureSpec");
251
252 int16 minimumSegment = 0;
253
254 if(VirtualList::getCount(this->textures))
255 {
256 BgmapTexture bgmapTexture = BgmapTexture::safeCast(VirtualList::back(this->textures));
257
258 minimumSegment = BgmapTexture::getSegment(bgmapTexture);
259
260 // This allows to have bgmaps sprites that are smaller than 512 pixels high
261 // But depends on all the segments being free
262 if(((MBgmapSpriteSpec*)this->componentSpec)->xLoop && 64 > Texture::getRows(bgmapTexture))
263 {
264 minimumSegment += 1;
265 }
266 }
267
268 BgmapTexture bgmapTexture =
269 BgmapTexture::safeCast
270 (
271 Texture::get
272 (
273 typeofclass(BgmapTexture), textureSpec, minimumSegment,
274 isFirstTextureAndHasMultipleTextures, ((MBgmapSpriteSpec*)this->componentSpec)->scValue
275 )
276 );
277
278 NM_ASSERT(!isDeleted(bgmapTexture), "MBgmapSprite::loadTexture: texture not loaded");
279 NM_ASSERT(!isDeleted(this->textures), "MBgmapSprite::loadTexture: null textures list");
280 NM_ASSERT
281 (
282 !isFirstTextureAndHasMultipleTextures || 0 == (BgmapTexture::getSegment(bgmapTexture) % 2),
283 "MBgmapSprite::loadTexture: first texture not loaded in even segment"
284 );
285
286 if(!isDeleted(bgmapTexture))
287 {
288 BgmapTexture::addEventListener
289 (
290 bgmapTexture, ListenerObject::safeCast(this), (EventListener)BgmapSprite::onTextureRewritten, kEventTextureRewritten
291 );
292
293 VirtualList::pushBack(this->textures, bgmapTexture);
294
295 this->texture = Texture::safeCast(bgmapTexture);
296 NM_ASSERT(this->texture, "MBgmapSprite::loadTexture: null texture");
297 }
298}

302void MBgmapSprite::releaseTextures()
303{
304 // Free the texture
305 if(!isDeleted(this->texture))
306 {
307 // If affine or bgmap
308 if(((__WORLD_AFFINE | __WORLD_HBIAS) & this->head) && this->param)
309 {
310 // Free param table space
311 ParamTableManager::free(BgmapSprite::safeCast(this));
312 }
313 }
314
315 this->texture = NULL;
316
317 if(!isDeleted(this->textures))
318 {
319 VirtualNode node = this->textures->head;
320
321 for(; NULL != node; node = node->next)
322 {
323 BgmapTexture bgmapTexture = BgmapTexture::safeCast(node->data);
324
325 if(!isDeleted(bgmapTexture))
326 {
327 BgmapTexture::removeEventListener
328 (
329 bgmapTexture, ListenerObject::safeCast(this), (EventListener)BgmapSprite::onTextureRewritten, kEventTextureRewritten
330 );
331
332 Texture::release(Texture::safeCast(bgmapTexture));
333 }
334 }
335
336 delete this->textures;
337 this->textures = NULL;
338 }
339}

343void MBgmapSprite::calculateSize()
344{
345 VirtualNode node = this->textures->head;
346
347 int32 cols = 0;
348 int32 rows = 0;
349
350 for(; NULL != node; node = node->next)
351 {
352 // Free the texture
353 int32 textureCols = (Texture::safeCast(node->data))->textureSpec->cols;
354 int32 textureRows = (Texture::safeCast(node->data))->textureSpec->rows;
355
356 if(cols < textureCols)
357 {
358 cols = textureCols;
359 }
360
361 if(rows < textureRows)
362 {
363 rows = textureRows;
364 }
365 }
366
367 this->halfWidth = cols << 2;
368 this->halfHeight = rows << 2;
369
370 if(0 < ((MBgmapSpriteSpec*)this->componentSpec)->width)
371 {
372 this->halfWidth = ((MBgmapSpriteSpec*)this->componentSpec)->width >> 1;
373 }
374
375 if(0 < ((MBgmapSpriteSpec*)this->componentSpec)->height)
376 {
377 this->halfHeight = ((MBgmapSpriteSpec*)this->componentSpec)->height >> 1;
378 }
379}
