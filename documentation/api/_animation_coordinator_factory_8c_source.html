42AnimationCoordinator AnimationCoordinatorFactory::getCoordinator(AnimationController animationController, ListenerObject scope, const CharSetSpec* charSetSpec)
43{
44 NM_ASSERT(NULL != charSetSpec, "AnimationCoordinatorFactory::getCoordinator: null charSetSpec");
45
46 if(NULL != charSetSpec && charSetSpec->shared)
47 {
48 // Try to find an already created coordinator
49 for(VirtualNode node = this->animationCoordinators->head; NULL != node; node = node->next)
50 {
51 AnimationCoordinator animationCoordinator = AnimationCoordinator::safeCast(node->data);
52
53 if(AnimationCoordinator::getCharSetSpec(animationCoordinator) == charSetSpec)
54 {
55 AnimationCoordinator::addAnimationController(animationCoordinator, animationController);
56 return animationCoordinator;
57 }
58 }
59
60 AnimationCoordinator animationCoordinator = new AnimationCoordinator(charSetSpec, scope);
61
62 // Create a new coordinator
63 AnimationCoordinator::addAnimationController(animationCoordinator, animationController);
64
65 VirtualList::pushBack(this->animationCoordinators, animationCoordinator);
66
67 return animationCoordinator;
68 }
69
70 return NULL;
71}
