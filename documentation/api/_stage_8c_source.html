105static uint32 Stage::computeDistanceToOrigin(StageActorDescription* stageActorDescription)
106{
107
108 int32 x =
109 stageActorDescription->positionedActor->onScreenPosition.x -
110 __METERS_TO_PIXELS(stageActorDescription->rightBox.x1 - stageActorDescription->rightBox.x0) / 2;
111
112 int32 y =
113 stageActorDescription->positionedActor->onScreenPosition.y -
114 __METERS_TO_PIXELS(stageActorDescription->rightBox.y1 - stageActorDescription->rightBox.y0) / 2;
115
116 int32 z =
117 stageActorDescription->positionedActor->onScreenPosition.z -
118 __METERS_TO_PIXELS(stageActorDescription->rightBox.z1 - stageActorDescription->rightBox.z0) / 2;
119
120 return x * x + y * y + z * z;
121}

156void Stage::destructor()
157{
158 this->deleteMe = true;
159
160 Stage::setFocusActor(this, NULL);
161
162 if(!isDeleted(this->actorLoadingListeners))
163 {
164 VirtualList::deleteData(this->actorLoadingListeners);
165 delete this->actorLoadingListeners;
166 this->actorLoadingListeners = NULL;
167 }
168
169 if(!isDeleted(this->sounds))
170 {
171 // Do not need to release sound wrappers here,
172 // They are taken care by the SoundManager when
173 // I called SoundManager::stopAllSounds
174 for(VirtualNode node = this->sounds->head; NULL != node; node = node->next)
175 {
176 Sound sound = Sound::safeCast(node->data);
177
178 if(!isDeleted(sound))
179 {
180 Sound::removeEventListenerScopes(sound, ListenerObject::safeCast(this), kEventSoundReleased);
181 }
182 }
183
184 delete this->sounds;
185 this->sounds = NULL;
186 }
187
188 if(!isDeleted(this->actorFactory))
189 {
190 delete this->actorFactory;
191 this->actorFactory = NULL;
192 }
193
194 if(!isDeleted(this->stageActorDescriptions))
195 {
196 VirtualList::deleteData(this->stageActorDescriptions);
197 delete this->stageActorDescriptions;
198 this->stageActorDescriptions = NULL;
199 }
200
201 // Always explicitly call the base's destructor
202 Base::destructor();
203}

207void Stage::suspend()
208{
209 // Save the camera position for resume reconfiguration
210 this->cameraTransformation = Camera::getTransformation(Camera::getInstance());
211
212 // Stream all pending actors to avoid having manually recover
213 // The stage actor registries
214 while(ActorFactory::createNextActor(this->actorFactory));
215
216 Base::suspend(this);
217
218 // Relinquish camera focus priority
219 if(!isDeleted(this->focusActor))
220 {
221 if(this->focusActor == Camera::getFocusActor(Camera::getInstance()))
222 {
223 // Relinquish focus actor
224 Camera::setFocusActor(Camera::getInstance(), NULL);
225 }
226 }
227 else
228 {
229 Stage::setFocusActor(this, Camera::getFocusActor(Camera::getInstance()));
230 }
231
232 delete this->actorFactory;
233 this->actorFactory = NULL;
234}

282void Stage::registerActors(VirtualList positionedActorsToIgnore)
283{
284 if(!isDeleted(this->stageActorDescriptions))
285 {
286 return;
287 }
288
289 this->stageActorDescriptions = new VirtualList();
290
291 // Register actors ordering them according to their distances to the origin
292 int32 i = 0;
293
294 for(;this->stageSpec->actors.children[i].actorSpec; i++)
295 {
296 if(positionedActorsToIgnore)
297 {
298 VirtualNode node = positionedActorsToIgnore->head;
299
300 for(; NULL != node; node = node->next)
301 {
302 if(&this->stageSpec->actors.children[i] == (PositionedActor*)node->data)
303 {
304 break;
305 }
306 }
307
308 if(node)
309 {
310 continue;
311 }
312 }
313
314 StageActorDescription* stageActorDescription = Stage::registerActor(this, &this->stageSpec->actors.children[i]);
315
316 Vector3D stageActorPosition = (Vector3D)
317 {
318 (__PIXELS_TO_METERS(stageActorDescription->positionedActor->onScreenPosition.x) -
319 (stageActorDescription->rightBox.x1 - stageActorDescription->rightBox.x0) / 2),
320 (__PIXELS_TO_METERS(stageActorDescription->positionedActor->onScreenPosition.y) -
321 (stageActorDescription->rightBox.y1 - stageActorDescription->rightBox.y0) / 2),
322 (__PIXELS_TO_METERS(stageActorDescription->positionedActor->onScreenPosition.z) -
323 (stageActorDescription->rightBox.z1 - stageActorDescription->rightBox.z0) / 2)
324 };
325
326 VirtualNode closestEnitryDescriptionNode = NULL;
327 VirtualNode auxNode = this->stageActorDescriptions->head;
328 StageActorDescription* auxStageActorDescription = (StageActorDescription*)auxNode->data;
329
330 fixed_ext_t closestDistance = 0;
331
332 for(; auxNode; auxNode = auxNode->next)
333 {
334 auxStageActorDescription = (StageActorDescription*)auxNode->data;
335
336 Vector3D auxStageActorPosition = (Vector3D)
337 {
338 (
339 __PIXELS_TO_METERS(auxStageActorDescription->positionedActor->onScreenPosition.x) -
340 (auxStageActorDescription->rightBox.x1 - auxStageActorDescription->rightBox.x0) / 2
341 ),
342 (
343 __PIXELS_TO_METERS(auxStageActorDescription->positionedActor->onScreenPosition.y) -
344 (auxStageActorDescription->rightBox.y1 - auxStageActorDescription->rightBox.y0) / 2
345 ),
346 (
347 __PIXELS_TO_METERS(auxStageActorDescription->positionedActor->onScreenPosition.z) -
348 (auxStageActorDescription->rightBox.z1 - auxStageActorDescription->rightBox.z0) / 2
349 )
350 };
351
352 fixed_ext_t squaredDistance = Vector3D::squareLength(Vector3D::get(stageActorPosition, auxStageActorPosition));
353
354 if(NULL == closestEnitryDescriptionNode || closestDistance > squaredDistance)
355 {
356 closestEnitryDescriptionNode = auxNode;
357 closestDistance = squaredDistance;
358 }
359 }
360
361 if(NULL == auxNode)
362 {
363 VirtualList::pushBack(this->stageActorDescriptions, stageActorDescription);
364 }
365 else
366 {
367 uint32 stageActorDistanceToOrigin = Stage::computeDistanceToOrigin(stageActorDescription);
368 uint32 auxStageActorDistanceToOrigin = Stage::computeDistanceToOrigin(auxStageActorDescription);
369
370 if(stageActorDistanceToOrigin > auxStageActorDistanceToOrigin)
371 {
372 VirtualList::insertAfter(this->stageActorDescriptions, auxNode, stageActorDescription);
373 }
374 else
375 {
376 VirtualList::insertBefore(this->stageActorDescriptions, auxNode, stageActorDescription);
377 }
378 }
379 }
380}

418
419void Stage::destroyChildActor(Actor child)
420{
421 NM_ASSERT(!isDeleted(child), "Stage::removeActor: null child");
422
423 if(isDeleted(child))
424 {
425 return;
426 }
427
428 int16 internalId = Actor::getInternalId(child);
429
430 Stage::removeChild(this, Container::safeCast(child), true);
431
432 VirtualNode node = this->stageActorDescriptions->head;
433
434 for(; NULL != node; node = node->next)
435 {
436 StageActorDescription* stageActorDescription = (StageActorDescription*)node->data;
437
438 if(stageActorDescription->internalId == internalId)
439 {
440 stageActorDescription->internalId = -1;
441 break;
442 }
443 }
444
445 if(NULL != node)
446 {
447 if(this->streamingHeadNode == node)
448 {
449 this->streamingHeadNode = this->reverseStreaming ? this->streamingHeadNode->next : this->streamingHeadNode->previous;
450 }
451
452 delete node->data;
453 VirtualList::removeNode(this->stageActorDescriptions, node);
454 }
455}

459void Stage::streamAll()
460{
461 Stage::transform(this, &_neutralEnvironmentTransformation, __INVALIDATE_TRANSFORMATION);
462
463 this->streamingHeadNode = NULL;
464
465 // Make sure that the actor factory doesn't have any pending operations
466 while(ActorFactory::createNextActor(this->actorFactory));
467
468 do
469 {
470 this->streamingHeadNode = NULL;
471
472 Stage::purgeChildren(this);
473
474 } while(Stage::unloadOutOfRangeActors(this, false));
475
476 this->streamingHeadNode = NULL;
477 this->streamingAmplitude = (uint16)-1;
478
479 while(Stage::loadInRangeActors(this, false));
480
481 while(ActorFactory::createNextActor(this->actorFactory))
482 {
483 Stage::transform(this, &_neutralEnvironmentTransformation, __INVALIDATE_TRANSFORMATION);
484 }
485
486 this->streamingAmplitude = this->stageSpec->streaming.streamingAmplitude;
487 this->streamingHeadNode = NULL;
488}

570bool Stage::stream()
571{
572 if(NULL == this->stageActorDescriptions->head)
573 {
574 return false;
575 }
576
577 if(Stage::updateActorFactory(this))
578 {
579 return true;
580 }
581
582 int32 streamingPhases = sizeof(_streamingPhases) / sizeof(StreamingPhase);
583
584 if(++this->streamingPhase >= streamingPhases)
585 {
586 this->streamingPhase = 0;
587 }
588
589 return _streamingPhases[this->streamingPhase](this, this->stageSpec->streaming.deferred);
590}

620//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
621
622bool Stage::unloadOutOfRangeActors(int32 defer __attribute__((unused)))
623{
624 if(isDeleted(this->children))
625 {
626 return false;
627 }
628
629#ifdef __PROFILE_STREAMING
630 _renderingProcessTimeHelper = 0;
631 timeBeforeProcess = TimerManager::getElapsedMilliseconds();
632#endif
633
634 bool unloadedActors = false;
635
636 VirtualNode node = this->children->head;
637
638 CACHE_RESET;
639
640 // Check which entites must be unloaded
641 for(; NULL != node; node = node->next)
642 {
643 // Get next actor
644 Actor actor = Actor::safeCast(node->data);
645
646 if(actor->dontStreamOut)
647 {
648 continue;
649 }
650
651 // If the actor isn't visible inside the view field, unload it
652 if(!actor->deleteMe && actor->parent == Container::safeCast(this))
653 {
654 if(Actor::isInCameraRange(actor, this->stageSpec->streaming.loadPadding + this->stageSpec->streaming.unloadPadding, true))
655 {
656 continue;
657 }
658
659 VirtualNode auxNode = this->stageActorDescriptions->head;
660 StageActorDescription* stageActorDescription = NULL;
661
662 for(; auxNode; auxNode = auxNode->next)
663 {
664 StageActorDescription* auxStageActorDescription = (StageActorDescription*)auxNode->data;
665
666 if(auxStageActorDescription->internalId == actor->internalId)
667 {
668 stageActorDescription = auxStageActorDescription;
669 break;
670 }
671 }
672
673 if(NULL != stageActorDescription)
674 {
675 if(stageActorDescription->positionedActor->loadRegardlessOfPosition)
676 {
677 continue;
678 }
679
680 stageActorDescription->internalId = -1;
681 }
682
683 // Unload it
684 Stage::destroyChildActor(this, actor);
685
686 // Remove from list of actors that are to be loaded by the streaming,
687 // If the actor is not to be alwaysStreamIned
688 if(!Actor::alwaysStreamIn(actor))
689 {
690 VirtualList::removeNode(this->stageActorDescriptions, auxNode);
691 }
692
693 unloadedActors = true;
694 }
695 }
696
697#ifdef __PROFILE_STREAMING
698 uint32 processTime =
699 -_renderingProcessTimeHelper + TimerManager::getElapsedMilliseconds() - timeBeforeProcess;
700
701 unloadOutOfRangeActorsHighestTime =
702 processTime > unloadOutOfRangeActorsHighestTime ? processTime : unloadOutOfRangeActorsHighestTime;
703#endif
704
705 return unloadedActors;
706}

708//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
709
710bool Stage::loadInRangeActors(int32 defer)
711{
712#ifdef __PROFILE_STREAMING
713 _renderingProcessTimeHelper = 0;
714 timeBeforeProcess = TimerManager::getElapsedMilliseconds();
715#endif
716
717 bool loadedActors = false;
718
719 CACHE_RESET;
720
721 if(this->reverseStreaming)
722 {
723 if(NULL == this->streamingHeadNode)
724 {
725 this->streamingHeadNode = this->stageActorDescriptions->tail;
726 }
727
728 bool negativeStreamingAmplitude = 0 > ((int16)this->streamingAmplitude);
729
730 for
731 (
732 uint16 counter = 0; counter < this->streamingAmplitude;
733 this->streamingHeadNode = this->streamingHeadNode->previous,
734 counter++
735 )
736 {
737 if(NULL == this->streamingHeadNode)
738 {
739 this->streamingHeadNode = this->stageActorDescriptions->tail;
740
741 if(negativeStreamingAmplitude)
742 {
743 break;
744 }
745 }
746
747 StageActorDescription* stageActorDescription = (StageActorDescription*)this->streamingHeadNode->data;
748
749 if(0 > stageActorDescription->internalId)
750 {
751 // If actor in load range
752 if
753 (
754 Stage::isActorInLoadRange
755 (
756 this, stageActorDescription->positionedActor->onScreenPosition, stageActorDescription->validRightBox ?
757 &stageActorDescription->rightBox : NULL
758 )
759 )
760 {
761 loadedActors = true;
762
763 stageActorDescription->internalId = this->nextActorId++;
764
765 if(defer)
766 {
767 ActorFactory::spawnActor
768 (
769 this->actorFactory, stageActorDescription->positionedActor, Container::safeCast(this),
770 !isDeleted(this->actorLoadingListeners) ?
771 (EventListener)Stage::onActorLoaded
772 :
773 NULL, stageActorDescription->internalId
774 );
775 }
776 else
777 {
778 Stage::doAddChildActor
779 (
780 this, stageActorDescription->positionedActor, false, stageActorDescription->internalId
781 );
782 break;
783 }
784 }
785 }
786 }
787 }
788 else
789 {
790 if(NULL == this->streamingHeadNode)
791 {
792 this->streamingHeadNode = this->stageActorDescriptions->head;
793 }
794
795 bool negativeStreamingAmplitude = 0 > ((int16)this->streamingAmplitude);
796
797 for(uint16 counter = 0; counter < this->streamingAmplitude; this->streamingHeadNode = this->streamingHeadNode->next, counter++)
798 {
799 if(NULL == this->streamingHeadNode)
800 {
801 this->streamingHeadNode = this->stageActorDescriptions->head;
802
803 if(negativeStreamingAmplitude)
804 {
805 break;
806 }
807 }
808
809 StageActorDescription* stageActorDescription = (StageActorDescription*)this->streamingHeadNode->data;
810
811 if(0 > stageActorDescription->internalId)
812 {
813 // If actor in load range
814 if
815 (
816 Stage::isActorInLoadRange
817 (
