52static Component ComponentManager::createComponent(Entity owner, const ComponentSpec* componentSpec)
53{
54 if(NULL == componentSpec)
55 {
56 return NULL;
57 }
58
59 ComponentManager componentManager = ComponentManager::getManager(componentSpec->componentType);
60
61 if(NULL == componentManager)
62 {
63 return NULL;
64 }
65
66 Component component = ComponentManager::instantiateComponent(componentManager, owner, componentSpec);
67
68 if(!isDeleted(component) && !isDeleted(owner))
69 {
70 Entity::addedComponent(owner, component);
71 }
72
73 return component;
74}

78static void ComponentManager::destroyComponent(Entity owner, Component component)
79{
80 if(isDeleted(component))
81 {
82 return;
83 }
84
85 if(owner != component->owner)
86 {
87 return;
88 }
89
90 uint32 componentType = ComponentManager::getComponentType(component);
91
92 if(kComponentTypes <= componentType)
93 {
94 return;
95 }
96
97 ComponentManager componentManager = ComponentManager::getManager(componentType);
98
99 if(NULL == componentManager)
100 {
101 return;
102 }
103
104 if(!isDeleted(owner))
105 {
106 Entity::removedComponent(owner, component);
107 }
108
109 ComponentManager::deinstantiateComponent(componentManager, owner, component);
110}

167static void ComponentManager::removeComponents(Entity owner, uint32 componentType)
168{
169 void removeComponents(ComponentManager componentManager)
170 {
171 if(NULL == componentManager)
172 {
173 return;
174 }
175
176 for(VirtualNode node = componentManager->components->head, nextNode = NULL; NULL != node; node = nextNode)
177 {
178 nextNode = node->next;
179
180 Component component = Component::safeCast(node->data);
181
182 if(!component->deleteMe && owner == component->owner)
183 {
184 ComponentManager::removeComponent(owner, component);
185 }
186 }
187 }
188
189 if(kComponentTypes <= componentType)
190 {
191 for(int16 i = 0; i < kComponentTypes; i++)
192 {
193 removeComponents(ComponentManager::getManager(i));
194 }
195 }
196 else
197 {
198 removeComponents(ComponentManager::getManager(componentType));
199 }
200}

236static VirtualList ComponentManager::getComponents(Entity owner, uint32 componentType)
237{
238 if(NULL == owner->components)
239 {
240 owner->components =
241 (VirtualList*)
242 (
243 (uint32)MemoryPool::allocate(sizeof(VirtualList) * kComponentTypes + __DYNAMIC_STRUCT_PAD) + __DYNAMIC_STRUCT_PAD
244 );
245
246 for(int16 i = 0; i < kComponentTypes; i++)
247 {
248 owner->components[i] = NULL;
249 }
250 }
251
252 if(NULL == owner->components[componentType])
253 {
254 owner->components[componentType] = new VirtualList();
255 }
256 else
257 {
258 return owner->components[componentType];
259 }
260
261 ComponentManager componentManager = ComponentManager::getManager(componentType);
262
263 if(NULL == componentManager)
264 {
265 return NULL;
266 }
267
268 for(VirtualNode node = componentManager->components->head; NULL != node; node = node->next)
269 {
270 Component component = Component::safeCast(node->data);
271
272 if(owner == component->owner)
273 {
274 VirtualList::pushBack(owner->components[componentType], component);
275 }
276 }
277
278 return owner->components[componentType];

283static bool ComponentManager::getComponentsOfClass(Entity owner, ClassPointer classPointer, VirtualList components, uint32 componentType)
284{
285 if(kComponentTypes <= componentType)
286 {
287 return false;
288 }
289
290 ComponentManager componentManager = ComponentManager::getManager(componentType);
291
292 if(NULL == componentManager)
293 {
294 return false;
295 }
296
297 for(VirtualNode node = componentManager->components->head; NULL != node; node = node->next)
298 {
299 Component component = Component::safeCast(node->data);
300
301 if(owner != component->owner || component->deleteMe)
302 {
303 continue;
304 }
305
306 if(!classPointer || Object::getCast(component, classPointer, NULL))
307 {
308 VirtualList::pushBack(components, component);
309 }
310 }
311
312 if(NULL != components->head)
313 {
314 return true;
315 }
316
317 return false;
318}

322static uint16 ComponentManager::getComponentsCount(Entity owner, uint32 componentType)
323{
324 uint16 getCount(ComponentManager componentManager)
325 {
326 if(NULL == componentManager)
327 {
328 return 0;
329 }
330
331 uint16 count = 0;
332
333 for(VirtualNode node = componentManager->components->head, nextNode = NULL; NULL != node; node = nextNode)
334 {
335 nextNode = node->next;
336
337 Component component = Component::safeCast(node->data);
338
339 if(owner == component->owner && !component->deleteMe)
340 {
341 count++;
342 }
343 }
344
345 return count;
346 }
347
348 uint16 count = 0;
349
350 if(kComponentTypes <= componentType)
351 {
352 for(int16 i = 0; i < kComponentTypes; i++)
353 {
354 count += getCount(ComponentManager::getManager(i));
355 }
356 }
357 else
358 {
359 count = getCount(ComponentManager::getManager(componentType));
360 }
361
362 return count;
363}

367static void ComponentManager::propagateCommand(int32 command, Entity owner, uint32 componentType, ...)
368{
369
370 void propagateCommand(ComponentManager componentManager, va_list args)
371 {
372 for(VirtualNode node = componentManager->components->head; NULL != node; node = node->next)
373 {
374 Component component = Component::safeCast(node->data);
375
376 if(NULL != owner && owner != component->owner)
377 {
378 continue;
379 }
380
381 Component::handleCommand(component, command, args);
382 }
383 }
384
385 if(kComponentTypes <= componentType)
386 {
387 for(int16 i = 0; i < kComponentTypes; i++)
388 {
389 va_list args;
390 va_start(args, componentType);
391
392 propagateCommand(ComponentManager::getManager(i), args);
393
394 va_end(args);
395 }
396 }
397 else
398 {
399 va_list args;
400 va_start(args, componentType);
401
402 propagateCommand(ComponentManager::getManager(componentType), args);
403
404 va_end(args);
405 }
406}

443static bool ComponentManager::calculateRightBox(Entity owner, RightBox* rightBox)
444{
445 bool modified = false;
446
447 if(NULL == owner)
448 {
449 return false;
450 }
451
452 for(int16 i = 0; i < kComponentTypes; i++)
453 {
454 ComponentManager componentManager = ComponentManager::getManager(i);
455
456 if(NULL == componentManager || !ComponentManager::overrides(componentManager, isAnyVisible))
457 {
458 continue;
459 }
460
461 modified |= ComponentManager::getRightBoxFromComponents(componentManager, owner, rightBox);
462 }
463
464 return modified;

502static ComponentManager ComponentManager::getManager(uint32 componentType)
503{
504 if(kComponentTypes <= componentType)
505 {
506 return NULL;
507 }
508
509 switch (componentType)
510 {
511 case kSpriteComponent:
512
513 return ComponentManager::safeCast(SpriteManager::getInstance());
514 break;
515
516 case kColliderComponent:
517
518 return ComponentManager::safeCast(VUEngine::getColliderManager());
519 break;
520
521 case kPhysicsComponent:
522
523 return ComponentManager::safeCast(VUEngine::getBodyManager());
524 break;
525
526 case kWireframeComponent:
527
528 return ComponentManager::safeCast(WireframeManager::getInstance());
529 break;
530
531 case kBehaviorComponent:
532
533 return ComponentManager::safeCast(BehaviorManager::getInstance());
534 break;
535 }
536
537 return NULL;
538}

542static uint32 ComponentManager::getComponentType(Component component)
543{
544 if(isDeleted(component))
545 {
546 return kComponentTypes;

548

554 }
555
556 if(__GET_CAST(Sprite, component))
557 {
558 return kSpriteComponent;
559 }
560
561 if(__GET_CAST(Wireframe, component))
562 {
563 return kWireframeComponent;
564 }
565
566 if(__GET_CAST(Behavior, component))
567 {
568 return kBehaviorComponent;
569 }
570
571 if(__GET_CAST(Body, component))
572 {
573 return kPhysicsComponent;
574 }
575 }
576
577 return component->componentSpec->componentType;

598static bool ComponentManager::getRightBoxFromComponents(ComponentManager componentMananager, Entity owner, RightBox* rightBox)
599{
600 if(NULL == rightBox)
601 {
602 return false;
603 }
604
605 bool modified = false;
606
607 for(VirtualNode node = componentMananager->components->head; node; node = node->next)
608 {
609 Component component = Component::safeCast(node->data);
610
611 if(owner != component->owner)
612 {
613 continue;
614 }
615
616 modified = true;
617
618 RightBox componentRightBox = Component::getRightBox(component);
619
620 NM_ASSERT(componentRightBox.x0 < componentRightBox.x1, "ComponentManager::getRightBoxFromComponents: 0 width");
621 NM_ASSERT(componentRightBox.y0 < componentRightBox.y1, "ComponentManager::getRightBoxFromComponents: 0 height");
622 NM_ASSERT(componentRightBox.z0 < componentRightBox.z1, "ComponentManager::getRightBoxFromComponents: 0 depth");
623
624 if(rightBox->x0 > componentRightBox.x0)
625 {
626 rightBox->x0 = componentRightBox.x0;
627 }
628
629 if(rightBox->x1 < componentRightBox.x1)
630 {
631 rightBox->x1 = componentRightBox.x1;
632 }
633
634 if(rightBox->y0 > componentRightBox.y0)
635 {
636 rightBox->y0 = componentRightBox.y0;
637 }
638
639 if(rightBox->y1 < componentRightBox.y1)
640 {
641 rightBox->y1 = componentRightBox.y1;
642 }
643
644 if(rightBox->z0 > componentRightBox.z0)
645 {
646 rightBox->z0 = componentRightBox.z0;
647 }
648
649 if(rightBox->z1 < componentRightBox.z1)
650 {
651 rightBox->z1 = componentRightBox.z1;
652 }
653 }
654
655 return modified;
656}
