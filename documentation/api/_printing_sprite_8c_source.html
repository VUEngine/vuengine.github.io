49int16 PrintingSprite::doRender(int16 index)
50{
51 WorldAttributes* worldPointer = &_worldAttributesCache[index];
52
53 worldPointer->mx = this->bgmapTextureSource.mx;
54 worldPointer->mp = this->bgmapTextureSource.mp;
55 worldPointer->my = this->bgmapTextureSource.my;
56 worldPointer->gx = this->position.x;
57 worldPointer->gp = this->position.parallax;
58 worldPointer->gy = this->position.y;
59 worldPointer->w = this->halfWidth << 1;
60 worldPointer->h = this->halfHeight << 1;
61 worldPointer->head = __WORLD_ON | __WORLD_BGMAP | __WORLD_OVR | this->printingBgmapSegment;
62
63 return index;
64}

68void PrintingSprite::reset()
69{
70 this->position.x = 0;
71 this->position.y = 0;
72 this->position.parallax = 0;
73
74 this->bgmapTextureSource.mx = __PRINTING_BGMAP_X_OFFSET;
75 this->bgmapTextureSource.my = __PRINTING_BGMAP_Y_OFFSET;
76 this->bgmapTextureSource.mp = __PRINTING_BGMAP_PARALLAX_OFFSET;
77
78 this->halfWidth = __SCREEN_WIDTH >> 1;
79 this->halfHeight = __SCREEN_HEIGHT >> 1;
80}
