122static uint32 Texture::getTotalCols(TextureSpec* textureSpec)
123{
124 if(NULL == textureSpec->charSetSpec)
125 {
126 return 0;
127 }
128
129 if(!Texture::isSpecSingleFrame(textureSpec))
130 {
131 uint32 maximumNumberOfFrames = 64 / textureSpec->cols;
132
133 if(maximumNumberOfFrames > textureSpec->numberOfFrames)
134 {
135 return textureSpec->numberOfFrames * textureSpec->cols;
136 }
137
138 return maximumNumberOfFrames * textureSpec->cols;
139 }
140
141 return textureSpec->cols;
142}

144//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
145
146static uint32 Texture::getTotalRows(TextureSpec* textureSpec)
147{
148 if(NULL == textureSpec->charSetSpec)
149 {
150 return 0;
151 }
152
153 if(!Texture::isSpecSingleFrame(textureSpec))
154 {
155 int16 allocableFrames = 64 / textureSpec->cols;
156 int16 neededRows =
157 __FIXED_TO_I(__FIXED_DIV(__I_TO_FIXED(textureSpec->numberOfFrames), __I_TO_FIXED(allocableFrames)) + __05F_FIXED) - 1;
158
159 // Return the total number of chars
160 return textureSpec->rows + textureSpec->rows * (0 < neededRows ? neededRows : 0);
161 }
162
163 return textureSpec->rows;
164}

167
168static void Texture::doUpdateTextures(int16 maximumTextureRowsToWrite)
169{
170 if(NULL == _texturesToUpdate)
171 {
172 return;
173 }
174
175 for(VirtualNode node = _texturesToUpdate->head, nextNode = NULL; NULL != node; node = nextNode)
176 {
177 nextNode = node->next;
178
179 Texture texture = Texture::safeCast(node->data);
180
181 NM_ASSERT(__GET_CAST(Texture, texture), "Texture::updateTextures: invalid texture");
182 NM_ASSERT(NULL != texture->textureSpec, "Texture::updateTextures: invalid texture spec");
183
184 bool remove = NULL == texture->textureSpec || (texture->update && Texture::update(texture, maximumTextureRowsToWrite));
185
186 if(remove)
187 {
188 VirtualList::removeNode(_texturesToUpdate, node);
189 texture->update = false;
190 }
191 }
192}

195
196static void Texture::doUpdateTexturesDeferred()
197{
198 if(NULL == _texturesToUpdate)
199 {
200 return;
201 }
202
203 for(VirtualNode node = _texturesToUpdate->head, nextNode = NULL; NULL != node; node = nextNode)
204 {
205 nextNode = node->next;
206
207 Texture texture = Texture::safeCast(node->data);
208
209 NM_ASSERT(__GET_CAST(Texture, texture), "Texture::updateTextures: invalid texture");
210
211 bool remove = NULL == texture->textureSpec || (texture->update && Texture::update(texture, -1));
212
213 if(remove)
214 {
215 texture->update = false;
216 VirtualList::removeNode(_texturesToUpdate, node);
217
218 if(NULL != texture->textureSpec)
219 {
220 break;
221 }
222 }
223 else if(texture->update)
224 {
225 break;
226 }
227 }
228}

278void Texture::constructor(const TextureSpec* textureSpec, uint16 id)
279{
280 if(NULL == _texturesToUpdate)
281 {
282 _texturesToUpdate = new VirtualList();
283 }
284
285 // Always explicitly call the base's constructor
286 Base::constructor();
287
288 // Set id
289 this->id = id;
290
291 this->doUpdate = NULL;
292
293 this->mapDisplacement = 0;
294 this->usageCount = 1;
295
296 // Save the bgmap spec's address
297 this->textureSpec = textureSpec;
298 this->charSet = NULL;
299 // Set the palette
300 this->palette = textureSpec->palette;
301 this->status = kTextureInvalid;
302 this->frame = 0;
303 this->update = false;
304}

334void Texture::setSpec(TextureSpec* textureSpec)
335{
336 ASSERT(textureSpec, "Texture::setSpec: null textureSpec");
337
338 if(NULL == textureSpec)
339 {
340 return;
341 }
342
343 if(this->textureSpec != textureSpec || kTextureWritten != this->status)
344 {
345 if(NULL != this->charSet && textureSpec->charSetSpec != CharSet::getSpec(this->charSet))
346 {
347 Texture::releaseCharSet(this);
348 }
349
350 this->textureSpec = textureSpec;
351 this->frame = 0;
352 this->mapDisplacement = 0;
353 this->palette = this->textureSpec->palette;
354 this->status = kTexturePendingWriting;
355 Texture::rewrite(this);
356 }
357}

569
570bool Texture::write(int16 maximumTextureRowsToWrite __attribute__((unused)))
571{
572 ASSERT(this->textureSpec, "Texture::write: null textureSpec");
573 ASSERT(this->textureSpec->charSetSpec, "Texture::write: null charSetSpec");
574
575 if(isDeleted(this->charSet))
576 {
577 Texture::loadCharSet(this);
578
579 if(isDeleted(this->charSet))
580 {
581 this->status = kTextureInvalid;
582 return false;
583 }
584 }
585
586 CharSet::setFrame(this->charSet, this->frame);
587
588 if(CharSet::isOptimized(this->charSet))
589 {
590 this->mapDisplacement = this->textureSpec->cols * this->textureSpec->rows * this->frame;
591 }
592
593 this->status = kTextureWritten;
594 return true;
595}

598
599void Texture::rewrite()
600{
601 bool statusChanged = kTexturePendingRewriting != this->status;
602
603 this->status = this->status > kTexturePendingWriting ? kTexturePendingRewriting : this->status;
604
605 if(!this->update || (statusChanged && kTexturePendingRewriting == this->status))
606 {
607 // Prepare the texture right away just in case the call initiates
608 // At a defragmentation process
609 Texture::prepare(this);
610 }
611}

620
621void Texture::loadCharSet()
622{
623 if(!isDeleted(this->charSet))
624 {
625 return;
626 }
627
628 if(NULL == this->textureSpec || NULL == this->textureSpec->charSetSpec)
629 {
630 return;
631 }
632
633 this->charSet = CharSet::get(this->textureSpec->charSetSpec);
634
635 if(isDeleted(this->charSet))
636 {
637 return;
638 }
639
640 this->status = kTexturePendingWriting;
641
642 Texture::setupUpdateFunction(this);
643
644 CharSet::addEventListener
645 (
646 this->charSet, ListenerObject::safeCast(this), (EventListener)Texture::onCharSetChangedOffset, kEventCharSetChangedOffset
647 );
648
649 CharSet::addEventListener
650 (
651 this->charSet, ListenerObject::safeCast(this), (EventListener)Texture::onCharSetDeleted, kEventCharSetDeleted
652 );
653}

680
681 this->charSet = NULL;

685//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
686
687bool Texture::update(int16 maximumTextureRowsToWrite)
688{
689 switch(this->status)
690 {
691 case kTexturePendingWriting:
692
693 Texture::write(this, maximumTextureRowsToWrite);
694 break;
695
696 case kTexturePendingRewriting:
697
698 Texture::write(this, maximumTextureRowsToWrite);
699
700 if(kTextureWritten == this->status)
701 {
702 Texture::fireEvent(this, kEventTextureRewritten);
703
704 NM_ASSERT(!isDeleted(this), "Texture::prepare: deleted this during kEventTextureRewritten");
705 }
706
707 break;
708
709 case kTextureMapDisplacementChanged:
710
711 Texture::write(this, maximumTextureRowsToWrite);
712
713 if(kTextureWritten != this->status)
714 {
715 break;
716 }
717
718 Texture::fireEvent(this, kEventTextureRewritten);
719 NM_ASSERT(!isDeleted(this), "Texture::prepare: deleted this during kEventTextureRewritten");
720
721 // Intended fall through
722
723 default:
724
725 if(isDeleted(this->charSet))
726 {
727 Texture::write(this, maximumTextureRowsToWrite);
728 }
729 else if(NULL != this->doUpdate)
730 {
731 this->doUpdate(this, maximumTextureRowsToWrite);
732 }
733 else
734 {
735 this->status = kTextureWritten;
736 }
737
738 break;
739 }
740
741 return kTextureWritten == this->status;
742}
