37static int16 BgmapSprite::doApplyAffineTransformations(BgmapSprite bgmapSprite, int32 maximumParamTableRowsToComputePerCall)
38{
39 ASSERT(bgmapSprite->texture, "BgmapSprite::doApplyAffineTransformations: null texture");
40
41 if(0 < bgmapSprite->param)
42 {
43 return Affine::transform
44 (
45 maximumParamTableRowsToComputePerCall,
46 bgmapSprite->param,
47 bgmapSprite->paramTableRow,
48 // Geometrically accurate, but kills the CPU
49 // (0 > bgmapSprite->position.x? bgmapSprite->position.x : 0) + bgmapSprite->halfWidth,
50 // (0 > bgmapSprite->position.y? bgmapSprite->position.y : 0) + bgmapSprite->halfHeight,
51 // Don't do translations
52 // Provide a little bit of performance gain by only calculation transformation equations
53 // For the visible rows, but causes that some sprites not be rendered completely when the
54 // Camera moves vertically
55 // Int32 lastRow = height + worldPointer->gy >= _cameraFrustum->y1 ? _cameraFrustum->y1 - worldPointer->gy + myDisplacement: height;
56 // This->paramTableRow = this->paramTableRow ? this->paramTableRow : myDisplacement;
57 __I_TO_FIXED(bgmapSprite->halfWidth),
58 __I_TO_FIXED(bgmapSprite->halfHeight),
59 __I_TO_FIX13_3(bgmapSprite->bgmapTextureSource.mx),
60 __I_TO_FIX13_3(bgmapSprite->bgmapTextureSource.my),
61 __I_TO_FIXED(bgmapSprite->texture->textureSpec->cols << 2),
62 __I_TO_FIXED(bgmapSprite->texture->textureSpec->rows << 2),
63 &bgmapSprite->rotation
64 );
65 }
66
67 return bgmapSprite->paramTableRow;
68}

79{
80 NM_ASSERT(NULL != bgmapSpriteSpec, "BgmapSprite::constructor: NULL bgmapSpriteSpec");

139void BgmapSprite::processEffects(int32 maximumParamTableRowsToComputePerCall)
140{
141 // Set the world size according to the zoom
142 if(0 < this->param && (uint8)__NO_RENDER_INDEX != this->index)
143 {
144 if(0 != ((__WORLD_AFFINE | __WORLD_HBIAS) & this->head) && NULL != this->applyParamTableEffect)
145 {
146 if(0 <= this->paramTableRow)
147 {
148 // Apply affine transformation
149 this->paramTableRow = this->applyParamTableEffect(this, maximumParamTableRowsToComputePerCall);
150
151 if(0 > this->paramTableRow)
152 {
153 this->paramTableRow = -1;
154 }
155 }
156 }
157 }

162int16 BgmapSprite::doRender(int16 index)
163{
164 if(isDeleted(this->texture))
165 {
166 BgmapSprite::configureTexture(this);
167
168 return __NO_RENDER_INDEX;
169 }
170
171 NM_ASSERT(!isDeleted(this->texture), "BgmapSprite::doRender: null texture");
172
173 WorldAttributes* worldPointer = &_worldAttributesCache[index];
174
175 // Get coordinates
176 int16 gx = this->position.x + this->displacement.x - this->halfWidth;
177 int16 gy = this->position.y + this->displacement.y - this->halfHeight;
178 int16 gp = this->position.parallax + this->displacement.parallax;
179
180 int16 auxGp = __ABS(gp);
181
182 // Get sprite's size
183 int16 w = this->halfWidth << 1;
184 int16 h = this->halfHeight << 1;
185
186 // Set the head
187 int32 mx = this->bgmapTextureSource.mx;
188 int32 my = this->bgmapTextureSource.my;
189 int32 mp = this->bgmapTextureSource.mp;
190
191 // Cap coordinates to camera space
192 if(_cameraFrustum->x0 - auxGp > gx)
193 {
194 if(0 == this->param)
195 {
196 mx += (_cameraFrustum->x0 - auxGp - gx);
197 w -= (_cameraFrustum->x0 - auxGp - gx);
198 gx = _cameraFrustum->x0 - auxGp;
199 }
200 }
201
202 int16 myDisplacement = 0;
203
204 if(_cameraFrustum->y0 > gy)
205 {
206 myDisplacement = (_cameraFrustum->y0 - gy);
207
208 my += myDisplacement;
209 h -= (_cameraFrustum->y0 - gy);
210 gy = _cameraFrustum->y0;
211 }
212
213 if(w + gx >= _cameraFrustum->x1 + auxGp)
214 {
215 w = _cameraFrustum->x1 - gx + auxGp;
216 }
217
218 if (__WORLD_SIZE_DISPLACEMENT >= w)
219 {
220 return __NO_RENDER_INDEX;
221 }
222
223/*
224 if(_cameraFrustum->y0 > h + gy)
225 {
226 worldPointer->head = __WORLD_OFF;
227#ifdef __PROFILE_GAME
228 worldPointer->w = 0;
229 worldPointer->h = 0;
230#endif
231 }
232*/
233 if(h + gy >= _cameraFrustum->y1)
234 {
235 h = _cameraFrustum->y1 - gy;
236 }
237
238#ifdef __HACK_BGMAP_SPRITE_HEIGHT
239 if (__MINIMUM_BGMAP_SPRITE_HEIGHT >= h && 0 == gy)
240 {
241 if (__WORLD_SIZE_DISPLACEMENT >= h)
242 {
243 return __NO_RENDER_INDEX;
244 }
245
246 my -= __MINIMUM_BGMAP_SPRITE_HEIGHT - h;
247 }
248#else
249 if (__WORLD_SIZE_DISPLACEMENT >= h)
250 {
251 return __NO_RENDER_INDEX;
252 }
253#endif
254
255 worldPointer->gx = gx;
256 worldPointer->gy = gy;
257 worldPointer->gp = gp;
258
259 worldPointer->mx = mx;
260 worldPointer->my = my;
261 worldPointer->mp = mp;
262
263 worldPointer->w = w - __WORLD_SIZE_DISPLACEMENT;
264 worldPointer->h = h - __WORLD_SIZE_DISPLACEMENT;
265
266 worldPointer->head = this->head | (BgmapTexture::safeCast(this->texture))->segment;
267
268 if(0 < this->param)
269 {
270 worldPointer->param = (uint16)((((this->param + (myDisplacement << 4))) - 0x20000) >> 1) & 0xFFF0;
271 }
272
273 return index;
274}

278void BgmapSprite::setMultiframe(uint16 frame)
279{
280 int16 mx = BgmapTexture::getXOffset(this->texture);
281 int16 my = BgmapTexture::getYOffset(this->texture);
282
283 int16 cols = Texture::getCols(this->texture);
284
285 int16 allocableFrames = (64 - mx) / cols;
286 int16 usableCollsPerRow = allocableFrames * cols;
287 int16 usedCollsPerRow = frame * cols;
288
289 int16 col = usedCollsPerRow % usableCollsPerRow;
290 int16 row = Texture::getRows(this->texture) * (frame / allocableFrames);
291
292 this->bgmapTextureSource.mx = mx + (col << 3);
293 this->bgmapTextureSource.my = my + (row << 3);
294 this->rendered = false;
295}

299void BgmapSprite::setRotation(const Rotation* rotation)
300{
301 if(NULL == rotation)
302 {
303 return;
304 }
305
306 this->rotation = *rotation;
307
308 if(0 < this->param)
309 {
310 this->paramTableRow = -1 == this->paramTableRow ? 0 : this->paramTableRow;
311
312 // Scale the texture in the next render cycle
313 BgmapSprite::invalidateParamTable(this);
314 }
315 else if(!isDeleted(this->texture))
316 {
317 NormalizedDirection normalizedDirection =
318 {
319 __QUARTER_ROTATION_DEGREES < __ABS(rotation->y) || __QUARTER_ROTATION_DEGREES < __ABS(rotation->z) ? __LEFT : __RIGHT,
320 __QUARTER_ROTATION_DEGREES < __ABS(rotation->x) || __QUARTER_ROTATION_DEGREES < __ABS(rotation->z) ? __UP : __DOWN,
321 __FAR,
322 };
323
324 if(__LEFT == normalizedDirection.x)
325 {
326 BgmapTexture::setHorizontalFlip(this->texture, true);
327 }
328 else if(__RIGHT == normalizedDirection.x)
329 {
330 BgmapTexture::setHorizontalFlip(this->texture, false);
331 }
332
333 if(__UP == normalizedDirection.y)
334 {
335 BgmapTexture::setVerticalFlip(this->texture, true);
336 }
337 else if(__DOWN == normalizedDirection.y)
338 {
339 BgmapTexture::setVerticalFlip(this->texture, false);
340 }
341 }
342}

346void BgmapSprite::setScale(const PixelScale* scale)
347{
348 if(NULL == scale)
349 {
350 return;
351 }
352
353 this->scale = *scale;
354
355 if(0 != (__WORLD_AFFINE & this->head))
356 {
357 PixelScale scaleHelper = *scale;
358
359 NM_ASSERT(0 < scaleHelper.x, "Sprite::scale: 0 scale x");
360 NM_ASSERT(0 < scaleHelper.y, "Sprite::scale: 0 scale y");
361
362 extern Rotation _previousCameraInvertedRotation;
363
364 Vector3D vector = Vector3D::rotate(Vector3D::getRelativeToCamera(this->transformation->position), _previousCameraInvertedRotation);
365
366 fix7_9 ratio = __FIXED_TO_FIX7_9(Vector3D::getScale(vector.z, true));
367
368 ratio = 0 > ratio? __1I_FIX7_9 : ratio;
369 ratio = __I_TO_FIX7_9(__MAXIMUM_SCALE) < ratio? __I_TO_FIX7_9(__MAXIMUM_SCALE) : ratio;
370
371 scaleHelper.x = __FIX7_9_MULT(scaleHelper.x, ratio);
372 scaleHelper.y = __FIX7_9_MULT(scaleHelper.y, ratio);
373
374 this->rendered = false;
375
376 if(!isDeleted(this->texture))
377 {
378 // Add 1 pixel to the width and 7 to the height to avoid cutting off the graphics
379 this->halfWidth =
380 __FIXED_TO_I
381 (
382 __ABS
383 (
384 __FIXED_MULT
385 (
386 __FIX7_9_TO_FIXED(__COS(__FIXED_TO_I(this->transformation->rotation.y))),
387 __FIXED_MULT
388 (
389 __I_TO_FIXED((int32)this->texture->textureSpec->cols << 2),
390 __FIX7_9_TO_FIXED(scaleHelper.x)
391 )
392 )
393 )
394 ) + 1;
395
396 this->halfHeight =
397 __FIXED_TO_I
398 (
399 __ABS
400 (
401 __FIXED_MULT
402 (
403 __FIX7_9_TO_FIXED(__COS(__FIXED_TO_I(this->transformation->rotation.x))),
404 __FIXED_MULT
405 (
406 __I_TO_FIXED((int32)this->texture->textureSpec->rows << 2),
407 __FIX7_9_TO_FIXED(scaleHelper.y)
408 )
409 )
410 )
411 ) + 1;
412 }
413
414 if(0 < this->param)
415 {
416 this->paramTableRow = -1 == this->paramTableRow ? 0 : this->paramTableRow;
417 }
418 }
419}

435void BgmapSprite::configureTexture()
436{
437 if(!isDeleted(this->texture))
438 {
439 return;
440 }
441
442 TextureSpec* textureSpec =((BgmapSpriteSpec*)this->componentSpec)->spriteSpec.textureSpec;
443
444 if(NULL == textureSpec)
445 {
446 return;
447 }
448
449 this->texture = Texture::get(typeofclass(BgmapTexture), textureSpec, 0, false, __WORLD_1x1);
450
451 if(isDeleted(this->texture))
452 {
453 return;
454 }
455
456 this->bgmapTextureSource.mx = BgmapTexture::getXOffset(this->texture) << 3;
457 this->bgmapTextureSource.my = BgmapTexture::getYOffset(this->texture) << 3;
458 this->bgmapTextureSource.mp = 0;
459
460 BgmapSprite::setMode(this, ((BgmapSpriteSpec*)this->componentSpec)->display, ((BgmapSpriteSpec*)this->componentSpec)->bgmapMode);
461
462 if(0 != this->param && !isDeleted(this->texture))
463 {
464 Texture::addEventListener
465 (
466 this->texture, ListenerObject::safeCast(this), (EventListener)BgmapSprite::onTextureRewritten, kEventTextureRewritten
467 );
468 }
469}

472
473void BgmapSprite::setMode(uint16 display, uint16 mode)
474{
475 this->head &= ~(__WORLD_BGMAP | __WORLD_AFFINE | __WORLD_HBIAS);
476
477 if(((__WORLD_AFFINE | __WORLD_HBIAS) & this->head) && 0 != this->param)
478 {
479 // Free param table space
480 ParamTableManager::free(this);
481
482 this->param = 0;
483 }
484
485 if(0 == this->param && !isDeleted(this->texture))
486 {
487 switch(mode)
488 {
489 case __WORLD_BGMAP:
490
491 // Set map head
492 this->head = display | __WORLD_BGMAP;
493 break;
494
495 case __WORLD_AFFINE:
496
497 this->head = display | __WORLD_AFFINE;
498 this->param = ParamTableManager::allocate(this);
499 this->applyParamTableEffect =
500 NULL != this->applyParamTableEffect ? this->applyParamTableEffect : BgmapSprite::doApplyAffineTransformations;
501 break;
502
503 case __WORLD_HBIAS:
504
505 // Set map head
506 this->head = display | __WORLD_HBIAS;
507
508 if(NULL != this->applyParamTableEffect)
509 {
510 this->param = ParamTableManager::allocate(this);
511 }
512
513 break;
514 }
515 }
516
517 this->head &= ~__WORLD_END;
518}

604void BgmapSprite::releaseTexture()
605{
606 // Free the texture
607 if(!isDeleted(this->texture))
608 {
609 // If affine or bgmap
610 if(((__WORLD_AFFINE | __WORLD_HBIAS) & this->head) && 0 != this->param)
611 {
612 // Free param table space
613 ParamTableManager::free(this);
614 }
615
616 if(0 != this->param)
617 {
618 Texture::removeEventListener
619 (
620 this->texture, ListenerObject::safeCast(this), (EventListener)BgmapSprite::onTextureRewritten, kEventTextureRewritten
621 );
622 }
623
624 Texture::release(this->texture);
625 }
626
627 this->texture = NULL;
628}
