66bool Entity::handleMessage(Telegram telegram)
67{
68 switch(Telegram::getMessage(telegram))
69 {
70 case kMessageBodyStartedMoving:
71
72 Entity::checkCollisions(this, true);
73 return true;
74 break;
75
76 case kMessageBodyStopped:
77
78 if(NULL == this->body || __NO_AXIS == Body::getMovementOnAllAxis(this->body))
79 {
80 Entity::checkCollisions(this, false);
81 }
82 break;
83 }
84
85 return false;
86}

90void Entity::clearComponentLists(uint32 componentType)
91{
92 if(NULL == this->components)
93 {
94 return;
95 }
96
97 if(kComponentTypes <= componentType)
98 {
99 for(int16 i = 0; i < kComponentTypes; i++)
100 {
101 if(NULL != this->components[i])
102 {
103 delete this->components[i];
104 this->components[i] = NULL;
105 }
106 }
107
108 delete this->components;
109 this->components = NULL;
110 }
111 else if(NULL != this->components[componentType])
112 {
113 delete this->components[componentType];
114 this->components[componentType] = NULL;
115 }
116}

435
436void Entity::addedComponent(Component component)
437{
438 switch(Component::getType(component))
439 {
440 case kSpriteComponent:
441 case kWireframeComponent:
442 {
443 if(Entity::overrides(this, calculateSize))
444 {
445 Entity::calculateSize(this);
446 }
447
448 break;
449 }
450
451 case kPhysicsComponent:
452 {
453 NM_ASSERT(NULL == this->body, "Entity::addedComponent: adding multiple bodies");
454
455 if(NULL != this->body)
456 {
457 return;
458 }
459
460 this->body = Body::safeCast(Object::getCast(component, typeofclass(Body), NULL));
461
462 if(!isDeleted(this->body))
463 {
464 Body::setPosition(this->body, &this->transformation.position, Entity::safeCast(this));
465 }
466
467 break;
468 }
469 }
470}

607
608bool Entity::canMoveTowards(Vector3D direction)
609{
610 fixed_t collisionCheckDistance = __PIXELS_TO_METERS(4);
611
612 Vector3D displacement =
613 {
614 direction.x ? 0 < direction.x ? collisionCheckDistance : -collisionCheckDistance : 0,
615 direction.y ? 0 < direction.y ? collisionCheckDistance : -collisionCheckDistance : 0,
616 direction.z ? 0 < direction.z ? collisionCheckDistance : -collisionCheckDistance : 0
617 };
618
619 bool canMove = true;
620
621 VirtualList colliders = Entity::getComponents(this, kColliderComponent);
622
623 if(NULL != colliders)
624 {
625 VirtualNode node = colliders->head;
626
627 for(; NULL != node; node = node->next)
628 {
629 Collider collider = Collider::safeCast(node->data);
630 canMove &= Collider::canMoveTowards(collider, displacement);
631 }
632 }
633
634 return canMove;
635}

668//——————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————————
669
670bool Entity::collisionStarts(const CollisionInformation* collisionInformation)
671{
672 ASSERT(collisionInformation->otherCollider, "Entity::collisionStarts: otherColliders");
673
674 ASSERT(NULL != collisionInformation->otherCollider, "Particle::collisionStarts: otherColliders");
675
676 bool returnValue = false;
677
678 if(collisionInformation->collider && collisionInformation->otherCollider)
679 {
680 if(collisionInformation->solutionVector.magnitude)
681 {
682 Collider::resolveCollision(collisionInformation->collider, collisionInformation);
683
684 Entity collidingEntity = Collider::getOwner(collisionInformation->otherCollider);
685
686 fixed_t bounciness =
687 Entity::isSensibleToCollidingObjectBouncinessOnCollision(this, collidingEntity) ?
688 Entity::getBounciness(collidingEntity) : 0;
689
690 fixed_t frictionCoefficient =
691 Entity::isSensibleToCollidingObjectFrictionOnCollision(this, collidingEntity) ?
692 Entity::getSurroundingFrictionCoefficient(this) : 0;
693
694 if(!isDeleted(this->body))
695 {
696 Body::bounce
697 (
698 this->body,
699 ListenerObject::safeCast(collisionInformation->otherCollider),
700 collisionInformation->solutionVector.direction,
701 frictionCoefficient,
702 bounciness
703 );
704 }
705
706 returnValue = true;
707 }
708 }
709
710 return returnValue;
711}
