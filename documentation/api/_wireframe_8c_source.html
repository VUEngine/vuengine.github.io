35void Wireframe::constructor(Entity owner, const WireframeSpec* wireframeSpec)
36{
37 // Always explicitly call the base's constructor
38 Base::constructor(owner, (const VisualComponentSpec*)&wireframeSpec->visualComponentSpec);
39
40 if(NULL == wireframeSpec)
41 {
42 this->color = __COLOR_BRIGHT_RED;
43 this->displacement = Vector3D::zero();
44 this->interlaced = false;
45 this->transparency = __TRANSPARENCY_NONE;
46 }
47 else
48 {
49 this->color = wireframeSpec->color;
50 this->displacement = ((WireframeSpec*)this->componentSpec)->displacement;
51 this->interlaced = ((WireframeSpec*)this->componentSpec)->interlaced;
52 this->transparency = wireframeSpec->transparency;
53 }
54
55 this->bufferIndex = 0;
56 this->squaredDistanceToCamera = 0;
57 this->rendered = false;
58 this->drawn = false;
59}

87bool Wireframe::prepareForRender(Vector3D* relativePosition)
88{
89 Vector3D displacement = Vector3D::rotate(this->displacement, this->transformation->rotation);
90 *relativePosition = Vector3D::sub(Vector3D::sum(this->transformation->position, displacement), _previousCameraPosition);
91
92 if(NULL == ((WireframeSpec*)this->componentSpec))
93 {
94 this->color = __COLOR_BRIGHT_RED;
95 this->interlaced = false;
96 this->squaredDistanceToCamera = 0;
97 return __COLOR_BLACK != this->color;
98 }
99
100 if(__COLOR_BLACK != ((WireframeSpec*)this->componentSpec)->color)
101 {
102 this->color = ((WireframeSpec*)this->componentSpec)->color;
103 this->interlaced = ((WireframeSpec*)this->componentSpec)->interlaced;
104#ifdef __WIREFRAME_MANAGER_SORT_FOR_DRAWING
105 this->squaredDistanceToCamera = __WIREFRAME_MAXIMUM_SQUARE_DISTANCE_TO_CAMERA;
106#endif
107 return __COLOR_BLACK != this->color;
108 }
109
110 NM_ASSERT(NULL != ((WireframeSpec*)this->componentSpec), "Wireframe::setupRenderingMode: NULL wireframeSpec");
111
112 this->color = ((WireframeSpec*)this->componentSpec)->color;
113
114 fixed_ext_t distanceToCamera = Vector3D::squareLength(*relativePosition);
115
116 if(__COLOR_BLACK != this->color)
117 {
118 this->interlaced = ((WireframeSpec*)this->componentSpec)->interlaced;
119
120 if(__FIXED_SQUARE(__DIRECT_DRAW_INTERLACED_THRESHOLD) < distanceToCamera)
121 {
122 this->interlaced = true;
123 }
124 }
125 else
126 {
127 int16 cameraViewingAngle = __CAMERA_VIEWING_ANGLE;
128
129 if(__FIXED_SQUARE(__PIXELS_TO_METERS(__SCREEN_WIDTH << 1)) < distanceToCamera)
130 {
131 if
132 (
133 __COS(cameraViewingAngle)
134 >
135 __FIXED_EXT_TO_FIX7_9(Vector3D::dotProduct(Vector3D::normalize(*relativePosition), _cameraDirection))
136 )
137 {
138#ifdef __WIREFRAME_MANAGER_SORT_FOR_DRAWING
139 this->squaredDistanceToCamera = __WIREFRAME_MAXIMUM_SQUARE_DISTANCE_TO_CAMERA;
140#endif
141 this->color = __COLOR_BLACK;
142 return __COLOR_BLACK != this->color;
143 }
144 }
145
146 if(__FIXED_SQUARE((__DIRECT_DRAW_INTERLACED_THRESHOLD << 1) < distanceToCamera))
147 {
148 this->color = __COLOR_BLACK;
149 this->interlaced = true;
150 return __COLOR_BLACK != this->color;
151 }
152 else if(__FIXED_SQUARE(__DIRECT_DRAW_INTERLACED_THRESHOLD + (__DIRECT_DRAW_INTERLACED_THRESHOLD >> 1)) < distanceToCamera)
153 {
154 this->interlaced = true;
155 this->color = __COLOR_DARK_RED;
156 }
157 else if(__FIXED_SQUARE(__DIRECT_DRAW_INTERLACED_THRESHOLD + (__DIRECT_DRAW_INTERLACED_THRESHOLD >> 2)) < distanceToCamera)
158 {
159 this->interlaced = ((WireframeSpec*)this->componentSpec)->interlaced;
160 this->color = __COLOR_DARK_RED;
161 }
162 else if(__FIXED_SQUARE(__DIRECT_DRAW_INTERLACED_THRESHOLD + (__DIRECT_DRAW_INTERLACED_THRESHOLD >> 3)) < distanceToCamera)
163 {
164 this->interlaced = true;
165 this->color = __COLOR_MEDIUM_RED;
166 }
167 else if(__FIXED_SQUARE(__DIRECT_DRAW_INTERLACED_THRESHOLD) < distanceToCamera)
168 {
169 this->interlaced = true;
170 this->color = __COLOR_BRIGHT_RED;
171 }
172 else if(__FIXED_SQUARE(__DIRECT_DRAW_INTERLACED_THRESHOLD - (__DIRECT_DRAW_INTERLACED_THRESHOLD >> 1)) < distanceToCamera)
173 {
174 this->interlaced = ((WireframeSpec*)this->componentSpec)->interlaced;
175 this->color = __COLOR_MEDIUM_RED;
176 }
177 else
178 {
179 this->interlaced = ((WireframeSpec*)this->componentSpec)->interlaced;
180 this->color = __COLOR_BRIGHT_RED;
181 }
182 }
183
184 this->interlaced += ((WireframeSpec*)this->componentSpec)->interlaced;
185
186#ifdef __WIREFRAME_MANAGER_SORT_FOR_DRAWING
187 this->squaredDistanceToCamera = distanceToCamera;
188#endif
189
190 return __COLOR_BLACK != this->color;
191}
