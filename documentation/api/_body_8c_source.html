84static inline fixed_t Body::computeInstantaneousSpeed
85(
86 fixed_t forceMagnitude, fixed_t gravity, fixed_t mass, fixed_t friction, fixed_t maximumSpeed
87)
88{
89 fixed_t instantaneousSpeed =
90 __FIX7_9_EXT_TO_FIXED
91 (
92 Body::doComputeInstantaneousSpeed(forceMagnitude, gravity, mass, friction, BodyManager::getElapsedTimeStep())
93 );
94
95 return 0 != maximumSpeed && maximumSpeed < __ABS(instantaneousSpeed) ? maximumSpeed : instantaneousSpeed;
96}

106static inline fix7_9_ext Body::doComputeInstantaneousSpeed
107(
108 fixed_t forceMagnitude, fixed_t gravity, fixed_t mass, fixed_t friction, fix7_9_ext elapsedTime
109)
110{
111 fix7_9_ext acceleration =
112 __FIXED_TO_FIX7_9_EXT(gravity) + __FIX7_9_EXT_DIV(__FIXED_TO_FIX7_9_EXT(forceMagnitude + friction), __FIXED_TO_FIX7_9_EXT(mass));
113
114 return __FIX7_9_EXT_MULT(acceleration, elapsedTime);
115}

231void Body::clearNormal(ListenerObject referent)
232{
233 ASSERT(!isDeleted(referent), "Body::clearNormal: dead referent");
234
235 if(!isDeleted(referent) && !isDeleted(this->normals))
236 {
237 VirtualNode node = this->normals->head;
238
239 for(; NULL != node; node = node->next)
240 {
241 NormalRegistry* normalRegistry = (NormalRegistry*)node->data;
242
243 // Search if registry exists for the referent
244 // It is Ok if it doesn't exist
245 if(normalRegistry->referent == referent)
246 {
247 ASSERT(!isDeleted(normalRegistry), "Body::clearNormal: dead normal registry");
248 VirtualList::removeNode(this->normals, node);
249 delete normalRegistry;
250 break;
251 }
252 }
253
254 if(NULL != node)
255 {
256 Body::computeTotalNormal(this);
257 }
258 }
259}

263void Body::update(uint16 cycle, fix7_9_ext elapsedTime)
264{
265 if(!this->awake)
266 {
267 return;
268 }
269
270 MovementResult movementResult;
271
272 if(0 != this->skipCycles)
273 {
274 if(0 < this->skipCycles)
275 {
276 if(this->skipCycles > this->skipedCycles++)
277 {
278 return;
279 }
280
281 this->skipedCycles = 0;
282
283 movementResult = Body::updateMovement(this, cycle, elapsedTime);
284 }
285 else if(0 > this->skipCycles)
286 {
287 this->skipedCycles = 0;
288
289 while(this->skipCycles <= this->skipedCycles--)
290 {
291 movementResult = Body::updateMovement(this, cycle, elapsedTime);
292 }
293 }
294 }
295 else
296 {
297 movementResult = Body::updateMovement(this, cycle, elapsedTime);
298 }
299
300 if(!isDeleted(this->owner))
301 {
302 Entity::setPosition(this->owner, &this->position);
303 Entity::setDirection(this->owner, &this->direction);
304 }
305
306 // If stopped on any axis
307 if(0 != movementResult.axisStoppedMovement)
308 {
309 Body::stopMovement(this, movementResult.axisStoppedMovement);
310
311 if(!isDeleted(this->owner) && 0 != movementResult.axisStoppedMovement && this->sendMessages)
312 {
313 Body::sendMessageTo(this, ListenerObject::safeCast(this->owner), kMessageBodyStopped, 0, 0);
314 }
315 }
316
317 // Clear any force so the next update does not get influenced
318 Body::clearExternalForce(this);
319}

323uint8 Body::applyForce(const Vector3D* force)
324{
325 if(NULL == force)
326 {
327 return __NO_AXIS;
328 }
329
330 this->externalForce.x += force->x;
331 this->externalForce.y += force->y;
332 this->externalForce.z += force->z;
333
334 uint16 axisOfExternalForce = __NO_AXIS;
335
336 if(0 != force->x)
337 {
338 axisOfExternalForce |= __X_AXIS;
339 }
340
341 if(0 != force->y)
342 {
343 axisOfExternalForce |= __Y_AXIS;
344 }
345
346 if(0 != force->z)
347 {
348 axisOfExternalForce |= __Z_AXIS;
349 }
350
351 if(__NO_AXIS != axisOfExternalForce)
352 {
353 Body::clearNormalOnAxis(this, axisOfExternalForce);
354 Body::setMovementType(this, __ACCELERATED_MOVEMENT, axisOfExternalForce);
355 Body::awake(this, axisOfExternalForce);
356 }
357
358 return axisOfExternalForce;
359}

363void Body::bounce(ListenerObject bounceReferent, Vector3D bouncingPlaneNormal, fixed_t frictionCoefficient, fixed_t bounciness)
364{
365 Body::setSurroundingFrictionCoefficient(this, frictionCoefficient);
366
367 fixed_t cosAngle =
368 __I_TO_FIXED(bouncingPlaneNormal.x | bouncingPlaneNormal.y | bouncingPlaneNormal.z)
369 &&
370 (this->gravity.x | this->gravity.y | this->gravity.z) ?
371 __ABS(__FIXED_EXT_DIV(Vector3D::dotProduct(this->gravity, bouncingPlaneNormal),
372 Vector3D::lengthProduct(this->gravity, bouncingPlaneNormal)))
373 :
374 __1I_FIXED;
375
376 fixed_t normalMagnitude = __FIXED_EXT_MULT(Vector3D::length(Body::getWeight(this)), cosAngle);
377
378 Body::addNormal(this, bounceReferent, bouncingPlaneNormal, normalMagnitude);
379
380 if(0 < this->bounciness)
381 {
382 Vector3D velocity = this->velocity;
383
384 // Compute bouncing velocity vector
385 Vector3D u = Vector3D::scalarProduct(bouncingPlaneNormal, Vector3D::dotProduct(velocity, bouncingPlaneNormal));
386 Vector3D w = Vector3D::get(u, velocity);
387
388 bounciness += this->bounciness;
389
390 if(__I_TO_FIXED(0) > bounciness)
391 {
392 bounciness = 0;
393 }
394 else if(__F_TO_FIXED(__MAXIMUM_BOUNCINESS_COEFFICIENT) < bounciness)
395 {
396 bounciness = __F_TO_FIXED(__MAXIMUM_BOUNCINESS_COEFFICIENT);
397 }
398
399 if(0 > frictionCoefficient)
400 {
401 frictionCoefficient = 0;
402 }
403 else if(__MAXIMUM_FRICTION_COEFFICIENT < frictionCoefficient)
404 {
405 frictionCoefficient = __MAXIMUM_FRICTION_COEFFICIENT;
406 }
407
408 // Add bounciness and friction
409 // This is the physically correct computation, but causes
410 // Wrong angles of bouncing
411 u = Vector3D::scalarProduct(u, bounciness);
412 w = Vector3D::scalarProduct(w, (__MAXIMUM_FRICTION_COEFFICIENT - frictionCoefficient));
413
414 this->velocity = Vector3D::get(u, w);
415
416 this->internalVelocity.x = __FIXED_TO_FIX7_9_EXT(this->velocity.x);
417 this->internalVelocity.y = __FIXED_TO_FIX7_9_EXT(this->velocity.y);
418 this->internalVelocity.z = __FIXED_TO_FIX7_9_EXT(this->velocity.z);
419
420 if(__NO_MOVEMENT == this->movementType.x && 0 != this->velocity.x)
421 {
422 this->movementType.x = __ACCELERATED_MOVEMENT;
423 }
424
425 if(__NO_MOVEMENT == this->movementType.y && 0 != this->velocity.y)
426 {
427 this->movementType.y = __ACCELERATED_MOVEMENT;
428 }
429
430 if(__NO_MOVEMENT == this->movementType.z && 0 != this->velocity.z)
431 {
432 this->movementType.z = __ACCELERATED_MOVEMENT;
433 }
434
435 // Determine bouncing result
436 MovementResult movementResult = Body::getBouncingResult(this, velocity, bouncingPlaneNormal);
437
438 Body::clampVelocity(this, false);
439
440 // Stop over the axis where there is no bouncing
441 if(movementResult.axisStoppedMovement)
442 {
443 uint16 axisOfStopping = Body::stopMovement(this, movementResult.axisStoppedMovement);
444
445 if(!isDeleted(this->owner) && __NO_AXIS != axisOfStopping && this->sendMessages)
446 {
447 Body::sendMessageTo(this, ListenerObject::safeCast(this->owner), kMessageBodyStopped, 0, 0);
448 }
449 }
450
451 if(0 != movementResult.axisOfAcceleratedBouncing)
452 {
453 // Body::setSurroundingFrictionCoefficient(this, 0);
454 Body::clearNormalOnAxis(this, movementResult.axisOfAcceleratedBouncing);
455 }
456 }
457 else
458 {
459 Body::stopMovement(this, __ALL_AXIS);
460 }
461
462 if(__NO_AXIS == Body::getMovementOnAllAxis(this))
463 {
464 Body::sleep(this);
465 }
466 else
467 {
468 if(!isDeleted(this->owner))
469 {
470 Entity::setDirection(this->owner, &this->direction);
471 }
472 }
473}

477uint16 Body::stopMovement(uint16 axis)
478{
479 uint16 axisOfMovement = Body::getMovementOnAllAxis(this);
480 uint16 axisOfStopping = __NO_AXIS;
481
482 if(axis & __X_AXIS)
483 {
484 // Not moving anymore
485 this->velocity.x = 0;
486 this->internalVelocity.x = 0;
487 this->accelerating.x = false;
488 this->externalForce.x = 0;
489 axisOfStopping |= axisOfMovement & __X_AXIS;
490 }
491
492 if(axis & __Y_AXIS)
493 {
494 // Not moving anymore
495 this->velocity.y = 0;
496 this->internalVelocity.y = 0;
497 this->accelerating.y = false;
498 this->externalForce.y = 0;
499 axisOfStopping |= axisOfMovement & __Y_AXIS;
500 }
501
502 if(axis & __Z_AXIS)
503 {
504 // Not moving anymore
505 this->velocity.z = 0;
506 this->internalVelocity.z = 0;
507 this->accelerating.z = false;
508 this->externalForce.z = 0;
509 axisOfStopping |= axisOfMovement & __Z_AXIS;
510 }
511
512 Body::setMovementType(this, __NO_MOVEMENT, axisOfStopping);
513
514 if(!Body::getMovementOnAllAxis(this))
515 {
516 Body::sleep(this);
517 }
518
519 return axisOfStopping;
520}

525{
526 if(NULL == velocity)
527 {

531 uint16 axisOfUniformMovement = 0;

978
979MovementResult Body::getMovementResult(Vector3D previousVelocity)
980{
981 MovementResult movementResult = {__NO_AXIS, __NO_AXIS};
982
983 // Xor values, if result != 0, there is movement
984 // Xor values, if result >= 0, there is no change in direction
985 Vector3D movementChange =
986 {
987 this->velocity.x ^ previousVelocity.x,
988 this->velocity.y ^ previousVelocity.y,
989 this->velocity.z ^ previousVelocity.z,
990 };
991
992 if
993 (
994 !this->movesIndependentlyOnEachAxis
995 &&
996 0 != (__ACCELERATED_MOVEMENT & (this->movementType.x | this->movementType.y | this->movementType.z))
997 )
998 {
999 if
1000 (
1001 this->speed < __STOP_VELOCITY_THRESHOLD
1002 &&
1003 0 ==
1004 (
1005 this->externalForce.x | this->externalForce.y | this->externalForce.z |
1006 this->gravity.x | this->gravity.y | this->gravity.z
1007 )
1008 )
1009 {
1010 movementResult.axisStoppedMovement = __ALL_AXIS;
1011 }
1012
1013 return movementResult;
1014 }
1015
1016 // Stop if no external force or opposing normal force is present
1017 // And if the velocity minimum threshold is not reached
1018 if(0 != previousVelocity.x && 0 == this->externalForce.x && 0 == this->gravity.x && __ACCELERATED_MOVEMENT == this->movementType.x)
1019 {
1020 if
1021 (
1022 __STOP_VELOCITY_THRESHOLD > __ABS(this->velocity.x)
1023 ||
1024 (0 == this->externalForce.x && 0 == this->accelerating.x)
1025 ||
1026 (0 > movementChange.x)
1027 )
1028 {
1029 movementResult.axisStoppedMovement |= __X_AXIS;
1030 }
1031 }
1032
1033 if(0 != previousVelocity.y && 0 == this->externalForce.y && 0 == this->gravity.y && __ACCELERATED_MOVEMENT == this->movementType.y)
1034 {
1035 if
1036 (
1037 __STOP_VELOCITY_THRESHOLD > __ABS(this->velocity.y)
1038 ||
1039 (0 == this->externalForce.y && 0 == this->accelerating.y)
1040 ||
1041 (0 > movementChange.y)
1042 )
1043 {
1044 movementResult.axisStoppedMovement |= __Y_AXIS;
1045 }
1046 }
1047
1048 if(0 != previousVelocity.z && 0 == this->externalForce.z && 0 == this->gravity.z && __ACCELERATED_MOVEMENT == this->movementType.z)
1049 {
1050 if
1051 (
1052 __STOP_VELOCITY_THRESHOLD > __ABS(this->velocity.z)

1054 (0 == this->externalForce.z && 0 == this->accelerating.z)
1055 ||
1056 (0 > movementChange.z)
1057 )
1058 {
1059 movementResult.axisStoppedMovement |= __Z_AXIS;
1060 }
1061 }
1062
1063 return movementResult;

1068MovementResult Body::getBouncingResult(Vector3D previousVelocity, Vector3D bouncingPlaneNormal)
1069{
1070 MovementResult movementResult = {__NO_AXIS, __NO_AXIS};
1071
1072 // Xor values, if result != 0, there is movement
1073 // Xor values, if result >= 0, there is no change in direction

1077 this->velocity.y ^ previousVelocity.y,
1078 this->velocity.z ^ previousVelocity.z,

1080

1082 // And if there is possible movement in the other components
1083 if(__STOP_BOUNCING_VELOCITY_THRESHOLD > __ABS(this->velocity.x) && !__FIXED_INT_PART(bouncingPlaneNormal.y | bouncingPlaneNormal.z))
1084 {
1085 movementResult.axisStoppedMovement |= __X_AXIS;

1089 {
1090 movementResult.axisStoppedMovement |= __Y_AXIS;
1091 }
1092
1093 if(__STOP_BOUNCING_VELOCITY_THRESHOLD > __ABS(this->velocity.z) && !__FIXED_INT_PART(bouncingPlaneNormal.x | bouncingPlaneNormal.y))
1094 {
1095 movementResult.axisStoppedMovement |= __Z_AXIS;
1096 }
1097
1098 // Bounce accelerated if movement changed direction and the previous movement was not uniform
1099 if(__UNIFORM_MOVEMENT != this->movementType.x && 0 != movementChange.x)
1100 {
1101 movementResult.axisOfAcceleratedBouncing |= __X_AXIS;
1102 }
1103
1104 if(__UNIFORM_MOVEMENT != this->movementType.y && 0 != movementChange.y)
1105 {
1106 movementResult.axisOfAcceleratedBouncing |= __Y_AXIS;
1107 }
1108
1109 if(__UNIFORM_MOVEMENT != this->movementType.z && 0 != movementChange.z)
1110 {
1111 movementResult.axisOfAcceleratedBouncing |= __Z_AXIS;
1112 }
1113
1114 // Don't bounce if movement stopped on that axis
1115 movementResult.axisOfAcceleratedBouncing &= ~movementResult.axisStoppedMovement;
1116
1117 return movementResult;
