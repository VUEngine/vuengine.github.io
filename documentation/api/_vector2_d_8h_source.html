283{
284 if(left)
285 {
286 fixed_t aux = a.x;
287 a.x = -a.y;
288 a.y = aux;

294 a.y = -aux;

338static inline Vector2D Vector2D::projectOnto(Vector2D p, Vector2D a, Vector2D b)
339{
340 Vector2D ap = Vector2D::get(a, p);
341 Vector2D ab = Vector2D::get(a, b);
342 fixed_ext_t dotApAb = Vector2D::dotProduct(ap, ab);
343 fixed_ext_t dotAbAb = Vector2D::dotProduct(ab, ab);
344
345 if(!dotAbAb)
346 {
347 if(a.x == b.x)
348 {
349 p.x = a.x;
350 }
351
352 if(a.y == b.y)
353 {
354 p.y = a.y;
355 }
356
357 return p;
358 }
359
360 Vector2D projection =
361 {
362 a.x + __FIXED_EXT_TO_FIXED(__FIXED_EXT_MULT(__FIXED_TO_FIXED_EXT(ab.x), __FIXED_EXT_DIV(dotApAb, dotAbAb))),
363 a.y + __FIXED_EXT_TO_FIXED(__FIXED_EXT_MULT(__FIXED_TO_FIXED_EXT(ab.y), __FIXED_EXT_DIV(dotApAb, dotAbAb)))
364 };
365
366 return projection;
367}

376 fix19_13 dotAbAb = Vector2D::dotProductHighPrecision(ab, ab);
377
378 if(!dotAbAb)
379 {
380 if(a.x == b.x)

382 p.x = a.x;

384
385 if(a.y == b.y)

393 Vector2D projection =
394 {
395 a.x + __FIX19_13_TO_FIXED(__FIX19_13_MULT(__FIXED_TO_FIX19_13(ab.x), __FIX19_13_DIV(dotApAb, dotAbAb))),
396 a.y + __FIX19_13_TO_FIXED(__FIX19_13_MULT(__FIXED_TO_FIX19_13(ab.y), __FIX19_13_DIV(dotApAb, dotAbAb)))
397 };

404static inline Vector2D Vector2D::rotate(Vector2D vector, int16 degrees)
405{
406 Vector2D result = vector;
407
408 if(0 != degrees)
409 {
410 fix7_9_ext cos = __FIX7_9_TO_FIX7_9_EXT(__COS(degrees));
411 fix7_9_ext sin = __FIX7_9_TO_FIX7_9_EXT(__SIN(degrees));
412
413 fix7_9_ext x = __FIXED_TO_FIX7_9_EXT(vector.x);
414 fix7_9_ext y = __FIXED_TO_FIX7_9_EXT(vector.y);
415
416 result = (Vector2D)
417 {
418 __FIX7_9_EXT_TO_FIXED(__FIX7_9_EXT_MULT(x , cos) - __FIX7_9_EXT_MULT(y , sin)),
419 __FIX7_9_EXT_TO_FIXED(__FIX7_9_EXT_MULT(x , sin) + __FIX7_9_EXT_MULT(y , cos)),
420 };
421 }
422
423 return result;
424}

473static inline fixed_t Vector2D::lengthProduct(Vector2D vectorA, Vector2D vectorB)
474{
475 fixed_ext_t lengthSquareA = __FIXED_EXT_MULT(vectorA.x, vectorA.x) + __FIXED_EXT_MULT(vectorA.y, vectorA.y);
476 fixed_ext_t lengthSquareB = __FIXED_EXT_MULT(vectorB.x, vectorB.x) + __FIXED_EXT_MULT(vectorB.y, vectorB.y);
477
478 fixed_ext_t product = __FIXED_EXT_MULT(lengthSquareA, lengthSquareB);
479
480 return Math_squareRootFixed(product);

500static inline fixed_t Vector2D::getScale(fixed_t z, bool applyScalingMultiplier)
501{
502 if(0 == _optical->halfWidth)
503 {
504 return __1I_FIXED;
505 }
506
507 fixed_ext_t projectedWidth = 0;
508
509 if(applyScalingMultiplier)
510 {
511 if(0 == z + _optical->halfWidth)
512 {
513 return __1I_FIXED;
514 }
515
516 projectedWidth = __FIXED_EXT_DIV(__FIXED_EXT_MULT(_optical->halfWidth, _optical->scalingMultiplier), z + _optical->halfWidth);
517 }
518 else
519 {
520 if(0 == z + _optical->cameraNearPlane)
521 {
522 return __1I_FIXED;
523 }
524
525 projectedWidth =
526 __FIXED_EXT_DIV(__FIXED_EXT_MULT(_optical->halfWidth, _optical->projectionMultiplierHelper), z + _optical->cameraNearPlane);
527 }
528
529 return __FIXED_EXT_DIV(projectedWidth, _optical->halfWidth);
530}
